This button contains a basic label field to display.

Example :

     import lunas.display.button.LabelButton ;
     
     var bt:LabelButton = new LabelButton() ;
     
     bt.x               = 25 ;
     bt.y               = 50 ;
     bt.toggle          = true ;
     
     bt.label           = "hello world" ;
     
     addChild(bt) ;
