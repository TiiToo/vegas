    26  import vegas.logging.LogEventLevel;
    27  import vegas.logging.targets.LineFormattedTarget;
    28  
    29  /**
    30   * Provides a logger target that uses the XPanel console to output log messages. 
    31   * Thanks Farata System and <a href='http://www.faratasystems.com/xpanel/readme.pdf'>XPanel</a> console.
    32   * <p><b>Example :</b></p>
    33   * <p>
    34   * {@code
    35   * import vegas.logging.* ;
    36   * import vegas.logging.targets.XPanelTarget ;
    37   * 
    38   * var target:XPanelTarget = new XPanelTarget("Test") ;
    39   * target.includeLines = true ;
    40   * target.includeLevel = true ;
    41   * target.filters = ["namespace.*"] ;
    42   * target.level = LogEventLevel.ALL ; // level filter !
    43   * 
    44   * // create a log writer
    45   * var logger:ILogger = Log.getLogger("namespace.myApplication") ;
    46   * 
    47   * logger.log(LogEventLevel.DEBUG, "here is some myDebug info : {0} and {1}", 2.25 , true) ;
    48   * logger.debug("DEBUG !!") ;
    49   * logger.error("ERROR !!") ;
    50   * logger.fatal("FATAL !!") ;
    51   * logger.info("INFO !!") ;
    52   * logger.warn("WARNING !!") ;
    53   * logger.warn([3, 2, 4]) ;
    54   * }
    55   * </p>
    56   * @author eKameleon
    57   */
    58  class vegas.logging.targets.XPanelTarget extends LineFormattedTarget
    59  {
    60  
    61      /**
    62       * Creates a new XPanelTarget instance.
    63       */ 
    64  	public function XPanelTarget() 
    65  	{
    66  		this.includeLevel = false ;
    67  	}
    68  
    69  	/**
    70  	 * The id of the LocalConnection.
    71  	 */
    72      public static var CONNECTION_ID:String = "_xpanel1" ;
    73         
    74      /**
    75       * The dispatch message label.
    76       */
    77      public static var DISPATCH_MESSAGE:String = "dispatchMessage" ;
    78  
    79      /**
    80       * The 'all' level value of the XPanel console.
    81       */ 
    82  	public static var LEVEL_ALL:Number = 0x00 ;
    83  		
    84  	/**
    85  	 * The 'debug' level value of the XPanel console.
    86  	 */ 
    87  	public static var LEVEL_DEBUG:Number = 0x0001 ;
    88  
    89  	/**
    90  	 * The 'error' level value of the XPanel console.
    91  	 */ 
    92  	public static var LEVEL_ERROR:Number = 0x0008 ;
    93  
    94  	/**
    95  	 * The 'fatal' level value of the XPanel console.
    96  	 */ 
    97  	public static var LEVEL_FATAL:Number = 0x0010 ;
    98  
    99  	/**
   100       * The 'information' level value of the XPanel console.
   101       */ 
   102  	public static var LEVEL_INFORMATION:Number = 0x0002 ;
   103  
   104      /**
   105       * The 'none' level value of the XPanel console.
   106       */ 
   107  	public static var LEVEL_NONE:Number = 0xFF ;
   108  
   109      /**
   110       * The 'start' level value of the XPanel console.
   111       */ 
   112  	public static var LEVEL_START:Number = 0x0100 ;
   113  	
   114      /**
   115       * The 'warning' level value of the XPanel console.
   116       */ 
   117  	public static var LEVEL_WARNING:Number = 0x0004 ;
   118  
   119  	/**
   120  	 * The started value.
   121  	 */
   122  	public static var START:String = "Started" ;
   123  
   124  
   125  	/**
   126       * Descendants of this class should override this method to direct the specified message to the desired output.
   127       * @param message String containing preprocessed log message which may include time, date, category, etc. 
   128  	 */
   129  	public function internalLog( message , lv:Number ):Void
   130  	{
   131  		var lvl:Number ;
   132  	   	switch (lv)
   133  	   	{
   134  	   		case LogEventLevel.DEBUG:
   135  	   		{
   136  	   			lvl = LEVEL_DEBUG ;
   137  	   			break;
   138  	   		}
   139  	   		case LogEventLevel.ERROR:
   140  	   		{
   141  	   			lvl = LEVEL_ERROR ;
   142  	   			break;
   143  	   		}
   144  	   		case LogEventLevel.FATAL:
   145  	   		{
   146  	   			lvl = LEVEL_FATAL ;
   147  	   			break;
   148  	   		}
   149  	   		case LogEventLevel.INFO:
   150  	   		{
   151  	   			lvl = LEVEL_INFORMATION ;
   152  	  			break;
   153  	   		}
   154  	   		case LogEventLevel.WARN:
   155  	   		{
   156  	   			lvl = LEVEL_WARNING ;
   157  	   			break;
   158  	   		}
   159  	   		default:
   160  	   		{
   161  	  			lvl = LEVEL_ALL ;
   162  	   			break;
   163  	   		}
   164  	   	}
   165      		
   166      	if ( _lc == null)
   167      	{
   168      		_lc = new LocalConnection() ;
   169      		_lc.send( CONNECTION_ID , DISPATCH_MESSAGE , getTimer(), "START", LEVEL_START ) ;
   170      	}
   171  		
   172  		_lc.send( CONNECTION_ID , DISPATCH_MESSAGE , getTimer(), message.toString(), lvl ) ;
   173     		
   174  	        
   175  	}   
   176  
   177  	/**
   178  	 * Internal LocalConnection reference.
   179  	 */
   180  	private static var _lc:LocalConnection ;
   181  	
   182  }
