    26  import vegas.logging.AbstractTarget;
    27  import vegas.logging.LogEvent;
    28  import vegas.logging.LogLogger;
    29  
    30  /**
    31   * All logger target implementations that have a formatted line style output should extend this class. It provides default behavior for including date, time, category, and level within the output.
    32   * @author eKameleon
    33   */
    34  class vegas.logging.targets.LineFormattedTarget extends AbstractTarget 
    35  {
    36  	
    37  	/**
    38  	 * Creates a new LineFormattedTarget instance.
    39  	 */
    40  	public function LineFormattedTarget() 
    41  	{
    42  		//
    43  	}
    44  
    45  	/**
    46  	 * Indicates if the category for this target should added to the trace.
    47  	 */
    48  	public var includeCategory:Boolean = false ;
    49  	
    50  	/**
    51  	 * Indicates if the date should be added to the trace.
    52  	 */
    53  	public var includeDate:Boolean = false ;
    54  	
    55  	/**
    56  	 * Indicates if the level for the event should added to the trace.
    57  	 */
    58  	public var includeLevel:Boolean = false ;
    59  	
    60  	/**
    61  	 * Indicates if a line number should be added to the trace.
    62  	 */
    63  	public var includeLines:Boolean = false ; 
    64  
    65    	/**
    66  	 * Indicates if the milliseconds should be added to the trace. Only relevant when includeTime is {@code true}.
    67       */
    68    	public var includeMilliseconds:Boolean = false ;
    69  
    70  	/**
    71  	 * Indicates if the time should be added to the trace.
    72  	 */
    73  	public var includeTime:Boolean = false ;
    74  
    75     	/**
    76       * The separator string.
    77       */
    78     	public var separator:String = " " ;
    79  
    80  	/**
    81       * Descendants of this class should override this method to direct the specified message to the desired output.
    82       * @param message String containing preprocessed log message which may include time, date, category, etc. based on property settings, such as <code>includeDate</code>, <code>includeCategory</code>, etc.
    83       * @param level the LogEventLevel of the message.
    84  	 */
    85  	public function internalLog( message , level:Number ):Void
    86  	{
    87      	// override this method.
    88  	}
    89  
    90  	/**
    91  	 * This method handles a LogEvent from an associated logger.
    92  	 */
    93  	/*override*/ public function logEvent( e:LogEvent ) 
    94  	{
    95  		var message = e.message ;
    96  		var level:String = LogEvent.getLevelString( e.level ) ;
    97  		var category:String = LogLogger(e.getTarget()).category ;
    98  		message = formatMessage( message, level, category, new Date() ) ;
    99  		internalLog( message, e.level ) ;
   100  	}
   101  
   102  	/**
   103  	 * The current line number.
   104  	 */
   105  	private var __lineNumber:Number = 1 ;
   106  	
   107  	/**
   108  	 * This method format the passed Date in arguments.
   109  	 */
   110  	private function _formatDate(d:Date):String 
   111  	{
   112  		var date:String = "" ;
   113  		date += _getDigit(d.getDate()) ;
   114  		date += "/" + _getDigit(d.getMonth() + 1) ;
   115  		date += "/" + d.getFullYear();
   116  		return date ;
   117  	}
   118  
   119  	/**
   120  	 * This method format the passed level in arguments.
   121  	 */
   122  	private function _formatLevel(level:String):String 
   123  	{
   124  		return "[" + level + "]" ;
   125  	}	
   126  
   127  	/**
   128  	 * This method format the current line value.
   129  	 */
   130  	private function _formatLines():String 
   131  	{
   132  		return "[" + __lineNumber++ + "]" ; 
   133  	}	
   134  	
   135  	/**
   136  	 * This method format the log message.
   137  	 * @return the format log message.
   138  	 */
   139      private function formatMessage(message, level:String, category:String, date:Date):String 
   140      {
   141  	    	
   142  	    var msg:String = "" ;
   143         		
   144         	var d:Date = date || new Date ;
   145      	
   146      	if (includeLines) 
   147      	{
   148      	    msg += _formatLines() + separator ; // lines
   149      	}
   150      	
   151      	if (includeDate) 
   152      	{
   153      	    msg += _formatDate(d) + separator ; // date
   154      	}
   155      	
   156      	if (includeTime) 
   157      	{
   158      	    msg += _formatTime(d) + separator ; // time
   159      	}
   160      	
   161      	if (includeLevel)
   162      	{
   163      	    msg += _formatLevel(level || "" ) + separator ; // level  
   164      	} 
   165      	
   166      	if (includeCategory) 
   167      	{
   168          	msg += (category || "") + separator ; // category
   169      	}
   170      	
   171      	msg += message.toString() ;
   172  	    
   173  	    return msg ;
   174  	    	
   175      }
   176      
   177  	/**
   178  	 * This method format the current Date passed in argument.
   179  	 * @param d the Date object to format.
   180  	 * @return the format time in the message.
   181  	 */
   182  	private function _formatTime(d:Date):String 
   183  	{
   184  		var time:String = "" ;
   185  		time += _getDigit(d.getHours()) ;
   186  		time += ":" + _getDigit(d.getMinutes()) ;
   187  		time += ":" + _getDigit(d.getSeconds()) ;
   188  		if( includeMilliseconds ) 
   189  		{
   190        		time += ":" + _getDigit(d.getMilliseconds()) ;
   191      	}
   192  		return time ;
   193  	}
   194  	
   195  	/**
   196  	 * Returns the string representation of a number and use digit conversion.
   197  	 * @return the string representation of a number and use digit conversion.
   198  	 */
   199  	private function _getDigit(n:Number):String 
   200  	{
   201  		if (isNaN(n)) return "00" ;
   202  		return ((n < 10) ? "0" : "") + n ;
   203  	}
   204  
   205  }
