    24  import vegas.logging.ILogger;
    25  import vegas.logging.Log;
    26  import vegas.logging.targets.SOSTarget;
    27  
    28  /**
    29   * This static class create a tracer to use with MTASC and this {@code -trace} command.
    30   * @author eKameleon
    31   */
    32  class vegas.logging.tracer.SOSTracer 
    33  {
    34  
    35  	/**
    36  	 * Provides the category of the current tracer.
    37  	 */	
    38  	static public var category:String ;
    39  
    40  	/**
    41  	 * Launch the {@code clear} method of the internal SOSTarget instance.
    42  	 */
    43  	static public function clear():Void 
    44  	{
    45  		_instance.clear() ;
    46  	}
    47  	
    48  	/**
    49  	 * Launch the {@code exit} method of the internal SOSTarget instance.
    50  	 */
    51  	static public function exit():Void 
    52  	{
    53  		_instance.exit() ;
    54  	}
    55  
    56  	/**
    57  	 * Returns the include category value of the internal SOSTarget instance.
    58  	 */
    59  	static public function getIncludeCategory():Boolean 
    60  	{
    61  		return _instance.includeCategory ;
    62  	}
    63  
    64  	/**
    65  	 * Returns the include date value of the internal SOSTarget instance.
    66  	 */
    67  	static public function getIncludeDate():Boolean 
    68  	{
    69  		return _instance.includeDate ;
    70  	}
    71  
    72  	/**
    73  	 * Returns the include level value of the internal SOSTarget instance.
    74  	 */
    75  	static public function getIncludeLevel():Boolean 
    76  	{
    77  		return _instance.includeLines ;
    78  	}
    79  	
    80  	/**
    81  	 * Returns the include lines value of the internal SOSTarget instance.
    82  	 */
    83  	static public function getIncludeLines():Boolean 
    84  	{
    85  		return _instance.includeLines ;
    86  	}
    87  
    88  	/**
    89  	 * Returns the include time value of the internal SOSTarget instance.
    90  	 */
    91  	static public function getIncludeTime():Boolean 
    92  	{
    93  		return _instance.includeTime ;
    94  	}
    95  
    96  	/**
    97  	 * Returns the internal SOSTarget reference.
    98  	 */
    99  	static public function getTargetInstance():SOSTarget 
   100  	{
   101  		return _instance ;	
   102  	}
   103  	
   104  	/**
   105  	 * Initialize the SOSTracer.
   106  	 * @param name the default name of the application.
   107  	 * @param color the default color of the application.
   108  	 * @param includes a boolean, {@code true} if you wan includes time, date, level and lines in your logs.
   109  	 */
   110  	static public function initialize(namespace:String, color:Number, includes:Boolean):Void 
   111  	{
   112  		namespace = namespace || "SOSTracer" ;
   113  		_instance = new SOSTarget(color) ;
   114  		_instance.filters = [ namespace ] ;
   115  		_instance.setAppName(namespace ) ;
   116  		if (includes)
   117  		{
   118  			setIncludes(true, true, true, true, true) ;
   119  		}
   120  		_instance.getIdentify() ;
   121  		
   122  		Log.addTarget(_instance) ;
   123  		
   124  		_logger = Log.getLogger( namespace ) ;
   125  	}
   126  	
   127  	/**
   128  	 * Release the SOSTracer.
   129  	 */
   130  	static public function release():Void 
   131  	{
   132  		if (_instance != null)
   133  		{
   134  			Log.removeTarget(_instance) ;
   135  			_instance.close() ;
   136  		}
   137  		_instance = null ;
   138  		_logger = null ;
   139  	}
   140  
   141  	/**
   142  	 * Sets the include category value of the internal SOSTarget instance.
   143  	 */
   144  	static public function setIncludeCategory(b:Boolean):Void 
   145  	{
   146  		_instance.includeCategory = b ;
   147  	}
   148  
   149  	/**
   150  	 * Sets the include date value of the internal SOSTarget instance.
   151  	 */
   152  	static public function setIncludeDate(b:Boolean):Void 
   153  	{
   154  		_instance.includeDate = b ;
   155  	}
   156  
   157  	/**
   158  	 * Sets the include level value of the internal SOSTarget instance.
   159  	 */
   160  	static public function setIncludeLevel(b:Boolean):Void 
   161  	{
   162  		_instance.includeLines = b ;
   163  	}
   164  
   165  	/**
   166  	 * Sets the include lines value of the internal SOSTarget instance.
   167  	 */	
   168  	static public function setIncludeLines(b:Boolean):Void 
   169  	{
   170  		_instance.includeLines = b ;
   171  	}
   172  
   173  	/**
   174  	 * Sets all the include values of the internal SOSTarget instance.
   175  	 */
   176  	static public function setIncludes(lines:Boolean, level:Boolean, time:Boolean, date:Boolean, category:Boolean):Void 
   177  	{
   178  		if (lines != null ) 
   179  		{
   180  			_instance.includeLines = lines ;
   181  		}
   182  		if (level != null )
   183  		{
   184  			_instance.includeLevel = level ;
   185  		}
   186  		if (time != null )
   187  		{
   188  			_instance.includeTime = time ;
   189  		}
   190  		if (date != null )
   191  		{
   192  			_instance.includeDate = date ;
   193  		}
   194  		if (category != null )
   195  		{
   196  			_instance.includeCategory = category ;
   197  		}
   198  	}
   199  
   200  	/**
   201  	 * Sets the include time value of the internal SOSTarget instance.
   202  	 */	
   203  	static public function setIncludeTime(b:Boolean):Void 
   204  	{
   205  		_instance.includeTime = b ;
   206  	}
   207  
   208  	/**
   209  	 * Trace a message with the internal SOSTarget.
   210  	 */
   211  	static public function trace(o):Void 
   212  	{
   213  		_logger.debug.apply(_logger, [].concat(arguments)) ;
   214  	}
   215  
   216  	/**
   217  	 * This method is used in MTASC {@code -trace} command.
   218  	 */
   219  	static public function traceReplacer(o):Void
   220  	{
   221  		SOSTracer.trace( o ) ;
   222  	}
   223  
   224  	/**
   225  	 * Internal SOSTarget singleton.
   226  	 */	
   227  	static private var _instance:SOSTarget ;
   228  
   229  	/**
   230  	 * Internal logger.
   231  	 */
   232  	static private var _logger:ILogger ;
   233  	
   234  }
