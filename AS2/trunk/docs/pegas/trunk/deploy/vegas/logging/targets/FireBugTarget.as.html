    24  import vegas.logging.LogEventLevel;
    25  import vegas.logging.targets.LineFormattedTarget;
    26  
    27  /**
    28   * Provides a logger target that uses the FireBug console extension in Firefox to output log messages. 
    29   * You can download the FireBug and test this target : https://addons.mozilla.org/fr/firefox/addon/1843
    30   * @author eKameleon
    31   */
    32  class vegas.logging.targets.FireBugTarget extends LineFormattedTarget 
    33  {
    34  	
    35  	/**
    36  	 * Creates a new FireBugTarget instance.
    37  	 */
    38  	public function FireBugTarget() 
    39  	{
    40  		super();
    41  	}
    42  	
    43  	/**
    44       * Descendants of this class should override this method to direct the specified message to the desired output.
    45       * @param message String containing preprocessed log message which may include time, date, category, etc. based on property settings, such as <code>includeDate</code>, <code>includeCategory</code>, etc.
    46       * @param level the LogEventLevel of the message.
    47  	 */
    48  	public /*override*/ function internalLog( message , level:Number ):Void
    49  	{
    50  		
    51  		var methodName:String ;
    52  		
    53  		switch (level)
    54  		{
    55  			
    56  			case LogEventLevel.ERROR :
    57  			{
    58  				methodName = "console.error" ;
    59  				break ;	
    60  			}
    61  			case LogEventLevel.FATAL :
    62  			{
    63  				methodName = "console.error" ;
    64  				break ;	
    65  			}
    66  			case LogEventLevel.INFO :
    67  			{
    68  				methodName = "console.info" ;
    69  				break ;	
    70  			}
    71  			case LogEventLevel.WARN :
    72  			{
    73  				methodName = "console.warn" ;
    74  				break ;	
    75  			}
    76  			default :
    77  			{
    78  				methodName = "console.debug" ;
    79  				break ;	
    80  			}		
    81  			
    82  		}
    83  		if ( flash.external.ExternalInterface.available )
    84  		{
    85      		flash.external.ExternalInterface.call( methodName, [message] ) ;
    86  		}
    87  		else
    88  		{
    89  			getURL("javascript:" + methodName + "('"+ message +"');");	
    90  		}
    91  	}
    92  
    93  }
