    24  import vegas.data.Set;
    25  import vegas.events.EventListener;
    26  import vegas.events.EventListenerCollection;
    27  import vegas.events.EventTarget;
    28  
    29  /**
    30   * This interface implement class who stores the listeners object an notifies them.
    31   * @author eKameleon
    32   */
    33  interface vegas.events.IEventDispatcher extends EventTarget 
    34  {
    35  
    36  	/**
    37  	 * Allows the registration of global event listeners on the event target.
    38  	 * 
    39  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the <b>EventListener</b> interface.
    40  	 * @param priority Determines the priority level of the event listener.
    41  	 * @param autoRemove Apply a removeEventListener after the first trigger
    42  	 */
    43  	function addGlobalEventListener(listener:EventListener, priority:Number, autoRemove:Boolean):Void ;
    44  
    45  	/**
    46  	 * Returns the {@code EventListenerCollection} of the specified event name.
    47  	 * @return the {@code EventListenerCollection} of the specified event name.
    48  	 */
    49  	function getEventListeners(eventName:String):EventListenerCollection ;
    50  
    51  	/**
    52  	 * Returns the {@code EventListenerCollection} of this EventDispatcher.
    53  	 * @return the {@code EventListenerCollection} of this EventDispatcher.
    54  	 */
    55  	function getGlobalEventListeners():EventListenerCollection ;
    56  
    57  	/**
    58  	 * Returns a {@code Set} of all register event's name in this EventListener.
    59  	 * @return a {@code Set} of all register event's name in this EventListener.
    60  	 */
    61  	function getRegisteredEventNames():Set ;
    62  
    63  	/**
    64  	 * Checks whether the EventDispatcher object has any listeners registered for a specific type of event.
    65  	 * This allows you to determine where altered handling of an event type has been introduced in the event flow heirarchy by an EventDispatcher object.
    66  	 */ 
    67  	function hasEventListener(eventName:String):Boolean ;
    68  
    69  	/** 
    70  	 * Removes a global listener from the EventDispatcher object.
    71  	 * If there is no matching listener registered with the EventDispatcher object, then calling this method has no effect.
    72  	 * @param the string representation of the class name of the EventListener or a EventListener object.
    73  	 */
    74  	function removeGlobalEventListener( listener ):EventListener ;
    75  	
    76  	/**
    77  	 * Returns the string representation of this object.
    78  	 * @return the string representation of this object.
    79  	 */
    80  	function toString():String ;
    81  	
    82  }
