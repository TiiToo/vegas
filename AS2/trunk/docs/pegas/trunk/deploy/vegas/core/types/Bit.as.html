    24  import vegas.core.HashCode;
    25  import vegas.core.IFormattable;
    26  import vegas.core.IHashable;
    27  import vegas.core.ISerializable;
    28  import vegas.util.ConstructorUtil;
    29  import vegas.util.MathsUtil;

    31  /**
    32   * Bit represents a bit value.
    33   * This class is too an enumeration of all bit and bytes values. 
    34   * <p>{@code Bit} can be used for a different kind of formatting of a bit value. 
    35   * It allows to access the value as bit, kilo-bit, mega-bit, giga-bit, tera-bit, byte, kilo-byte, mega-byte, giga-byte and tera-byte.
    36   * </p>
    37   * <p><b>Examples:</b></p>
    38   * {@code 
    39   *   import vegas.core.types.Bit ;
    40   *   trace(new Bit(1).toString()) ; // 1b
    41   *   trace(new Bit(1234).toString()) ; // 1.21Kb
    42   *   trace(new Bit(15002344).toString()) ; // 14.31Mb
    43   * }
    44   * @author eKameleon
    45   */
    46  class vegas.core.types.Bit extends Number implements IFormattable, IHashable, ISerializable  
    47  {
    48  
    49  	/**
    50  	 * Creates a new Bit instance.
    51  	 * @param b value in bit
    52  	 */
    53  	public function Bit(n:Number, floatingPoints:Number ) 
    54  	{
    55  		_bit = n ;
    56  		if (isNaN(floatingPoints))
    57  		{
    58  			_comma = DEFAULT_FLOATING_POINTS ;
    59  		}
    60  		else
    61  		{
    62  			setFloatingPoints(floatingPoints) ;	
    63  		}
    64  	}
    65  
    66  	/**
    67  	 * Default floating points used.
    68  	 */
    69  	static public var DEFAULT_FLOATING_POINTS:Number = 2 ;
    70  
    71  	/**
    72  	 * Size of a byte.
    73  	 */
    74  	static public var BYTE:Number = 8 ; // BYTE
    75  
    76  	/**
    77  	 * Size of a kilo.
    78  	 */
    79  	static public var KBIT:Number = 1024 ;
    80  
    81  	/**
    82  	 * Size of a kilobyte.
    83  	 */
    84  	static public var KBYTE:Number = KBIT * BYTE ;
    85  
    86  	/**
    87  	 * Size of a megabit.
    88  	 */
    89  	static public var MBIT:Number = KBIT * KBIT ;
    90  
    91  	/**
    92  	 * Size of a megabyte
    93  	 */
    94  	static public var MBYTE:Number = KBYTE * KBIT ;
    95  	
    96  	/**
    97  	 * Size of a gigabit.
    98  	 */
    99  	static public var GBIT:Number = MBIT * KBIT ;
   100  	
   101  	/**
   102  	 * Size of a gigabyte.
   103  	 */
   104  	static public var GBYTE:Number = MBYTE * KBIT ;
   105  	
   106  	/**
   107  	 * Size of a terabit.
   108  	 */
   109  	static public var TBIT:Number = GBIT * KBIT ;
   110  
   111  	/**
   112  	 * Size of a terabyte.
   113  	 */	
   114  	static public var TBYTE:Number = GBYTE * KBIT ;
   115  	
   116  	/**
   117  	 * Shortname of bit.
   118  	 */
   119  	static public var SB:String = "b" ;
   120  
   121  	/**
   122  	 * Shortname of kilobit.
   123  	 */
   124  	static public var SKB:String = "Kb" ;
   125  	
   126  	/**
   127  	 * Shortname of megabit.
   128  	 */
   129  	static public var SMB:String = "Mb" ;
   130  
   131  	/**
   132  	 * Shortname of gigabit.
   133  	 */
   134  	static public var SGB:String = "Gb" ;
   135  
   136  	/**
   137  	 * Shortname of terabit.
   138  	 */
   139  	static public var STB:String = "Tb" ;
   140  
   141  	/**
   142  	 * Returns the value in bit.
   143  	 * @return value in bit
   144  	 */
   145  	public function getBit():Number 
   146  	{
   147  		return _bit ;
   148  	}
   149  
   150  	/**
   151  	 * Returns the value in bytes.
   152  	 * @return value in bytes
   153  	 */
   154  	public function getBytes():Number 
   155  	{
   156  		return MathsUtil.round( _bit / BYTE, _comma) ;
   157  	}
   158  
   159  	/**
   160  	 * Returns the value in kilobit.
   161  	 * @return value in kilobit
   162  	 */
   163  	public function getKBit():Number 
   164  	{
   165  		return MathsUtil.round( _bit / KBIT, _comma) ;
   166  	}
   167  
   168  	/**
   169  	 * Returns the value in kilobytes.
   170  	 * @return value in kilobytes
   171  	 */
   172  	public function getKBytes():Number 
   173  	{
   174  		return MathsUtil.round( _bit / KBYTE, _comma) ;
   175  	}
   176  
   177  	/**
   178  	 * Returns the value in megabit.
   179  	 * @return value in megabit
   180  	 */
   181  	public function getMegaBit():Number 
   182  	{
   183  		return MathsUtil.round( _bit / MBIT, _comma) ;
   184  	}
   185  
   186  	/**
   187  	 * Returns the value in megabytes.
   188  	 * @return value in megabytes
   189  	 */
   190  	public function getMegaBytes():Number 
   191  	{
   192  		return MathsUtil.round( _bit / MBYTE, _comma) ;
   193  	}
   194  
   195  	/**
   196  	 * Returns the value in gigabit.
   197  	 * @return value in gigabit
   198  	 */
   199  	public function getGigaBit():Number 
   200  	{
   201  		return MathsUtil.round( _bit / GBIT, _comma) ;
   202  	}
   203  
   204  	/**
   205  	 * Returns the value in gigabytes.
   206  	 * @return value in gigabytes
   207  	 */
   208  	public function getGigaBytes():Number 
   209  	{
   210  		return MathsUtil.round( _bit / GBYTE, _comma) ;
   211  	}
   212  	
   213  	/**
   214  	 * Returns the value in terabit.
   215  	 * @return value in terabit
   216  	 */
   217  	public function getTeraBit():Number 
   218  	{
   219  		return MathsUtil.round( _bit / TBIT, _comma);
   220  	}
   221  
   222  	/**
   223  	 * Returns the value in terabytes.
   224  	 * @return value in terabytes
   225  	 */
   226  	public function getTeraBytes():Number 
   227  	{
   228  		return MathsUtil.round(_bit / TBYTE, _comma);
   229  	}
   230  
   231  	/**
   232  	 * Returns a hash code value for the object.
   233  	 */
   234  	public function hashCode():Number 
   235  	{
   236  		return null ;
   237  	}
   238  	
   239  	/**
   240  	 * Sets the used amount of values after the comma.
   241  	 * <p>This method does not change anything if {@code fp} is smaller than 0 or not passed-in.</p>
   242  	 * @param n amount of characters after the floating point
   243  	 * @return the current instance
   244  	 */
   245  	public function setFloatingPoints(n:Number):Bit 
   246  	{
   247  		_comma = (n > 0) ? n : 0 ;
   248  		return this ;
   249  	}
   250  	
   251  	/**
   252  	 * Returns a Eden reprensation of the object.
   253  	 * @return a string representing the source code of the object.
   254  	 */
   255  	public function toSource(indent:Number, indentor:String):String
   256  	{
   257  		return "new " + ConstructorUtil.getPath(this) + "(" + _bit + "," + _comma  + ")" ;
   258  	}
   259  
   260  	/**
   261  	 * Returns a string representation of the object.
   262  	 * @return a string representation of the object.
   263  	 */
   264  	public function toString():String 
   265  	{
   266  		if (_bit < KBIT) 
   267  		{
   268  			return getBit() + SB ;
   269  		}
   270  		else if(_bit < MBIT)
   271  		{
   272  			return getKBit() + SKB ;
   273  		}
   274  		else if(_bit < GBIT) 
   275  		{
   276  			return getMegaBit() + SMB ;
   277  		}
   278  		else if(_bit < TBIT) 
   279  		{
   280  			return getGigaBit() + SGB ;
   281  		}
   282  		else 
   283  		{
   284  			return getTeraBit() + STB ;
   285  		}
   286  	}
   287  	
   288  	/**
   289  	 * Returns the real value of the object.
   290  	 */
   291  	public function valueOf() 
   292  	{
   293  		return getBytes().valueOf();
   294  	}
   295  	
   296  	/**
   297  	 * Internal amout value of bits.
   298  	 */
   299  	private var _bit:Number ;
   300  	
   301  	/**
   302  	 * Internal value for the comma seperation.
   303  	 */
   304  	private var _comma:Number ;
   305  	
   306  	static private var _initHashCode:Boolean = HashCode.initialize(Bit.prototype) ;	
   307  	
   308  
   309  }
