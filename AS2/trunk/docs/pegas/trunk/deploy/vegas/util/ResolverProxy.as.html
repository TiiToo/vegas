    24  import vegas.core.CoreObject;
    25  import vegas.util.Mixin;
    26  
    27  /**
    28   * Allow to link an object with another objet using {@code __resolve} method.
    29   * <p>This is a 1 to 1 relationship.
    30   * <li>EventBroadcaster -> 1 to *    + unidirectional</li>
    31   * <li>ResolverProxy    -> 1 to 1    + unidirectional or bidirectional</li>
    32   * </p>
    33   * <p><b>Note :</b>This can be used to connect a "Model" object to a "View" object as an alternative to  "Model-View-Controller" or "visual proxy".</p>
    34   * <p><b>Example 1 :</b>
    35   * {@code
    36   * var myModel = new Model();
    37   * createView( "myView" ) ;
    38   * myView.proxy = myModel ;
    39   * myModel.proxy = myView;
    40   * }
    41   * </p>
    42   * <p><b>Example 2 :</b>
    43   * {@code
    44   * myModel = new Model();
    45   * createView( "myView" );
    46   * myView.link( myModel );
    47   * }
    48   * </p>
    49   * <p>The View and the Model share their methods and properties as they were only one object.</p>
    50   * @author eKameleon
    51   * @see Mixin
    52   */
    53  class vegas.util.ResolverProxy extends CoreObject 
    54  {
    55  	
    56  	/**
    57  	 * Creates a new ResolverProxy instance.
    58  	 */
    59  	public function ResolverProxy ( proxy ) 
    60  	{
    61  		_proxy = proxy ;
    62      }
    63  	
    64  	/**
    65  	 * Returns the proxy reference of this instance.
    66  	 */
    67  	public function get proxy()
    68  	{
    69  		return getProxy() ;	
    70  	}
    71  
    72  	/**
    73  	 * Sets the proxy reference of this instance.
    74  	 */
    75  	public function set proxy( oTarget )
    76  	{
    77  		setProxy( oTarget ) ;	
    78  	}
    79  
    80  	/**
    81  	 * Returns the proxy reference of this ResolverProxy.
    82  	 */
    83  	public function getProxy() 
    84  	{
    85  		return _proxy ;
    86      }
    87  
    88  	/**
    89  	 * Initialize a proxy on a specific target.
    90  	 */
    91  	static public function initialize( target )  
    92  	{
    93  		var attributes:Array = [ "linkProxy", "getProxy", "_proxy", "proxy", "__resolve", "setProxy" ] ;
    94  		var mix:Mixin = new Mixin(ResolverProxy, target, attributes) ;
    95  		mix.run() ;
    96      }
    97  
    98  
    99  	/**
   100  	 * Creates a link proxy between 2 proxy object.
   101  	 */		
   102  	public function linkProxy( linkedProxy ) 
   103  	{
   104  		linkedProxy.proxy = this ;
   105  		_proxy = linkedProxy ;
   106      }
   107  
   108  	/**
   109  	 * Sets the proxy reference of this ResolverProxy.
   110  	 */
   111  	public function setProxy( o ) 
   112  	{
   113  		_proxy = o;
   114  	}
   115  
   116  	/**
   117  	 * Resolve the specified property name.
   118  	 */
   119  	public function __resolve( name:String ) 
   120  	{
   121  		
   122  		if ( _proxy == null ) 
   123  		{
   124  			return ;
   125  		}
   126  		
   127  		if( !_proxy.hasOwnProperty( name ) ) 
   128  		{
   129  			if( _proxy.__proto__[name] == undefined ) return ;
   130          }
   131  		
   132  		if( typeof(_proxy[name]) == "function" ) 
   133  		{
   134  			var p = _proxy ; 
   135  			return function() 
   136  			{
   137  				return p[name].apply( p, arguments );
   138  			};
   139  		} 
   140  		else 
   141  		{
   142  			return _proxy[name] ;
   143  		}
   144  		
   145      }
   146  
   147  	/**
   148  	 * Internal proxy reference.
   149  	 */	
   150  	private var _proxy ;
   151  	
   152  }
