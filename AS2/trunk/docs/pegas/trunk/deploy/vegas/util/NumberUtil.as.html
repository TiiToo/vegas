    78      /***
    79       * Returns the exponential string reprsentation of the number passsed in argument.
    80       * <p><b>Example :</b></p>
    81       * {@code
    82       * import vegas.util.NumberUtil ;
    83       * 
    84       * var num:Number = 77.1234;
    85       * 
    86       * trace( NumberUtil.toExponential(num) ) ; // 7.71234e+1
    87       * trace( NumberUtil.toExponential(num, 4)) ; // 7.7123e+1
    88       * trace( NumberUtil.toExponential(num, 2)) ; // 7.71e+1
    89       * trace( NumberUtil.toExponential(77.1234) ) ; // 7.71234e+1
    90       * trace( NumberUtil.toExponential(77) ) ; // 7.7e+1
    91       * }
    92       * @param n the number to format.
    93       * @param fractionDigits An integer specifying the number of digits after the decimal point. Defaults to as many digits as necessary to specify the number.
    94       * @return the exponential string reprsentation of the number passsed in argument.
    95       */
    96      public static function toExponential( n:Number, fractionDigits:Number):String 
    97      {
    98          var str:String ;
    99          var x:Number = n ;
   100             var s:String = "+" ;
   101          if( isNaN( x ) ) 
   102          {
   103              return "NaN" ;
   104          }
   105          if( x < 0 ) 
   106          {
   107              s = "-";
   108              x = -x ;
   109          }
   110          if( x == Infinity ) 
   111          {
   112              return s + "Infinity" ;
   113          }
   114          var l:Number = Math.floor( Math.log(x) / Math.LN10 ) ;
   115          var lm:Number = Math.pow( 10, l ) ;
   116             n = x/lm ;
   117          if( fractionDigits == null ) 
   118          {
   119              str = n.toString();
   120          }
   121          else 
   122          {
   123              fractionDigits = MathsUtil.clamp(fractionDigits, 0, 20) ;
   124              str = NumberUtil.toFixed( n, fractionDigits );
   125          }
   126          str += (l > 0) ? ("e" + s + l) : "" ;
   127          return str ;
   128      }

   130      /**
   131       * Returns a string representing the number in fixed-point notation (*ECMA-262*).
   132       * <p><b>Example :</b></p>
   133       * {@code
   134       * import vegas.util.NumberUtil ;
   135       * 
   136       * var n:Number = 12345.6789 ;
   137       * 
   138       * trace( NumberUtil.toFixed(n) ) ;
   139       * // Returns 12346: note rounding, no fractional part
   140       * 
   141       * trace( NumberUtil.toFixed( n , 1) ) ;
   142       * // Returns 12345.7: note rounding
   143       * 
   144       * trace( NumberUtil.toFixed( n , 6 ) );
   145       * // Returns 12345.678900 : note added zeros
   146       * }
   147       * @param n the number to fixed.
   148       * @param fractionDigits the number of digits to appear after the decimal point; this may be a value between 0 and 20, inclusive, and implementations may optionally support a larger range of values. If this argument is omitted, it is treated as 0.
   149       * @return A string representation of number that does not use exponential notation and has exactly {@code fractionDigits} digits after the decimal place. The number is rounded if necessary, and the fractional part is padded with zeros if necessary so that it has the specified length. If number is greater than 1e+21, this method simply calls Number.toString() and returns a string in exponential notation.
   150       */
   151      public static function toFixed( n:Number, fractionDigits:Number):String 
   152      {
   153          var x:Number = n ;
   154          
   155          if( isNaN( x )) 
   156          {
   157              return "NaN" ;
   158          }
   159          if (isNaN(fractionDigits)) 
   160          {
   161              fractionDigits = 0 ;
   162          }
   163             fractionDigits = MathsUtil.clamp(fractionDigits, 0, 20) ;
   164          var m:Number = Math.pow( 10, fractionDigits );
   165          var r:Number = Math.round( x*m ) / m;
   166          var str:String = r.toString() ;
   167          var d = str.split( "." )[1] ;
   168          if( d && (d.length < fractionDigits) ) 
   169          {
   170              var len:Number = fractionDigits - d.length  ;
   171              for( var i:Number=0; i< len ; i++ ) 
   172              {
   173                  str += "0" ;
   174              }
   175          }
   176          return str;
   177      }

   179      /**
   180       * Returns the hexadecimal string representation of the specified number value.
   181       * <p><b>Example :</b></p>
   182       * {@code
   183       * import vegas.util.NumberUtil  ;
   184       *
   185       * for (var i:Number =0 ; i<256 ; i++)
   186       * {
   187       *     trace( NumberUtil.toHex( i ) ) ; // without optional prefix argument (default "0x")
   188       * }
   189       *
   190       * trace("---") ; 
   191        * 
   192       * for (var i:Number =0 ; i<256 ; i++)
   193       * {
   194       *     trace( NumberUtil.toHex( i , "#" ) ) ; // with optional prefix argument
   195       * }
   196       * 
   197       * }
   198       * @param n the number to format.
   199       * @param prefix Optional string represention of the prefix of the return format string. If this argument is undefined the prefix is "0x".
   200       * @return the hexadecimal string representation of the specified number value.
   201       */
   202      public static function toHex( n:Number, prefix:String ):String
   203      {
   204          if ( prefix == null )
   205          {
   206              prefix = "0x" ;    
   207          }
   208          var temp:String = n.toString( 16 ) ;
   209          if(n < 16) 
   210          {
   211              temp = "0" + temp ;
   212          }
   213          return (prefix || "") + temp ;
   214      }

   235      /**
   236       * Returns a string representing the number to a specified precision in fixed-point notation (*ECMA-262*).
   237       * <p><b>Example :</b></p>
   238       * {@code
   239       * import vegas.util.NumberUtil ;
   240       * var num:Number = 5.123456 ;
   241       * 
   242       * trace( NumberUtil.toPrecision(num) )  ; // 5.123456
   243       * trace( NumberUtil.toPrecision(num, 4) ) ; // 5.123
   244       * trace( NumberUtil.toPrecision(num, 2) ) ; // 5.1
   245       * trace( NumberUtil.toPrecision(num, 1) ) ; // 5
   246       * trace( NumberUtil.toPrecision(1250, 2) ) ; // 1.3e+3
   247       * trace( NumberUtil.toPrecision(1250, 5) ) ; // 1250.0
   248       * }
   249       * @param n The number to format.
   250       * @param precision An integer specifying the number of digits after the decimal point.
   251       * @return A string representing a Number object in fixed-point or exponential notation rounded to precision significant digits.
   252       */
   253      public static function toPrecision(n:Number, precision:Number):String 
   254      {
   255          var x:Number = n ;
   256          var str:String = x.toString() ;
   257          if ( isNaN(x) ) 
   258          {
   259              return "NaN" ;
   260          }
   261          if( (precision == null) || (x == Infinity) || (x == -Infinity) ) 
   262          {
   263              return x.toString();
   264          }
   265          precision = MathsUtil.clamp(precision, 1, 21) ;
   266          if (str.length > precision) 
   267          {
   268              return NumberUtil.toExponential( x, precision - 1 ) ;    
   269          }
   270          var l:Number = Math.floor( Math.log(x) / Math.LN10 ) ;
   271          var m:Number = Math.pow( 10 , l+1 - precision ) ;
   272          var r:Number = Math.round(x/m) * m ;
   273          str = r.toString() ;
   274          var d = str.split( "." ).join( "" ) ;
   275          if( d && (d.length < precision) ) 
   276          {
   277              if( str.indexOf( "." ) == -1 ) 
   278              {
   279                  str += "." ;
   280              }
   281              var len:Number = precision - d.length ;
   282              for( var i:Number = 0 ; i < len; i++ ) 
   283              {
   284                  str += "0" ;
   285              }
   286          }
   287          return str ;
   288      }
