    24  import vegas.data.iterator.ArrayIterator;
    25  import vegas.data.iterator.Iterator;
    26  import vegas.events.Delegate;
    27  import vegas.util.ArrayUtil;
    28  
    29  /**
    30   * This singleton create a virtual Frame interval with a onEnterFrame event. 
    31   * The use can register object listeners to receive onEnterFrame event.
    32   * <p><b>Example :</b></p>
    33   * {@code
    34   * import vegas.util.FrameBeacon ;
    35   * 
    36   * var i:Number = 0 ;
    37   * var max:Number = 25 ;
    38   * 
    39   * var o = {} ;
    40   * o.onEnterFrame = function () 
    41   * {
    42   *     trace (i) ;
    43   *     i++ ;
    44   *     if (i == max) 
    45   *     {
    46   *         FrameBeacon.removeFrameListener(this) ;
    47   *     }
    48   * }
    49   * 
    50   * FrameBeacon.addFrameListener(o) ;
    51   * 
    52   * this.onKeyDown = function () 
    53   * {
    54   *     if (FrameBeacon.running)
    55   *     {
    56   *         FrameBeacon.stop() ;
    57   *     }
    58   *     else
    59   *     {
    60   *         FrameBeacon.start() ;
    61   *     }
    62   * }
    63   * Key.addListener(this) ;
    64   * }  
    65   * @author eKameleon
    66   */
    67  class vegas.util.FrameBeacon 
    68  {
    69  
    70  	private static var __initBroadcaster = AsBroadcaster.initialize(FrameBeacon) ;	
    71  
    72  	private static var __initConstructor = FrameBeacon.initialize() ;
    73  
    74  	/**
    75  	 * Returns {@code true} if the {@code FrameBeacon} is in progress (read only property).
    76  	 * <p><b>Example :</b></p>
    77  	 * {@code
    78  	 * import vegas.util.FrameBeacon ;
    79  	 * FrameBeacon.start() ;
    80  	 * trace(FrameBeacon.running) ; // true
    81  	 * } 
    82  	 * @return {@code true} if the {@code FrameBeacon} is in progress.
    83  	 */
    84  	public static function get running():Boolean 
    85  	{
    86  		return FrameBeacon._mc.onEnterFrame == FrameBeacon._proxy ;
    87  	}
    88  
    89  	/**
    90  	 * Registers a listener to receive the onEnterFrame event.
    91  	 */
    92  	public static function addFrameListener(listener):Void  
    93  	{
    94  		if (!FrameBeacon.running) 
    95  		{
    96  			FrameBeacon.start() ;
    97  		}
    98  		FrameBeacon.addListener(listener) ;
    99  	}
   100  
   101  	/**
   102  	 * Returns {@code true} if the listener is registered in the FrameBeacon tool.
   103  	 * @return {@code true} if the listener is registered in the FrameBeacon tool.
   104  	 */
   105  	public static function containsListener(listener):Boolean
   106  	{
   107  		return ArrayUtil.contains( FrameBeacon._listeners, listener ) ;
   108  	}
   109  
   110  	/**
   111  	 * Initialize the protected MovieClip who notify the onEnterFrame event.
   112  	 * @return the internal reference of the movieclip used by the FrameBeacon tool class.
   113  	 */
   114  	public static function initialize():MovieClip 
   115  	{
   116  		FrameBeacon._mc = _level0.createEmptyMovieClip ("__mcFBeacon__", -9998) ;
   117  		return FrameBeacon._mc ;
   118  	}
   119  
   120  	/**
   121  	 * Returns {@code true} if this instance is empty of listeners.
   122  	 * @return {@code true} if this instance is empty of listeners.
   123  	 */
   124  	public static function isEmpty():Boolean 
   125  	{
   126  		return FrameBeacon._listeners.length == 0 ;
   127  	}
   128  
   129  	/**
   130  	 * Returns the iterator of this FrameBeacon singleton.
   131  	 * @return the iterator of this FrameBeacon singleton.
   132  	 */
   133  	public static function iterator():Iterator 
   134  	{
   135  		return new ArrayIterator(FrameBeacon._listeners) ;
   136  	}
   137  
   138  	/**
   139  	 * Release the {@code FrameBeacon} singleton.
   140  	 */
   141  	public static function release():Void 
   142  	{
   143  		FrameBeacon.stop() ;
   144  		FrameBeacon._mc.swapDepths(_level0.getNextHighestDepth()) ;
   145  		FrameBeacon._mc.removeMovieClip () ;
   146  		FrameBeacon._mc = undefined ;
   147  	}
   148  
   149  	/**
   150  	 * Removes a listener who receive the onEnterFrame event.
   151  	 * If the {@code FrameBeacon} is empty the event broadcast is stopped.
   152  	 */
   153  	public static function removeFrameListener(listener):Void 
   154  	{
   155  		FrameBeacon.removeListener(listener) ;
   156  		if (FrameBeacon.isEmpty())
   157  		{
   158  			FrameBeacon.stop();
   159  		}
   160  	}
   161  	
   162  	/**
   163  	 * Returns the number of listeners in the tool.
   164  	 * @return the number of listeners in the tool.
   165  	 */
   166  	public static function size():Number 
   167  	{
   168  		return FrameBeacon._listeners.length ;
   169  	}
   170  	
   171  	/**
   172  	 * Start the onEnterFrame broadcast.
   173  	 */
   174  	public static function start():Void 
   175  	{
   176  		FrameBeacon._mc.onEnterFrame = FrameBeacon._proxy ;
   177  	}
   178  
   179  	/**
   180  	 * Stop the onEnterFrame broadcast.
   181  	 */
   182  	public static function stop():Void 
   183  	{
   184  		delete FrameBeacon._mc.onEnterFrame ;
   185  	}
   186  
   187  	/**
   188  	 * Returns the string representation of this singleton.
   189  	 * @return the string representation of this singleton.
   190  	 */
   191  	public static function toString():String 
   192  	{
   193  		return "[FrameBeacon]" ;
   194  	}
   195  
   196  	private static var addListener:Function ;
   197  
   198  	private static var broadcastMessage:Function;
   199  
   200  	private static var _listeners:Array = new Array() ;
   201  
   202  	private static var _mc:MovieClip ;
   203  
   204  	private static var _proxy:Function = Delegate.create( FrameBeacon, FrameBeacon.broadcastMessage , "onEnterFrame" ) ; 
   205  
   206  	private static var removeListener:Function ;
   207  	
   208  }
