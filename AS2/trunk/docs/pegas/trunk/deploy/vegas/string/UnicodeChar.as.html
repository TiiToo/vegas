    27  /**
    28   * ECMA 262 Unicode IFormat-Control Characters tools.
    29   * <p><b>Example :</b></p>
    30   * {@code
    31   * import vegas.string.UnicodeChar ;
    32   * 
    33   * trace( "## The White space chars array." ) ;
    34   * 
    35   * // The length of the UnicodeChar.WHITE_SPACE_CHARS array : 5
    36   * trace ("The length of the UnicodeChar.WHITE_SPACE_CHARS array : " + UnicodeChar.WHITE_SPACE_CHARS.length) ;
    37   * 
    38   * trace("---------") ;
    39   * 
    40   * trace( "## Use the UnicodeChar constants." ) ;
    41   * // | UnicodeChar.TAB hello world : |	hello world
    42   * trace ("| UnicodeChar.TAB hello world : " +  "|" + UnicodeChar.TAB + "hello world" ) ;
    43   * 
    44   * trace("---------") ;
    45   * 
    46   * trace( "## Test if a charactere is a whitespace." ) ;
    47   * // UnicodeChar.TAB isWhiteSpace : true
    48   * trace ("UnicodeChar.TAB isWhiteSpace : " + UnicodeChar.isWhiteSpace( UnicodeChar.TAB ) ) ;
    49   * 
    50   * trace("---------") ;
    51   * 
    52   * trace( "## Converts a String unicode number representation in this character representation." ) ;
    53   * // UnicodeChar.toChar("0040" ) : @
    54   * trace( 'UnicodeChar.toChar("0040" ) : ' + UnicodeChar.toChar( "0040" ) ) ;
    55   * 
    56   * trace("---------") ;
    57   * 
    58   * trace( "## Use a UnicodeChar instance and this proxy process." ) ;
    59   * var u:UnicodeChar = new UnicodeChar() ;
    60   * // u.u5c0f() + u.u98fc() + u.u5f3e() + u.u0040() ) : 小飼弾@
    61   * trace( 'u.u5c0f() + u.u98fc() + u.u5f3e()) : ' + u.u5c0f() + u.u98fc() + u.u5f3e()  ) ;
    62   * }
    63   * @author eKameleon
    64   * @see <a href='http://www.ecma-international.org/publications/files/ECMA-ST/Ecma-262.pdf'>ECMAScript 262 specifications</a>
    65   */
    66  dynamic class vegas.string.UnicodeChar

    77  	public function UnicodeChar()

    95  	public static var SIMPLE_QUOTE:String = "u0027" ;
    96  	
    97  	/**

   100  	public static var DOUBLE_QUOTE:String = "u0022" ;

   110  	public static var VT:String = "u000B" ;

   130  	public static var WHITE_SPACE_CHARS:Array = [ TAB, VT, FF, SP, NBSP ] ;

   135  	public static var LF:String = "u000A" ;

   152  	/**

   154  	 */
   155  	public static var LINE_TERMINATOR_CHARS:Array = [ LF, CR, LS, PS ] ;
   156  	
   157  	/**
   158  	 * Returns {@code true} of the specified character is a whitespace.
   159  	 * @return {@code true} of the specified character is a whitespace.
   160  	 */
   161  	public static function isWhiteSpace( char:String ):Boolean 
   162  	{
   163  		var c:String = (char || "").substring(0, 1) ;
   164  		var ar:Array = WHITE_SPACE_CHARS ;
   165  		var l:Number = ar.length ;
   166  		while (--l > -1) 
   167  		{
   168  			if (ar[l] == c) 
   169  			{
   170  				return true ;
   171  			}
   172  		}
   173  		return false ;
   174  	}
   175  
   176  	/**
   177  	 * Returns {@code true} of the specified character is a line terminator.
   178  	 * @return {@code true} of the specified character is a line terminator.
   179  	 */
   180  	public static function isLineTerminators( char:String ):Boolean 
   181  	{
   182  		var c:String = (char || "").substring(0, 1) ;
   183  		var ar:Array = LINE_TERMINATOR_CHARS ;
   184  		var l:Number = ar.length ;
   185  		while (--l > -1) 
   186  		{
   187  			if (ar[l] == c) 
   188  			{
   189  				return true ;
   190  			}
   191  		}
   192  		return false ;
   193  	}

   195  	/**
   196  	 * Converts a unicode representation in a Char representation and returns this char's string.
   197  	 * @return The char of the unicode representation.
   198  	 */
   199  	public static function toChar( unicode:String ):Char 
   200  	{
   201  		return new Char( toCharString(unicode) ) ;
   202  	}
   203  
   204  	/**
   205  	 * Converts a unicode representation in a String representation and returns the new String.
   206  	 * @return The char of the unicode representation.
   207  	 */
   208  	public static function toCharString( unicode:String ):String 
   209  	{
   210  		return String.fromCharCode( parseInt( unicode, 16) )  ;
   211  	}
   212  
   213  	/**
   214  	 * Converts an unicode number value and returns this string representation.
   215  	 * @return The string representation of a unicode number.
   216  	 */
   217      public static function toUnicode(n:Number):String 
   218      {
   219  		var hex:String = n.toString(16) ;
   220  		while( hex.length < 4 )
   221  		{
   222  			hex = "0" + hex ;
   223  		}
   224  		return hex ;
   225  	}
   226  	
   227  	/**
   228  	 * Resolve the methods used over this instance.
   229  	 * If this instance receive an unknow method with the format "uXXXX" this method returns the unicode Char.  
   230  	 */
   231  	private function __resolve( name:String ) 
   232  	{
   233    		return function() 
   234    		{ 
   235    			return UnicodeChar.toCharString( "" + StringUtil.replace( name, "u" , "" ) ) ; 
   236    		} ;
   237  	};
   238  }
