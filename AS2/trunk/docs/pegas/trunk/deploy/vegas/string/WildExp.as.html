    25  import vegas.string.UnicodeChar;
    26  import vegas.util.StringUtil;

    28  /**
    29   * A wild expression is constructed like a regular expression but is based on a globbing algorithm like the one you found for searching files on a hard-drive.
    30   * <p><b>Syntax :</b>
    31   * 	<p>1. ? : match any unique char - exactly 1 char must be found</p>
    32   * 	<p>2. * : match any number of chars - at least 0 or more any chars can be found</p>
    33   * 	<p>3. ! : ignore next char</p>
    34   * </p>
    35   * <p><b>Rules :</b>
    36   *	<p>1. "*" is equal to "**" to "***" to an infinity number of "*" chars</p>
    37   *	<p>2. "*?" is concatened to "*" (more precisely "*?*")</p>
    38   *	<p>3. "*" if for zero char to infinity chars</p>
    39   *	<p>4. "!*" ignore the pattern and look for the char "*" in string</p>
    40   *	<p>5. "!?" ignore the pattern and look for the char "?" in string</p>
    41   *	</p>
    42   *	<p><b>Tips and tricks :</b>
    43   *	
    44   *	<p><b>1. find all the words in a string</b>
    45   *	{@code 
    46   *	import vegas.string.WildExp ;
    47   *	var we:WildExp = new WildExp( "*", WildExp.IGNORECASE | WildExp.MULTIWORD );
    48   *	result = we.test( "any phrases with words inside" );
    49   *	//result = [ "any", "phrases", "with", "words", "inside" ];
    50   *	}
    51   *	</p>
    52   *	<p><b>2. find comments in a string</b>
    53   *	{@code 
    54   *	import vegas.string.WildExp ;
    55   *	var we:WildExp = new WildExp( "*\/!**!*!/\*", WildExp.IGNORECASE | WildExp.MULTIWORD );
    56   *	result = we.test( "toto = \"123\"; /\*hello world*\/" );
    57   *	//result = [ "toto = \"123\"; ", "hello world" ] ;
    58   *	}
    59   *	</p>
    60   *	<p><b>3. find the name, arguments and body of a function</b>
    61   *	{@code 
    62   *	import vegas.string.WildExp ;
    63   *	var we:WildExp = new WildExp( "function *(*)*{*}", WildExp.IGNORECASE | WildExp.MULTIWORD );
    64   *	result = we.test( "function toto( a, b, c )\r\n{\treturn \"hello world\";\r\n\t}" );
    65   *	//result = [ "toto", " a, b, c ", "\r\n", "\treturn \"hello world\";\r\n\t" ];
    66   *	}
    67   *	</p>
    68   *	<p>Thanks Zwetan : <a href='http://groups.google.com/group/FCNG/browse_thread/thread/467e11b594bb180d/4baedfcc0d2cd95f?lnk=st&q=%5BFMX%5D+WildExp&rnum=1#4baedfcc0d2cd95f'>WildExp in FCNG</a></p>
    69   *	
    70   *	@see vegas.string.StringUtil replace() method
    71   *	@see vegas.string.UnicodeChar WHITE_SPACE_CHARS and LINE_TERMINATOR_CHARS constants
    72   *	@author eKameleon
    73   */
    74  class vegas.string.WildExp extends CoreObject 
    75  {
    76  
    77  	/**
    78  	 * Creates a new WildExp instance.
    79  	 */
    80  	public function WildExp (pattern:String, flag:Number) 
    81  	{
    82  		
    83  		if( isNaN(flag) ) 
    84  		{
    85  			flag = NONE ;
    86  		}
    87         	wildcards = [] ;
    88  		questionMarks = [] ;
    89  		
    90  		_caseSensitive = true ;
    91  		_multiline = false ;
    92  		_multiword = false ;
    93  		_wildcardFound = false;
    94  		_questionMarksFound = false ;
    95      
    96  		switch(flag) 
    97  		{
    98          	case IGNORECASE : // 1
    99          	{
   100  				_caseSensitive = false ;
   101  				break ;
   102          	}
   103          	case MULTILINE :
   104          	{
   105  				_multiline = true ; // 2
   106  				break ;
   107          	}
   108  			case IGNORECASE | MULTILINE : // 3
   109  			{
   110  				_caseSensitive = false ;
   111  				_multiline = true ;
   112  				break ;
   113  			}
   114  			case MULTIWORD : // 4
   115  			{
   116  				_multiword = true ;
   117  				break ;
   118  			}
   119  			case IGNORECASE | MULTIWORD : // 5
   120  			{
   121  				_caseSensitive = false ;
   122  				_multiword = true ;
   123  				break ;
   124  			}
   125  			case MULTILINE | MULTIWORD : // 6
   126  			{
   127  				_multiline = true ;
   128  				_multiword = true ;
   129  				break ;
   130  			}
   131  			case IGNORECASE | MULTILINE | MULTIWORD : // 7
   132  			{
   133  				_caseSensitive = false ;
   134  				_multiline = true ;
   135  				_multiword = true ;
   136  				break ;
   137  			}
   138  			case NONE : 
   139  			default :
   140  			{
   141  				
   142  			}
   143          }
   144  		
   145  		if( _caseSensitive ) {
   146  			pattern = pattern.toLowerCase();
   147  		}
   148          
   149  		source = pattern ;
   150      
   151      }

   153  	/**
   154  	 * const the NONE value (0).
   155  	 */	
   156  	static public var NONE:Number = 0 ;
   157  
   158  	/**
   159  	 * const the IGNORECASE value (1).
   160  	 */	
   161  	static public var IGNORECASE:Number = 1 ;
   162  
   163  	/**
   164  	 * const the MULTILINE value (2).
   165  	 */	
   166  	static public var MULTILINE:Number = 2 ;
   167  	
   168  	/**
   169  	 * const the MULTIWORD value (4).
   170  	 */	
   171  	static public var MULTIWORD:Number = 4 ;
   172  
   173  	static private var __ASPF__ = _global.ASSetPropFlags(WildExp, null , 7, 7) ;

   175  	/**
   176  	 * The array of all wildcards.
   177  	 */		
   178  	public var wildcards:Array ;
   179  	
   180  	/**
   181  	 * The array of all question marks.
   182  	 */
   183  	public var questionMarks:Array ;
   184  	
   185  	/**
   186  	 * The source of this wildcard.
   187  	 */
   188  	public var source:String ;
   189  	
   190  	/**
   191  	 * Adds a caracter in the questionMarks array, only if the array isn't empty.
   192  	 */
   193  	public function addToQuestionMarks(chr:String):Void 
   194  	{
   195  		var l:Number = questionMarks.length ;
   196  		if( l == 0 ) return ;
   197          questionMarks[l-1] += chr ;
   198      }
   199  	
   200  	/**
   201  	 * Adds a caracter in the wildcards array, only if the array isn't empty.
   202  	 */
   203  	public function addToWildcards(chr:String):Void 
   204  	{
   205  		var l:Number = wildcards.length ;
   206  		if( l == 0 ) return ;
   207  		wildcards[l-1] += chr ;
   208      }

   210  	/**
   211  	 * Test the specific expression in argument.
   212  	 */
   213  	public function test(str:String) 
   214  	{
   215  		var segment, result:Array ;
   216  		var i, j, k, l:Number ;
   217  		var ORC:String  = "uFFFC";
   218  		var CRLF:String = "r\n";
   219  		if( _caseSensitive ) str = str.toLowerCase();
   220          if(_multiline) {
   221  			var lineTerminatorChars:Array = UnicodeChar.LINE_TERMINATOR_CHARS ;
   222  			if( str.indexOf( CRLF ) > -1 ) 
   223  			{
   224  				str = StringUtil.replace(str, CRLF, ORC ) ;
   225  			}
   226  			l = lineTerminatorChars.length ;
   227  			for( i=0 ; i<l ; i++ ) 
   228  			{
   229  				if( str.indexOf( lineTerminatorChars[i] ) > -1 ) 
   230  				{
   231  					str = StringUtil.replace(str, lineTerminatorChars[i], ORC ) ;
   232  				}
   233              }
   234          }
   235  		if( _multiword ) 
   236  		{
   237  			var whiteSpaceChars:Array = UnicodeChar.WHITE_SPACE_CHARS ; 
   238  			l = whiteSpaceChars.length ;
   239  			for( j=0 ; j<l ; j++ ) 
   240  			{
   241  				if( str.indexOf( whiteSpaceChars[j] ) > -1 ) 
   242  				{
   243  					str = StringUtil.replace(str, whiteSpaceChars[j], ORC ) ;
   244  				}
   245              }
   246          }
   247  		if( str.indexOf( ORC ) > -1 ) 
   248  		{
   249  			segment = str.split( ORC ) ;
   250  			result = [] ;
   251  			l = segment.length ;
   252  			for( k=0 ; k<l ; k++ ) 
   253  			{
   254  				if( _testMatch( segment[k], source ) ) result.push( segment[k] ) ;
   255  		    }
   256  			return ( result.length > 0 ) ? result : false ;
   257          }
   258  		return _testMatch( str, source ) ;
   259      }

   261  	/**
   262  	 * Returns the string representation of this object.
   263  	 * @return the string representation of this object.
   264  	 */	
   265  	public function toString():String 
   266  	{
   267  		return source ;
   268  	}

   270  	private var _caseSensitive:Boolean ;
   271  
   272  	private var _multiline = false ;
   273  
   274  	private var _multiword = false ;

   276  	private var _wildcardFound = false;
   277  
   278  	private var _questionMarksFound = false;
   279  	
   280  	private function _testMatch(str:String, pattern:String, ignoreChar:String ):Boolean 
   281  	{
   282  		
   283  		str = new String(str) ;
   284  		pattern = new String(pattern) ;
   285  				
   286  		var c, c1, pat, pat1:String ;
   287  		c = str.charAt( 0 ) ;
   288  		c1 = str.charAt( 1 ) ;
   289  		pat = pattern.charAt( 0 ) ;
   290  		pat1 = pattern.charAt( 1 ) ;
   291      
   292  		if( pat != "?" ) _questionMarksFound = false ;
   293          if( pat == "!" ) return _testMatch( str, pattern.substr( 1 ), pat1 );
   294          
   295  		if( pat == "?" && ignoreChar != "?" ) 
   296  		{
   297  			if( c != "" ) 
   298  			{
   299  				if( _questionMarksFound ) addToQuestionMarks( c );
   300                  else questionMarks.push( c ) ;
   301  				_questionMarksFound = true ;
   302  				if ((pat1 == "") && (_wildcardFound == true)) pattern += "*" ;
   303                  return _testMatch( str.substr( 1 ), pattern.substr( 1 ) ) ;
   304              } 
   305              else 
   306              {
   307  				return false ;
   308              }
   309          }
   310  		
   311  		if( pat == "*" && ignoreChar != "*" ) 
   312  		{
   313  			_wildcardFound = true ;
   314  			if( pat1 != "*" ) wildcards.push( "" );
   315              if( pat1 == "" ) 
   316              {
   317  				addToWildcards( str ) ;
   318  				return true ;
   319              }
   320  			if( pattern.substr(1).indexOf( "*" ) > -1 ) 
   321  			{
   322  				while( str != "" ) 
   323  				{
   324  					if( pat1 == "*" ) break ;
   325  					if( pat1 == "?" ) 
   326  					{
   327  						pattern = pattern.substr(1) ;
   328  						break ;
   329  					}
   330  					if( str == "" ) return false ;
   331  					if( str.charAt(0) == pat1 ) break ;
   332  					if( pat1 == "!" ) 
   333  					{
   334  						ignoreChar = pattern.charAt( 2 ) ;
   335  						pattern = pattern.substr( 2 ) ;
   336  						pat1 = pattern.charAt( 1 ) ;
   337  					}
   338  					if( str.charAt(0) == ignoreChar ) 
   339  					{
   340  						str = str.substr(1) ; continue ;
   341  					}
   342  					addToWildcards( str.charAt( 0 ) ) ;
   343  					str = str.substr(1) ;
   344  				}
   345  				return _testMatch( str, pattern.substr( 1 ), ignoreChar ) ;
   346  			} 
   347  			else 
   348  			{
   349  				var found:Number = str.lastIndexOf( pat1 ) ;
   350  				if( found != -1 ) 
   351  				{
   352  					addToWildcards( str.substring( 0, found ) ) ;
   353  					return _testMatch( str.substr( found ), pattern.substr( 1 ) ) ;
   354                  }
   355                  else if( pat1 == "?" ) 
   356                  {
   357  					if( pattern.charAt(2) == "" ) return str.length >= 1 ;
   358                      else return _testMatch( str.substr( 1 ), pattern.substr( 1 ) );
   359                  }
   360  				return false;
   361              }
   362          }
   363  		
   364  		if( pat == "" ) return( c == "" );
   365          if( c != pat ) return false ;
   366  		
   367          if( c != "" ) return _testMatch( str.substr( 1 ), pattern.substr( 1 ) ) ;
   368          else return false ;
   369          
   370      }
