    24  import vegas.core.CoreObject;
    25  import vegas.core.IEquality;
    26  import vegas.util.Comparater;
    27  import vegas.util.ConstructorUtil;
    28  
    29  /**
    30   * Base object for representing an errors.
    31   * @author eKameleon
    32   */
    33  class vegas.errors.ErrorElement extends CoreObject implements IEquality 
    34  {
    35      
    36  	/**
    37  	 * Creates a new ErrorElement instance.
    38  	 */
    39  	public function ErrorElement(thrower, methodName:String, args:Array, fileName:String, lineNumber:Number, isNative:Boolean) 
    40  	{
    41  		_args = args || null ;
    42  		_fileName = fileName || null ;
    43  		_lineNumber = lineNumber || null ;
    44  		_isNative = isNative || null ;
    45  		_methodName = methodName || null ;
    46  		_thrower = thrower || null ;
    47  	}
    48  	
    49  	/**
    50  	 * Indicates whether some other object is "equal to" this one.
    51  	 */
    52  	public function equals(o):Boolean 
    53  	{
    54  		if (!o instanceof ErrorElement) 
    55  		{
    56  			return false ;
    57  		}
    58  		var b1:Boolean = o.getConstructorName() == getConstructorName()  ;
    59  		var b2:Boolean = o.getConstructorPath() == getConstructorPath() ;
    60  		var b3:Boolean = Comparater.compare(o.getArguments(), getArguments()) ;
    61  		var b4:Boolean = o.getFileName() == getFileName() ;
    62  		var b5:Boolean = o.getLineNumber() == getLineNumber() ;
    63  		var b6:Boolean = o.getMethodName() == getMethodName() ;
    64  		var b7:Boolean = o.getThrower() == getThrower() ;
    65  		return b1 && b2 && b3 && b4 && b5 && b6 && b7 ;
    66  	}
    67  	
    68  	/**
    69  	 * Returns an array with parameters of the thrower method.
    70  	 */
    71  	public function getArguments():Array 
    72  	{
    73  		return _args ;
    74  	}
    75  	
    76  	/**
    77  	 * Returns the fully qualified name of the class containing the execution point represented by this stack trace element.
    78  	 */
    79  	public function getConstructorName():String 
    80  	{
    81  		return ConstructorUtil.getName(_thrower) ;
    82  	}
    83  	
    84  	/**
    85  	 * Returns the fully qualified path of the class containing the execution point represented by this stack trace element.
    86  	 */
    87  	public function getConstructorPath():String 
    88  	{
    89  		return ConstructorUtil.getPath(_thrower) ;
    90  	}
    91  	
    92  	/**
    93  	 * Returns the name of the source file containing the execution point represented by this stack trace element.
    94  	 */
    95  	public function getFileName():String 
    96  	{
    97  		return _fileName ;
    98  	}
    99  	
   100  	/**
   101  	 * Returns the line number of the source line containing the execution point represented by this stack trace element.
   102  	 */
   103  	public function getLineNumber():Number 
   104  	{
   105  		return _lineNumber ;
   106  	}
   107  	
   108  	/**
   109  	 * Returns the name of the method containing the execution point represented by this stack trace element.
   110  	 */
   111  	public function getMethodName():String 
   112  	{
   113  		return _methodName ;
   114  	}
   115  	
   116  	/**
   117  	 * Return the thrower object reference.
   118  	 */
   119  	public function getThrower() 
   120  	{
   121  		return _thrower ;
   122  	}
   123  
   124  	/**
   125  	 * Returns true if the method containing the execution point represented by this stack trace element is a native method.
   126  	 */
   127  	public function isNativeMethod():Boolean 
   128  	{
   129  		return _isNative ;
   130  	}
   131  	
   132  	/**
   133  	 * Internal array parameters of the thrower method.
   134  	 */
   135  	private var _args:Array ;
   136  
   137  	/**
   138  	 * Internal file name.
   139  	 */
   140  	private var _fileName:String ;
   141  
   142  	/**
   143  	 * Internal line number.
   144  	 */
   145  	private var _lineNumber ;
   146  
   147  	/**
   148  	 * Internal value to represented is the trace is a native method.
   149  	 */
   150  	private var _isNative:Boolean ;
   151  
   152  	/**
   153  	 * The name of the method.
   154  	 */
   155  	private var _methodName:String ;
   156  
   157  	/**
   158  	 * The internal thrower.
   159  	 */
   160  	private var _thrower ;
   161  	
   162  }
