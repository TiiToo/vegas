    43  /**
    44   * Linked list implementation of the List and Queue interface. 
    45   * <p>Implements all optional list operations, and permits all elements (including null).</p>
    46   * <p>In addition to implementing the List interface, the {@code LinkedList} class provides uniformly named methods to get, remove and insert an element at the beginning and end of the list.</p>
    47   * <p>These operations allow linked lists to be used as a stack, queue, etc.</p>
    48   * <p><b>Example :</b></p>
    49   * {@code
    50   * import vegas.data.Collection ;
    51   * import vegas.data.collections.SimpleCollection ;
    52   * import vegas.data.iterator.Iterator ;
    53   * import vegas.data.iterator.ListIterator ;
    54   * import vegas.data.list.LinkedList ;
    55   * 
    56   * var c1:Collection = new SimpleCollection() ;
    57   * c1.insert("item0") ;
    58   * c1.insert("item1") ;
    59   * c1.insert("item2") ;
    60   * c1.insert("item3") ;
    61   * c1.insert("item4") ;
    62   * c1.insert("item5") ;
    63   * c1.insert("item6") ;
    64   * c1.insert("item7") ;
    65   * c1.insert("item8") ;
    66   * c1.insert("item9") ;
    67   * 
    68   * trace("c1 : " + c1) ;
    69   * 
    70   * var c2:Collection = new SimpleCollection() ;
    71   * c2.insert("item7") ;
    72   * c2.insert("item8") ;
    73   * 
    74   * trace("c2 : " + c1) ;
    75   * 
    76   * var list:LinkedList = new LinkedList(c1) ;
    77   * 
    78   * trace ("list toSource : " + list.toSource()) ;
    79   * 
    80   * trace ("-----") ;
    81   * 
    82   * trace ("create list : " + list) ;
    83   * 
    84   * trace("list.contains('item1') : " + list.contains("item1") ) ;
    85   * 
    86   * list.insertAt(2,'test')
    87   * trace("list.insertAt(2,'test')" + list ) ;
    88   * 
    89   * list.removeAt(2) ;
    90   * trace ("list.removeAt(2) : " + list) ;
    91   * 
    92   * list.remove("item4") ;
    93   * trace ("list.remove('item4') : " + list) ;
    94   * 
    95   * list.removeFirst() ;
    96   * trace ("list.removeFirst() : " + list) ;
    97   * 
    98   * list.removeLast() ;
    99   * trace ("list.removeLast() : " + list) ;
   100   * 
   101   * list.removesAt(1, 4) ;
   102   * trace ("list.removesAt(1, 4) : " + list) ;
   103   * 
   104   * list.removeRange(1, 3) ;
   105   * trace ("list.removeRange(1, 3) : " + list) ;
   106   * 
   107   * trace ("-----") ;
   108   * 
   109   * list.insertAllAt(0, c2) ;
   110   * trace ("list.insertAllAt(0, c2) : " + list) ;

   116   * while (it.hasNext())
   117   * {
   118   *     it.next() ;
   119   *     it.set("element" + i++) ;
   120   * }
   121   * trace (">> " + list) ;
   122   * 
   123   * trace ("---") ;
   124   * 
   125   * var it:ListIterator = list.listIterator(list.size()) ;
   126   * while(it.hasPrevious())
   127   * {
   128   *     trace (it.previous()) ;
   129   * }
   130   * 
   131   * //*
   132   * trace ("-----") ;
   133   * 
   134   * list.insertFirst("begin") ;
   135   * trace ("list.insertFirst('begin') : " + list) ;
   136   *  
   137   * list.insertLast("end") ;
   138   * trace ("list.insertLast('end') : " + list) ;
   139   * 
   140   * trace ("list.getFirst() : " + list.getFirst()) ;

   161  	 * Creates a new LinkedList instance.
   162  	 */
   163  	function LinkedList( c:Collection )
   164  	{
   165  		_header = new LinkedListEntry(null, null, null) ;
   166  		_header.next = _header.previous = _header ;	
   167  		if (c != null && c.size() > 0)
   168  		{
   169  			insertAll(c) ;
   170  		}
   171  	}

   174  	 * Removes all of the elements from this queue (optional operation).
   175  	 */
   176  	public function clear():Void
   177  	{
   178  		var e:LinkedListEntry = _header.next;
   179          while (e != _header) 
   180          {
   181              var next:LinkedListEntry = e.next;
   182              e.next = e.previous = null;
   183              e.element = null;
   184              e = next;
   185          }
   186  		_header.next = _header.previous = _header ;
   187  		_size = 0 ;
   188  		_modCount++ ;
   189  	}

   392  	public function indexOf( o ):Number
   393  	{
   394  		var index:Number = 0 ;
   395  		var e:LinkedListEntry ;
   396          if ( o == null ) 
   397          {
   398              for ( e = _header.next ; e != _header ; e = e.next) 
   399              {
   400                  if ( e.element == null )
   401                  {
   402                      return index ;
   403                  }
   404                  index++ ;
   405              }
   406          } 
   407          else 
   408          {
   409              for ( e = _header.next ; e != _header ; e = e.next ) 
   410              {
   411              	if (o instanceof IEquality)
   412              	{
   413                  	if (o.equals(e.element))
   414                  	{
   415                      	return index;
   416                  	}
   417              	}
   418              	else
   419              	{
   420              		if (o == e.element)
   421              		{
   422              			return index ;	
   423              		}
   424              	}	
   425                  index++ ;
   426              }
   427          }
   428          return -1 ;
   429  	}

   446       * @param c the elements to be inserted into this list.

   455  	/**
   456  	 * Inserts all of the elements in the specified collection into this list at the specified position (optional operation).
   457  	 */
   458      public function insertAllAt( index:Number, c:Collection):Boolean
   459      {
   460  		
   461  		if (index < 0 || index > _size)
   462  		{
   463  			throw new IndexOutOfBoundsError("LinkedList insertAllAt method failed, index:"+index+ ", size: " + _size + "." ) ;
   464  		}
   465          
   466          var a:Array = c.toArray() ;
   467          var l:Number = a.length ;
   468          
   469          if (l == 0)
   470          {
   471              return false;
   472          }
   473  		
   474  		_modCount ++ ;
   475  
   476          var successor:LinkedListEntry = (index == _size) ? _header : _entry(index) ;
   477          var predecessor:LinkedListEntry = successor.previous ;
   478          
   479  		for ( var i:Number = 0 ; i < l ; i++) 
   480  		{
   481              var e:LinkedListEntry = new LinkedListEntry( a[i] , successor, predecessor ) ;
   482              predecessor.next = e ;
   483              predecessor = e ;
   484          }
   485  
   486          successor.previous = predecessor ;
   487  
   488          _size += l ;
   489  
   490          return true;
   491  
   492      }

   667  	/**
   668  	 * Removes all elements defined in the specified Collection in the list.
   669  	 * <p><b>Example :</b></p>
   670  	 * {@code
   671  	 * import vegas.data.collections.SimpleCollection ;
   672  	 * import vegas.data.list.LinkedList ;
   673  	 * var list:LinkedList = new LinkedList() ;
   674  	 * list.insert("item1") ;
   675  	 * list.insert("item2") ;
   676  	 * trace(list.removeAll(new SimpleCollection(["item1"])) ; // true
   677  	 * trace(list) ; // {item2}
   678  	 * }
   679  	 * @return {@code true} if all elements are find and remove in the list.
   680  	 */
   681  	public function removeAll(c:Collection) : Boolean 
   682  	{
   683  		if (c instanceof vegas.data.Collection)
   684  		{
   685  			var result:Boolean = true ;
   686  			var i:Iterator = c.iterator() ;
   687  			while( i.hasNext() )
   688  			{
   689  				var b:Boolean = remove( i.next()) ;
   690  				if ( !b )
