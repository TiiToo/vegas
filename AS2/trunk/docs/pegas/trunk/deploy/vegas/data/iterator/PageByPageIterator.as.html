    24  import vegas.core.CoreObject;
    25  import vegas.data.iterator.OrderedIterator;
    26  import vegas.errors.UnsupportedOperation;
    27  import vegas.util.MathsUtil;
    28  
    29  /**
    30   * An iterator page by page over an array who return an new array of elements.
    31   * If the step size value is {@code 1} the next and previous methods returns the single value element in the data array.
    32   * <p><b>Example :</b></p>
    33   * {@code
    34   * import vegas.data.iterator.PageByPageIterator
    35   * 
    36   * var ar:Array = [1, 2, 3, 4, 5, 6, 7, 8] ;
    37   * 
    38   * var it:PageByPageIterator = new PageByPageIterator(1, ar) ;
    39   *
    40   * var next:Function = function()
    41   * {
    42   *     if (!it.hasNext())
    43   *     {
    44   *         it.reset() ;
    45   *     }
    46   *     var next = it.next()
    47   *     trace( "> " + next + " : " + it.currentPage() ) ;
    48   * }
    49   * 
    50   * var prev:Function = function()
    51   * {
    52   *     if ( !it.hasPrevious())
    53   *     {
    54   *         it.lastPage() ;
    55   *     }
    56   *     var prev = it.previous() ;
    57   *     trace( "> " + prev + " : " + it.currentPage() ) ;
    58   * }
    59   * 
    60   * Key.addListener(this) ;
    61   * 
    62   * onKeyDown = function()
    63   * {
    64   *     var code:Number = Key.getCode() ;
    65   *     switch (code)
    66   *     {
    67   *         case Key.LEFT :
    68   *         {
    69   *             prev() ;
    70   *             break ;
    71   *         }
    72   *         case Key.RIGHT :
    73   *         {
    74   *             next() ;
    75   *             break ;
    76   *         }
    77   *     }
    78   * }
    79   * }
    80   * @author eKameleon
    81   */
    82  class vegas.data.iterator.PageByPageIterator extends CoreObject implements OrderedIterator 
    83  {
    84  	
    85  	/**
    86  	 * Creates a new PageByPageIterator.
    87  	 * @param stepSize the step size value.
    88  	 * @param data the array to enumerate.
    89  	 * @throws UnsupportedOperation if the data array is empty.
    90  	 */
    91  	public function PageByPageIterator( stepSize:Number, data:Array )
    92  	{
    93  		super();
    94  		var len:Number = data.length ;
    95  		if (len > 0)
    96  		{
    97  			_data = data ;
    98  			_key = 0 ;
    99  			_currentPage = 0 ;
   100  			_step = (stepSize > 1) ? stepSize : DEFAULT_STEP ; 
   101  			_pageCount =  Math.ceil(len / _step) ;
   102  		}
   103  		else
   104  		{
   105  			throw new UnsupportedOperation(this + " constructor failed, data length not must be empty") ;
   106  		}
   107  		
   108  	}
   109  
   110  	/**
   111  	 * The default step value in all the PageByPageIterators.
   112  	 */
   113  	static public var DEFAULT_STEP:Number = 1 ;
   114  
   115  	/**
   116  	 * Returns the current page value.
   117  	 * @return the number of the current page.
   118  	 */
   119  	public function currentPage():Number
   120  	{
   121  		return _currentPage ;
   122  	}
   123  	
   124  	/**
   125  	 * Returns the step size of this PageByPageIterator.
   126  	 * @return the step size of this PageByPageIterator.
   127  	 */
   128  	public function getStepSize():Number
   129  	{
   130  		return _step ;	
   131  	}
   132  
   133  	/**
   134  	 * Returns {@code true} if the iteration has more elements.
   135  	 * @return {@code true} if the iterator has more elements.
   136  	 */
   137  	public function hasNext() : Boolean 
   138  	{
   139  		return _key < _pageCount ;
   140  	}
   141  
   142  	/**
   143  	 * Checks to see if there is a previous element that can be iterated to.
   144  	 * @return {@code true} if the iterator has more elements.
   145  	 */
   146  	public function hasPrevious() : Boolean 
   147  	{
   148  		return _key > 1 ;
   149  	}
   150  
   151  	/**
   152  	 * Returns the current page number.
   153  	 * @return the current page number.
   154  	 */
   155  	public function key() 
   156  	{
   157  		return _currentPage ;
   158  	}
   159  	
   160  	/**
   161  	 * Seek the iterator in the last page of this object.
   162  	 */
   163  	public function lastPage():Void
   164  	{
   165  		seek( _pageCount + 1 ) ;
   166  	}
   167  
   168  	/**
   169  	 * Returns the next Array page of elements or the next element in the Array if the getStepSize() value is 1.
   170  	 * @return the next Array page of elements or the next element in the Array if the getStepSize() value is 1.
   171  	 */
   172  	public function next() 
   173  	{
   174  		var index:Number = _step * _key++ ;
   175  		_currentPage = _key ;
   176  		if (_step > 1)
   177  		{
   178  			return _data.slice(index, index + _step) ;
   179  		}
   180  		else
   181  		{
   182  			return _data[index] ;
   183  		}
   184  	}
   185  
   186  	/**
   187  	 * Returns the numbers of page of this iterator.
   188  	 * @return the numbers of page of this iterator.
   189  	 */
   190  	public function pageCount():Number
   191  	{
   192  		return _pageCount ;	
   193  	}
   194  
   195  	/**
   196  	 * Returns the previous Array page of elements or the previous element in the Array if the getStepSize() value is 1.
   197  	 * @return the previous element from the collection.
   198  	 */
   199  	public function previous() 
   200  	{
   201  		_currentPage -- ;
   202  		_key -- ;
   203  		var index:Number = _step * (_key-1) ;
   204  		if (_step > 1)
   205  		{
   206  			return _data.slice(index, index + _step) ;
   207  		}
   208  		else
   209  		{
   210  			return _data[index] ;	
   211  		}
   212  	}
   213  
   214  	/**
   215  	 * Unsupported operation in a PageByPageIterator.
   216  	 * @throws UnsupportedOperation the method remove() in this iterator is unsupported. 
   217  	 */
   218  	public function remove() 
   219  	{
   220  		throw new UnsupportedOperation(this + " remove method is unsupported by this instance.") ;
   221  	}
   222  
   223  	/**
   224  	 * Resets the key pointer of the iterator.
   225  	 */
   226  	public function reset():Void 
   227  	{
   228  		_key = 0 ;
   229  		_currentPage = 0 ;
   230  	}
   231  
   232  	/**
   233  	 * Seek the key pointer of the iterator.
   234  	 */
   235  	public function seek(n : Number):Void 
   236  	{
   237  		_key = MathsUtil.clamp( n++, 0, _pageCount+1 ) ;
   238  		_currentPage = _key ;
   239  	}
   240  
   241  	/**
   242  	 * The currentPage value.
   243  	 */
   244  	private var _currentPage:Number ;
   245  
   246  	/**
   247  	 * The data to iterate.
   248  	 */
   249  	private var _data:Array ;
   250  
   251  	/**
   252  	 * The current key of the iterator. 
   253  	 */
   254  	private var _key:Number ;
   255  	
   256  	/**
   257  	 * Count of pages.
   258  	 */
   259  	private var _pageCount:Number ;
   260  		
   261  	/**
   262  	 * The numbers of lines in a page.
   263  	 */
   264  	private var _step:Number ;
   265  	
   266  }
