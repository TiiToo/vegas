    24  import vegas.core.CoreObject;
    25  import vegas.data.bag.AbstractBag;
    26  import vegas.data.iterator.Iterator;
    27  import vegas.errors.ConcurrentModificationError;
    28  import vegas.errors.UnsupportedOperation;
    29  
    30  /**
    31   * Converts an Bag to an iterator.
    32   * @author eKameleon
    33   */
    34  class vegas.data.iterator.BagIterator extends CoreObject implements Iterator 
    35  {
    36  
    37  	/**
    38  	 * Creates a new BagIterator instance.
    39  	 * @param parent the bag (@code AbstractBag) used in this iterator.
    40  	 * @param support the iterator to support this iterator.
    41  	 */
    42      public function BagIterator( parent:AbstractBag, support:Iterator ) 
    43      {
    44  		_parent = parent;
    45  		_support = support ;
    46  		_current = null;
    47  		_mods = parent.getModCount() ;
    48  	}
    49  
    50  	/**
    51  	 * Returns {@code true} if the iteration has more elements.
    52  	 * @return {@code true} if the iteration has more elements.
    53  	 */	
    54  	public function hasNext():Boolean 
    55  	{
    56  		return _support.hasNext() ;
    57  	}
    58  
    59  	/**
    60  	 * Unsupported method in all BagIterator.
    61  	 * @throws UnsupportedOperation {@ode key} method in unsupported.
    62  	 */
    63  	public function key() 
    64  	{
    65  		throw new UnsupportedOperation(this + " 'key' method in unsupported.") ;
    66  	}
    67  
    68  	/**
    69  	 * Returns the next element in the iteration.
    70  	 * @return the next element in the iteration.
    71  	 */
    72  	public function next() 
    73  	{
    74  		if (_parent.getModCount() != _mods) 
    75  		{
    76  			throw new ConcurrentModificationError();
    77  		}
    78          _current = _support.next() ;
    79  		return _current;
    80      }
    81  
    82  	/**
    83  	 * Removes from the underlying collection the last element returned by the iterator (optional operation).
    84  	 */
    85  	public function remove() 
    86  	{
    87  		if (_parent.getModCount() != _mods) 
    88  		{
    89              throw new ConcurrentModificationError();
    90          }
    91          _support.remove() ;
    92          _parent.removeCopies(_current, 1);
    93  		_mods++ ;
    94      }
    95  
    96  	/**
    97  	 * Unsupported method in all BagIterator.
    98  	 * @throws UnsupportedOperation {@code reset} method in unsupported.
    99  	 */
   100  	public function reset():Void 
   101  	{
   102  		throw new UnsupportedOperation(this + " 'reset' method in unsupported.") ;
   103  	}
   104  
   105  	/**
   106  	 * Unsupported method in all BagIterator.
   107  	 * @throws UnsupportedOperation {@code seek} method in unsupported.
   108  	 */
   109  	public function seek(n:Number):Void 
   110  	{
   111  		throw new UnsupportedOperation(this + " 'seek' method in unsupported.") ;
   112  	}
   113  
   114  	private var _current = null ;
   115  
   116  	private var _mods:Number = 0;
   117  
   118  	private var _parent:AbstractBag = null ;
   119  
   120  	private var _support:Iterator = null ;
   121  
   122  }
