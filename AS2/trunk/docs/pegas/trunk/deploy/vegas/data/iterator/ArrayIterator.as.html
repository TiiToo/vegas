    24  import vegas.core.CoreObject;
    25  import vegas.data.iterator.Iterator;
    26  import vegas.util.MathsUtil;
    27  import vegas.util.serialize.Serializer;
    28  
    29  /**
    30   * Converts an {@code Array} to an iterator.
    31   * <p><b>Example :</b></p>
    32   * {@code
    33   * import vegas.data.iterator.ArrayIterator ;
    34   * import vegas.data.iterator.Iterator ;
    35   * 
    36   * var ar:Array = ["item1", "item2", "item3", "item4"] ;
    37   * var it:Iterator = new ArrayIterator(ar) ;
    38   * 
    39   * while (it.hasNext())
    40   * {
    41   *     trace (it.next()) ;
    42   * }
    43   * 
    44   * trace ("--- it reset") ;
    45   * 
    46   * it.reset() ;
    47   * while (it.hasNext()) 
    48   * {
    49   *     trace (it.next() + " : " + it.key()) ;
    50   * }
    51   * 
    52   * trace ("--- it seek 2") ;
    53   * 
    54   * it.seek(2) ;
    55   * while (it.hasNext())
    56   * {
    57   *     trace (it.next()) ;
    58   * }
    59   * 
    60   * trace ("---") ;
    61   * 
    62   * }
    63   * @author eKameleon
    64   */
    65  class vegas.data.iterator.ArrayIterator extends CoreObject implements Iterator 
    66  {
    67  
    68  	/**
    69  	 * Creates a new ArrayIterator instance.
    70  	 * @param a the array to enumerate with the iterator.
    71  	 */
    72  	public function ArrayIterator(a:Array) {
    73  		_a = a ;
    74  		_k = -1 ;
    75  	}
    76  	
    77  	/**
    78  	 * Returns {@code true} if the iteration has more elements.
    79  	 * @return {@code true} if the iteration has more elements.
    80  	 */	
    81  	public function hasNext():Boolean 
    82  	{
    83  		return _k < (_a.length - 1);
    84  	}
    85  
    86  	/**
    87  	 * Returns the current key of the internal pointer of the iterator (optional operation).
    88  	 * @return the current key of the internal pointer of the iterator (optional operation).
    89  	 */
    90  	public function key() 
    91  	{
    92  		return _k ;
    93  	}
    94  
    95  	/**
    96  	 * Returns the next element in the iteration.
    97  	 * @return the next element in the iteration.
    98  	 */
    99  	public function next() 
   100  	{
   101  		return _a[++_k] ;
   102  	}
   103  	
   104  	/**
   105  	 * Removes from the underlying collection the last element returned by the iterator (optional operation).
   106  	 */
   107  	public function remove() 
   108  	{
   109  		return _a.splice(_k--, 1);
   110  	}
   111  
   112  	/**
   113  	 * Reset the internal pointer of the iterator (optional operation).
   114  	 */
   115  	public function reset():Void {
   116  		_k = -1 ;
   117  	}
   118  
   119  	/**
   120  	 * Change the position of the internal pointer of the iterator (optional operation).
   121  	 */		
   122  	public function seek(n:Number):Void {
   123  		_k = MathsUtil.clamp ((n-1), -1, _a.length) ;
   124  	}
   125  
   126  	/**
   127  	 * Returns a Eden reprensation of the object.
   128  	 * @return a string representing the source code of the object.
   129  	 */
   130      /*override*/ public function toSource():String 
   131      {
   132      	return "new vegas.data.iterator.ArrayIterator(" + Serializer.toSource(_a) + ")" ;
   133      }
   134  			
   135  	/**
   136  	 * Current internal array.
   137  	 */
   138  	private var _a:Array ;
   139  	
   140  	/**
   141  	 * Current internal key pointer.
   142  	 */
   143  	private var _k:Number ;
   144  
   145  }
