    24  import vegas.core.IComparator;
    25  import vegas.data.queue.LinearQueue;
    26  
    27  /**
    28   * This queue orders elements according to an order specified at construction time, which is specified either according to their natural order or according to a IComparator object.
    29   * @author eKameleon
    30   * @see IComparator
    31   */
    32  class vegas.data.queue.PriorityQueue extends LinearQueue 
    33  {
    34  
    35  	/**
    36  	 * Creates a new PriorityQueue instance.
    37  	 * <p><b>Example :</b>
    38  	 * {@code
    39  	 * import vegas.data.queue.PriorityQueue ;
    40  	 * import vegas.util.comparators.StringComparator ;
    41  	 * 
    42  	 * var q:PriorityQueue = new PriorityQueue(null, ["item0", "item1"])  ;
    43  	 * 
    44  	 * trace ("queue size : " + q.size()) ;
    45  	 * trace ("enqueue item4 : " + q.enqueue ("item4")) ;
    46  	 * trace ("enqueue item3 : " + q.enqueue ("item2")) ;
    47  	 * trace ("enqueue item2 : " + q.enqueue ("item3")) ;
    48  	 * trace ("enqueue item2 : " + q.enqueue ("item1")) ;
    49  	 * 
    50  	 * trace("> " + q) ;
    51  	 * 
    52  	 * q.setComparator( new StringComparator() ) ;
    53  	 * 
    54  	 * trace("> " + q) ;
    55  	 * }
    56  	 * </p>
    57  	 * @param comp a IComparator object used in the PriorityQueue to defined the sort model when enqueue or modify the queue.
    58  	 * @param ar an Array with values to fill the queue.
    59  	 * @see IComparator
    60  	 */
    61  	public function PriorityQueue( comp:IComparator , ar:Array ) 
    62  	{
    63  		setComparator(comp) ;
    64  		var l:Number = ar.length ;
    65  		if (l > 0)
    66  		{
    67  			for (var i:Number = 0 ; i<l ; i++)
    68  			{
    69  				enqueue(ar[i]) ;	
    70  			}		
    71  		}
    72  	}
    73  
    74  	/**
    75  	 * Returns a shallow copy of this object.
    76  	 * @return a shallow copy of this object.
    77  	 */
    78  	public function clone() 
    79  	{
    80  		return new PriorityQueue(_comparator);
    81  	}
    82  	
    83  	/**
    84  	 * Returns the internal IComparator reference.
    85  	 */
    86  	public function comparator():IComparator 
    87  	{
    88  		return _comparator ;
    89  	}
    90  	
    91  	/**
    92  	 * Inserts the specified element into this queue, if possible.
    93  	 */
    94  	public function enqueue(o):Boolean 
    95  	{
    96  		var isEnqueue:Boolean = super.enqueue(o) ;
    97  		if ( isEnqueue && (_comparator != null) ) 
    98  		{
    99  			_a.sort(_comparator.compare) ;
   100  		}
   101  		return isEnqueue ;
   102  	}
   103  	
   104  	/**
   105  	 * Sets the IComparator reference of this object.
   106  	 * @param comp the IComparator reference of this object.
   107  	 */
   108  	public function setComparator( comp:IComparator ):Void
   109  	{
   110  		_comparator = comp ;
   111  		if (_comparator != null)
   112  		{
   113  			_a.sort(_comparator.compare) ;
   114  		}
   115  	}
   116  	
   117  	/**
   118  	 * The internal IComparator reference.
   119  	 */
   120  	private var _comparator:IComparator = null ;
   121  	
   122  }
