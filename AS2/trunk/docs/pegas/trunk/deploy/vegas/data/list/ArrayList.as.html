    29  /**
    30   * Resizable-array implementation of the List interface.
    31   * Implements all optional list operations, and permits all elements, including null.
    32   * <p><b>Example :</b></p>
    33   * {@code
    34   * import vegas.data.list.ArrayList ;
    35   * import vegas.data.List ;
    36   * import vegas.data.iterator.ListIterator ;
    37   * 
    38   * import vegas.util.ConstructorUtil ;
    39   * 
    40   * var a:Array = ["item0", "item1", "item2", "item3", "item4"] ;
    41   * 
    42   * var list:ArrayList = new ArrayList(a) ;
    43   * trace("constructor :: " + ConstructorUtil.getPath(list)) ;
    44   * trace ("list : " + list.size() + " >> " + list) ;
    45   * trace ("list.toSource : " + list.toSource()) ;
    46   * trace ("----") ;
    47   * 
    48   * var it:ListIterator = list.listIterator(2) ;
    49   * while(it.hasNext())
    50   * {
    51   *     trace(">> " + it.next() + " : " + it.key()) ;
    52   *     it.remove() ;
    53   * }
    54   * trace ("next : " + list) ;
    55   * 
    56   * trace ("---- ListIterator hasPrevious") ;
    57   * 
    58   * var cpt = list.size() ;
    59   * while(it.hasPrevious())
    60   * {
    61   *     it.previous() ;
    62   *     it.set("changeItem" +  cpt--) ;
    63   * }
    64   * trace ("list : " + list) ;
    65   * }
    66   * @author eKameleon
    67   */
    68  class vegas.data.list.ArrayList extends AbstractList 
    69  {
    70  	
    71  	/**
    72  	 * Creates a new ArrayList instance.
    73  	 */
    74  	function ArrayList() 
    75  	{
    76  		
    77  		var arg = arguments[0] ;
    78  		var it:Iterator ;
    79  		if (arg instanceof Array) 
    80  		{
    81  			
    82  			it = new ArrayIterator(arg) ;
    83  			while (it.hasNext()) 
    84  			{
    85  				insert(it.next()) ;
    86  			}
    87  			
    88  		} 
    89  		else if (arg instanceof Collection) 
    90  		{
    91  			
    92  			it = arg.iterator() ;
    93  			while (it.hasNext()) 
    94  			{
    95  				insert(it.next()) ;
    96  			}
    97  			
    98  		}
    99  		else if (typeof (arg) == "number") 
   100  		{
   101  			
   102  			_a.length = arg ;
   103  			
   104  		}
   105  	}
   106  
   107  	/**
   108  	 * Creates and returns a shallow copy of the object.
   109  	 * @return A new object that is a shallow copy of this instance.
   110  	 */	
   111  	public function clone() 
   112  	{
   113  		return new ArrayList(toArray()) ;
   114  	}
   115  	
   116  	/**
   117  	 * Increases the capacity of this ArrayList instance, if necessary, to ensure that it can hold at least the number of elements specified by the minimum capacity argument.
   118  	 */
   119  	public function ensureCapacity(n:Number):Void 
   120  	{
   121  		_a.length = n ;
   122  	}
   123  	
   124  }
