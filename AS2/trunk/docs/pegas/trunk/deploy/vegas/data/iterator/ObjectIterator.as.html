    24  import vegas.core.CoreObject;
    25  import vegas.data.iterator.Iterator;
    26  import vegas.util.MathsUtil;
    27  
    28  /**
    29   * Converts an object to an iterator.
    30   * <p><b>Example :</b></p>
    31   * {@code
    32   * import vegas.data.iterator.ObjectIterator ;
    33   * import vegas.data.iterator.Iterator ;
    34   * 
    35   * var o:Object = {} ;
    36   * for (var i:Number = 0 ; i<5; i++)
    37   * {
    38   *      o["prop"+i] = "item"+i ;
    39   * }
    40   * 
    41   * var it:Iterator = new ObjectIterator(o) ;
    42   * trace ("-- object iterator") ;
    43   * 
    44   * while (it.hasNext())
    45   * {
    46   *     var next = it.next() ;
    47   *     var index = ObjectIterator(it).index() ;
    48   *     var key = it.key() ;
    49   *     trace ("it >> " + index + " :: " + key + " : " + next) ;
    50   * }
    51   * 
    52   * trace ("-- it seek 1") ;
    53   * 
    54   * it.seek(1) ;
    55   * 
    56   * while (it.hasNext())
    57   * {
    58   *     it.next() ;
    59   *     trace ("it remove : " + it.remove()) ;
    60   * }
    61   * 
    62   * trace ("-- it reset") ;
    63   * 
    64   * it.reset() ;
    65   * 
    66   * while (it.hasNext())
    67   * {
    68   *     var next = it.next() ;
    69   *     var index = ObjectIterator(it).index() ;
    70   *     var key = it.key() ;
    71   *     trace ("it >> " + index + " :: " + key + " : " + next) ;
    72   * }
    73   * }
    74   * @author eKameleon
    75   */
    76  class vegas.data.iterator.ObjectIterator extends CoreObject implements Iterator 
    77  {
    78  
    79  	/**
    80  	 * Creates a new ObjectIterator instance.
    81  	 */
    82  	public function ObjectIterator(o) 
    83  	{
    84  		_o = o ;
    85  		_a = new Array() ;
    86  		_k = -1 ;
    87  		for (var each:String in o) 
    88  		{
    89  			if (typeof(o[each]) != "function") 
    90  			{
    91  				_a.push(each) ;
    92  			}
    93  		}
    94  		_len = _a.length ;
    95  	}
    96  	
    97  	/**
    98  	 * Returns {@code true} if the iteration has more elements.
    99  	 */	
   100  	public function hasNext():Boolean 
   101  	{
   102  		return _k <  (_len - 1) ;
   103  	}
   104  
   105  	/**
   106  	 * Returns the current index of the internal pointer of the iterator (optional operation).
   107  	 */
   108  	public function index():Number 
   109  	{
   110  		return _k ;
   111  	}
   112  
   113  	/**
   114  	 * Returns the current key value of the internal pointer of the iterator (optional operation).
   115  	 */
   116  	public function key() 
   117  	{
   118  		return _a[_k] ;
   119  	}
   120  
   121  	/**
   122  	 * Returns the next element in the iteration.
   123  	 */
   124  	public function next() 
   125  	{
   126  		return _o[ _a[++_k] ] ;
   127  	}
   128  
   129  	/**
   130  	 * Removes from the object the last element returned by the iterator (optional operation).
   131  	 */
   132  	public function remove() 
   133  	{
   134  		var p:String = _a[_k] ;
   135  		_a.splice(_k--, 1) ;
   136  		delete _o[p] ;
   137  		_len -- ;
   138  		return p ;
   139  	}
   140  
   141  	/**
   142  	 * Reset the internal pointer of the iterator (optional operation).
   143  	 */
   144  	public function reset():Void 
   145  	{
   146  		_k = -1 ;
   147  	}
   148  	
   149  	/**
   150  	 * Change the position of the internal pointer of the iterator (optional operation).
   151  	 */
   152  	public function seek(n:Number):Void 
   153  	{
   154  		_k = MathsUtil.clamp ((n-1), -1, _len) ;
   155  	}
   156  	
   157  	/**
   158  	 * The internal array.
   159  	 */
   160  	private var _a:Array ;
   161  	
   162  	/**
   163  	 * The internal key.
   164  	 */
   165  	private var _k:Number ;
   166  	
   167  	/**
   168  	 * The internal object reference.
   169  	 */
   170  	private var _o ;
   171  	
   172  	/**
   173  	 * The internal size of the array.
   174  	 */
   175  	private var _len:Number ;
   176  	
   177  }
