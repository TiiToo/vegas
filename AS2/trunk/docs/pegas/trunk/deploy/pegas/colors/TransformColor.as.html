    24  import pegas.colors.BasicColor;
    25  import pegas.colors.ColorRGB;
    26  
    27  /**
    28   * Transformation class to modify the filters on a Color object with Photoshop filters.
    29   * @author eKameleon
    30   */
    31  class pegas.colors.TransformColor extends BasicColor 
    32  {
    33  
    34  	/**
    35  	 * Creates a new TransformColor instance.
    36  	 */	
    37  	public function TransformColor (mc:MovieClip) 
    38  	{ 
    39  		super (mc) ;
    40  	}
    41  
    42  	/**
    43  	 * Corresponding a filter in mode "addition" in photoshop over a movieclip. 
    44  	 */
    45  	public function addition(n:Number):Void 
    46  	{
    47  		var o:Object = ColorRGB.hex2rgb(n) ;
    48  		setTransform ( { rb:o.r , gb:o.g , bb:o.b });
    49  	}
    50  
    51  	/**
    52  	 * Corresponding a filter in mode "colorDodge" in photoshop over a movieclip. 
    53  	 */
    54  	public function colorDodge (n:Number):Void 
    55  	{
    56  		var o:Object = ColorRGB.hex2rgb(n) ;
    57  		setTransform ({ ra:_dodge(o.r) , ga:_dodge(o.g) , ba:_dodge(o.b) } );
    58  	}
    59  
    60  	/**
    61  	 * Corresponding a filter in mode "difference" in photoshop over a movieclip. 
    62  	 */
    63  	public function difference (n:Number):Void 
    64  	{
    65  		var o:Object = ColorRGB.hex2rgb(n) ;
    66  		setTransform ( { rb:-o.r , gb:-o.g , bb:-o.b });
    67  	}
    68  	
    69  	/**
    70  	 * Corresponding a filter in mode "divide" in photoshop over a movieclip. 
    71  	 */
    72  	public function divide(n:Number):Void 
    73  	{
    74  		var o:Object = ColorRGB.hex2rgb(n) ;
    75  		setTransform ({ ra:_div(o.r) , ga:_div(o.g) , ba:_div(o.b) } );
    76  	}
    77  	
    78  	/**
    79  	 * Corresponding a filter in mode "linearBurn" in photoshop over a movieclip. 
    80  	 */
    81  	public function linearBurn (n:Number):Void 
    82  	{
    83  		var o:Object = ColorRGB.hex2rgb(n) ;
    84  		setTransform ({ rb:o.r-255 , gb:o.g-255, bb:o.b-255 });
    85  	}
    86  	
    87  	/**
    88  	 * Corresponding a filter in mode "linearDodge" in photoshop over a movieclip. 
    89  	 */
    90  	public function linearDodge (n:Number):Void 
    91  	{
    92  		var o:Object = ColorRGB.hex2rgb(n) ;
    93  		setTransform ({ rb:o.r , gb:o.g, bb:o.b });
    94  	}
    95  	
    96  	/**
    97  	 * Corresponding a filter in mode "multiply" in photoshop over a movieclip. 
    98  	 */
    99  	public function multiply(n:Number):Void 
   100  	{
   101  		var o:Object = ColorRGB.hex2rgb(n) ;
   102  		setTransform ({ ra:o.r/2.55 , ga:o.g/2.55 , ba:o.b/2.55 });
   103  	}
   104  
   105  	/**
   106  	 * Corresponding a filter in mode "negative" in photoshop over a movieclip. 
   107  	 */
   108  	public function negative (n:Number):Void 
   109  	{
   110  		setTransform ({ ra:-100, ga:-100, ba:-100, rb:255, gb:255, bb:255 });
   111  	}
   112  
   113  	/**
   114  	 * Corresponding a filter in mode "screen" in photoshop over a movieclip. 
   115  	 */
   116  	public function screen (n:Number):Void 
   117  	{
   118  		var o:Object = ColorRGB.hex2rgb(n) ;
   119  		setTransform ({ ra:_scr(o.r), ga:_scr(o.g), ba:_scr(o.b), rb:o.r, gb:o.g, bb:o.b } );
   120  	}
   121  
   122  	/**
   123  	 * Corresponding a filter in mode "substraction" in photoshop over a movieclip. 
   124  	 */
   125  	public function substraction (n:Number):Void 
   126  	{
   127  		var o:Object = ColorRGB.hex2rgb(n) ;
   128  		setTransform ({ ra:-100, ga:-100, ba:-100, rb:o.r, gb:o.g, bb:o.b } );
   129  	}
   130  	
   131  	private function _div(n:Number):Number 
   132  	{
   133  		return 100/((n + 1)/256);
   134  	}
   135  
   136  	private function _dodge(n:Number):Number 
   137  	{
   138  		return 100 / ((258 - n) / 256);
   139  	}
   140  	
   141  	private function _scr(n:Number):Number 
   142  	{
   143  		return 100 * (255 - n) / 255;
   144  	}
   145  	
   146  }
