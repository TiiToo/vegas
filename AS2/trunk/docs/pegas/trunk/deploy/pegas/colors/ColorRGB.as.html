    24  /**
    25   * This static tool class is used to transform the RGB color space on a Color instances.
    26   * <p>Thanks 2003 Robert Penner - Use freely, giving credit where possible.</p>
    27   * <p>This code is based on the book : Robert Penner's Programming Macromedia Flash MX. More informations in :
    28   * <ul>
    29   * <li>http://www.robertpenner.com/profmx
    30   * <li>http://www.amazon.com/exec/obidos/ASIN/0072223561/robertpennerc-20
    31   * </ul>
    32   * </p>
    33   * @author eKameleon
    34   */
    35  class pegas.colors.ColorRGB 
    36  {
    37  	
    38  	/**
    39  	 * Returns the string representation of the passed color with ECMAScript formatting (0xrrggbb).
    40  	 * @return the string representation of the passed color with ECMAScript formatting (0xrrggbb).
    41  	 */
    42  	public static function getRGBStr(c:Color):String 
    43  	{
    44  		var str:String = c.getRGB().toString(16);
    45  		var toFill:Number = 6 - str.length;
    46  		while (toFill--) 
    47  		{
    48  			str = "0" + str ;
    49  		}
    50  		return str.toUpperCase();
    51  	}
    52  	
    53  	/**
    54  	 * Returns the object representation of the color number passed in argument. 
    55  	 * This object contains the properties r, g and b for the red, the green and the blue component of the color.
    56  	 * @return the object representation of the color number passed in argument.
    57  	 */
    58  	public static function hex2rgb(hex:Number):Object 
    59  	{
    60  		var r,g,b,gb:Number ;
    61  		r = hex>>16 ;
    62  		gb = hex ^ r << 16 ;
    63  		g = gb>>8 ;
    64  		b = gb^g<<8 ;
    65  		return {r:r,g:g,b:b} ;
    66  	}
    67  	
    68  	/**
    69  	 * Converts the red, green, blue arguments in a HTML number color (base 10).
    70  	 */
    71  	public static function rgb2hex(r:Number, g:Number, b:Number):Number  
    72  	{
    73  		return ((r << 16) | (g << 8) | b);
    74  	}
    75  	
    76  	/**
    77  	 * Sets the color of the passed Color instance with the specified string expression of the color.
    78  	 */
    79  	public static function setRGBStr(c:Color, str:String):Void 
    80  	{
    81  		c.setRGB (parseInt (str.substr (-6, 6), 16));
    82  	}
    83  	
    84  }
