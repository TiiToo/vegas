    24  /**
    25   * Implements the static behaviours of the Prime Class.
    26   * @author eKameleon
    27   */
    28  class pegas.maths.Prime 
    29  {
    30  
    31  	/**
    32  	 * Defines an array of all primes between 'from' and 'n' inclusive, positive but restricted integer value, ignores decimals.
    33  	 * @return a array representation of all primes between {@code from} and {@code n} inclusive}
    34  	 */
    35  	static public function findPrimeFrom(n:Number,from:Number):Array 
    36  	{
    37  		n |= 0 ;
    38          from |= 0 ;
    39          var i,j:Number;
    40          var aOut:Array = [];
    41          var aCount:Array = [];
    42          if (!from) from = 0 ;
    43          else from -= 1 ;
    44          n += 1 ;
    45          for (i=0 ; i<n; i++) 
    46          {
    47          	aCount[i] = 0;
    48          }
    49          var sqrtN:Number = Math.round(Math.sqrt(n+1));
    50          var last:Number = 2 ;
    51          for (i=2 ; i<=sqrtN ;i++ ) 
    52          {
    53              if (aCount[i]==0) 
    54              {
    55                  for (j=last*i ; j<n ; j+=i ) 
    56                  {
    57                  	aCount[j] = 1 ;
    58                  }
    59                  last = i ;
    60              }
    61          }
    62          for (i=n-1 ;i>from ; i--) 
    63          {
    64          	if (aCount[i] == 0) 
    65          	{
    66          		aOut.push(i);
    67          	}
    68          }
    69          return aOut;
    70      }
    71  
    72  	/**
    73  	 * Defines an array of all primes between 2 and 'n' inclusive.
    74  	 */
    75  	static public function generatePrimes(limit:Number):Array 
    76  	{
    77  		var b:Boolean ;
    78  		var a:Array = new Array() ;
    79  		var i:Number = 1 ;
    80  		while (++i<=limit) if (isPrime(i)) a.push(i) ;
    81  		return a ;
    82  	}
    83  
    84  	/**
    85  	 * Boolean for 'isPrime' integer condition, ignores decimals.
    86  	 * Successive division.
    87  	 */
    88  	static public function isPrime(n:Number):Boolean 
    89  	{ 
    90  		if (n<3) return ( n == 2 ) ; 
    91  		else if ((n%2) == 0) return false ;
    92  		var i:Number = 3 ;
    93  		while (i*i <= n) {
    94  			if ( (n%i) == 0 ) return false ;
    95  			i+= 2 ;
    96  		}
    97  		return true ;
    98  	}
    99  	
   100  	/**
   101  	 * Defines prime factors of 'n', positive but restricted integer value, ignores decimals.
   102  	 * @return a string representation of the multiplication of primes of 'n'.
   103  	 */
   104  	static public function primeFactor(n:Number):String 
   105  	{
   106          var bFlag:Boolean;
   107          n |= 0;
   108          if (n==1)
   109          {
   110          	return "1";
   111          }
   112          var temp:Number = n;
   113          var delim:String = "*";
   114          var sFactor:String = "";
   115          while (1) 
   116          {
   117              if (temp%2==0) 
   118              {
   119                  temp /= 2;
   120                  sFactor += 2+delim;
   121              }
   122              else 
   123              {
   124              	break;
   125              }
   126          }
   127          var num:Number = 3;
   128          while (1<temp) 
   129          {
   130              bFlag = true;
   131              while (bFlag) 
   132              {
   133                  if (temp%num==0) 
   134                  {
   135                      temp /= num;
   136                      sFactor += num+delim;
   137                  }
   138                  else 
   139                  {
   140                  	bFlag = false;
   141                  }
   142              }
   143              num += 2;
   144          }
   145          return sFactor.substr(0,-1);
   146      }
   147  	
   148  	/**
   149  	 * Defines total of relative primes of 'n'.
   150  	 */
   151  	static public function totient(n:Number):Number 
   152  	{
   153          var k:Number = 1;
   154          var j:Number;
   155          if (n%2==0) j++;
   156          for (j=3;j<=n;j+=2) 
   157          {
   158          	if (Prime.isPrime(j)) 
   159          	{
   160          		k++;
   161          	}
   162          }
   163          return k;
   164      }
   165  	
   166  }
