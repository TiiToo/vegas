    24  import pegas.events.ActionEvent;
    25  import pegas.process.AbstractAction;
    26  import pegas.process.Action;
    27  
    28  import vegas.data.iterator.Iterator;
    29  import vegas.data.queue.LinearQueue;
    30  import vegas.data.queue.TypedQueue;
    31  import vegas.events.Delegate;
    32  import vegas.events.EventListener;
    33  import vegas.events.EventTarget;
    34  import vegas.util.serialize.Serializer;
    35  
    36  /**
    37   * A Sequencer of Action process.
    38   * @author eKameleon
    39   */
    40  class pegas.process.Sequencer extends AbstractAction 
    41  {
    42  	
    43  	/**
    44  	 * Creates a new Sequencer instance.
    45  	 * @param ar An Array of {@code Action} objects.
    46  	 */
    47  	public function Sequencer( ar:Array ) 
    48  	{
    49  		_queue = new TypedQueue(Action, new LinearQueue()) ; 
    50  		_runner = new Delegate(this, run) ;
    51  		var l:Number = ar.length ;
    52  		if (l>0) 
    53  		{
    54  			for (var i:Number = 0 ; i < l ; i++) 
    55  			{
    56  				var a:Action = ar[i] ;
    57  				if (a instanceof Action) 
    58  				{
    59  					addAction(ar[i]) ;
    60  				}		
    61  			}
    62  		}
    63  	}
    64  
    65  	/**
    66  	 * Adds a process(Action) in the Sequencer.
    67  	 * @return {@code true} if the method success.
    68  	 */
    69  	public function addAction(action:Action, isClone:Boolean):Boolean 
    70  	{
    71  		var a:Action = isClone ? action.clone() : action ;
    72  		var isEnqueue:Boolean = _queue.enqueue(a) ;
    73  		if (isEnqueue)
    74  		{
    75  			EventTarget(a).addEventListener( ActionEvent.FINISH, _runner ) ;
    76  		}
    77  		return isEnqueue ;
    78  	}
    79  	
    80  	/**
    81  	 * Removes all process in the Sequencer.
    82  	 * @param noEvent A boolean flag to disabled the events dispatched by this method if is {@code true}.
    83  	 * @param callback Function to map and check the current process in progress in the sequencer before reset it.
    84  	 */
    85  	public function clear( noEvent:Boolean, callback:Function ):Void 
    86  	{
    87  		if (running) 
    88  		{
    89  			EventTarget(_cur).removeEventListener(ActionEvent.FINISH, _runner) ;
    90  			if (callback != null)
    91  			{
    92  				callback.call( this, _cur ) ;
    93  			}
    94  			_setRunning(false) ;
    95  		}
    96  		_cur = null ;
    97  		_queue.clear() ;
    98  		if (noEvent) return ;
    99  		notifyCleared() ;
   100  	}
   101  
   102  	/**
   103  	 * Returns a shallow copy of this object.
   104  	 * @return a shallow copy of this object.
   105  	 */
   106  	public function clone() 
   107  	{
   108  		var s:Sequencer = new Sequencer() ;
   109  		var it:Iterator = _queue.iterator() ;
   110  		while (it.hasNext()) 
   111  		{
   112  			s.addAction(it.next().clone()) ;
   113  		}
   114  		return s ;
   115  	}
   116  
   117  	/**
   118  	 * Returns the current process in progress.
   119  	 * @return the current process in progress.
   120  	 */
   121  	public function getCurrent()
   122  	{
   123  		return _cur ;	
   124  	}
   125  
   126  	/**
   127  	 * Launchs the Sequencer with the first element in the internal Queue of this Sequencer.
   128  	 */
   129  	public function run():Void 
   130  	{
   131  		if (_queue.size() > 0) 
   132  		{
   133  			if (!running) 
   134  			{
   135  				notifyStarted() ;
   136  			}
   137  			else 
   138  			{
   139  				notifyProgress() ;
   140  			}
   141  			_setRunning(true) ;
   142  			_cur = _queue.poll() ;
   143  			_cur.run() ;
   144  		}
   145  		else 
   146  		{
   147  			EventTarget(_cur).removeEventListener(ActionEvent.FINISH, _runner) ;
   148  			_cur = null ;
   149  			if (running) 
   150  			{
   151  				_setRunning(false) ;
   152  				notifyFinished() ;
   153  			}
   154  		}
   155  	}
   156  	
   157  	/**
   158  	 * Returns the numbers of process in this Sequencer.
   159  	 * @return the numbers of process in this Sequencer.
   160  	 */
   161  	public function size():Number 
   162  	{
   163  		return _queue.size() ;
   164  	}
   165  	
   166  	/**
   167  	 * Starts the Sequencer if is not in progress.
   168  	 */
   169  	public function start():Void 
   170  	{
   171  		if (!running) 
   172  		{
   173  			run() ;
   174  		}
   175  	}
   176  	
   177  	/**
   178  	 * Stops the Sequencer. Stop only the last process if is running.
   179  	 * @param noEvent A boolean flag to disabled the events dispatched by this method if is {@code true}.
   180  	 * @param callback Function to map and check the current process in progress in the sequencer before reset it.
   181  	 */
   182  	public function stop( noEvent:Boolean , callback:Function )
   183  	{
   184  		if (running) 
   185  		{
   186  			EventTarget(_cur).removeEventListener(ActionEvent.FINISH, _runner) ;
   187  			if (callback != null)
   188  			{
   189  				callback.call( this, _cur ) ;
   190  			}
   191  			_cur = null ;
   192  			_setRunning(false) ;
   193  			if (noEvent) return ;
   194  			notifyStopped() ;
   195  			notifyFinished() ;
   196  		}
   197  	}
   198  	
   199  	/**
   200  	 * Returns the array representation of all process in this Sequencer.
   201  	 * @return the array representation of all process in this Sequencer.
   202  	 */
   203  	public function toArray():Array 
   204  	{
   205  		return _queue.toArray() ;	
   206  	}
   207  
   208  	/**
   209  	 * Returns the source of the specified object passed in argument.
   210  	 * @return the source of the specified object passed in argument.
   211  	 */
   212  	public function toSource():String 
   213  	{
   214  		var sourceA:String = Serializer.toSource(toArray()) ;
   215  		return Serializer.getSourceOf(this, [sourceA]) ;
   216  	}
   217  
   218  	private var _cur:Action ;
   219  	private var _queue:TypedQueue  ;
   220  	private var _runner:EventListener ;
   221  	
   222  }
