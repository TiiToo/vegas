    24  import vegas.core.IRunnable;
    25  import vegas.data.collections.SimpleCollection;
    26  import vegas.data.collections.TypedCollection;
    27  import vegas.data.iterator.Iterator;
    28  
    29  /**
    30   * A batch is a collection of {@code Action} objects. All {@code Action} objects are processed as a single unit.
    31   * This class use an internal typed {@code SimpleCollection} to register all {@code Action} objects.  
    32   * @author eKameleon
    33   */
    34  class pegas.process.Batch extends TypedCollection implements IRunnable 
    35  {
    36  
    37  	/**
    38  	 * Creates a new Batch instance.
    39  	 */
    40   	public function Batch() 
    41   	{
    42  		super(IRunnable, new SimpleCollection()) ;
    43    	}
    44  
    45  	/**
    46  	 * Returns a shallow copy of the object.
    47  	 * @return a shallow copy of the object.
    48  	 */
    49  	public function clone() 
    50  	{
    51  		var b:Batch = new Batch() ;
    52  		var it:Iterator = iterator() ;
    53  		while (it.hasNext()) b.insert(it.next()) ;
    54  		return b ;
    55  	}
    56  	
    57  	/**
    58  	 * Runs the process.
    59  	 */
    60   	public function run():Void 
    61   	{
    62  		var ar:Array = toArray() ;
    63  		var i:Number = -1 ;
    64  		var l:Number = ar.length ;
    65  		if (l>0) while (++i < l) 
    66  		{ 
    67  			ar[i].run() ; 
    68  		}
    69    	}
    70  	
    71  }
