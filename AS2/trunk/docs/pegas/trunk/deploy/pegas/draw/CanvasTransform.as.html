    24  /**
    25   * This tool class transform a {@code Canvas} object with static methods.
    26   * @author eKameleon
    27   * Thanks : Peter Hall <www.peterjoel.com> ASVDrawing Class v1.0
    28   */
    29  class pegas.draw.CanvasTransform {
    30  
    31  	
    32  	/**
    33  	 * Creates a free form effect over a {@code Canvas} object}.
    34  	 * @return the method to be used to creates a free form effect over a {@code Canvas} object}.
    35  	 */
    36  	static public function createFreeform 
    37  	(
    38  		xMin:Number, yMin:Number, xMax:Number, yMax:Number,
    39  		x0:Number, y0:Number, x1:Number, y1:Number, x2:Number, y2:Number, x3:Number, y3:Number ):Function  
    40  	{
    41  		var w:Number = xMax - xMin ;
    42  		var h:Number = yMax - yMin;
    43  		var w2_0:Number = x1 - x0 ;
    44  		var w2_1:Number = x2 - x3 ;
    45  		var h2_0:Number = y1 - y0 ;
    46  		var h2_1:Number = y2 - y3 ;
    47  		return function(x:Number, y:Number):Object{
    48  			var gx:Number = ( x - xMin ) / w ;
    49  			var gy:Number = ( y - yMin ) / h ;
    50  			var bx:Number = x0 + gy * (x3-x0) ;
    51  			var by:Number = y0 + gy * (y3-y0) ;
    52  			return {x: bx + gx*((x1+gy*(x2-x1))-bx), y: by + gx*((y1+gy*(y2-y1))-by)};
    53  		} ;
    54  	}
    55  	
    56  	/**
    57  	 * Creates a pinch effect over a {@code Canvas} object}.
    58  	 * @return the method to be used to creates a pinch effect over a {@code Canvas} object}.
    59  	 */
    60  	static public function createPinch(xMin:Number, yMin:Number, xMax:Number, yMax:Number, xAmount:Number, yAmount:Number):Function 
    61  	{
    62  		var w:Number = xMax - xMin ;
    63  		var h:Number = yMax - yMin ;
    64  		var h2:Number = h / 2 ;
    65  		var w2:Number = w / 2 ;
    66  		var ax:Number = xAmount * w / 200 ;
    67  		var ay:Number = yAmount * h / 200 ;
    68  		return function(x:Number,y:Number):Object 
    69  		{
    70  			var newX:Number ;
    71  			var newY:Number ;
    72  			if(x<xMin || x>xMax || y<yMin || y>yMax) return {x:x,y:y};
    73  			if(!isNaN(xAmount)) {
    74  				var y2:Number = (y-yMin)/h2-1;
    75  				var gx:Number = (1-y2*y2)*ax;
    76  				newX = xMin+gx+(x-xMin)*(w-2*gx)/w;
    77  			} else {
    78  				newX = x;
    79  			}
    80  			if(!isNaN(yAmount)) {
    81  				var x2:Number = (x-xMin)/w2-1;
    82  				var gy:Number = (1-x2*x2)*ay;
    83  				newY = yMin+gy+(y-yMin)*(h-2*gy)/h;
    84  			} else {
    85  				newY = y;
    86  			}
    87  			return { x : newX ,  y : newY } ;
    88  		} ;
    89  	}
    90  
    91  	/**
    92  	 * Creates a waves effect over a {@code Canvas} object}.
    93  	 * @return the method to be used to creates a waves effect over a {@code Canvas} object}.
    94  	 */
    95  	static public function createWaves
    96  	( 
    97  		xMin:Number, yMin:Number, xMax:Number, yMax:Number,
    98  		xAmount:Number, yAmount:Number, xCount:Number, yCount:Number, xOffset:Number, yOffset:Number):Function 
    99  	{
   100  		var w:Number = xMax-xMin;
   101  		var h:Number = yMax-yMin;
   102  		var jx:Number = Math.PI*xCount/h;
   103  		var jy:Number = Math.PI*yCount/w;
   104  		xOffset = xOffset*Math.PI/180;
   105  		yOffset = yOffset*Math.PI/180;
   106  		return function(x:Number,y:Number):Object 
   107  		{
   108  			if(x<xMin || x>xMax || y<yMin || y>yMax) return {x:x,y:y};
   109  			var newX:Number = xAmount ? x+xAmount*Math.sin(jx*y+xOffset) : x;
   110  			var newY:Number = yAmount ? y+yAmount*Math.sin(jy*x+yOffset) : y;
   111  			return {x:newX, y:newY};
   112  		} ;
   113  	}
   114  
   115  	/**
   116  	 * Creates a Whirl effect over a {@code Canvas} object}.
   117  	 * @return the method to be used to creates a Whirl effect over a {@code Canvas} object}.
   118  	 */
   119  	static public function createWhirl(centreX:Number, centreY:Number, radius:Number, amount:Number):Function 
   120  	{
   121  		var radius2:Number = radius * radius ;
   122  		var rotate:Number = amount*Math.PI/180 ;
   123  		return function (x:Number,y:Number):Object 
   124  		{
   125  			var x2:Number = x - centreX ;
   126  			var y2:Number = y - centreY ;
   127  			var r2:Number = x2*x2 + y2*y2 ;
   128  			if(r2>radius2) return {x:x,y:y} ;
   129  			var th:Number = Math.cos(Math.PI/2 + Math.PI*Math.sqrt(r2)/radius)*rotate ;
   130  			var cosTh:Number = Math.cos(th);
   131  			var sinTh:Number = Math.sin(th);
   132  			return {x:x2*cosTh+y2*sinTh+centreX, y:y2*cosTh-x2*sinTh+centreY} ;
   133  		} ;
   134  	}

   138  	 * @return the string representation of this singleton.
   139  	 */
   140  	static public function toString():String

   142  		return "[CanvasTransform]" ;
   143  	}
