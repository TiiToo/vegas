    27  /**
    28   * This tool class is used to transform the fill color in a shap draw with the APIs of draw of the MovieClip class.
    29   * @author eKameleon
    30   * Thanks : ascb framework <http://www.person13.com/ascblibrary/>
    31   */
    32  class pegas.draw.FillMatrix extends CoreObject 
    33  {
    34  
    35  	/**
    36  	 * Creates a new FillMatrix instance.
    37  	 */
    38  	public function FillMatrix() 
    39  	{
    40  		var first = arguments[0] ;
    41  		if(!first || TypeUtil.typesMatch(first, String) ) 
    42  		{
    43     			matrixType = first || "box" ;
    44     			x = isNaN(arguments[1]) ? 0 : arguments[1] ;
    45  			y = isNaN(arguments[2]) ? 0 : arguments[2] ;
    46  			w = isNaN(arguments[3]) ? 100 : arguments[3] ;
    47  			h = isNaN(arguments[4]) ? 100 : arguments[4] ;
    48  			r = isNaN(arguments[5]) ? 0 : arguments[5] ;
    49  		} 
    50  		else
    51  		{
    52  			a = arguments[0] ;
    53  			b = arguments[1] ;
    54  			c = arguments[2] ;
    55  			d = arguments[3] ;
    56  			e = arguments[4] ;
    57  			f = arguments[5] ;
    58  			g = arguments[6] ;
    59  			h = arguments[7] ;
    60  			i = arguments[8] ;
    61  		}
    62  	}
    63  
    64  	public var x:Number ;
    65  	public var y:Number ;
    66  	public var w:Number ;
    67  	public var h:Number ;
    68  	public var r:Number ;
    69  	public var a:Number ;
    70  	public var b:Number ;
    71  	public var c:Number ;
    72  	public var d:Number ; 
    73  	public var e:Number ;
    74  	public var f:Number ;
    75  	public var g:Number ;
    76  	public var i:Number ;
    77  	public var matrixType:String ;
    78  
    79  }
