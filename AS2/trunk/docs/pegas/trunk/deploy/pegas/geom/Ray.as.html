    24  import pegas.geom.Vector3;
    25  import pegas.util.Vector3Util;
    26  
    27  import vegas.core.CoreObject;
    28  import vegas.core.ICloneable;
    29  import vegas.core.ICopyable;
    30  import vegas.core.IEquality;
    31  import vegas.util.ConstructorUtil;
    32  import vegas.util.serialize.Serializer;
    33  
    34  /**
    35   * This means half of a line, it is infinite in one direction, but ends at a certain point in the other direction.
    36   * In Euclidean geometry, a ray (or half-line) given two distinct points A (the origin) and B on the ray, is the set of points C on the line containing points A and B such that A is not strictly between C and B. 
    37   * In geometry, a ray starts at one point, then goes on forever in one direction : (A) -- (B) -- ((C)) -->
    38   * @author eKameleon
    39   */
    40  class pegas.geom.Ray extends CoreObject implements ICloneable, ICopyable, IEquality
    41  {
    42  	
    43  	/**
    44  	 * Creates a new Ray instance.
    45  	 * <p><b>Usage :</b></p>
    46  	 * <p>With a Ray object passed in the argument of the constructor :</p>
    47  	 * {@code
    48  	 * var r:Ray = new Ray( r:Ray) ;
    49  	 * }
    50  	 * <p>With 2 Vector3 objects passed in the arguments of the constuctor : </p>
    51  	 * {@code
    52  	 * var r:Ray = new Ray( r:Vector3 , p:Vector3 ) ;
    53  	 * }
    54  	 */
    55  	public function Ray() 
    56  	{
    57  		p = new Vector3() ;
    58  		v = new Vector3() ;
    59  		if ( arguments[0] instanceof Ray )
    60  		{
    61  			var r:Ray = arguments[0] ;
    62  			p = r.p ;
    63  			v = r.v ;
    64  		}
    65  		else if ( arguments.length == 2)
    66  		{
    67  			if (arguments[0] instanceof Vector3)
    68  			{
    69  				p = arguments[0].clone() ;
    70  			}			
    71  			if (arguments[1] instanceof Vector3)
    72  			{
    73  				v = arguments[1].clone() ;
    74  			}
    75  		}
    76  		update() ;
    77  	}
    78  
    79  	/**
    80  	 * Determinates the p {@code Vector3} of the Ray object.
    81  	 */
    82  	public var p:Vector3 ;
    83  
    84  	/**
    85  	 * Determinates the q {@code Vector3} of the Ray object.
    86  	 */
    87  	public var q:Vector3 ;	
    88  	
    89  	/**
    90  	 * Determinates the v {@code Vector3} of the Ray object.
    91  	 */	
    92  	public var v:Vector3 ;
    93  	
    94  	/**
    95  	 * Returns a shallow copy of this instance.
    96  	 * @return a shallow copy of this instance.
    97  	 */
    98  	public function clone()
    99  	{
   100  		return new Ray( this ) ;
   101  	}
   102  
   103  	/**
   104  	 * Returns a deep copy of this instance.
   105  	 * @return a deep copy of this instance.
   106  	 */
   107  	public function copy()
   108  	{
   109  		var r:Ray = new Ray() ;
   110  		r.p = p.clone() ;
   111  		r.q = q.clone() ;
   112  		r.v = v.clone() ;
   113  		return r ;
   114  	}
   115  
   116  	/**
   117  	 * Compares the specified object with this object for equality.
   118  	 * @return {@code true} if the the specified object is equal with this object.
   119  	 */
   120  	public function equals(o):Boolean 
   121  	{
   122  		if (o instanceof Ray)
   123  		{
   124  			return p.equals(o.p) && v.equals(o.v) && q.equals(o.q) ;  
   125  		}
   126  		else
   127  		{
   128  			return false ;
   129  		}
   130  	}
   131  
   132  	/**
   133  	 * Returns a Eden reprensation of the object.
   134  	 * @return a string representing the source code of the object.
   135  	 */
   136  	public function toSource(indent : Number, indentor : String):String 
   137  	{
   138  		return Serializer.getSourceOf(this, [p.toSource(), v.toSource()]) ;
   139  	}
   140  
   141  	/**
   142  	 * Returns the string representation of the object.
   143  	 * @return the string representation of the object.
   144  	 */ 	
   145  	public function toString():String
   146  	{
   147  		return "[" + ConstructorUtil.getName(this) + ":" + p + " + t * " + v + "]" ;
   148  	}
   149  
   150  	/**
   151  	 * Updates the Ray object.
   152  	 */
   153  	public function update():Void
   154  	{
   155  		if (q != null)
   156  		{
   157  			Vector3Util.setByVector3(q, p) ;
   158  			Vector3Util.addition(q, v) ;
   159  		}
   160  		else
   161  		{
   162  			q = Vector3Util.getAddition(p, v) ;	
   163  		}
   164  	}
   165  
   166  }
