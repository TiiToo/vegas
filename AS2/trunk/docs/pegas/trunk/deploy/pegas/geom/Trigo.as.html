    24  /**
    25   * Implements the static behaviours of the trigometric manipulations.
    26   * <p>Thanks : <b>Robert Penner</b>, Robert Penner's Programming Macromedia Flash MX
    27   * <ul>
    28   * <li><a href="http://www.robertpenner.com/profmx">http://www.robertpenner.com/profmx</a></li>
    29   * <li><a href="http://www.amazon.com/exec/obidos/ASIN/0072223561/robertpennerc-20">http://www.amazon.com/exec/obidos/ASIN/0072223561/robertpennerc-20</a></li>
    30   * </ul>
    31   * </p>
    32   */
    33  class pegas.geom.Trigo 
    34  {
    35  	
    36  	/**
    37  	 * Returns the inverse cosine of a slope ratio and returns its angle in degrees.
    38  	 * @param ratio a value between -1 and 1 inclusive.
    39  	 * @return the inverse cosine of a slope ratio and returns its angle in degrees.
    40  	 */
    41  	public static function acosD (ratio:Number) : Number 
    42  	{
    43  		return Math.acos(ratio) * (180 / Math.PI) ;
    44  	}
    45  
    46  	/**
    47  	 * Returns the angle in degrees between 2 points with this coordinates passed in argument.
    48  	 * @param x1 the x coordinate of the first point.
    49  	 * @param y1 the y coordinate of the first point.
    50  	 * @param x2 the x coordinate of the second point.
    51  	 * @param y2 the y coordinate of the second point.
    52  	 * @return the angle in degrees between 2 points with this coordinates passed in argument.
    53  	 */
    54  	public static function angleOfLine (x1:Number, y1:Number, x2:Number, y2:Number):Number 
    55  	{
    56  		return Trigo.atan2D (y2 - y1, x2 - x1) ;
    57  	}
    58  
    59  	/**
    60  	 * Calculates the arcsine of the passed angle.
    61  	 * @param ratio a value between -1 and 1 inclusive.
    62  	 * @return the arcsine of the passeds angle in degrees.
    63  	 */
    64  	public static function asinD (ratio:Number) : Number 
    65  	{
    66  		return Math.asin(ratio) * (180 / Math.PI) ;
    67  	}
    68  
    69  	/**
    70  	 * Calculates the arctangent of the passed angle.
    71  	 * @param n a real number
    72  	 * @return the arctangent of the passed angle, a number between -Math.PI/2 and Math.PI/2 inclusive.
    73  	 */
    74  	public static function atanD( angle:Number ):Number 
    75  	{
    76  		return Math.atan( angle ) * (180 / Math.PI);
    77  	}
    78  
    79  	/**
    80  	 * Calculates the arctangent2 of the passed angle.
    81  	 * @param y a value representing y-axis of angle vector.
    82  	 * @param x a value representing x-axis of angle vector.
    83  	 * @return the arctangent2 of the passed angle.
    84  	 */
    85  	public static function atan2D( y:Number , x:Number ):Number 
    86  	{
    87  		return Math.atan2(y, x) * (180 / Math.PI) ;
    88  	}
    89  	
    90  
    91  	/**
    92  	 * Converts a vector in cartesian in a polar vector.
    93  	 * @return a vector in cartesian in a polar vector.
    94  	 */
    95  	public static function cartesianToPolar (p:Object) : Object 
    96  	{
    97  		return ( { r : Math.sqrt (p.x * p.x + p.y * p.y) , t : Trigo.atan2D (p.y , p.x) } ) ;
    98  	}	
    99  
   100  	/**
   101  	 * Calculates the cosine of the passed angle.
   102  	 * @param angle a value representing angle in degrees.
   103  	 * @return the cosine of the passed angle, a number between -1 and 1 inclusive.
   104  	 */
   105  	public static function cosD(angle:Number):Number 
   106  	{
   107  		return Math.cos( angle * (Math.PI / 180) ) ;
   108  	}
   109  
   110  	/**
   111  	 * Converts an angle in degrees in radians
   112  	 * @return an angle in degrees in radians.
   113  	 */
   114  
   115  	public static function degreesToRadians (angle:Number) : Number 
   116  	{
   117  		return angle * (Math.PI / 180) ;
   118  	}
   119  
   120  	/**
   121  	 * Returns the distance between 2 points with the coordinates of the 2 points.
   122  	 * @param x1 the x coordinate of the first point.
   123  	 * @param y1 the y coordinate of the first point.
   124  	 * @param x2 the x coordinate of the second point.
   125  	 * @param y2 the y coordinate of the second point.
   126  	 * @return the length between 2 points.
   127  	 */
   128  	public static function distance (x1:Number, y1:Number, x2:Number, y2:Number):Number 
   129  	{
   130  		var dx = x2 - x1 ;
   131  		var dy = y2 - y1 ;
   132  		return Math.sqrt (dx * dx + dy * dy ) ;
   133  	}
   134  
   135  	/**
   136  	 * Returns the distance between 2 points with the coordinates of the 2 points.
   137  	 * @param p1 the first point to determinate the distance.
   138  	 * @param p2 the second point to determinate the distance.
   139  	 * @return the length between 2 points.
   140  	 */
   141  	public static function distanceP(p1, p2 ) : Number 
   142  	{
   143  		return distance(p1.x, p1.y, p2.x, p2.y) ;
   144  	}
   145  	
   146  	/**
   147  	 * Fixs an angle in degrees between 0 and 360 degrees.
   148  	 * @param angle the passed angle value in degrees.
   149  	 * @return an angle in degrees between 0 and 360 degrees. 
   150  	 */
   151  	public static function fixAngle (angle:Number):Number 
   152  	{
   153  		if (isNaN(angle)) 
   154  		{
   155  			angle = 0 ;
   156  		}
   157  		angle %= 360 ;
   158  		return (angle < 0) ? angle + 360 : angle ;
   159  	}
   160  
   161  	/**
   162  	 * Converts an angle in radians in degrees.
   163  	 * @return an angle in radians in degrees.
   164  	 */
   165  	public static function radiansToDegrees (angle:Number) : Number 
   166  	{
   167  		return angle * (180 / Math.PI) ;
   168  	}
   169  	
   170  	/**
   171  	 * Converts a vector in polar in a cartesian vector.
   172  	 * @return a vector in polar in a cartesian vector.
   173  	 */
   174  	public static function polarToCartesian (p:Object) : Object 
   175  	{
   176  		return ( { x : p.r * Trigo.cosD (p.t) , y : p.r * Trigo.sinD (p.t) } ) ;
   177  	}
   178  
   179  	/**
   180  	 * Calculates the sine of the passed angle.
   181  	 * @param angle a value representing angle in degrees.
   182  	 * @return the sine of the passed angle, a number between -1 and 1 inclusive.
   183  	 */
   184  	public static function sinD(angle:Number):Number 
   185  	{
   186  		return Math.sin( angle * (Math.PI / 180) ) ;
   187  	}
   188  
   189  	/**
   190  	 * Calculates the tangent of the passed angle.
   191  	 * @param angle a value representing angle in degrees.
   192  	 * @return the tangent of the passed angle.
   193  	 */
   194  	public static function tanD(angle:Number):Number 
   195  	{
   196  		return Math.tan( angle * (Math.PI / 180) ) ;
   197  	}
   198  
   199  }
