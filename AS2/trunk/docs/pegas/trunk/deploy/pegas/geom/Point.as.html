    29  /**
    30   * The Point class represents a location in a two-dimensional coordinate system, where x represents the horizontal axis and y represents the vertical axis.
    31   * <p>The following code creates a point at (0,0) :</p>
    32   * {@code var myPoint:Point = new Point() ; }
    33   * @author eKameleon
    34   */

    38  	/**
    39  	 * Creates a new Point instance. 
    40  	 * <p><b>Parameters</b>
    41  	 * <ul>
    42  	 * <li>If you pass no parameters to this method, a point is created at (0,0).</li>
    43  	 * <li>If you pass a Point object the x and y properties define the properties of the new Point.</li>
    44  	 * <li>if you pass two Numbers (x and y) you can define the horizontal and vertical coordinates.</li>
    45  	 * </ul>
    46  	 * <p><b>Example :</b></p>
    47  	 * {@code
    48  	 * import pegas.geom.Point ;
    49  	 * 
    50  	 * var p1:Point = new Point() ;
    51  	 * trace(p1) ; // [Point:{0,0}]
    52  	 * 
    53  	 * var p2:Point = new Point( new Point(25,30) ) ;
    54  	 * trace(p2) ; // [Point:{25,30}]
    55  	 * 
    56  	 * var p3:Point = new Point( 25,30 ) ;
    57  	 * trace(p3) ; // [Point:{25,30}]
    58  	 * }
    59  	 */
    60  	public function Point() 
    61  	{
    62  		x = y = 0 ;
    63  		var l:Number = arguments.length ;
    64  		if ( l > 0 ) 
    65  		{
    66  			var o1 = arguments[0] ; 
    67  			if( l == 1 && o1 instanceof Point) 
    68  			{
    69  				x = o1.x ;
    70  				y = o1.y ;
    71  			}
    72  			else if (l == 2)
    73  			{
    74  				var o2 = arguments[1] ;
    75  				x = isNaN(o1) ? 0 : o1 ;
    76  				y = isNaN(o2) ? 0 : o2 ;
    77  			}
    78  		}
    79  	}

   167  	/**
   168  	 * Returns the angle value between this Point object and the specified Point passed in arguments.
   169  	 * <p><b>Example :</b></p>
   170  	 * {@code
   171  	 * var p1:Point = new Point(10, 20) ;
   172  	 * var p2:point = new Point(50, 200) ;
   173  	 * var angle:Number = p1.angleBetween(p2) ;
   174  	 * }
   175  	 * @return the angle value between this Point object and the specified Point passed in arguments.
   176  	 */
   177  	public function angleBetween(p:Point):Number 
   178  	{
   179  		var dp:Number = dot(p) ;
   180  		var a:Number = dp / (getLength() * p.getLength()) ;
   181  		return Trigo.acosD(a) ;
   182  	}

   375  	/**
   376  	 * Determines a point between two specified points. 
   377  	 * The parameter f determines where the new interpolated point is located relative to the two end points specified by parameters {@code p1} and {@code p2}. 
   378  	 * The closer the value of the parameter f is to 1.0, the closer the interpolated point is to the first point (parameter {@code p1}). 
   379  	 * The closer the value of the parameter f is to 0, the closer the interpolated point is to the second point (parameter {@code p2}).
   380  	 * <p><b>Example :</b></p>
   381  	 * {@code
   382  	 * var p1:Point = new Point(10,10) ;
   383  	 * var p2:Point = new Point(40,40) ;
   384  	 * var p3:Point ;
   385  	 * 
   386  	 * p3 = Point.interpolate( p1 , p2, 0 ) ;
   387  	 * trace(p3) ; // [Point:{40,40}]
   388  	 * 
   389   	 * p3 = Point.interpolate( p1 , p2, 1 ) ;
   390  	 * trace(p3) ; // [Point:{10,10}]
   391  	 * 
   392   	 * p3 = Point.interpolate( p1 , p2, 0.5 ) ;
   393  	 * trace(p3) ; // [Point:{25,25}]
   394  	 * }
   395  	 * @param p1 The first point.
   396  	 * @param p2 The second Point.
   397  	 * @param f the The level of interpolation between the two points. Indicates where the new point will be, along the line between {@code p1} and {@code p2}. If f=1, pt1 is returned; if f=0, pt2 is returned.
   398  	 * @return The new interpolated point.
   399  	 */
   400  	public static function interpolate(p1:Point, p2:Point, f:Number):Point 
   401  	{
   402  		return new Point( p2.x + f * (p1.x - p2.x) , p2.y + f * (p1.y - p2.y) ) ;
   403  	}

   490  	/**
   491  	 * Scales the line segment between (0,0) and the current point to a set length.
   492  	 * <p><b>Example :</b></p>
   493  	 * {@code
   494  	 * var p:Point = new Point(0,5) ;
   495  	 * p.normalize() ;
   496  	 * trace(p) ; // [Point:{0,1}]
   497  	 * }
   498  	 * @param thickness The scaling value. For example, if the current point is (0,5), and you normalize it to 1, the point returned is at (0,1).
   499  	 * @see length
   500  	 * @throws IllegalArgumentError if a zero-length vector or a illegal NaN value is calculate in this method.
   501  	 */
   502  	public function normalize( thickness:Number ):Void 
   503  	{
   504  		if ( isNaN(thickness) )
   505  		{
   506  			thickness = 1 ;	
   507  		}
   508  		var l:Number = getLength() ;
   509  		if (l > 0) 
   510  		{
   511  			l = thickness / l ;
   512  			x *= l ;
   513  			y *= l ;
   514  		}
   515  		else
   516  		{
   517  			throw new IllegalArgumentError(this + " normalize method failed with a zero-length vector or a illegal NaN value.") ;
   518  		}
   519  	}

   531  	 * @param dx The amount by which to offset the horizontal coordinate, x.

   547  	 * @param len The length coordinate of the polar pair.
