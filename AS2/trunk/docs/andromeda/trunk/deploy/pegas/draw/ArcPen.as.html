    24  import pegas.draw.Align;
    25  import pegas.draw.ArcType;
    26  import pegas.draw.EasyPen;
    27  import pegas.geom.Trigo;

    29  /**
    30   * This pen draw a pie or chord arc shape in a MovieClip reference.
    31   * @author eKameleon
    32   */
    33  class pegas.draw.ArcPen extends EasyPen 
    34  {
    35  
    36  	/**
    37  	 * Creates a new ArcPen instance.
    38  	 * @param target the movieclip reference.
    39  	 * @param isNew if this flag is {@code true} the draw is localize in a new movieclip in the specified {@code target}.
    40  	 */
    41  	public function ArcPen(target:MovieClip, isNew:Boolean) 
    42  	{
    43  		initialize(target, isNew) ;
    44  		setAlign(Align.TOP_LEFT) ;
    45  	}
    46  
    47  	/**
    48  	 * (read-write) Returns the value of the angle used to draw an arc shape in the movieclip reference.
    49  	 * @return the value of the angle used to draw an arc shape in the movieclip reference.
    50  	 */
    51  	public function get angle():Number 
    52  	{
    53  		return getAngle() ;	
    54  	}
    55  	
    56  	/**
    57  	 * (read-write) Sets the value of the angle used to draw an arc shape in the movieclip reference.
    58  	 */
    59  	public function set angle(n:Number):Void 
    60  	{
    61  		setAngle(n) ;	
    62  	}
    63  
    64  	/**
    65  	 * Defines the radius value of the arc shape.
    66  	 */
    67  	public var radius:Number = 100;
    68  
    69  	/**
    70  	 * Defines the type of the arc, can be a chord or a pie arc.
    71  	 */
    72  	public var type:String = ArcType.CHORD ; // CHORD || PIE
    73  
    74  	/**
    75  	 * Defines the x origin position of the arc shape.
    76  	 */
    77  	public var x:Number = 0 ;
    78  
    79  	/**
    80  	 * Defines the y origin position of the arc shape.
    81  	 */
    82  	public var y:Number = 0 ;
    83  
    84  	/**
    85  	 * Defines the y origin position of the arc radius.
    86  	 */
    87  	public var yRadius:Number ;
    88  
    89  	/**
    90  	 * (read-write) Returns the value of the start angle to draw the arc in the movieclip reference.
    91  	 * @return the value of the start angle to draw the arc in the movieclip reference.
    92  	 */
    93  	public function get startAngle():Number 
    94  	{
    95  		return getStartAngle() ;	
    96  	}
    97  	
    98  	/**
    99  	 * (read-write) Sets the value of the start angle to draw the arc in the movieclip reference.
   100  	 */
   101  	public function set startAngle(n:Number):Void 
   102  	{
   103  		setStartAngle(n) ;	
   104  	}
   105  
   106  	/**
   107  	 * Returns a shallow copy of this object.
   108  	 * @return a shallow copy of this object.
   109  	 */
   110  	public function clone() 
   111  	{
   112  		var arc:ArcPen = new ArcPen(_target) ;
   113  		arc.radius = radius ;
   114  		arc.x = x ;
   115  		arc.y = y ;
   116  		arc.yRadius = yRadius ;
   117  		arc.type = ArcType.CHORD ;
   118  		arc.setAlign(_align, true) ;
   119  		arc.setAngle(_angle, true) ;
   120  		arc.setStartAngle(_startAngle, true) ;
   121  		return arc ;
   122  	}
   123  
   124  	/**
   125  	 * Draws the shape in the movieclip reference of this pen.
   126  	 */
   127  	public function draw(p_angle:Number, p_startAngle:Number, p_x:Number, p_y:Number, p_align:Number):Void 
   128  	{
   129  		if (arguments.length > 0) 
   130  		{
   131  			setArc.apply(this, arguments) ;
   132  		}
   133  		init() ;
   134  		moveTo(_nX, _nY);
   135  		var ax:Number ;
   136  		var ay:Number ;
   137  		var bx:Number ;
   138  		var by:Number ;
   139  		var cx:Number ;
   140  		var cy:Number ;
   141  		var angleMid:Number ;
   142  		var nR:Number = isNaN(yRadius) ? radius : yRadius ;
   143  		var segs:Number = Math.ceil ( Math.abs(_angle) / 45 ) ;
   144  		var segAngle:Number = _angle / segs ;
   145  		var theta:Number = - Trigo.degreesToRadians(segAngle) ;
   146  		var a:Number = - _startAngle  ;
   147  		if (segs>0) 
   148  		{
   149  			ax = _nX + Math.cos (_startAngle) * radius ;
   150  			ay = _nY + Math.sin (-_startAngle) * nR ;
   151  			if (_angle < 360 && _angle > -360 && type == ArcType.PIE) lineTo (ax, ay) ;
   152  			moveTo (ax, ay) ;
   153  			for (var i:Number = 0 ; i<segs ; i++) 
   154  			{
   155  				a += theta ;
   156  				angleMid = a - ( theta / 2 ) ;
   157  				bx = _nX + Math.cos ( a ) * radius ;
   158  				by = _nY + Math.sin ( a ) * nR ;
   159  				cx = _nX + Math.cos( angleMid ) * ( radius / Math.cos ( theta / 2 ) ) ;
   160  				cy = _nY + Math.sin( angleMid ) * ( nR / Math.cos( theta / 2 ) ) ;
   161  				curveTo(cx, cy, bx, by) ;
   162  			}
   163  			if(type == ArcType.PIE) 
   164  			{
   165  				if (_angle < 360 && _angle > -360) lineTo(_nX, _nY);
   166  			}
   167  			else 
   168  			{ 
   169  				lineTo(ax, ay); // CHORD or other value
   170  			}
   171  		}
   172  		if (isEndFill) endFill() ;	
   173  	}
   174  
   175  	/**
   176  	 * Returns the value of the angle used to draw an arc shape in the movieclip reference.
   177  	 * @return the value of the angle used to draw an arc shape in the movieclip reference.
   178  	 */
   179  	public function getAngle():Number 
   180  	{ 
   181  		return _angle ;
   182  	}
   183  	
   184  	public function getStartAngle():Number 
   185  	{ 
   186  		return Trigo.radiansToDegrees(_startAngle)  ;
   187  	}
   188  
   189  	/**
   190  	 * Initialize the pen.
   191  	 */
   192  	public function init():Void 
   193  	{
   194  		if (isNaN(x)) x = 0 ;
   195  		if (isNaN(y)) y = 0 ;
   196  		_nX = x ; 
   197  		_nY = y ;
   198  		var nR:Number = (isNaN(yRadius)) ? radius : yRadius ;
   199  		switch (_align) {
   200  			case Align.TOP : // Top
   201  				_nY += nR ;
   202  				break ;
   203  			case Align.BOTTOM : // Bottom
   204  				_nY -= nR ;
   205  				break ;
   206  			case Align.LEFT : // Left
   207  				_nX += radius ;
   208  				break ;
   209  			case Align.RIGHT : // Right
   210  				_nX -= radius ;
   211  				break ;
   212  			case Align.TOP_LEFT : // Top Left
   213  				_nX += radius ;
   214  				_nY = nR ;
   215  				break ;
   216  			case Align.TOP_RIGHT : // Top Right
   217  				_nX -= radius ;
   218  				_nY = nR ;
   219  				break ;
   220  			case Align.BOTTOM_LEFT : // Bottom Left
   221  				_nX += radius ;
   222  				_nY -=  nR ;
   223  				break ;
   224  			case Align.BOTTOM_RIGHT : // Bottom Right
   225  				_nX -= radius ;
   226  				_nY -= nR ;
   227  				break ;
   228  			default : // Center
   229  				break ;
   230  		}
   231  	}
   232  
   233  	/**
   234  	 * Sets the value of the angle used to draw an arc shape in the movieclip reference.
   235  	 */
   236  	public function setAngle(n:Number):Void 
   237  	{
   238  		_angle = Trigo.fixAngle(n) ;
   239  	}
   240  
   241  	/**
   242  	 * Sets the arc options to defined all values to draw the arc shape in the movieclip reference of this pen.
   243  	 */
   244  	public function setArc(p_angle:Number, p_startAngle:Number, p_x:Number, p_y:Number, p_align:Number):Void 
   245  	{
   246  		if (!isNaN(p_angle) ) setAngle(p_angle) ;
   247  		if (!isNaN(p_startAngle) ) setStartAngle(p_startAngle) ;
   248  		if (!isNaN(p_align)) setAlign(p_align) ;
   249  		x = (isNaN(p_x)) ? 0 : p_x ;
   250  		y = (isNaN(p_y)) ? 0 : p_y ;
   251  	}
   252  
   253  	/**
   254  	 * Sets the value of the start angle to draw the arc in the movieclip reference.
   255  	 */
   256  	public function setStartAngle(n:Number):Void 
   257  	{
   258  		_startAngle = Trigo.degreesToRadians(n) ;
   259  	}
   260  
   261  	private var _angle:Number = 0 ;
   262  	
   263  	private var _angleMid:Number ;
   264  	
   265  	private var _nX:Number ;
   266  	
   267  	private var _nY:Number ;
   268  	
   269  	private var _startAngle:Number = 360 ;
   270  	
   271  }
