    61  	public function drawCubicBezier(p0:Vector2, p1:Vector2, p2:Vector2, p3:Vector2, nSegment:Number, moveto:Boolean):Number 
    62  	{
    63  		if (nSegment < 2) 
    64  		{
    65  			nSegment = 4 ;
    66  		}
    67  		var nextT ; 
    68  		var total:Number = 0 ; 
    69  		var tStep:Number = 1 / nSegment ;
    70  		var curT:Object  = {
    71  			p : p0 ,
    72  			l : Line.getLine( p0, p1 ) 
    73  		} ;
    74  		if (moveto) 
    75  		{
    76  			moveTo( p0.x, p0.y );
    77  		}
    78  		for (var i:Number = 1 ; i <= nSegment ; i++) 
    79  		{
    80  			nextT = Bezier.getCubicTgt( p0, p1, p2, p3, i * tStep ) ;
    81  			total += sliceCubicBezierSegment( p0, p1, p2, p3, (i - 1) * tStep, i * tStep, curT, nextT, 0 ) ;
    82  			curT = nextT ; 
    83  		}
    84  		return total ;
    85  	}

    87  	public function drawCubicBezier2(p0, p1, p2, p3):Void 
    88  	{
    89  		var pA:Vector2 = Line.getPointOnSegment( p0, p1, 3 / 4 );
    90  		var pB:Vector2 = Line.getPointOnSegment( p3, p2, 3 / 4 );
    91  		var dx:Number = (p3.x - p0.x) / 16 ;
    92  		var dy:Number = (p3.y - p0.y) / 16 ;
    93  		var pc_1:Vector2 = Line.getPointOnSegment( p0, p1, 3 / 8 );
    94  		var pc_2:Vector2 = Line.getPointOnSegment( pA, pB, 3 / 8 );
    95  		pc_2.x -= dx;
    96  		pc_2.y -= dy;
    97  		var pc_3 = Line.getPointOnSegment( pB, pA, 3 / 8 );
    98  		pc_3.x += dx;
    99  		pc_3.y += dy;
   100  		var pc_4:Vector2 = Line.getPointOnSegment( p3, p2, 3 / 8 );
   101  		var pa_1:Vector2 = Line.getMiddle( pc_1, pc_2 );
   102  		var pa_2:Vector2 = Line.getMiddle( pA, pB );
   103  		var pa_3:Vector2 = Line.getMiddle( pc_3, pc_4 );
   104  		curveTo( pc_1.x, pc_1.y, pa_1.x, pa_1.y );
   105  		curveTo( pc_2.x, pc_2.y, pa_2.x, pa_2.y );
   106  		curveTo( pc_3.x, pc_3.y, pa_3.x, pa_3.y );
   107  		curveTo( pc_4.x, pc_4.y, p3.x, p3.y );
   108  	}

   110  	public function drawCubicBezierSpline(p0:Vector2, p1:Vector2, p2:Vector2, p3:Vector2):Void 
   111  	{
   112  		var mid:Vector2 = Line.getMiddle( p1, p2 ) ;
   113  		curveTo( p1.x, p1.y, mid.x, mid.y );
   114  		curveTo( p2.x, p2.y, p3.x, p3.y );
   115  	}

   117  	public function drawPoints( /* p1:Vector2, p2:Vector2, p3:Vector2, p4:Vector2, tolerance:Number */ ):Void 
   118  	{
   119  		if (arguments.length == 1 && arguments[0] instanceof Array) 
   120  		{
   121  			var pts:Array = arguments[0] ;
   122  			moveTo( pts[0].x, pts[0].y );
   123  			for (var i:Number = 0; i < pts.length ;i++)
   124  			{
   125  				lineTo( pts[i].x, pts[i].y );
   126  			}
   127  		} 
   128  		else 
   129  		{
   130  			var p1:Vector2 = arguments[0] ;
   131  			var p2:Vector2 = arguments[1] ;
   132  			var p3:Vector2 = arguments[2] ;
   133  			var p4:Vector2 = arguments[3] ;
   134  			var tolerance:Number = arguments[4] ;
   135  			if (tolerance == undefined) tolerance = 5 ;
   136  			moveTo( p1.x, p1.y );
   137  			_create( p1, p2, p3, p4, Math.pow( tolerance, 2 ) ) ;
   138  		}
   139  	}

   141  	/**
   142  	 * Simulate the bezier pen result with a speed algorithm.
   143  	 */
   144  	public function simulate(p1:Vector2, p2:Vector2, p3:Vector2, p4:Vector2):Void 
   145  	{
   146  		var a,b,c,d,e,f,g,h,i,j:Number ;
   147  		a = (p1.x + p2.x) / 2 ;
   148  		b = (p1.y + p2.y) / 2 ;
   149  		c = (p2.x + p3.x) / 2 ;
   150  		d = (p2.y + p3.y) / 2 ;
   151  		e = (p3.x + p4.x) / 2 ;
   152  		f = (p3.y + p4.y) / 2 ;
   153  		g = (a + c) / 2 ;
   154  		h = (b + d) / 2 ;
   155  		i = (c + e) / 2 ;
   156  		j = (d + f) / 2 ;
   157  		curveTo( a, b, g, h ) ;
   158  		curveTo( c, d, i, j ) ;
   159  		curveTo( e, f, p4.x, p4.y ) ;
   160  	}

   162  	/**
   163  	 * Slice a cubic bezier segment and return the value.
   164  	 */
   165  	public function sliceCubicBezierSegment(p0:Vector2, p1:Vector2, p2:Vector2, p3:Vector2, u1, u2, tu1, tu2, recurs:Number):Number 
   166  	{
   167  		var p ;
   168  		if (recurs > 10) 
   169  		{
   170  			p = tu2.p;
   171  			lineTo( p.x, p.y );
   172  			return 1 ;
   173  		}
   174  		var ctrlpt:Vector2 = Line.getLineCross( tu1.l, tu2.l ) ;
   175  		var d:Number = 0 ;
   176  		if ( (ctrlpt == null) || (Line.distance( tu1.p, ctrlpt ) > (d = Line.distance( tu1.p, tu2.p ))) || (Line.distance( tu2.p, ctrlpt ) > d) ) 
   177  		{
   178  			var result:Number = 0 ;
   179  			var uMid:Number = (u1 + u2) / 2 ;
   180  			var tuMid = Bezier.getCubicTgt( p0, p1, p2, p3, uMid ) ;
   181  			result += sliceCubicBezierSegment( p0, p1, p2, p3, u1, uMid, tu1, tuMid, recurs + 1 ) ;
   182  			result += sliceCubicBezierSegment( p0, p1, p2, p3, uMid, u2, tuMid, tu2, recurs + 1 ) ;
   183  			return result ;
   184  		} 
   185  		else 
   186  		{
   187  			p = tu2.p ;
   188  			curveTo( ctrlpt.x, ctrlpt.y, p.x, p.y ) ;
   189  			return 1 ;
   190  		}
   191  	}

   193  	/**
   194  	 * @private
   195  	 */
   196  	private function _create(a:Vector2, b:Vector2, c:Vector2, d:Vector2, k:Number):Void 
   197  	{
   198  		var l1:Line = Line.getLine( a, b ) ;
   199  		var l2:Line = Line.getLine( c, d ) ;
   200  		var s:Vector2 = Line.getLineCross( l1, l2 ) ;
   201  		var dx:Number = (a.x + d.x + s.x * 4 - (b.x + c.x) * 3) * .125 ;
   202  		var dy:Number = (a.y + d.y + s.y * 4 - (b.y + c.y) * 3) * .125 ;
   203  		if (dx * dx + dy * dy > k) 
   204  		{
   205  			var halves = Bezier.split( a, b, c, d ) ;
   206  			var b0 = halves.b0 ;
   207  			var b1 = halves.b1 ;
   208  			_create( a, b0.b, b0.c, b0.d, k ) ;
   209  			_create( b1.a, b1.b, b1.c, d, k ) ;
   210  		} 
   211  		else 
   212  		{
   213  			curveTo( s.x, s.y, d.x, d.y ) ;
   214  		}
   215  	}
