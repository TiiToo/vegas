    24  import vegas.core.CoreObject;
    25  import vegas.core.ICloneable;
    26  import vegas.core.IEquality;
    27  import vegas.errors.ArgumentOutOfBoundsError;
    28  import vegas.errors.IllegalArgumentError;
    29  import vegas.util.MathsUtil;
    30  
    31  /**
    32   * Represents an immutable range of values.
    33   * <p><b>Example :</b></p>
    34   * {@code
    35   * import pegas.maths.Range ;
    36   * 
    37   * var r1:Range = new Range(10, 120) ;
    38   * var r2:Range = new Range(100, 150) ;
    39   * 
    40   * trace ("r1 : " + r1) ; // r1 : [Range<10,120>]
    41   * trace ("r2 : " + r2) ; // r2 : [Range<100,150>]
    42   * 
    43   * trace ("r2.toSource() : " + r2.toSource()) ; // r2.toSource() : new pegas.maths.Range(100,150)
    44   * trace ("r1 contains 50 : " + r1.contains(50)) ; // r1 contains 50 : true
    45   * trace ("r1 isOutOfRange 5 : " + r1.isOutOfRange(5)) ; // r1 isOutOfRange 5 : true
    46   * trace ("r1 overlap r2 : " + r1.overlap(r2)) ; // r1 overlap r2 : true
    47   * trace ("r1 clamp 5 : " + r1.clamp(5)) ; // r1 clamp 5 : 10
    48   * trace ("r1 clamp 121 : " + r1.clamp(121)) ; // r1 clamp 121 : 120
    49   * }
    50   * @author eKameleon
    51   */
    52  class pegas.maths.Range extends CoreObject implements ICloneable, IEquality
    53  {
    54  
    55  	/**
    56  	 * Creates a new Range instance.
    57  	 * use <pre>var r:Range = new Range(p_min:Number, p_max:Number) ;</pre>
    58  	 */ 
    59  	public function Range(p_min:Number, p_max:Number) 
    60  	{
    61  		if (p_max < p_min) 
    62  		{
    63  			throw new ArgumentOutOfBoundsError("Range constructor : 'max' argument is < of 'min' argument") ;
    64  		}
    65  		min = p_min ;
    66  		max = p_max ;
    67  	}
    68  
    69  	/**
    70  	 * Range between 0 and 100.
    71  	 */
    72  	static public var PERCENT_RANGE:Range = new Range(0, 100) ;
    73  	
    74  	/**
    75  	 * Range between -255 and 255.
    76  	 */
    77  	static public var COLOR_RANGE:Range = new Range(-255, 255) ;
    78  
    79  	/**
    80  	 * Range reference between 0 and 1.
    81  	 */
    82      static public var UNITY_RANGE:Range = new Range(0, 1) ;
    83  
    84  
    85  	static private var __ASPF__ = _global.ASSetPropFlags(Range, null , 7, 7) ;
    86  	
    87  	/**
    88  	 * The max value of the range.
    89  	 */	
    90  	public var max:Number ;
    91  	
    92  	/**
    93  	 * The min value of the range.
    94  	 */
    95  	public var min:Number ;
    96  		
    97  	/**
    98  	 * Clamp a value in the current range.
    99  	 */	
   100  	public function clamp(value:Number):Number 
   101  	{
   102  		return MathsUtil.clamp(value, min, max) ;
   103  	}
   104  	
   105  	/**
   106  	 * Returns a shallow copy of the object.
   107  	 * @return a shallow copy of the object.
   108  	 */
   109  	public function clone() 
   110  	{
   111  		return new Range(min, max) ;
   112  	}
   113  
   114  	/**
   115  	 * Creates a new range by combining two existing ranges.
   116  	 * <li>either range can be {@code null}, in which case the other range is returned.</li>
   117       * <li>if both ranges are {@code null} the return value is {@code null}.</li>
   118  	 * 
   119  	 * @param range1 the first range, {@code null} permitted.
   120  	 * @param range2 the second range, {@code null} permitted.
   121  	 */
   122  	static public function combine( range1:Range, range2:Range ):Range
   123  	{
   124  		if (range1 == null)
   125  		{
   126  			return range2 ;	
   127  		}
   128  		else
   129  		{
   130  			if (range2 == null)
   131  			{
   132  				return range1 ;	
   133  			}	
   134  			else
   135  			{
   136  				var lower:Number = Math.min( range1.min , range2.min ) ;
   137  				var upper:Number = Math.max( range1.max , range2.max ) ;
   138  				return new Range(lower, upper) ;	
   139  			}
   140  		}
   141  	}
   142  	
   143  	/**
   144  	 * Returns {@code true} if the Range instance contains the value passed in argument.
   145  	 * @return {@code true} if the Range instance contains the value passed in argument.
   146  	 */
   147  	public function contains(value:Number):Boolean 
   148  	{
   149  		return !isOutOfRange(value) ;
   150  	}
   151  	
   152  	/**
   153  	 * Returns a deep copy of the object.
   154  	 * @return a deep copy of the object.
   155  	 */
   156  	public function copy() 
   157  	{
   158  		return new Range(min, max) ;
   159  	}
   160  	
   161  	/**
   162  	 * Indicates whether some other object is "equal to" this one.
   163  	 */
   164  	public function equals(o):Boolean 
   165  	{
   166  		return (o instanceof Range) && (o.min == min) && (o.max == o.max) ;
   167  	}
   168  	
   169  	/**
   170  	 * Creates a new range by adding margins to an existing range.
   171  	 * @param range the range {@code null} not permitted.
   172  	 * @param lowerMargin the lower margin (expressed as a percentage of the range length).
   173  	 * @param upperMargin the upper margin (expressed as a percentage of the range length).
   174  	 * @return The expanded range.
   175  	 * @throws IllegalArgumentError if the range argument is {@code null}
   176  	 */
   177  	static public function expand(range:Range, lowerMargin:Number, upperMargin:Number):Range
   178  	{
   179  		if (range == null)
   180  		{
   181  			throw new IllegalArgumentError("Range.expand method failed, the range argument not must be 'null' or 'undefined'.");  
   182  		}
   183  		var size:Number = range.size() ;
   184  		var lower:Number = size * lowerMargin ;
   185  		var upper:Number = size * upperMargin ;
   186  		return new Range( range.min - lower , range.max + upper ) ;
   187  	}
   188  	
   189  	/**
   190  	 * Returns the central value for the range.
   191  	 * @return The central value.
   192  	 */
   193  	public function getCentralValue():Number
   194  	{
   195  		return (min + max) / 2 ;
   196  	}
   197  	
   198  	/**
   199  	 * Returns {@code true} if the value is out of the range.
   200  	 * @return {@code true} if the value is out of the range.
   201  	 */
   202  	public function isOutOfRange(value:Number):Boolean 
   203  	{
   204  		return (value > max ) || (value < min) ;
   205  	}
   206  
   207  	/**
   208  	 * Returns {@code true} if the range in argument overlap the current range.
   209  	 * @return {@code true} if the range in argument overlap the current range.
   210  	 */
   211  	public function overlap(r:Range):Boolean 
   212  	{
   213  		return max > r.min && r.max > min ;
   214  	}
   215  
   216  	/**
   217  	 * Returns the length of the range.
   218  	 * @return the length of the range.
   219  	 */
   220  	public function size():Number
   221  	{
   222  		return max - min ;	
   223  	}
   224  
   225  	/**
   226  	 * Returns the Eden reprensation of the object.
   227  	 * @return a string representing the source code of the object.
   228  	 */
   229  	public function toSource(indent:Number, indentor:String):String 
   230  	{
   231  		return "new pegas.maths.Range(" + min + "," + max + ")";
   232  	}
   233  
   234  	/**
   235  	 * Returns the string representation of this instance.
   236  	 * @return the string representation of this instance.
   237  	 */
   238  	public function toString():String 
   239  	{
   240  		return "[Range<" + min + "," + max + ">]";
   241  	}
   242  	
   243  }
