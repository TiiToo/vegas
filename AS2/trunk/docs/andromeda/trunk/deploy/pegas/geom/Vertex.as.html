    24  import pegas.geom.Vector3;
    25  
    26  import vegas.util.ConstructorUtil;
    27  import vegas.util.serialize.Serializer;
    28  
    29  /**
    30   * A Vertex is a point which can be represented in differents coordinates (local, world, screen).
    31   * @author eKameleon
    32   */
    33  class pegas.geom.Vertex extends Vector3 
    34  {
    35  	
    36  	/**
    37  	 * Creates a new {@code Vertex} instance.
    38  	 * @param x the x coordinate.
    39  	 * @param y the y coordinate.
    40  	 * @param z the z coordinate.
    41  	 * @param tx (optional) the transformed x position number. If this argument is 'null' or 'undefined' the default value is x.
    42  	 * @param ty (optional) the transformed y position number. If this argument is 'null' or 'undefined' the default value is y.
    43  	 * @param tz (optional) the transformed z position number. If this argument is 'null' or 'undefined' the default value is z. 
    44  	 */
    45  	public function Vertex( x:Number, y:Number, z:Number , tx:Number, ty:Number, tz:Number ) 
    46  	{
    47  
    48  		super(x, y, z);
    49  		
    50  		this.tx = isNaN(tx) ? x : tx; 
    51  		this.ty = isNaN(ty) ? y : ty; 
    52  		this.tz = isNaN(tz) ? z : tz; 
    53  
    54  		this.wx = tx ; 
    55  		this.wy = ty ; 
    56  		this.wz = tz ;
    57  
    58  		this.sy = 0 ; 
    59  		this.sx = 0 ;
    60  	}
    61  
    62  	/**
    63  	 * Returns a shallow copy of this instance.
    64  	 * @return a shallow copy of this instance.
    65  	 */
    66  	public function clone()
    67  	{
    68  		return new Vertex(x, y, z, tx, ty, tz) ;
    69  	}
    70  
    71  	/**
    72  	 * Returns a deep copy of this instance.
    73  	 * @return a deep copy of this instance.
    74  	 */
    75  	public function copy()
    76  	{
    77  		var v:Vertex = new Vertex(x, y, z, tx, ty, tz) ;
    78  		v.wx = wx ;
    79  		v.wy = wy ;
    80  		v.wz = wz ;
    81  		v.sx = sx ;
    82  		v.sy = sy ;	
    83  		return v ;
    84  	}
    85  
    86  	/**
    87  	 * Compares the specified object with this object for equality.
    88  	 * @return {@code true} if the the specified object is equal with this object.
    89  	 */
    90  	public function equals( o ):Boolean
    91  	{
    92  		if ( o instanceof Vertex)
    93  		{
    94  			var b1:Boolean = (o.x == x) && (o.y == y) && (o.z == z) ;
    95  			var b2:Boolean = (o.tx == tx) && (o.ty == ty) && (o.tz == tz) ;
    96  			var b3:Boolean = (o.wx == wx) && (o.wy == wy) && (o.wz == wz) ;
    97  			var b4:Boolean = (o.sx == sx) && (o.sy == sy) ;
    98  			return b1 && b2 && b3 && b4 ;
    99  		}
   100  		else
   101  		{
   102  			return false ;	
   103  		} 
   104  	}
   105  
   106  
   107  	/**
   108  	 * Defined the x coordinates in the screen World.
   109  	 */
   110  	public var sx:Number;
   111  	
   112  	/**
   113  	 * Defined the y coordinates in the screen World.
   114  	 */
   115  	public var sy:Number;
   116  
   117  	/**
   118  	 * Defined the x coordinates in the local coordinates.
   119  	 */
   120  	public var tx:Number;
   121  
   122  	/**
   123  	 * Defined the y coordinates in the local coordinates.
   124  	 */
   125  	public var ty:Number;
   126  	
   127  	/**
   128  	 * Defined the z coordinates in the local coordinates.
   129  	 */
   130  	public var tz:Number;
   131  
   132  	/**
   133  	 * Defined the x coordinates in the World coordinates.
   134  	 */
   135  	public var wx:Number;
   136  	
   137  	/**
   138  	 * Defined the y coordinates in the World coordinates.
   139  	 */
   140  	public var wy:Number;
   141  	
   142  	/**
   143  	 * Defined the z coordinates in the World coordinates.
   144  	 */
   145  	 public var wz:Number;	
   146  
   147  	/**
   148  	 * Returns the {@code Vector3} representation of the transformed coordinate system of this Vertex.
   149  	 * @return the {@code Vector3} representation of the transformed coordinate system of this Vertex.
   150  	 */
   151  	public function getTransformVector3():Vector3
   152  	{
   153  		return new Vector3( tx, ty, tz ) ;	
   154  	}
   155  
   156  	/**
   157  	 * Returns the {@code Vector3} representation of this Vertex in the world coordinate.
   158  	 * @return the {@code Vector3} representation of this Vertex in the world coordinate.
   159  	 */
   160  	public function getWorldVector3():Vector3
   161  	{
   162  		return new Vector3( wx, wy, wz );
   163  	}
   164  
   165  	/**
   166  	 * Returns the Object representation of this object.
   167  	 * @return the Object representation of this object.
   168  	 */
   169  	public function toObject():Object 
   170  	{
   171  		return { x:x , y:y , z:z , tx:tx , ty:ty , tz:tz } ;
   172  	}
   173  
   174  	/**
   175  	 * Returns a Eden reprensation of the object.
   176  	 * @return a string representing the source code of the object.
   177  	 */
   178  	public function toSource():String 
   179  	{
   180  		return Serializer.getSourceOf(this, [x, y, z, tx, ty, tz]) ;
   181  	}
   182  
   183  	/**
   184  	 * Returns the string representation of the object.
   185  	 * @return the string representation of the object.
   186  	 */ 	
   187  	public function toString():String
   188  	{
   189  		return "[" + ConstructorUtil.getName(this) + ":{x:" + x + ",y:" + y + ",z:" + z + ",tx:" + tx + ",ty:" + ty + ",tz:" + tz + "}]" ;
   190  	}
   191  
   192  
   193  }
