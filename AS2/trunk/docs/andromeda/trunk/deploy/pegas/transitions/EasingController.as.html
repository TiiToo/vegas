    24  import vegas.core.CoreObject;
    25  import vegas.data.Map;
    26  import vegas.data.map.HashMap;
    27  
    28  /**
    29   * The EasingController register easing methods. 
    30   * This controller can be use with an external configuration or a dynamic easing engine.
    31   * This controller centralize all easing method in the application.
    32   * <p><b>Example :</b></p>
    33   * {@code
    34   * import pegas.transitions.easing.* ;
    35   * import pegas.transitions.EasingController ;
    36   * import pegas.transitions.Tween ;
    37   * 
    38   * var controller:EasingController = new EasingController() ;
    39   * controller.insert( "elastic_ease_out", Elastic.easeOut ) ;
    40   * controller.insert( "elastic_ease_in", Elastic.easeIn ) ;
    41   * 
    42   * // in the code
    43   * 
    44   * var easing:Function = controller.getEasing( "elastic_ease_out" ) ;
    45   * 
    46   * var tw:Tween = new Tween( mc, "_x" , easing, mc._x, 500, 24 , false, true) ; // mc a movieclip on the _root.
    47   * }
    48   * @author eKameleon
    49   */
    50  class pegas.transitions.EasingController extends CoreObject 
    51  {
    52  	
    53  	/**
    54  	 * Creates a new EasingController instance.
    55  	 *  <p><b>Example :</b> {@code var ec:EasingController = new EasingController() ;}</p>
    56  	 */	
    57  	public function EasingController() 
    58  	{
    59  		_map = new HashMap() ;
    60  	}
    61  
    62  	/**
    63  	 * Removes all elements in this controller.
    64  	 */
    65  	public function clear():Void 
    66  	{
    67  		_map.clear() ;	
    68  	}
    69  
    70  	/**
    71  	 * Returns {@code true} if the {@code id} passed in argument is registered in the EasingController.
    72  	 * @param id the {@code id} of the easing method mapped in the EasingController.
    73  	 * @return {@code true} if the {@code id} passed in argument is registered in the EasingController.
    74  	 */
    75  	public function contains( id:String  ):Boolean 
    76  	{
    77  		return _map.containsKey( id ) ;	
    78  	}
    79  	
    80  	/**
    81  	 * Returns a easing method register in the EasingController with the specified {@code id} passed in argument.
    82  	 * If the id isn't find in the controller, this method return a Regular.easeOut function.
    83  	 * @param  id the {@code id} of the easing method mapped in the EasingController.
    84  	 * @return the easing method mapped in the EasingController.  
    85  	 */
    86  	public function getEasing( id:String , defaultEasing:Function ):Function
    87  	{
    88  		return contains(id) ? _map.get( id ) : (defaultEasing || null) ;
    89  	}
    90  	
    91  	/**
    92  	 * Returns the singleton instance of EasingController.
    93  	 * @return singleton instance of EasingController.
    94  	 */
    95  	public static function getInstance():EasingController 
    96  	{
    97  		if ( _instance == null )
    98  		{
    99  			_instance = new EasingController() ;
   100  		}
   101  		return _instance ;
   102  	}
   103  	
   104  	/**
   105  	 * Adds a new entry into the EasingController.
   106  	 * @param id the {@code id} of the easing method mapped in the EasingController.
   107  	 * @param easingMethod the function to register in the EasingController.
   108  	 */
   109  	public function insert( id:String, easingMethod:Function ):Void 
   110  	{
   111  		_map.put( id, easingMethod ) ;
   112  	}
   113  	
   114  	/**
   115  	 * Remove an entry into the EasingController.
   116  	 * @param id the name of the easing method mapped in the EasingController.
   117  	 */
   118  	public function remove( id:String ):Void 
   119  	{
   120  		_map.remove(id) ;
   121  	}
   122  
   123  	/**
   124  	 * Returns the number of elements in the EasingController.
   125  	 * @return the number of elements in the EasingController.
   126  	 */
   127  	public function size():Number
   128  	{
   129  		return _map.size() ;
   130  	}	
   131  
   132  	/**
   133  	 * The internal singleton reference.
   134  	 */
   135  	private static var _instance:EasingController;
   136  
   137  	/**
   138  	 * Internal HashMap instance.
   139  	 */
   140  	private var _map:Map ;
   141  }
