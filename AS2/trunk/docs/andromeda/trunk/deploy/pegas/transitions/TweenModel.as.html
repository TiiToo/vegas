    36  /**
    37   * The model of the Tween class.
    38   * @author eKameleon
    39   */
    40  class pegas.transitions.TweenModel extends AbstractModel implements Iterable 
    41  {
    42  	
    43  	/**
    44  	 * Creates a new TweenModel instance.
    45  	 * @param id the id of the model.
    46  	 * @param tweens The array to initialize the model with some TweenEntry objects. All no TweenEntry objects are ignored.
    47  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    48  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    49  	 * @param tweens an array of TweenEntry objects. 
    50  	 */
    51  	public function TweenModel( id , tweens:Array, bGlobal:Boolean , sChannel:String ) 
    52  	{
    53  		super( id,  bGlobal , sChannel ) ;
    54  		_map = new HashMap() ;
    55  		initEvent() ;
    56  		var size:Number = tweens.length ;
    57  		if (size > 0) 
    58  		{
    59  			for (var i:Number = 0 ; i < size ; i++) 
    60  			{
    61  				var t:TweenEntry = tweens[i] ;
    62  				if ( t != null && t instanceof TweenEntry )
    63  				{
    64  					insert( t.clone() ) ;
    65  				}
    66  			}
    67  		}
    68  	}
    69  
    70  	/**
    71  	 * Returns a shallow copy of this object. This method keep all Tween entries and the id of the original object.
    72  	 * {@code  var clone:TweenModel = tp.clone() ;}
    73  	 * @return a shallow copy of this object.
    74  	 */
    75  	public function clone() 
    76  	{
    77  		return new TweenModel( null , toArray( ) ) ;
    78  	}
    79  
    80  	/**
    81  	 * Clear the model.
    82  	 */
    83  	public function clear():Void 
    84  	{
    85  		_map.clear( ) ;
    86  		dispatchEvent( _eClear ) ;
    87  	}
    88  	
    89  	/**
    90  	 * Returns {@code true} if the specified property exist in this model.
    91  	 * @return {@code true} if the specified property exist in this model.
    92  	 */
    93  	public function contains( prop:String ):Boolean 
    94  	{
    95  		return _map.containsKey( prop ) ;
    96  	}
    97  	
    98  	/**
    99  	 * Returns the specified TweenEntry reference.
   100  	 * @return the specified TweenEntry reference.
   101  	 */
   102  	public function get( prop:String ):TweenEntry 
   103  	{
   104  		return _map.get( prop ) ;
   105  	}
   106  
   107  	/**
   108  	 * Returns the event name use in the {@code insert} method.
   109  	 * @return the event name use in the {@code insert} method.
   110  	 */
   111  	public function getEventTypeADD():String
   112  	{
   113  		return _eAdd.getType() ;
   114  	}
   115  
   116  	/**
   117  	 * Returns the event name use in the {@code clear} method.
   118  	 * @return the event name use in the {@code clear} method.
   119  	 */
   120  	public function getEventTypeCLEAR():String
   121  	{
   122  		return _eClear.getType() ;
   123  	}
   124  
   125  
   126  	/**
   127  	 * Returns the event name use in the {@code remove} method.
   128  	 * @return the event name use in the {@code remove} method.
   129  	 */
   130  	public function getEventTypeREMOVE():String
   131  	{
   132  		return _eRemove.getType() ;
   133  	}
   134  
   135  	/**
   136  	 * Returns the Array representation of all property names in this model.
   137  	 * @return the Array representation of all property names in this model.
   138  	 */
   139  	public function getProperties():Array 
   140  	{
   141  		return _map.getKeys( ) ;
   142  	}
   143  	
   144  	/**
   145  	 * Initialize the events of this model.
   146  	 */
   147  	public function initEvent():Void
   148  	{
   149  		_eAdd    = new TweenEntryEvent( TweenEntryEvent.ADD_ENTRY    , this ) ;
   150  		_eClear  = new TweenEntryEvent( TweenEntryEvent.CLEAR_ENTRY  , this ) ;
   151  		_eRemove = new TweenEntryEvent( TweenEntryEvent.REMOVE_ENTRY , this ) ;
   152  	}
   153  
   154  	/**
   155  	 * Inserts a new TweenEntry in the model.
   156  	 */
   157  	public function insert( entry:TweenEntry ):Boolean 
   158  	{
   159  		var p:String = entry.prop ;
   160  		if ( p ) 
   161  		{
   162  			_map.put( p, entry ) ;
   163  			_eAdd.setTweenEntry(entry) ;
   164  			dispatchEvent( _eAdd ) ;
   165  			return true ;
   166  		}
   167  		else 
   168  		{
   169  			return false ;
   170  		}
   171  	}
   172  
   173  	/**
   174  	 * Returns an iterator of this model.
   175  	 * @return an iterator of this model.
   176  	 */
   177  	public function iterator():Iterator 
   178  	{
   179  		return _map.iterator( ) ;
   180  	}
   181  
   182  	/**
   183  	 * Removes an entry in the model.
   184  	 */
   185  	public function remove( entry:Object ):Boolean 
   186  	{
   187  		var p:String ;
   188  		if (TypeUtil.typesMatch( entry, String )) 
   189  		{
   190  			p = String( entry ) ;
   191  		}
   192  		else if (TypeUtil.typesMatch( entry, TweenEntry )) 
   193  		{
   194  			p = entry.prop ;
   195  		}
   196  		else 
   197  		{
   198  			throw new IllegalArgumentError(this + " remove method failed with an unknow argument value : " + entry ) ;
   199  		} 
   200  		if (p) 
   201  		{
   202  			var t:TweenEntry = _map.remove( p ) ;
   203  			if (t != null) 
   204  			{
   205  				_eRemove.setTweenEntry( t ) ;
   206  				dispatchEvent( _eRemove ) ;
   207  				return true ;
   208  			}
   209  		} 
   210  		else 
   211  		{
   212  			return false ;
   213  		}
   214  	}
   215  
   216  
   217  	/**
   218  	 * Sets the event name use in the {@code insert} method.
   219  	 */
   220  	public function setEventTypeADD( type:String ):Void
   221  	{
   222  		_eAdd.setType( type ) ;
   223  	}
   224  
   225  	/**
   226  	 * Sets the event name use in the {@code clear} method.
   227  	 */
   228  	public function setEventTypeCLEAR( type:String ):Void
   229  	{
   230  		_eClear.setType( type ) ;
   231  	}
   232  
   233  	/**
   234  	 * Sets the event name use in the {@code remove} method.
   235  	 */
   236  	public function setEventTypeREMOVE( type:String ):Void
   237  	{
   238  		_eRemove.setType( type ) ;
   239  	}
   240  	
   241  	/**
   242  	 * Returns the number of elements in the model.
   243  	 * @return the number of elements in the model.
   244  	 */
   245  	public function size():Number 
   246  	{
   247  		return _map.size( ) ;
   248  	}
   249  
   250  	/**
   251  	 * Returns the Array representation of all TweenEntry objects in this model.
   252  	 * @return the Array representation of all TweenEntry objects in this model.
   253  	 */
   254  	public function toArray():Array 
   255  	{
   256  		return _map.getValues( ) ;
   257  	}
   258  	
   259  	/**
   260  	 * Returns the Map representation of all TweenEntry objects in this model.
   261  	 * @return the Map representation of all TweenEntry objects in this model.
   262  	 */
   263  	public function toMap():Map 
   264  	{
   265  		return _map.clone() ;		
   266  	}
   267  
   268  	/**
   269  	 * Returns the String representation of this object.
   270  	 * @return the String representation of this object.
   271  	 */
   272  	public function toString():String 
   273  	{
   274  		var s:String = "[TweenModel" ; 
   275  		if ( _map.size() > 0 )
   276  		{
   277  			s += ":" + _map.toString() ;
   278  		}
   279  		s += "]" ;
   280  		return s ;
   281  	}
   282

   285  	 */

   287  	
   288  	/**
   289  	 * @private
   290  	 */
   291  	private var _eClear:TweenEntryEvent ;
   292  	
   293  	/**
   294  	 * @private
   295  	 */
   296  	private var _eRemove:TweenEntryEvent ;
   297  	
   298  	/**
   299  	 * @private
   300  	 */
   301  	private var _map:HashMap ;
