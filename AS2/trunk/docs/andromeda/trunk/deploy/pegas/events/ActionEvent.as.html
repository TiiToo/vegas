    24  import vegas.events.DynamicEvent;
    25  import vegas.util.serialize.Serializer;
    26  
    27  /**
    28   * The ActionEvent is notify by all the objects who implements the Action interface.
    29   * @author eKameleon
    30   */
    31  class pegas.events.ActionEvent extends DynamicEvent 
    32  {
    33  
    34  	/**
    35  	 * Creates a new ActionEvent instance.
    36  	 * @param type the string type of the instance. 
    37  	 * @param target the target of the event.
    38  	 * @param info The information object of this event.
    39  	 * @param context the optional context object of the event.
    40  	 * @param bubbles indicates if the event is a bubbling event.
    41  	 * @param eventPhase the current EventPhase of the event.
    42  	 * @param time this parameter is used in the Eden deserialization.
    43  	 * @param stop this parameter is used in the Eden deserialization.
    44  	 */
    45  	public function ActionEvent( type:String, target:Object, info, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number )
    46  	{
    47  		super(type, target, context, bubbles, eventPhase, time, stop) ;
    48  		_oInfo = info  ;
    49  	}
    50  	
    51  	/**
    52  	 * The event type when the event notify a change.
    53  	 */
    54  	public static var CHANGE:String = "onChanged" ;
    55  
    56  	/**
    57  	 * The event type when the event notify a cleanup process.
    58  	 */
    59  	public static var CLEAR:String = "onCleared" ;
    60  
    61  	/**
    62  	 * The event type when the event notify the end of the process.
    63  	 */
    64  	public static var FINISH:String = "onFinished" ;
    65  
    66  	/**
    67  	 * The event type when the event notify an info process.
    68  	 */
    69  	public static var INFO:String = "onInfo" ;
    70  
    71  	/**
    72  	 * The event type when the process is looped.
    73  	 */
    74  	public static var LOOP:String = "onLooped" ;
    75  
    76  	/**
    77  	 * The event type when the event notify a process in progress.
    78  	 */
    79  	public static var PROGRESS:String = "onProgress" ;
    80  
    81  	/**
    82  	 * The event type when the process is resumed.
    83  	 */
    84  	public static var RESUME:String = "onResumed" ;
    85  	
    86  	/**
    87  	 * The event type when the process is started.
    88  	 */
    89  	public static var START:String = "onStarted" ;
    90  	
    91  	/**
    92  	 * The event type when the process is stopped.
    93  	 */
    94  	public static var STOP:String = "onStopped" ;	
    95  
    96  	/**
    97  	 * The event type when the process is out of time.
    98  	 */
    99  	public static var TIMEOUT:String = "onTimeOut" ;
   100  
   101  	/**
   102  	 * Returns the shallow copy of this object.
   103  	 * @return the shallow copy of this object.
   104  	 */
   105  	public function clone() 
   106  	{
   107  		return new ActionEvent(getType(), getTarget()) ;
   108  	}
   109  
   110  	/**
   111  	 * Returns the info object of this event.
   112  	 * @return the info object of this event.
   113  	 */
   114  	public function getInfo() 
   115  	{
   116  		return _oInfo ;
   117  	}
   118  
   119  	/**
   120  	 * Sets the info object of this event.
   121  	 */
   122  	public function setInfo( oInfo ):Void 
   123  	{
   124  		_oInfo = oInfo ;	
   125  	}
   126  	
   127  	/**
   128  	 * @private
   129  	 */
   130  	private var _oInfo ;
   131  
   132  	/**
   133  	 * @private
   134  	 */
   135  	private /*protected*/ function _getParams():Array 
   136  	{
   137  		var ar:Array = super._getParams() ;
   138  		ar.splice(2, null, Serializer.toSource(_oInfo)) ;
   139  		return ar ;
   140  	}
   141  
   142  }
