    24  import vegas.events.DynamicEvent;
    25  import vegas.util.serialize.Serializer;
    26  
    27  /**
    28   * @author eKameleon
    29   */
    30  class pegas.events.MouseEvent extends DynamicEvent 
    31  {
    32  
    33  	/**
    34  	 * Creates a new MouseEvent instance.
    35  	 */
    36  	public function MouseEvent
    37  	(
    38  		type:String, target, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number
    39  		, p_localX:Number, p_localY:Number, p_relatedObject:Number, p_ctrlKey:Boolean, p_altKey:Boolean
    40  		, p_shiftKey:Boolean, p_buttonDown:Boolean, p_delta:Number
    41  	)
    42  	{
    43  			
    44  		super(type, target, context, bubbles, eventPhase, time, stop) ;
    45  		
    46  		localX = isNaN(p_localX) ? 0 : p_localX ;
    47  		localY = isNaN(p_localY) ? 0 : p_localY ;
    48  		relatedObject = p_relatedObject || null ;
    49  		ctrlKey = p_ctrlKey || false ;
    50  		altKey = p_altKey || false ;
    51  		shiftKey = p_shiftKey || false ;
    52  		buttonDown = p_buttonDown || false ;
    53  		delta = isNaN(p_delta) ? 0 : p_delta ;
    54  		
    55  	}
    56  
    57  	/**
    58  	 * Reserved for future use (not currently functional).
    59  	 */
    60  	public var altKey:Boolean = false ;
    61  
    62  	/**
    63  	 * Indicates whether the left mouse button is depressed.
    64  	 */
    65  	public var buttonDown:Boolean = false ;
    66  
    67  	/**
    68  	 * Indicates whether the control(Ctrl) key modifier is activated.
    69  	 */
    70  	public var ctrlKey:Boolean = false ;
    71  
    72  	/**
    73  	 * A number indicating how many lines should be scrolled for each notch the user rolls the mouse wheel.
    74  	 * A positive delta value indicates an upward scroll.
    75  	 * A negative value indicates a downward scroll.
    76  	 * Typical values are 1 to 3, but faster scrolling may produce larger values.
    77  	 * This parameter is used only for the MouseEventType.mouseWheel event.
    78  	 */
    79  	public var delta:Number = 0 ;
    80  
    81  	/**
    82  	 * The horizontal coordinate at which the event occurred relative to the containing sprite.
    83  	 */
    84  	public var localX:Number = 0 ;
    85  
    86  	/**
    87  	 * The vertical coordinate at which the event occurred relative to the containing sprite.
    88  	 */
    89  	public var localY:Number = 0 ;
    90  
    91  	/**
    92  	 * Indicates the complementary InteractiveObject instance that is affected by the event.
    93  	 * For example, when a mouseOut event occurs, the relatedObject represents the display list object to which the pointing device now points.
    94  	 */
    95  	public var relatedObject:Number = null ;
    96  
    97  	/**
    98  	 * Indicates whether the Shift(Shift) key modifier is activated.
    99  	 */
   100  	public var shiftKey:Boolean = false ;
   101  
   102  	public function clone() 
   103  	{
   104  		var prop:String ;
   105  		var props:Array = ["localX", "localY", "relatedObject", "ctrlKey", "altKey", "shiftKey", "buttonDown", "delta"] ;
   106  		var me:MouseEvent = new MouseEvent(getType(), getTarget()) ;
   107  		var l:Number = props.length ;
   108  		while(--l > -1) 
   109  		{
   110  			prop = props[l] ;
   111  			me[prop] = this[prop];
   112  		}
   113  		return me ;
   114  	}
   115  
   116  	/*protected*/ private function _getParams():Array 
   117  	{
   118  		var ar1:Array = super._getParams() ;
   119  		var ar2:Array = [] ;
   120  		var props:Array = ["localX", "localY", "relatedObject", "ctrlKey", "altKey", "shiftKey", "buttonDown", "delta"] ;
   121  		var len:Number = props.length ;
   122  		for (var i:Number = 0 ; i<len ; i++) {
   123  			ar2[i] = Serializer.toSource( this[props[i]] ) ;
   124  		} ;
   125  		return ar1.concat(ar2) ;
   126  	}
   127  	
   128  }
