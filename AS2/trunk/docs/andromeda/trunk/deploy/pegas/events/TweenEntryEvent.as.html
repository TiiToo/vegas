    24  import pegas.transitions.TweenEntry;
    25  
    26  import vegas.events.BasicEvent;
    27  
    28  /**
    29   * This event dispatch a TweenEntry.
    30   * @author eKameleon
    31   */
    32  class pegas.events.TweenEntryEvent extends BasicEvent 
    33  {
    34  
    35  	/**
    36  	 * Creates a new TweenEntryEvent instance.
    37  	 * @param type the string type of the instance. 
    38  	 * @param target the target of the event.
    39  	 * @param context the optional context object of the event.
    40  	 * @param bubbles indicates if the event is a bubbling event.
    41  	 * @param eventPhase the current EventPhase of the event.
    42  	 * @param time this parameter is used in the Eden deserialization.
    43  	 * @param stop this parameter is used in the Eden deserialization.
    44  	 */
    45  	public function TweenEntryEvent(type:String, target:Object, entry:TweenEntry, context, bubbles:Boolean, eventPhase : Number, time : Number, stop : Number) 
    46  	{
    47  		super(type, target, context, bubbles, eventPhase, time, stop);
    48  		if ( entry != null )
    49  		{
    50  			setTweenEntry( entry ) ;	
    51  		}
    52  	}
    53  	
    54  	/**
    55  	 * Default event type when a tween entry is added.
    56  	 */
    57  	public static var ADD_ENTRY:String = "onAddEntry" ;
    58  
    59  	/**
    60  	 * Default event type when all tween entries are removed.
    61  	 */
    62  	public static var CLEAR_ENTRY:String = "onClearEntry" ;
    63  
    64  	/**
    65  	 * Default event type when a tween entry is removed.
    66  	 */
    67  	public static var REMOVE_ENTRY:String = "onRemoveEntry" ;
    68  
    69  	/**
    70  	 * Returns the TweenEntry reference of this event.
    71  	 * @return the TweenEntry reference of this event.
    72  	 */
    73  	public function getTweenEntry():TweenEntry
    74  	{
    75  		return _entry ;
    76  	}
    77  	
    78  	/**
    79  	 * Sets the TweenEntry reference of this event.
    80  	 */
    81  	public function setTweenEntry( entry:TweenEntry ):Void
    82  	{
    83  		_entry = entry ;
    84  	}
    85  	
    86  	/**
    87  	 * The internal TweenEntry reference of this event.
    88  	 */
    89  	private var _entry:TweenEntry ;
    90  
    91  }
