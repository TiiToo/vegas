    24  import pegas.process.AbstractAction;
    25  
    26  import vegas.events.Delegate;
    27  
    28  /**
    29   * This {@code Action} object run a proxy method.
    30   * <p><b>Example :</b></p>
    31   * {@code
    32   * import pegas.process.ActionProxy ;
    33   * import vegas.core.CoreObject     ;
    34   * import vegas.events.Delegate     ;
    35   * import vegas.events.Event        ;
    36   * 
    37   * var debug:Function = function( e:Event ):Void
    38   * {
    39   *     trace( this + " debug : " + e ) ;
    40   * }
    41   * 
    42   * var method:Function = function( message:String ):Void
    43   * {
    44   *      trace( this + " message: " + message ) ;
    45   * }
    46   * 
    47   * var scope:CoreObject    = new CoreObject() ;
    48   * var process:ActionProxy = new ActionProxy( scope, method ) ;
    49   * 
    50   * process.addGlobalEventListener( new Delegate( this, debug ) ) ;
    51   * 
    52   * process.run() ;
    53   * }
    54   */
    55  class pegas.process.ActionProxy extends AbstractAction 
    56  {
    57  
    58  	/**
    59  	 * Creates a new ActionProxy instance.
    60  	 * @param scope The scope of the proxy.
    61  	 * @param method The method of the proxy.
    62  	 * @param ...arguments All the optionals arguments to pass in the proxy method when is invoked.
    63  	 */
    64  	public function ActionProxy( scope:Object, method:Function ) 
    65  	{
    66  		obj  = scope ;
    67  		func = method ;
    68  		args = arguments.splice(2) ;
    69  	}
    70  
    71  	/**
    72  	 * The Array representation of all arguments passed-in the proxy method.
    73  	 */
    74  	public var args:Array ;
    75  	
    76  	/**
    77  	 * The method reference of this proxy process.
    78  	 */
    79  	public var func:Function ;
    80  	
    81  	/**
    82  	 * The scope reference of this proxy process.
    83  	 */
    84  	public var obj:Object ;
    85  	
    86  	/**
    87  	 * Returns a shallow copy of this object.
    88  	 * @return a shallow copy of this object.
    89  	 */
    90  	public /*override*/ function clone() 
    91  	{
    92  		var p:ActionProxy = new ActionProxy(obj, func);
    93  		p.args = args ;
    94  		return p ;
    95  	}
    96  	
    97  	/**
    98  	 * Run the process.
    99  	 */
   100  	public /*override*/ function run():Void 
   101  	{
   102  		notifyStarted() ;
   103  		_setRunning(true) ;
   104  		Delegate.create.apply(this, [obj, func].concat(args)) ();
   105  		_setRunning(false) ;
   106  		notifyFinished() ;
   107  	}
   108  
   109  }
