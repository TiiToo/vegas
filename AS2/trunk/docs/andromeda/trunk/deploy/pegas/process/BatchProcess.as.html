    24  import pegas.events.ActionEvent;
    25  import pegas.process.Action;
    26  import pegas.process.Batch;
    27  import pegas.process.SimpleAction;
    28  
    29  import vegas.data.iterator.Iterator;
    30  import vegas.events.Delegate;
    31  import vegas.events.EventListener;
    32  import vegas.events.EventTarget;
    33  
    34  /**
    35   * This {@code Action} object register {@code Action} objects in a batch process.
    36   * @author eKameleon
    37   */
    38  class pegas.process.BatchProcess extends SimpleAction
    39  {
    40  	
    41  	/**
    42  	 * Creates a new BatchTransition instance.
    43  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    44  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    45  	 */
    46  	function BatchProcess( bGlobal:Boolean, sChannel:String ) 
    47  	{
    48  		super(bGlobal, sChannel);
    49  		_eProgress = new ActionEvent(ActionEvent.PROGRESS, this) ;
    50  		_batch = new Batch() ;
    51  		_finishListener = new Delegate(this, _onFinished) ;
    52  		initialize() ;
    53  	}
    54  
    55     	/**
    56       * Inserts an IAction object in the batch process collection.
    57       */
    58  	public function addAction( action:Action ):Void
    59  	{
    60  		EventTarget(action).addEventListener( ActionEvent.FINISH, _finishListener) ;
    61  		_batch.insert( action ) ;	
    62  	}
    63  
    64  	/**
    65  	 * Clear the layout manager.
    66  	 */
    67  	public function clear():Void
    68  	{
    69  
    70  		if ( running && !_batch.isEmpty() )
    71  		{
    72  			var it:Iterator = _batch.iterator() ;
    73  			while(it.hasNext())
    74  			{
    75  				clearProcess( it.next()) ;
    76  			}
    77  		}
    78  		_batch.clear() ;
    79  		_cpt = 0 ;
    80  	}
    81  	
    82  	/**
    83  	 * Clear the specified process. Overrides this method in a custom BatchProcess implementation. 
    84  	 */
    85  	public function clearProcess( action:Action ):Void
    86  	{
    87  		// overrides
    88  	}
    89  
    90  	/**
    91  	 * Invoqued before the notifyFinished method is invoqued.
    92  	 * Overrides this method.
    93  	 */
    94  	public function finish():Void
    95  	{
    96  		// overrides
    97  	}
    98  
    99  	/**
   100  	 * Returns the internal Batch reference of this layout manager.
   101  	 * @return the internal Batch reference of this layout manager.
   102  	 */
   103  	public function getBatch():Batch
   104  	{
   105  		return _batch ;	
   106  	}
   107  	
   108  	/**
   109  	 * Initialize the layout. Overrides this method.
   110  	 */
   111  	public function initialize():Void
   112  	{
   113  		//	
   114  	}
   115  
   116  	/**
   117  	 * Notify an ActionEvent when the process is in progress.
   118  	 */
   119  	public function notifyProgress( a:ActionEvent ):Void 
   120  	{
   121  		_eProgress.context = a ;
   122  		dispatchEvent(_eProgress) ;
   123  	}
   124  
   125  	/**
   126  	 * Removes an {@code Action} object in the internal batch collection.
   127  	 */
   128  	public function removeAction( action:Action ):Void
   129  	{
   130  		if (_batch.contains(action))
   131  		{
   132  			EventTarget(action).removeEventListener( ActionEvent.FINISH, _finishListener) ;
   133  			_batch.remove( action ) ;
   134  		}	
   135  	}
   136  
   137  	/**
   138  	 * Run the process.
   139  	 */
   140  	public function run():Void 
   141  	{
   142  		start() ;
   143  		notifyStarted() ;
   144  		_cpt = 0 ;
   145  		_batch.run() ;
   146  	}
   147  
   148  	/**
   149  	 * Invoqued before the notifyStarted method is invoqued.
   150  	 * Overrides this method.
   151  	 */
   152  	public function start():Void
   153  	{
   154  		// overrides
   155  	}
   156  
   157  	/**
   158  	 * The internal batch process of this manager.
   159  	 */
   160  	private var _batch:Batch ;
   161  
   162  	/**
   163  	 * Internal count use in the _onFinished method.
   164  	 */
   165  	private var _cpt:Number ;
   166  
   167  	/**
   168  	 * Invoqued during the progress of the process.
   169  	 */
   170  	private var _eProgress:ActionEvent ;
   171  
   172  	/**
   173  	 * The EventListener invoqued when the process is finished. 
   174  	 */
   175  	private var _finishListener:EventListener ;
   176  
   177  	/**
   178  	 * Invoqued when a tween finish this movement.
   179  	 * If all tweens are finished the notifyFinished method is invoqued.
   180  	 */
   181  	private function _onFinished( e:ActionEvent ):Void
   182  	{
   183  		_cpt ++ ;
   184  		notifyProgress( e.getTarget() ) ;		
   185  		if (_cpt == _batch.size())
   186  		{
   187  			finish() ;		
   188  			notifyFinished() ;
   189  		}
   190  	}
   191  
   192  }
