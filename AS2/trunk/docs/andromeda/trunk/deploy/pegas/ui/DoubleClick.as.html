    24  import pegas.events.MouseEvent;
    25  
    26  import vegas.events.Event;
    27  import vegas.events.FastDispatcher;
    28  
    29  /**
    30   * This class defines a double click manager in your applications.
    31   * <p><b>Example :</b></p>
    32   * {@code
    33   * import pegas.events.MouseEvent ;
    34   * import pegas.ui.DoubleClick ;
    35   * 
    36   * // Test with broadcaster
    37   * 
    38   * var doubleClick:Function = function (ev:MouseEvent):Void
    39   * {
    40   *     trace ( this + " -> " + ev.type) ;
    41   * }
    42   * 
    43   * DoubleClick.addListener( this )  ;
    44   * 
    45   * // Test with the static property ISDOUBLE.
    46   * 
    47   * bt.onPress = function ()
    48   * {
    49   *     if ( DoubleClick.ISDOUBLE )
    50   *     {
    51   *        trace (this + " -> " + MouseEvent.DOUBLE_CLICK) ;
    52   *     }
    53   * }
    54   * 
    55   * }
    56   * @author eKameleon
    57   */
    58  class pegas.ui.DoubleClick 
    59  {
    60  
    61  	/**
    62  	 * Returns the delay between 2 clicks.
    63  	 * @return the delay between 2 clicks.
    64  	 */
    65  	static public function get delay():Number
    66  	{
    67  		return _delay ;
    68  	}
    69     
    70     	/**
    71     	 * Sets the delay between 2 clicks.
    72     	 */
    73  	static public function set delay(n:Number):Void 
    74  	{
    75  		_delay = n > 0 ? n : 0 ;
    76  	}
    77  
    78  	/**
    79  	 * Returns {@code true} if the last click with the mouse is a double click.
    80  	 * @return {@code true} if the last click with the mouse is a double click.
    81  	 */
    82  	static public function get ISDOUBLE():Boolean 
    83  	{
    84  		if (!_dispatcher) initialize() ;
    85  		return __ISDOUBLE__ ;
    86  	}
    87  	
    88  	/**
    89  	 * Registers an object to receive event notification messages.
    90  	 * @return {@code true} of the listener is register.
    91  	 */
    92  	static public function addListener(listener):Boolean 
    93  	{
    94  		if ( _dispatcher == null ) 
    95  		{
    96  			initialize() ;
    97  		}
    98  		return _dispatcher.addListener(listener) ;
    99  	}
   100  	
   101  	/**
   102  	 * Initialize the double click dispatcher.
   103  	 * @return {@code true} if the initialize process is success.
   104  	 */
   105  	static public function initialize():Boolean 
   106  	{
   107  		if (_dispatcher == null)
   108  		{
   109  			_dispatcher = new FastDispatcher() ;
   110  			Mouse.addListener(DoubleClick) ;
   111  			reset() ;
   112  			return true ;
   113  		}
   114  		else
   115  		{
   116  			return false ;
   117  		}
   118  	}
   119  	
   120  	/**
   121  	 * Release the double click dispatcher.
   122  	 * @return {@code true} if the release process is success.
   123  	 */
   124  	static public function release():Boolean
   125  	{
   126  		if ( _dispatcher != null )
   127  		{
   128  			_dispatcher = null ;
   129  			reset() ;
   130  			Mouse.removeListener(DoubleClick) ;
   131  			return true ;
   132  		}
   133  		else
   134  		{
   135  			return false ;
   136  		}
   137  	}
   138  
   139  	/**
   140  	 * Removes an object from the list of objects that receive event notification messages.
   141  	 * @return {@code true} of the listener is removed.
   142  	 */
   143  	static public function removeListener(listener):Boolean 
   144  	{
   145  		return _dispatcher.removeListener(listener) ;
   146  	}
   147  	
   148  	/**
   149  	 * Removes all objects from the list of objects that receive event notification messages.
   150  	 */
   151  	static public function removeAllListeners():Void 
   152  	{
   153  		_dispatcher.removeAllListeners() ;
   154  	}
   155  	
   156  	/**
   157  	 * Returns the string representation of this object.
   158  	 * @return the string representation of this object.
   159  	 */
   160  	static public function toString():String 
   161  	{
   162  		return "[DoubleClick]" ;
   163  	}
   164  
   165  	static private var _delay:Number = 250 ;
   166  	static private var _itv:Number ;
   167  	static private var _cpt:Number ;
   168  	static private var _dispatcher:FastDispatcher ;
   169  	static private var __ISDOUBLE__:Boolean = false ;
   170  	
   171  	/**
   172  	 * Invoqued when the mouse is down.
   173  	 */
   174  	static private function onMouseDown():Void
   175  	{
   176  		_cpt ++ ;
   177  		__ISDOUBLE__ = _cpt == 2 ;
   178  		clearInterval(_itv) ;
   179  		var type:String = (_cpt == 2)  ? MouseEvent.DOUBLE_CLICK : MouseEvent.CLICK ;
   180  		var ev:Event = new MouseEvent(type, DoubleClick) ;
   181  		_dispatcher.dispatch( ev ) ;
   182  		_itv = setInterval( reset , _delay) ;
   183  	}
   184  
   185  	/**
   186  	 * Reset the effect.
   187  	 */	
   188  	static private function reset():Void
   189  	{
   190  		clearInterval(_itv);
   191  		__ISDOUBLE__ = false ;
   192  		_cpt = 0 ;
   193  	}
   194  	
   195  }
