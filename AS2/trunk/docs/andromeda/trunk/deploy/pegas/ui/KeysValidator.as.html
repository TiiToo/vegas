    35  		
    36  		- ASCII:Number (1)
    37  
    38  	PROPERTY SUMMARY
    39  	
    40  		- codePolicy[R/W] [Number] : vaut CODE(0) ou ASCII(1) (par défaut vaut CODE)
    41

    43  	
    44  	METHOD SUMMARY
    45

    83  import vegas.core.CoreObject;
    84  import vegas.core.IValidator;
    85  import vegas.data.iterator.Iterator;
    86  import vegas.data.set.HashSet;
    87  import vegas.events.EventDispatcher;
    88  import vegas.events.EventListener;
    89  import vegas.events.ValidatorEvent;
    90  import vegas.events.ValidatorEventType;

    92  /**
    93   * Set a collection of keys to validate an action.
    94   * @author eKameleon
    95   * @see asgard.ui.Keyboard (key manager with F1..F12, etc. key codes)
    96   */
    97  class pegas.ui.KeysValidator extends CoreObject implements IValidator 
    98  {
    99  	
   100  	/**
   101  	 * Creates a new KeysValidator instance.
   102  	 */
   103  	public function KeysValidator( keys:Array ) 
   104  	{
   105  		Key.addListener(this) ;
   106  		_dispatcher = new EventDispatcher() ;
   107  		_set = new HashSet(keys) ;
   108  		_reset() ;
   109  	}
   110  
   111  	static public var CODE:Number = 0 ;
   112  
   113  	static public var ASCII:Number = 1 ;
   114  	
   115  	static private var __ASPF__ = _global.ASSetPropFlags(KeysValidator, null, 7, 7) ;
   116  	
   117  	public function get codePolicy():Number 
   118  	{
   119  		return _cp ;
   120  	}
   121  	
   122  	public function set codePolicy(n:Number):Void 
   123  	{
   124  		setCodePolicy(n) ;
   125  	}
   126  	
   127  	public function addEventListener( eventName:String, listener, autoRemove:Boolean, priority:Number ):Void 
   128  	{
   129  		_dispatcher.addEventListener(eventName, listener, autoRemove, priority) ;
   130  	}
   131  
   132  	public function clear():Void 
   133  	{
   134  		_set.clear() ;
   135  	}
   136  	
   137  	public function countKeys():Number 
   138  	{
   139  		return _set.size() ;
   140  	}
   141  
   142  	public function insert(keyCode:Number):Boolean 
   143  	{
   144  		var b:Boolean = _set.insert(keyCode) ;
   145  		if (b) _reset() ;
   146  		return b ;
   147  	}
   148  	
   149  	public function remove(keyCode:Number):Boolean 
   150  	{
   151  		var b:Boolean = _set.remove(keyCode) ;
   152  		if (b) _reset() ;
   153  		return b ;
   154  	}
   155  	
   156  	public function removeEventListener(eventName:String, listener):EventListener 
   157  	{
   158  		return _dispatcher.removeEventListener(eventName, listener) ;
   159  	}
   160  	
   161  	public function setCodePolicy(code:Number):Void 
   162  	{
   163  		_cp = code ;
   164  	}
   165  	
   166  	public function supports(value):Boolean 
   167  	{
   168  		return _it.next() == value ;
   169  	}
   170  	
   171  	public function validate(value):Void 
   172  	{
   173  		if ( supports(value) ) 
   174  		{
   175  			if (!_it.hasNext()) 
   176  			{
   177  				_reset() ;
   178  				_dispatcher.dispatchEvent(new ValidatorEvent(ValidatorEventType.VALID, this)) ;
   179  			}
   180  		} 
   181  		else 
   182  		{
   183  			_reset() ;
   184  			_dispatcher.dispatchEvent(new ValidatorEvent(ValidatorEventType.INVALID, this)) ;
   185  		}
   186  	}
   187  	
   188  
   189  	
   190  	private var _cp:Number = KeysValidator.CODE ; // codePolicy property
   191  	private var _dispatcher:EventDispatcher ;
   192  	private var _set:HashSet ;
   193  	private var _it:Iterator ;
   194  	
   195  	private function _reset():Void 
   196  	{
   197  		_it = _set.iterator() ;
   198  	}
   199  	
   200  	private function onKeyDown():Void 
   201  	{
   202  		var code:Number ;
   203  		if (_cp == KeysValidator.ASCII) {
   204  			code = Key.getAscii() ;
   205  			if (code == 0 || code == undefined ) code = Key.getCode() ;
   206  		} else {
   207  			code = Key.getCode() ;
   208  		}
   209  		validate(code) ;
   210  	}
   211  	
   212  	private function onKeyUp():Void 
   213  	{
   214  		_reset() ;
   215  	}
   216  
   217  }
