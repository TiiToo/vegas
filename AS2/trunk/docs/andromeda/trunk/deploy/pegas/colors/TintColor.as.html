    24  import pegas.colors.BasicColor;
    25  import pegas.colors.ColorRGB;
    26  
    27  /**
    28   * Control the tint of a Color object.
    29   * <p>Thanks 2003 Robert Penner - Use freely, giving credit where possible.</p>
    30   * <p>This code is based on the book : Robert Penner's Programming Macromedia Flash MX. More informations in :
    31   * <ul>
    32   * <li>http://www.robertpenner.com/profmx
    33   * <li>http://www.amazon.com/exec/obidos/ASIN/0072223561/robertpennerc-20
    34   * </ul>
    35   * </p>
    36   * @author eKameleon
    37   */
    38  class pegas.colors.TintColor extends BasicColor 
    39  {
    40  
    41  	/**
    42  	 * Creates a new TintColor instance.
    43  	 * @param mc the movieclip target of the TintColor object.
    44   	 */
    45  	public function TintColor (mc:MovieClip) 
    46  	{ 
    47  		super(mc) ;
    48  	}
    49  
    50  	/**
    51  	 * Returns the tint of a Color object.
    52  	 * <p><b>Example</b></b>
    53  	 * {@code
    54  	 * var cTint:TintColor = new TintColor(my_mc);
    55  	 * cTint.setTint( 0, 0, 128, 50 ) ;
    56  	 * var tint:Object = cTint.getTint();
    57  	 * trace ("r : " + tint.r);
    58  	 * trace ("g : " + tint.g);
    59  	 * trace ("b : " + tint.b);
    60  	 * trace ("percent : " + tint.percent);
    61  	 * }
    62  	 * @return The tint value object with r, g, b, and percent properties.
    63  	 */
    64  	public function getTint():Object 
    65  	{
    66  		var t:Object = getTransform();
    67  		var percent:Number = 100 - t.ra ;
    68  		var ratio:Number = 100 / percent;
    69  		return { 
    70  			percent: percent ,
    71  			r : t.rb * ratio ,
    72  			g : t.gb * ratio ,
    73  			b : t.bb * ratio 
    74  		} ;
    75  	}
    76  
    77  	/**
    78  	 * Returns the tint of a Color object.
    79  	 * <p><b>Example</b></b>
    80  	 * {@code
    81  	 * var cTint:TintColor = new TintColor(my_mc);
    82  	 * cTint.setTint2( 0x0000FF, 100 ) ;
    83  	 * var tint:Object = cTint.getTint2();
    84  	 * trace ("rgb     : " + tint.rgb);
    85  	 * trace ("percent : " + tint.percent);
    86  	 * }
    87  	 * @return The tint value object with rgb and percent properties.
    88  	 */
    89  	public function getTint2():Object 
    90  	{
    91  		var t:Object = getTransform();
    92  		var percent:Number = 100 - t.ra ;
    93  		var ratio:Number = 100 / percent ;
    94  		return { 
    95  			percent:percent ,
    96  			rgb:ColorRGB.rgb2hex(t.rb*ratio, t.gb*ratio, t.bb*ratio) 
    97  		} ;
    98  	}
    99  
   100  	/**
   101  	 * Returns the tint offset of a Color object.
   102  	 * <p><b>Example</b></b>
   103  	 * {@code
   104  	 * var cTint:TintColor = new TintColor(my_mc);
   105  	 * cTint.setTintOffset(0, 0, 128);
   106  	 * var tint:Object = cTint.getTintOffset();
   107  	 * trace ("r : " + tint.r);
   108  	 * trace ("g : " + tint.g);
   109  	 * trace ("b : " + tint.b);
   110  	 * }
   111  	 * @return The tint offset value object with r, g, and b properties.
   112  	 */
   113  	public function getTintOffset():Object 
   114  	{
   115  		var t:Object = getTransform() ;
   116  		return {r:t.rb, g:t.gb, b:t.bb} ;
   117  	}
   118  
   119  	/**
   120  	 * Tints a color object with a Color according to a certain percentage.
   121  	 * <p><b>Example</b></b>
   122  	 * {@code
   123  	 * var cTint:TintColor = new TintColor(my_mc);
   124  	 * cTint.setTint( 0, 0, 128, 50 ) ;
   125  	 * }
   126  	 * @param r The red color value.
   127  	 * @param g The green color value.
   128  	 * @param b The blue color value.
   129  	 * @param percent
   130  	 */
   131  	public function setTint(r:Number, g:Number, b:Number, percent:Number):Void 
   132  	{
   133  		var ratio:Number = percent / 100;
   134  		var t:Object = { rb:r*ratio, gb:g*ratio, bb:b*ratio } ;
   135  		t.ra = t.ga = t.ba = 100-percent ;
   136  		setTransform (t);
   137  	}
   138  
   139  	/**
   140  	 * Tints a color object with a Color according to a certain percentage.
   141  	 * <p><b>Example</b></b>
   142  	 * {@code
   143  	 * var cTint:TintColor = new TintColor(my_mc);
   144  	 * cTint.setTint2( 0x0000FF, 100 ) ;
   145  	 * }
   146  	 * @param hex The rgb value.
   147  	 * @param percent The tint percentage.
   148  	 */
   149  	public function setTint2(hex:Number, percent:Number):Void 
   150  	{
   151  		var c:Object = ColorRGB.hex2rgb (hex) ;
   152  		var ratio:Number = percent / 100 ;
   153  		var t:Object = {rb:c.r*ratio, gb:c.g*ratio, bb:c.b*ratio};
   154  		t.ra = t.ga = t.ba = 100-percent;
   155  		setTransform (t);
   156  	}
   157  
   158  	/**
   159  	 * Tints a Color object with a color according to red, green, and blue values.
   160  	 * <p><b>Example</b></b>
   161  	 * {@code
   162  	 * var cTint:TintColor = new TintColor(my_mc);
   163  	 * cTint.setTintOffset( 0, 0, 128 ) ;
   164  	 * }
   165  	 * @param r The red color value.
   166  	 * @param g The green color value.
   167  	 * @param b The blue color value.
   168  	 */
   169  	public function setTintOffset(r:Number, g:Number, b:Number):Void 
   170  	{
   171  		var t:Object = getTransform();
   172  		t.rb = r ; 
   173  		t.gb = g ; 
   174  		t.bb = b ;
   175  		setTransform (t);
   176  	}
   177  
   178  }
