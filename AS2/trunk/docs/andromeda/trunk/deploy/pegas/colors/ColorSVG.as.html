   358  	/**
   359  	 * Returns true if the object passed in argument is in the map of ColorSVG elements.
   360  	 * @return true if the object passed in argument is in the map of ColorSVG elements.
   361  	 */
   362  	public static function contains( o ):Boolean
   363  	{
   364  		if (o instanceof ColorSVG)
   365  		{
   366  			return ColorSVG.ELEMENTS.containsKey( o.name.toLowerCase() ) ;
   367  		}
   368  		else if (TypeUtil.typesMatch(o, String))
   369  		{
   370  			return ColorSVG.ELEMENTS.containsKey(o.toLowerCase()) ;
   371  		}
   372  		else
   373  		{
   374  			return false ;	
   375  		}
   376  	}
   377  	
   378  	/**
   379  	 * Returns the current ColorSVG reference specified with the name passed in parameter.
   380  	 * @return the current ColorSVG reference specified with the name passed in parameter.
   381  	 */
   382  	public static function get( name:String ):ColorSVG 
   383  	{
   384  		return ColorSVG.ELEMENTS.get(name.toLowerCase()) ;
