   282  	/**
   283  	 * Returns the R+G+B values currently in use by the Color object as individual red, green, and blue values.
   284  	 * <p><b>Example :</b></p>
   285  	 * {@code 
   286  	 * var my_color:Color = new Color(my_mc);
   287  	 * my_color.setRGB2(255, 0, 255);
   288  	 * var rgb:Object = my_color.getRGB2();
   289  	 * trace (rgb.r);
   290  	 * trace (rgb.g);
   291  	 * trace (rgb.b);
   292  	 * }
   293  	 */
   294  	public function getRGB2():Object 
   295  	{
   296  		var t:Object = getTransform() ;
   297  		return {r:t.rb, g:t.gb, b:t.bb} ;
   298  	}
