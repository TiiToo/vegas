    26  /**
    27  	NameSpace: config
    28     	Configure the eden library.
    29  */
    30  class buRRRn.eden.config extends Object
    31      {
    32      
    33      /**
    34         StaticProperty: compress
    35         Parameter to remove (true) or add (false) all unecessary spaces,
    36         tabs, carriages returns, lines feeds etc.
    37         to optimize (more or less) packets of datas when they are transfered.
    38         
    39         note:
    40         use "compress = false" when you want
    41         to have a better view or debug packets of datas.
    42      */
    43      static var compress:Boolean = true;
    44      
    45      /**
    46         StaticProperty: copyObjectByValue
    47         Parameter allowing to copy objects by value
    48         if true or by reference if false.
    49         
    50         exemple:
    51         foo = {a:1, b:2, c:3};
    52         bar = foo;
    53         
    54         in this case with copyObjectByValue = false
    55         bar will be a reference to the foo object
    56         but if copyObjectByValue = true
    57         bar will be an exact copy of foo object
    58      */
    59      static var copyObjectByValue:Boolean = false;
    60      
    61      /**
    62         StaticProperty: strictMode
    63         Allows to define the case-sensitivy of the parsers.
    64         If true, variable names that differ only in case are
    65         considered different.
    66      */
    67      static var strictMode:Boolean = true;
    68      
    69      /**
    70         StaticProperty: undefineable
    71         Value assigned to a variable
    72         when this one is not found or not authorized.
    73         
    74         note:
    75         Depending on your environment you can
    76         override it with a more suitable one
    77         for exemple on C# you could set it
    78         to null.
    79      */
    80      static var undefineable = undefined ;
    81      
    82      /**
    83         StaticProperty: verbose
    84         Parameter allowing to trace messages
    85         in the console if the environment permit it.
    86      */
    87      static var verbose:Boolean = true ;
    88      
    89      /**
    90         StaticProperty: security
    91         Parameter setting on (true) or off (false) the security.
    92         If true, all object path, function or constructor will
    93         be scanned at interpretation time against the
    94         authorized list (see: buRRRn.eden.config.authorized).
    95      */
    96      static var security:Boolean = true ;
    97      
    98      /**
    99      	StaticProperty: authorized
   100         	List of authorized keywords, objects path and constructors
   101         	that the parser is allowed to interpret.
   102  	       
   103         	note:
   104         	you can add full path
   105         	ex: "blah.foobar"
   106         	and/or starting path
   107         	ex: "toto.titi.*"
   108  	       
   109         	the difference is
   110         	with a full path you can only
   111         	create/use/define/assign value to this exact path
   112         	and
   113         	with a starting path you can
   114         	create/use/define/assign value to this path and its child paths
   115  	       
   116         	attention:
   117         	special values as
   118         	NaN, true, false, null, undefined
   119         	are always authorized
   120      	*/
   121      static var authorized:Array = [ 
   122  		"Array", "Boolean", "Date", "Error",
   123  		"Math.*", "Number.*", "Object", "String.*",
   124          "Infinity" 
   125  	] ;
   126      
   127      /**
   128       * StaticProperty: allowFunctionCall
   129       * Allows to execute function call is set to true,
   130       * set to false it blocks any functrion call and return undefined.
   131       *  
   132       * exemple:
   133       * "titi = \"hello world\";
   134       * toto = titi.toUpperCase();"
   135       *  
   136       * with allowFunctionCall= true
   137       * toto will equal "HELLO WORLD"
   138       * but with allowFunctionCall = false
   139       *  toto will equal undefined
   140       */
   141      static var allowFunctionCall:Boolean = false ;
   142      
   143      /* StaticProperty: autoAddScopePath
   144      */
   145      static var autoAddScopePath:Boolean = false;
   146      
   147      /* StaticProperty: arrayIndexAsBracket
   148         When set to false array index are evaluated without bracket
   149         eval( test.0 )
   150         for Flash ActionScript
   151         
   152         When set to true array index are evaluated with bracket
   153         eval( test[0] )
   154         for JavaScript, JScript, JSDB etc.
   155      */
   156      static var arrayIndexAsBracket:Boolean = false;
   157      
   158      }
