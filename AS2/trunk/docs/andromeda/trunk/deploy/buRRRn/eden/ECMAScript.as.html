    40  	function ECMAScript( source : String, scope, callback )
    41  	{
    42          
    43  		super(source, callback) ;
    44          
    45  		if( scope == null )
    46  		{
    47  			scope = _global; //default assignment scope
    48  		}
    49          
    50  		var scopepath;
    51          
    52  		if( scope != _global )
    53  		{
    54  			scopepath = ConstructorUtil.getPath(scope);
    55  		}
    56          
    57  		if( config.autoAddScopePath && (scopepath != undefined) && (scopepath != "_global") )
    58  		{
    59  			addAuthorized(scopepath + ".*");
    60  		}
    61          
    62  		this._ORC = "uFFFC" ; 
    63  		// Object Replacement Character
    64  		this.inAssignement = false ;
    65  		this.inConstructor = false ;
    66  		this.inFunction = false ;
    67  		this.scope = scope ;
    68  		this.scopepath = scopepath ;
    69  	}

   101  	/**
   102  	 * Evaluates the source of this parser.
   103  	 */
   104  	public function eval()
   105  	{
   106  		var value, tmp;
   107  		value = _ORC;
   108  		while( hasMoreChar() )
   109  		{
   110  			next();
   111  			scanSeparators();
   112  			tmp = scanValue();
   113  			if( tmp != _ORC )
   114  			{
   115  				value = tmp ;
   116  			}
   117  			/* note: poor man semicolon auto-insertion */
   118  			if( ch == " ")
   119  			{
   120  				ch = ";";
   121  			}
   122  		}
   123          
   124  		return onParsed(value);
   125  	}

   135  	/**
   136  	 * Returns {@code true} if the specified value is an octal number.
   137  	 */
   138  	public function isOctalNumber( num:String ) : Boolean
   139  	{
   140          
   141  		if( ( num.indexOf(".") > -1) || (num.indexOf("e") > -1) || (num.indexOf("E") > -1) )
   142  		{
   143  			return false;
   144  		}
   145          
   146  		var aNum:Array = num.split("") ;
   147          
   148  		var l : Number = aNum.length ;
   149  		while(--l > -1)
   150  		{
   151  			if( !isOctalDigit(aNum[l]) )
   152  			{
   153  				return false;
   154  			}
   155  		}
   156  		
   157  		return true;
   158  	}

   162  	function isDigitNumber( num : String ) : Boolean
   163  	{
   164  		var numarr:Array = num.split(""); 
   165          var size:Number = numarr.length ;
   166  		for( var i:Number = 0 ; i < size ; i++ )

   168  			if( !this.isDigit(numarr[i]) )
   169  			{
   170  				return false;
   171  			}
   172  		}
   173          
   174  		return true;
   175  	}

   177  	/* Method: isIdentifierStart
   178         
   179  	note:
   180  	identifiers
   181  	see: ECMA-262 spec 7.6 (PDF p26/188)
   182  	 */
   183  	function isIdentifierStart( /*char*/ c : String ) : Boolean
   184  	{
   185  		if( isAlpha(c) || (c == "_") || (c == "$" ) )
   186  		{
   187  			return true;
   188  		}
   189              
   190  		if( c.charCodeAt(0) < 128 )
   191  		{
   192  			return false;
   193  		}
   194              
   195  		return false;
   196  	}

   211

   213  		{
   214  			return false;
   215  		}
   216          
   217  		return false;
   218  	}

   358  	function isValidPath( path : String ) : Boolean
   359  	{
   360  		var i:Number ;
   361  		var subpath  ;
   362          var paths:Array ;
   363  		var formatter:StringFormatter = new vegas.string.StringFormatter() ;
   364  		
   365  		if( path.indexOf(".") > -1 )
   366  		{
   367  			paths = path.split(".");
   368  		}
   369          else
   370  		{
   371  			paths = [path];
   372  		}
   373          
   374  		for( i = 0;i < paths.length; i++ )
   375  		{
   376  			subpath = paths[i];
   377              
   378  			if( isReservedKeyword(subpath) || isFutureReservedKeyword(subpath) )
   379  			{
   380  				formatter.pattern = strings.notValidPath ;
   381  				log(formatter.format(path)) ;
   382  				return false;
   383  			}
   384  		}
   385          
   386  		if( config.autoAddScopePath && (scopepath != undefined) && (scope != _global) )
   387  		{
   388  			path = scopepath + "." + path;
   389  		}
   390          
   391  		if( config.security && !isAuthorized(path) )
   392  		{
   393  			formatter.pattern = strings.notAuthorizedPath ;
   394  			log(formatter.format(path)) ;
   395  			return config.undefineable;
   396  		}
   397  		//         else
   398  		//             {
   399  		//             trace( path + " is authorized (isValidPath)" );
   400  		//             }
   401  		return true;
   402  	}

   406  	function doesExistInGlobal( path : String ) : Boolean
   407  	{
   408  		var paths:Array ;
   409  		var i, subpath, scope;    
   410  		
   411  		scope = _global;
   412          
   413  		if( path.indexOf(".") > -1 )
   414  		{
   415  			paths = path.split(".");
   416  		}
   417          else
   418  		{
   419  			paths = [path];
   420  		}
   421          
   422          var size:Number = paths.length ;
   423  		for( i = 0;i < size ; i++ )
   424  		{
   425  			subpath = paths[i];
   426              
   427  			if( isDigitNumber(subpath) )
   428  			{
   429  				subpath = parseInt(subpath);
   430  			}
   431              
   432  			if( scope[ subpath ] == undefined )
   433  			{
   434  				return false;
   435  			}
   436              
   437  			scope = scope[ subpath ];
   438  		}
   439          
   440  		return true;
   441  	}

   476  	/* Method: scanComments
   477  	 */
   478  	function scanComments()
   479  	{
   480  		next();
   481          
   482  		switch( ch )
   483  		{
   484  			case "/":
   485  				while( !isLineTerminator(ch) )
   486  				{
   487  					next();
   488  				}
   489  				break;
   490              
   491  			case "*":
   492  				var ch_ = next();
   493              
   494  				while( (ch_ != "*") && (ch != "/") )
   495  				{
   496  					ch_ = ch;
   497  					next();
   498                  
   499  					if( ch == "" )
   500  					{
   501  						log(strings.unterminatedComment);
   502  						break;
   503  					}
   504  				}
   505      
   506  				next();
   507  				delete ch_;
   508  				break;
   509              
   510  			case "":
   511  			default:
   512  				log(strings.errorComment);
   513  		}
   514  	}

   516  	/* Method: scanWhiteSpace
   517         
   518  	note:
   519  	White Space
   520  	"\t" - \u0009 - TAB
   521  	"\v" - \u000B - VT
   522  	"\f" - \u000C - FF
   523  	" "  - \u0020 - SP
   524  	???  - \u00A0 - NBSP
   525  	see: ECMA-262 spec 7.2 (PDF p23/188)
   526  	 */
   527  	function scanWhiteSpace()
   528  	{
   529  		var scan = true;
   530          
   531  		while( scan )
   532  		{
   533  			switch( ch )
   534  			{
   535  				case "u0009": 
   536  				case "u000B": 
   537  				case "u000C": 
   538  				case "u0020": 
   539  				case "u00A0":
   540  					next();
   541  					break;
   542                  
   543  				case "/":
   544  					scanComments();
   545  					break;
   546                  
   547  				default:
   548  					scan = false;
   549  			}
   550  		}
   551  	}

   553  	/* Method: scanSeparators
   554  	 */
   555  	function scanSeparators()
   556  	{
   557  		var scan = true;
   558          
   559  		while( scan )
   560  		{
   561  			switch( ch )
   562  			{
   563                  /* note:
   564  				White Space
   565  				"\t" - \u0009 - TAB
   566  				"\v" - \u000B - VT
   567  				"\f" - \u000C - FF
   568  				" "  - \u0020 - SP
   569  				???  - \u00A0 - NBSP
   570  				see: ECMA-262 spec 7.2 (PDF p23/188)
   571  				 */
   572  				case "u0009": 
   573  				case "u000B": 
   574  				case "u000C": 
   575  				case "u0020": 
   576  				case "u00A0":
   577  					next();
   578  					break;
   579                  
   580                  /* note:
   581  				line terminators
   582  				"\n" - \u000A - LF
   583  				"\R" - \u000D - CR
   584  				???  - \u2028 - LS
   585  				???  - \u2029 - PS
   586  				see: ECMA-262 spec 7.3 (PDF p24/188)
   587  				 */
   588  				case "u000A": 
   589  				case "u000D": 
   590  				case "u2028": 
   591  				case "u2029":
   592  					next();
   593  					break;
   594                  
   595  				case "/":
   596  					scanComments();
   597  					break;
   598                  
   599  				default:
   600  					scan = false;
   601  			}
   602  		}
   603  	}

   605  	/* Method: scanIdentifier
   606  	 */
   607  	function scanIdentifier() : String
   608  	{
   609  		var id = "";
   610          
   611  		if( isIdentifierStart(ch) )
   612  		{
   613  			id += ch;
   614  			next();
   615              
   616  			while( isIdentifierPart(ch) )
   617  			{
   618  				id += ch;
   619  				next();
   620  			}
   621  		}
   622          else
   623  		{
   624  			log(strings.errorIdentifier);
   625  		}
   626          
   627  		return id;
   628  	}

   630  	/* Method: scanPath
   631  	 */
   632  	function scanPath() : String
   633  	{
   634  		
   635  		var path:String = "" ;
   636  		
   637  		var subpath;
   638          
   639  		if( isIdentifierStart(ch) )
   640  		{
   641  			path += ch;
   642  			next();
   643              
   644  			while( isIdentifierPart(ch) || (ch == ".") || (ch == "[") )
   645  			{
   646                  
   647  				if( ch == "[" )
   648  				{
   649  					next();
   650  					scanWhiteSpace();
   651                      
   652  					if( isDigit(ch) )
   653  					{
   654  						subpath = String(scanNumber());
   655  						scanWhiteSpace();
   656  						path += "." + subpath;
   657  					}
   658                      else if( (this.ch == "\"") || (this.ch == "'") )
   659  					{
   660  						subpath = scanString(ch);
   661  						scanWhiteSpace();
   662  						path += "." + subpath;
   663  					}
   664                      
   665  					if( ch == "]" )
   666  					{
   667  						next();
   668  						continue;
   669  					}
   670  				}
   671                  
   672                  
   673  				path += ch;
   674  				next();
   675  			}
   676  		}
   677          
   678  		if( StringUtil.startsWith( path , "_global." ) )
   679  		{
   680  			path = path.substr("_global.".length);
   681  		}
   682          
   683  		if( !inConstructor && ch == "(" )
   684  		{
   685  			inFunction = true;
   686  			return scanFunction(path);
   687  		}
   688          
   689  		return path;
   690  	}

   692  	/* Method: scanFunction
   693  	 */
   694  	function scanFunction( fcnPath : String )
   695  	{
   696  		var fcnName:String ;
   697  		var fcnObj:Function ;
   698  		var fcnObjScope;
   699          var formatter:StringFormatter = new vegas.string.StringFormatter() ;
   700          
   701  		if( fcnPath.indexOf( "." ) > -1 )
   702          {
   703              fcnName = String( fcnPath.split( "." ).pop() );
   704          }
   705          else
   706          {
   707              fcnName = fcnPath;
   708          }
   709          
   710  		scanWhiteSpace();
   711  
   712  		var args:Array = [] ;
   713          
   714  		next();
   715  		scanSeparators();
   716          
   717  		while( ch != "" )
   718              {
   719              if( ch == ")" )
   720                  {
   721                  next();
   722                  break;
   723                  }
   724              
   725              args.push( scanValue() );
   726              scanSeparators();
   727              
   728              if( ch == "," )
   729                  {
   730                  next();
   731                  scanSeparators();
   732                  }
   733              }
   734          
   735  		if( !config.allowFunctionCall )
   736              {
   737              formatter.pattern = strings.notFunctionCallAllowed ;
   738              log( formatter.format( fcnPath, args ) ) ;
   739              return config.undefineable;
   740              }
   741          
   742  		if( !isReservedKeyword( fcnPath ) && !isFutureReservedKeyword( fcnPath ) )
   743  		{
   744              fcnObj = eval( "_global." + fcnPath );
   745              
   746              if( fcnPath.indexOf( "." ) > -1 )
   747  			{
   748                  fcnObjScope = eval( "_global." + StringUtil.replace( fcnPath, "." + fcnName , "" ) );
   749              }
   750              else
   751              {
   752                  fcnObjScope = null;
   753              }
   754  		}
   755          else
   756  		{
   757              formatter.pattern = strings.notValidFunction ;
   758              log( formatter.format( fcnPath ) ) ;
   759              return config.undefineable;
   760  		}

   913      /* Method: scanString
   914      */
   915      function scanString( quote:String ):String
   916          {
   917          var str = "";
   918          
   919          if( ch == quote )
   920              {
   921              while( next() )
   922                  {
   923                  switch( ch )
   924                  {
   925                      
   926                      case quote:
   927                      	
   928                      	next() ;
   929                      	return str;
   930                      
   931                      case "\\":
   932                      	/* note:
   933                         		Escape Sequence
   934                         		\ followed by one of ' " \ b f n r t v
   935                         		or followed by x hexdigit hexdigit
   936                         		or followed by u hexdigit hexdigit hexdigit hexdigit
   937                         		see: ECMA-262 specs 7.8.4 (PDF p30 to p32/188)
   938                      	*/
   939                      
   940                      	switch( next() )
   941                      	{
   942                      		case "b": //backspace       - \u0008
   943                          		str += "b";
   944  	                        	break;
   945                          	
   946                          	case "t": //horizontal tab  - \u0009
   947  	                        	str += "\t";
   948                          		break;
   949  	                        
   950                          	case "n": //line feed       - \u000A
   951                          		str += "\n";
   952                          		break;
   953                          
   954  	                       //TODO check \v bug 
   955  	                       case "v" : //vertical tab    - \u000B
   956                          		
   957                          		//str += "\v";
   958                          		break;
   959                          
   960                          	case "f" : //form feed       - \u000C
   961                          		str += "f";
   962                          		break;
   963                          
   964                         		 case "r": //carriage return - \u000D
   965                          		str += "r";
   966                          		break;
   967                          
   968                          	case "\"": //double quote   - \u0022
   969                          		str += "\"";
   970                          		break;
   971                          
   972  		                    case "'": //single quote   - \u0027
   973                          		str += "'";
   974                          		break;
   975                          
   976                          	case "\\": //backslash      - \u005c
   977                          		str += "\\";
   978                          		break;
   979                          
   980                          	case "u": //unicode escape sequence \uFFFF
   981                          		var ucode = source.substring( pos, pos+4 );
   982                          		str  += String.fromCharCode( parseInt( ucode, 16 ) );
   983                          		pos      += 4;
   984                          		break;
   985                          
   986  	                        case "x": //hexadecimal escape sequence \xFF
   987                          		var xcode = source.substring( pos, pos+2 );
   988                          		str      += String.fromCharCode( parseInt( xcode, 16 ) );
   989                          		pos      += 2;
   990                          		break;
   991                          
   992                          	default:
   993                          		str += ch;
   994                          }
   995                      	break;
   996                      
   997                      default:
   998                      	
   999                      	if( !isLineTerminator( ch ) )
  1000                          {
  1001                          	str += ch;
  1002                          }
  1003                      	else
  1004                          {
  1005                          	log( strings.errorLineTerminator );
  1006                          }
  1007                      }
  1008                  
  1009                  }
  1010              }
  1011          
  1012          log( strings.errorString );
  1013          }
