    24  import buRRRn.ASTUce.AssertionFailedError;
    25  import buRRRn.ASTUce.IProtectable;
    26  import buRRRn.ASTUce.ITest;
    27  import buRRRn.ASTUce.ITestListener;
    28  import buRRRn.ASTUce.TestCase;
    29  import buRRRn.ASTUce.TestFailure;
    30  
    31  import vegas.core.CoreObject;
    32  import vegas.util.ArrayUtil;
    33  
    34  /**
    35   * A TestResult collects the results of executing a test case.
    36   * It is an instance of the Collecting Parameter pattern.
    37   * The test framework distinguishes between failures and errors.
    38   * A failure is anticipated and checked for with assertions.
    39   * Errors are unanticipated problems like an ArrayIndexOutOfBoundsException.
    40   * @author eKameleon
    41   */
    42  class buRRRn.ASTUce.TestResult extends CoreObject 
    43  {
    44  
    45  	/**
    46  	 * Creates a new TestResult instance.
    47  	 */
    48  	public function TestResult() 
    49  	{
    50  		super();
    51  		_failures  = [] ;
    52          _errors    = [] ;
    53          _listeners = [] ;
    54          _runTests  = 0 ;
    55          _stop      = false ;
    56  	}
    57  
    58      /**
    59       * (read-only) Returns an Array for the errors.
    60       */
    61      public function get errors():Array
    62  	{
    63  		return _errors;
    64  	}
    65  
    66      /**
    67       * (read-only) Returns the number of detected errors.
    68       */
    69  	public function get errorCount():Number
    70  	{
    71  		return _errors.length;
    72  	}
    73      
    74      /**
    75       * (read-only) Returns the number of detected failures.
    76       */
    77      public function get failureCount():Number
    78  	{
    79  		return _failures.length;
    80  	}
    81      
    82      /**
    83       * (read-only) Returns an Array for the failures.
    84       */
    85  	public function get failures():Array
    86  	{
    87  		return _failures;
    88  	}
    89  
    90      /**
    91       * (read-only) Returns the number of run tests.
    92       */
    93      public function get runCount():Number
    94  	{
    95  		return _runTests;
    96  	}
    97      
    98      /**
    99       * (read-only) Checks whether the test run should stop.
   100       */
   101      public function get shouldStop():Boolean
   102  	{
   103  		return _stop;
   104  	}
   105  
   106      /**
   107       * Adds an error to the list of errors.
   108       * The passed in exception caused the error.
   109       */
   110      public function addError( test:ITest, e:Error ):Void
   111  	{
   112  		_errors.push( new TestFailure( test, e ) );
   113          
   114  		var listeners:Array = cloneListeners() ;
   115          var len:Number = listeners.length ;
   116          for( var i:Number = 0 ; i<len ; i++ )
   117  		{
   118  			listeners[i].addError( test, e );
   119  		}
   120  		
   121  	}
   122  
   123  	/**
   124  	 * Adds a failure to the list of failures. 
   125  	 * The passed in exception caused the failure.
   126  	 */
   127  	public function addFailure( test:ITest, afe:AssertionFailedError ):Void
   128  	{
   129  		_failures.push( new TestFailure( test, afe ) ) ;
   130          var listeners:Array = cloneListeners() ;
   131          for( var i:Number = 0 ; i<listeners.length ; i++ )
   132  		{
   133  			listeners[i].addFailure( test, afe );
   134  		}
   135  	}
   136  	
   137  	/**
   138  	 * Registers a TestListener.
   139  	 */
   140  	public function addListener( listener:ITestListener ):Void
   141  	{
   142  		_listeners.push( listener );
   143  	}
   144  
   145      /**
   146       * Returns a copy of the listeners.
   147       */
   148      public function cloneListeners():Array
   149  	{
   150          return [].concat(_listeners) ;
   151  	}
   152  
   153      /**
   154       * Informs the result that a test was completed.
   155       */
   156  	public function endTest( test:ITest ):Void
   157  	{
   158  		var listeners:Array = cloneListeners();
   159  		var len:Number = listeners.length ;
   160          for( var i:Number = 0 ; i<len ; i++ )
   161  		{
   162  			listeners[i].endTest( test );
   163  		}
   164  	}
   165  
   166  	/**
   167  	 * Unregisters a TestListener.
   168  	 */
   169      public function removeListener( listener:ITestListener ):Void
   170  	{
   171  		var index:Number ;
   172  		index = ArrayUtil.indexOf( _listeners, listener );
   173  		if( index > -1 )
   174          {
   175  			_listeners.splice( index, 1 );
   176  		}
   177  	}
   178  	
   179  	/**
   180  	 * Runs a TestCase.
   181  	 */
   182  	public function run( test:TestCase ):Void
   183  	{
   184  		
   185  		startTest( test ) ;
   186  
   187          var p:IProtectable = new IProtectable();
   188          p.protect = function()
   189  		{
   190  			return test.runBare();
   191  		} ;
   192          
   193          runProtected( test, p );
   194  
   195          endTest( test );
   196  	}
   197  
   198  	/**
   199  	 * Runs a TestCase.
   200  	 */
   201      public function runProtected( test:ITest, p:IProtectable ):Void
   202  	{
   203  		try
   204  		{
   205  			p.protect() ;
   206  		}
   207          catch( e )
   208          {
   209  			if( e instanceof AssertionFailedError )
   210  			{
   211  				addFailure( test, e );
   212  			}
   213              else if( e instanceof Error )
   214  			{
   215  				addError( test, e );
   216  			}
   217  		}    
   218  	}
   219  
   220      /**
   221       * Informs the result that a test will be started.
   222       */
   223  	public function startTest( test:ITest ):Void
   224  	{
   225  		
   226  		var count:Number = test.countTestCases();
   227  		
   228  		_runTests += count;
   229          
   230  		var listeners:Array = cloneListeners() ;
   231  		
   232  		for( var i:Number = 0 ; i<listeners.length ; i++ )
   233  		{
   234  			listeners[i].startTest( test );
   235  		}
   236  	}
   237      
   238  	/**
   239  	 * Marks that the test run should stop.
   240  	 */
   241  	public function stop():Void
   242  	{
   243  		_stop = true ;	
   244  	}
   245  	
   246      /**
   247       * Returns whether the entire test was successful or not.
   248       * @return whether the entire test was successful or not.
   249       */
   250    	public function wasSuccessful():Boolean
   251  	{
   252  		return( (failureCount == 0) && (errorCount == 0) );
   253  	}
   254  
   255  	private var _errors:Array;
   256  
   257      private var _failures:Array;
   258      
   259      private var _listeners:Array;
   260      
   261      private var _runTests:Number;
   262      
   263      private var _stop:Boolean;
   264  
   265  }
