    25  import buRRRn.ASTUce.Config;

    35  /**
    36   * A test case defines the fixture to run multiple tests. To define a test case :
    37   * <ul>
    38   * <li>implement a subclass of TestCase.</li>
    39   * <li>define instance variables that store the state of the fixture.</li>
    40   * <li>initialize the fixture state by overriding setUp</li>
    41   * <li>clean-up after a test by overriding tearDown.</li>
    42   * </ul>
    43   * <p>Each test runs in its own fixture so there can be no side effects among test runs.</p>
    44   * @author eKameleon
    45   */
    46  class buRRRn.ASTUce.TestCase extends Assertion implements ITest 
    47  {
    48  	
    49  	/**
    50  	 * Constructs a test case with the given name or null if name is not provided.
    51  	 * @param name the name of the test case.
    52  	 */
    53  	function TestCase( name:String ) 
    54  	{
    55  		_name = name ;
    56  	}
    57  
    58      /**
    59       * (read-only) Returns the name of a TestCase.
    60       */
    61      public function get name():String
    62  	{
    63  		return _name;
    64  	}
    65      
    66  	/**
    67  	 * (read-only) Sets the name of a TestCase.
    68  	 */
    69      public function set name( value:String ):Void
    70  	{
    71  		_name = value;
    72  	}
    73  
    74  	/**
    75  	 * Counts the number of test cases executed by run (TestResult result).
    76  	 */
    77  	public function countTestCases():Number 
    78  	{
    79  		return 1 ;
    80  	}
    81  
    82      /** 
    83       * Creates a default TestResult object.
    84       */
    85  	public function createResult():TestResult
    86  	{
    87  		return new TestResult();
    88  	}
    89  
    90  	/**
    91  	 * Runs the test case and collects the results in TestResult.
    92  	 */
    93  	public function run( result:TestResult ):Void 
    94  	{
    95  		if( result == null )
    96  		{
    97  			result = createResult();
    98  		}
    99          result.run( this );
   100  	}
   101  
   102      /**
   103       * Runs the bare test sequence.
   104       */
   105  	public function runBare():Void
   106  	{
   107          
   108          setUp();
   109          try
   110  		{
   111              runTest() ;
   112  		}
   113  		/*catch( e )
   114  		{
   115              //trace( e ); // debug only.
   116  		}
   117  		*/
   118  		finally
   119  		{
   120  			tearDown() ;
   121  		}
   122          
   123  	}
   124  
   125      /**
   126       * A convenience method to run this test, collecting the results with a default TestResult object.
   127       * @see TestResult
   128       */
   129  	public function runEmpty():TestResult
   130  	{
   131  		var result:TestResult = createResult();
   132  		run( result );
   133  		return result;
   134  	}
   135  
   136      /**
   137       * Override to run the test and assert its state.
   138       */
   139      public function runTest():Void
   140  	{
   141  
   142  		var runMethod:Function ;
   143          
   144  		assertNotNull( _name, Strings.methodNameNull ) ;
   145          
   146  		assertNotUndefined( _name, Strings.methodNameUndef ) ;
   147          
   148  		try
   149  		{
   150  			
   151              if( ! ObjectUtil.hasProperty( this, _name ) )
   152  			{
   153  				throw new Error();
   154  			}
   155              
   156  			runMethod = this[_name];
   157  		}
   158          catch( e )
   159  		{
   160  			fail( (new StringFormatter( Strings.methodNotFound )).format(_name) ) ;
   161  		}
   162          
   163          if( StringUtil.startsWith(_name, "_" ) && ( Config.testPrivateMethods != true) ) 
   164  		{
   165  			fail( (new StringFormatter( Strings.methodshouldBePublic)).format( _name ) ) ; 
   166  		}
   167          
   168          try
   169  		{
   170  			runMethod.call( this ) ;
   171  		}
   172  		catch( e:Error )
   173  		{
   174  			throw e ;
   175  		}
   176  	}
   177  
   178      /**
   179       * Sets up the fixture, for example, open a network connection.
   180       * This method is called before a test is executed.
   181       */
   182      public function setUp():Void
   183  	{
   184      	//    
   185  	}
   186  
   187      /**
   188       * Tears down the fixture, for example close a network connection. 
   189       * This method is called after a test is executed.
   190  	 */
   191  	public function tearDown():Void
   192  	{
   193  		//    
   194  	}
   195  	
   196  	/**
   197  	 * Returns the string representation of this object.
   198  	 * @return the string representation of this object.
   199  	 */
   200      public function toString():String
   201  	{
   202          var className = ConstructorUtil.getName( this["__constructor__"] ) ;
   203          return className + "(" + _name + ")" ;
   204  	}
   205  
   206  	private var _name:String ;
   207  
   208  }
