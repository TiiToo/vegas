    24  /**
    25   * Configure the ASTUce framework.
    26   * @author eKameleon
    27   */
    28  class buRRRn.ASTUce.Config 
    29  {
    30  
    31      /**
    32       * Boolean configuring the getInfo method behaviour, true for more verbose and false for less verbose.
    33       */
    34      public static var verbose:Boolean = true;
    35  
    36      /**
    37       * Boolean option to display all the constructors being tested.
    38       */
    39      public static var showConstructorList:Boolean = true;
    40      
    41      /**
    42       * Boolean option to display the source of objects being compared.
    43       * Note : It help you to debug to see 
    44       * {@code
    45       * ## AssertionFailedError : expected:<{a:1,b:2,c:3}> but was:<{a:1,b:2,c:4}> ##
    46       * }
    47       * instead of
    48       * {@code
    49       * ## AssertionFailedError : expected:<[object Object]> but was:<[object Object]> ##
    50       * }
    51       */
    52      public static var showObjectSource:Boolean = true;
    53      
    54      /**
    55       * Boolean option to invert the order of the arguments: expected, actual in buRRRn.ASTUce.Assertion.
    56       * <p>true  - the argument order is: actual, expected. (inverted)</p>
    57       * <p>false - the argument order is: expected, actual. (default)</p>
    58       */
    59      public static var invertExpectedActual:Boolean = false;
    60      
    61      /**
    62       * Boolean configuring the behaviour of ASTUce regarding private methods.
    63       * <p>true  - test private methods</p>
    64       * <p>false - don't test private methods</p>
    65       * <p>Note : By default in ECMAScript all methods are public, but by convention we use an underscore before the name
    66       * of a method to indicate its private nature.</p>
    67       * <p>ASTUce will not test methods starting with an underscore except if you force testPrivateMethods = true, 
    68       * then methods as _testSomething will be tested by the framework.</p>
    69       */
    70      public static var testPrivateMethods:Boolean = false;
    71      
    72      /**
    73       * Boolean option allowing to iterate or not trough inherited tests.
    74       * <p>true iterate inherited tests</p>
    75       * <p>false does NOT iterate inherited tests</p>
    76       * <p>Note : If you set this option to false the following test SuiteTest( testInheritedTests ) will fail.</p>
    77       */
    78      public static var testInheritedTests:Boolean = true;
    79      
    80      /**
    81       * Boolean option allowing the ASTUce framework to test itself.
    82       * <p>true  - add to tests *buRRRn.Tests.AllTests.suite()*.</p>
    83       * <p>false - add nothing</p>
    84  	 */
    85  	public static var testMyself:Boolean = false;
    86  
    87  }
