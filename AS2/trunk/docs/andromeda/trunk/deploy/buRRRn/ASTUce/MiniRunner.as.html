    24  import buRRRn.ASTUce.ResultPrinter;
    25  import buRRRn.ASTUce.TestResult;
    26  import buRRRn.ASTUce.TestSuite;
    27  
    28  import vegas.core.CoreObject;
    29  import vegas.core.IRunnable;
    30  
    31  /**
    32   * A trace based tool to run tests.
    33   * @author eKameleon
    34   */
    35  class buRRRn.ASTUce.MiniRunner extends CoreObject implements IRunnable 
    36  {
    37  	
    38  	/**
    39  	 * Creates a new MiniRunner instance.
    40  	 */
    41  	public function MiniRunner( suiteName:String ) 
    42  	{
    43  		super();
    44  		
    45  		if( suiteName == null )
    46          {
    47          	suiteName = "MAIN" ;
    48  		}
    49          
    50          suite   = new TestSuite( suiteName ) ;
    51          result  = new TestResult();
    52          printer = new ResultPrinter();
    53          
    54  	}
    55  
    56  	/**
    57  	 * The TestSuite reference of this test tracer.
    58  	 */
    59      public var suite:TestSuite;
    60      
    61      /**
    62       * The TestResult reference of this test tracer.
    63       */
    64      public var result:TestResult;
    65      
    66      /**
    67       * The ResultPrinter reference of this test tracer.
    68       */
    69      public var printer:ResultPrinter;
    70  
    71  	/**
    72  	 * Run the tracer.
    73  	 */
    74  	public function run():Void 
    75  	{
    76  		
    77  		setUp();
    78          
    79          var time:Number = (new Date()).valueOf() ;
    80          
    81          suite.run( result );
    82          
    83          var runtime:Number = (new Date()).valueOf() - time ;
    84          
    85          printer.print( result, runtime );
    86          
    87  	}
    88  
    89  	/**
    90  	 * Overrides this method to setup the Runner.
    91  	 */
    92      public function setUp():Void
    93      {
    94          
    95      }
    96  
    97  }
