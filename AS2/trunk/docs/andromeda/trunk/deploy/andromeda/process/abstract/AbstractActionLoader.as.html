    53  	public function notifyStarted():Void

    59  	/**

    61  	 */
    62  	public function registerLoader( loader:ILoader )
    63  	{
    64  		EventTarget(loader).addEventListener( LoaderEvent.COMPLETE , new Delegate(this, _onLoadComplete) ) ;
    65   		EventTarget(loader).addEventListener( LoaderEvent.IO_ERROR , new Delegate(this, _onLoadError) ) ;
    66   		EventTarget(loader).addEventListener( LoaderEvent.PROGRESS , new Delegate(this, _onLoadProgress ) ) ;
    67   		EventTarget(loader).addEventListener( LoaderEvent.TIMEOUT  , new Delegate(this, _onLoadTimeout ) ) ;
    68   		return loader ;
    69  	}

    71  	private function _onLoadError(ev:LoaderEvent):Void
    72  	{
    73  	    getLogger().error(  this + " " + ev.getType() + " : " + ev.error ) ;
    74  	   	notifyFinished() ;
    75  	}
    76  	
    77  	private function _onLoadComplete(ev:LoaderEvent):Void
    78  	{
    79  	    getLogger().debug( this + " load complete." ) ;
    80  	}	
    81  	
    82  	private function _onLoadProgress (ev:LoaderEvent):Void
    83  	{
    84  	    getLogger().info(  this + " load progress : " + ev.getPercent() + "%") ;
    85  	   	ApplicationCommand.changeLoader( this + " " + ev.getPercent() + "%", ev.getPercent() ) ;
    86  	}
    87  	
    88  	private function _onLoadTimeout (ev:LoaderEvent):Void
    89  	{
    90  		getLogger().debug(  this + " load timeout." ) ;
    91  	   	notifyFinished() ;
    92  	}
