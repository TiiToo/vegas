    15  /**
    16   * This process load an external swf library of sounds use in this application.
    17   * Use SoundLibrary model with model.getSound( SoundList.ID_SOUND ).start() to launch a sound in the library when the library is loading.
    18   * @author eKameleon
    19   * @see andromeda.media.SoundLibrary
    20   */
    21  class andromeda.process.application.RunSound extends AbstractActionLoader
    22  {
    23  
    24  	/**
    25  	 * Creates a new RunSound instance.
    26  	 * @param model the SoundLibrary model of this process.
    27  	 * @param applicationID the name of the application display.
    28  	 */	
    29  	public function RunSound( model:SoundLibrary, applicationID:String ) 
    30  	{
    31  		this.applicationID = applicationID ;
    32  		this.model = model ;
    33  	}
    34  
    35  	/**
    36  	 * The display application reference.
    37  	 */
    38  	public var applicationID:String ;
    39  
    40  	/**
    41  	 * The SoundLibrary reference of this process.
    42  	 */
    43  	public var model:SoundLibrary ;
    44  
    45  	/**
    46  	 * initialize the sound library of the application.
    47  	 */
    48  	public function initialize():Void
    49  	{
    50  		
    51  		var config:Config = getConfig() ;
    52  		var sounds:Array = config.sound.sounds ;
    53  		
    54  		getLogger().debug(this + " initialize : " + sounds ) ;
    55  		
    56  		if (model == null)
    57  		{
    58  			getLogger().warn(this + " the initialize of this process failed with an empty SoundLibrary model.") ;	
    59  		}
    60  		else if (sounds instanceof Array && sounds.length > 0)
    61  		{
    62  			model.initialize( DisplayLoaderCollector.get( ApplicationList.SOUND ).getContent() ) ;
    63  			model.addSounds(Config.getInstance().sound.sounds) ;
    64  		}
    65  		
    66  		notifyFinished() ;
    67  
    68  	}
    69  
    70  	/**
    71  	 * Run the process.
    72  	 */
    73  	public function run() 
    74  	{
    75  		
    76  		notifyStarted() ;
    77  		
    78  		getLogger().debug( this + " run." ) ;
    79  
    80  		if ( !DisplayLoaderCollector.contains( ApplicationList.SOUND ) )
    81  		{
    82    			var view:MovieClip = DisplayObjectCollector.get( applicationID ).view ;
    83    			try
    84    			{
    85  				DisplayLoaderCollector.insert( ApplicationList.SOUND, new DisplayLoader(view, ApplicationDepthList.SOUND_DEPTH , false) ) ;
    86    			}
    87    			catch(e:Error)
    88    			{
    89    				getLogger().warn( this + ", the application can't creates and register the sound display : " + e.toString ) ;
    90    				notifyFinished() ;
    91    				return ;	
    92    			}
    93  		}		
    94  
    95   		var config:Config = getConfig() ;
    96   		var path:String = config.libraryPath || "" ;
    97   		var sounds:Array = config.sound.sounds ;
    98   		var url:String = config.sound.url || "" ;
    99   		
   100   		getLogger().info(this + " sound url : " + url ) ;
   101   		getLogger().info(this + " sounds : " + sounds ) ;
   102   		
   103   		if (sounds instanceof Array && sounds.length > 0)
   104   		{
   105   			
   106  	 	 	if (url.length > 0)
   107   	 		{
   108  	 			
   109  	 			var request:URLRequest = new URLRequest(path + url);
   110   				var loader:DisplayLoader = DisplayLoaderCollector.get( ApplicationList.SOUND ) ;
   111   				
   112   				loader.addEventListener( LoaderEvent.INIT , new Delegate( this, initialize ) , false, 1000, true ) ;
   113  				registerLoader( loader ) ;
   114   				loader.load(request) ;
   115   	 		}
   116   		}
   117   		else
   118   		{
   119   			notifyFinished() ;
   120   		}
   121   	 	
   122  	}
   123  	
   124  }
