    24  import vegas.core.CoreObject;
    25  import vegas.core.IRunnable;
    26  import vegas.errors.IllegalArgumentError;
    27  import vegas.events.BasicEvent;
    28  import vegas.events.EventDispatcher;
    29  import vegas.events.FrontController;
    30  import vegas.util.serialize.Serializer;
    31  import vegas.util.TypeUtil;

    33  /**
    34   * A command is a easy entry with name and value property to launch a global command in the Commands static tool class.
    35   * @author eKameleon
    36   */
    37  dynamic class andromeda.core.Command extends CoreObject implements IRunnable
    38  {
    39  
    40  	/**
    41  	 * Creates a new Command instance.
    42  	 * <p><b>Example :</b></p>
    43  	 * {@code
    44  	 * var c1:Command = new Command() ;
    45  	 * var c2:Command = new Command([name:String [, value:*[, channel:String]]]) ;
    46  	 * var c3:Command = new Command([initObject:Object]) ; 
    47  	 * }
    48  	 * @throws IllegalArgumentError if the optional passed-in arguments failed the initialization of the new {@code Command} instance. 
    49  	 */ 
    50  	public function Command( /*args*/ ) 
    51  	{
    52  		
    53  		if (arguments.length == 1)
    54  		{
    55  			
    56  			var arg /*Object*/ = arguments[0] ;
    57  			
    58  			if (TypeUtil.typesMatch(arg, String) )
    59  			{
    60  				this.name = arg ;
    61  			}
    62  			else if (TypeUtil.typesMatch(arg, Object))
    63  			{
    64  				
    65  				if (arg.hasOwnProperty("name") && TypeUtil.typesMatch(arg["name"], String) )
    66  				{
    67  					this.name = arg.name ;	
    68  				}
    69  				else
    70  				{
    71  					throw new IllegalArgumentError( toString() + CONSTRUCTOR_ERROR ) ;	
    72  				}
    73  				
    74  				this.value   = arg.value   || null ;
    75  				this.channel = arg.channel || null ;
    76  				
    77  			}
    78  			else
    79  			{
    80  				throw IllegalArgumentError(this + CONSTRUCTOR_ERROR) ;	
    81  			}
    82  		}
    83  		else if (arguments.length > 2)
    84  		{
    85  			
    86  			if (TypeUtil.typesMatch(arguments[0], String) )
    87  			{
    88  					this.name = arguments[0] ;	
    89  			}
    90  			else
    91  			{
    92  				throw IllegalArgumentError(this + CONSTRUCTOR_ERROR) ;	
    93  			}
    94  			this.value   = arguments[1] || null ;
    95  			this.channel = arguments[2] || null ;
    96  			
    97  		}
    98  		else
    99  		{
   100  			// throw IllegalArgumentError(this + CONSTRUCTOR_ERROR) ;	
   101  		}
   102  	}
   103  
   104  	/**
   105  	 * The internal string message used in the constructor if the constructor notify an IllegalArgumentError.
   106  	 */
   107  	public static var CONSTRUCTOR_ERROR:String = ", you can't create this instance without 'name' definition." ;
   108  	
   109  	/**
   110  	 * The channel of this command.
   111  	 */
   112  	public var channel:String ;
   113  
   114  	/**
   115  	 * The command's type name.
   116  	 */
   117  	public var name:String ;
   118  	
   119  	/**
   120  	 * The value of this command.
   121  	 */
   122  	public var value ;
   123  
   124  	/**
   125  	 * Register the class to AMF communication.
   126  	 * @param id The optional default id to register this class in the 'Object.registerClass' method. If this argument is 'null' the default id is 'Command'. 
   127  	 */
   128  	public static function register( id:String ):Boolean
   129  	{
   130  		return Object.registerClass( id || "Command" , Command ) ;
   131  	}
   132  
   133  	/**
   134  	 * Run the process of this command.
   135  	 */
   136  	public function run():Void
   137  	{
   138  		if ( FrontController.getInstance( channel ).contains( name ) )
   139  		{
   140  			getLogger().info(this + " run : " + name + " : " + value ) ;
   141  			EventDispatcher.getInstance( channel ).dispatchEvent( new BasicEvent( name , this, value ) ) ;
   142  		}
   143  		else
   144  		{
   145  			getLogger().warn(this + " run failed with an unknow command '" + name + "' in the Front Controller of this application.") ;
   146  		}
   147  	}
   148  
   149  	/**
   150  	 * Returns a simple Object representation of the Commands's instance. 
   151  	 * @return a simple Object representation of the Commands's instance.
   152  	 */
   153  	public function toObject():Object
   154  	{
   155  		return { name:name , value:value , channel:channel } ;	
   156  	}
   157  
   158  	/**
   159  	 * Returns the string EDEN representation of this object.
   160  	 * @return the string EDEN representation of this object.
   161  	 */
   162  	public function toSource():String
   163  	{
   164  		var params:Array = [ Serializer.toSource(name) , Serializer.toSource(value) , Serializer.toSource(channel) ] ;
   165  		return Serializer.getSourceOf( this , params ) ;
   166  	}
   167  	
   168  	/**
   169  	 * Returns the string representation of this object.
   170  	 * @return the string representation of this object.
   171  	 */
   172  	public function toString():String
   173  	{
   174  		var txt:String = "[Command" ;
   175  		if (name != null) 
   176  		{
   177  			txt += " name:" + name ;
   178  		}
   179  		if (channel != null) 
   180  		{
   181  			txt += " channel:" + channel ;
   182  		}
   183  		txt += "]";
   184  		return txt ; 
   185  	}
   186  
   187  
   188  }
