    26  /**
    27   * Describes an object instance, which has property values, constructor argument values, and further information supplied by concrete implementations.
    28   * @author eKameleon
    29   */
    30  interface andromeda.ioc.core.IObjectDefinition 
    31  {
    32  
    33  	/**
    34  	 * Returns the constructor arguments values of this object in a Array list.
    35  	 * @return the constructor arguments values of this object in a Array list.
    36  	 */	
    37  	function getConstructorArguments():Array ;
    38  
    39  	/**
    40  	 * Returns the name of the method invoqued when the object is destroyed.
    41  	 * @return the name of the method invoqued when the object is destroyed.
    42  	 */	
    43  	function getDestroyMethodName():String ; 
    44  
    45  	/**
    46  	 * Returns the name of the method call when the object is instanciate.
    47  	 * @return the name of the method call when the object is instanciate.
    48  	 */	
    49  	function getInitMethodName():String ; 
    50  
    51  	/**
    52  	 * Returns the Map of all properties of this Definition.
    53  	 * @return the Map of all properties of this Definition.
    54  	 */	
    55  	function getProperties():Map ; 
    56  
    57  	/**
    58  	 * Returns the type of the object (the class name).
    59  	 * @return the type of the object (the class name).
    60  	 */	
    61  	function getType():String  ;
    62  
    63  	/**
    64  	 * Returns {@code true} if the object in a Sigleton else the object is a prototype.

    66  	 */		
    67  	function isSingleton():Boolean ; 
    68

    71  	 */

    74  	/**
    75  	 * Sets the name of the method invoqued when the object is destroyed.
    76  	 */

    78  
    79  	/**
    80  	 * Init the name of the method.
    81  	 */

    83  
    84  	/**
    85  	 * Sets the Map of all properties of this Definition.
    86  	 */	
    87  	function setProperties( value:Map ):Void ; 
    88  
    89  	/**
    90  	 * Sets the type of the object (the class name).
    91  	 */	
    92  	function setType( value:String ):Void ;

    94  }
