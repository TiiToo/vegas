     8   * This builder use a eden object to creates all Objects with the IObjectDefinitionContainer.
     9   * <p><b>Example :</b></p>
    10   * 1 - The eden application file : <b>"application.eden"</b>
    11   * {@code 
    12   *   {
    13   *        objects : 
    14   *        [
    15   *            {   
    16   *                id            : "address"  ,
    17   *                type          : "test.Address" ,
    18   *                properties    : 
    19   *                [ 
    20   *                    { name : "city"   , value : "Marseille" } ,
    21   *                    { name : "street" , value : "xx xxx xxxxxxxxxxx" } ,
    22   *                    { name : "zip"    , value : 13004 } 
    23   *                ]
    24   *            }
    25   *            ,
    26   *            {   
    27   *                id         : "job_dev"  ,
    28   *                type       : "test.Job" ,
    29   *                properties : [ { name:"name" , value:"AS Developper" } ]
    30   *            }
    31   *            ,
    32   *            {   
    33   *                id            : "user" , 
    34   *                type          : "test.User" , 
    35   *                arguments     : 
    36   *                [ 
    37   *                    { value :"eKameleon" } , 
    38   *                    { value :"ALCARAZ"   } , 
    39   *                    { ref   :"address"   } 
    40   *                ] ,
    41   *                singleton     : true ,
    42   *                assemblyName  : "" , // not implemented yet
    43   *                destroy       : "destroy" ,  // not fully implemented yet
    44   *                init          : "initialize" ,
    45   *                properties : 
    46   *                [
    47   *                    { name:"age"       , value : 30          } ,
    48   *                    { name:"firstName" , value : "Marc"      } , 
    49   *                    { name:"job"       , ref   : "job_dev"      } , 
    50   *                    { name:"setMail"   , arguments  :  [ { value :"vegas@ekameleon.net" } ] } , // method
    51   *                    { name:"url"       , value : "http://www.ekameleon.net/blog" }
    52   *                ]
    53   *             }
    54   *         ]
    55   *    }
    56   * }
    57   * 2 - The ActionScript test code :
    58   * {@code
    59   *  import test.User ;
    60   * 
    61   *  import andromeda.ioc.factory.EdenObjectFactory ;

    63   *  var loader:LoadVars = new LoadVars() ;

    65   *  {
    66   *     app = buRRRn.eden.Application.deserialize(source) ;

    68   *     var factory:EdenObjectFactory = new EdenObjectFactory() ;
    69   *     factory.create( app.objects ) ;
    70   *     
    71   *     trace("--- User") ;
    72   *     
    73   *     var user:User = factory.getObject("user") ;
    74   *     
    75   *     trace("User : " + user) ;
    76   *     trace("User pseudo : " + user.pseudo ) ; // ekameleon
    77   *     trace("User firstName : " + user.firstName ) ; // Marc
    78   *     trace("User name : " + user.name ) ; // ALCARAZ
    79   *     trace("User age : " + user.age ) ; // 30
    80   *     trace("User mail : " + user.mail ) ; // vegas@ekameleon.net
    81   *     trace("User url : " + user.url  ) ; // http://www.ekameleon.net/blog
    82   *     trace("User job : " + user.job ) ; // [Job AS Developper]
    83   *     trace("User address : " + user.address) ; // [Address]

    85   *     trace("User address street : " + user.address.street) ; // xx xxx xxxxxxxxxxx

   170  	/**
   171  	 * Creates a new EdenObjectFactory instance.
   172  	 */
   173  	public function EdenObjectFactory() 
   174  	{
   175  		_assemblies = new HashMap() ;
   176  	}

   178  	/**
   179  	 * Defines the label of the arguments in a method or a constructor object.
   180  	 */
   181  	static public var ARGUMENTS:String = "arguments" ;  
   182  
   183  	/**
   184  	 * Defines the label of the assembly name property of the object.
   185  	 */
   186  	static public var ASSEMBLY_NAME:String = "assemblyName" ;
   187  
   188  	/**
   189  	 * Defines the label of the name in a property object.
   190  	 */
   191  	static public var NAME:String = "name" ;  
   192  
   193  	/**
   194  	 * The name of the external object property to register the destroy method name.
   195  	 */
   196  	static public var OBJECT_DESTROY_METHOD_NAME:String = "destroy" ;  
   197  
   198  	/**
   199  	 * The name of the external object property to define the id of the object.
   200  	 */
   201  	static public var OBJECT_ID:String = "id" ;  
   202  
   203  	/**
   204  	 * The name of the external object property to register the init method name.
   205  	 */
   206  	static public var OBJECT_INIT_METHOD_NAME:String = "init" ;  
   207  	
   208  	/**
   209  	 * The name of the external object property to register the properties.
   210  	 */
   211  	static public var OBJECT_PROPERTIES:String = "properties" ;  
   212  
   213  	/**
   214  	 * The name of the external object property to define the singleton flag of the object.
   215  	 */
   216  	static public var OBJECT_SINGLETON:String = "singleton" ;  
   217  
   218  	/**
   219  	 * Defines the label of the type of the object.
   220  	 */
   221  	static public var TYPE:String = "type" ;  
   222  
   223  	/**
   224  	 * Defines the label of the reference in a property object.
   225  	 */
   226  	static public var REFERENCE:String = "ref" ;

   228  	/**
   229  	 * Defines the label of the value in a property object.
   230  	 */
   231  	static public var VALUE:String = "value" ;

   233  	/**
   234  	 * Create the objects and fill the IObjectDefinitionContainer.
   235  	 * <p><b>Parameters</b></p>
   236  	 * {@code edenObject} An object who contains all the "objects" settings.
   237  	 */
   238  	public function create( /* ...args */ )
   239  	{
   240  		var arg:Array = arguments[0] ;
   241  		if ( arg instanceof Array && arg.length > 0)
   242  		{
   243  			while ( arg.length > 0)
   244  			{
   245  				_createNewObjectDefinition( arg.shift() ) ;
   246  			}
   247  		}
   248  	}
   249  
   250  	private var _assemblies:HashMap ;
   251  
   252  	/**
   253  	 * Returns and creates a new IObjectDefinition instance.
   254  	 * @return and creates a new IObjectDefinition instance.
   255  	 */
   256  	private function _createNewObjectDefinition( o:Object ):Void
   257  	{
   258  		if ( o != null )
   259  		{
   260  			
   261  			var args:Array          =  o[ ARGUMENTS ] ;
   262  			var assemblyName:String =  o[ ASSEMBLY_NAME ] ;
   263  			var destroy:String      =  o[ OBJECT_DESTROY_METHOD_NAME ] ;
   264  			var id:String           =  o[ OBJECT_ID ] ;
   265  			var init:String         =  o[ OBJECT_INIT_METHOD_NAME ] ;
   266  			var properties:Array    =  o[ OBJECT_PROPERTIES ] ;
   267  			var singleton:Boolean   =  o[ OBJECT_SINGLETON ] ;
   268  			var type:String         =  o[ TYPE ] ;
   269  
   270  			var definition:ObjectDefinition = new ObjectDefinition( type , singleton ) ;
   271  			definition.setConstructorArguments( args ) ;
   272  			definition.setDestroyMethodName( destroy ) ;
   273  			definition.setInitMethodName( init ) ;
   274  			definition.setProperties( _createNewProperties( properties ) ) ;
   275  			
   276  			addObjectDefinition( id , definition ) ;
   277  			
   278  		}
   279  	}
   280  
   281  	/**
   282  	 * Returns and creates the Map of all properties.
   283  	 * @return and creates the Map of all properties.
   284  	 */
   285  	private function _createNewProperties( a:Array ):HashMap
   286  	{
   287  		var len:Number = a.length ;
   288  		if ( len > 0 )
   289  		{
   290  			var args:Array  ;
   291  			var prop:Object ;
   292  			var name:String ;
   293  			var properties:HashMap = new HashMap() ;
   294  			var ref:String  ;
   295  			var value ;
   296  			for (var i:Number = 0 ; i<len ; i++)
   297  			{
   298  				prop  = a[i] ;
   299  				args  = prop[ ARGUMENTS ] ;
   300  				ref   = prop[ REFERENCE ]   ;
   301  				name  = prop[ NAME ] ;
   302  				value = prop[ VALUE ] ;
   303  				
   304  				if (args instanceof Array && args.length > 0) 
   305  				{
   306  					properties.put( name , _createArguments( args ) ) ; // methods
   307  				}
   308  				else if (ref != null) 
   309  				{
   310  					properties.put( name , ref ) ; // ref property	
   311  				}
   312  				else 
   313  				{
   314  					properties.put( name, value ) ; // property	
   315  				}
   316  			}
   317  			return (properties.size() > 0) ? properties : null ;
   318  		}
   319  		else
   320  		{
   321  			return null ;	
   322  		}	
   323  	}
