    24  import andromeda.ioc.core.IObjectDefinition;
    25  
    26  import vegas.core.CoreObject;
    27  import vegas.data.Map;
    28  import vegas.errors.IllegalArgumentError;
    29  
    30  /**
    31   * The concrete implementation of the IObjectDefinition interface.
    32   * @author eKameleon
    33   */
    34  class andromeda.ioc.core.ObjectDefinition extends CoreObject implements IObjectDefinition 
    35  {
    36  
    37  	/**
    38  	 * Creates a new ObjectDefinition instance.
    39  	 * @param type the type of the {@code ObjectDefinition} object.
    40  	 * @param singleton the boolean flag to indicate if the object is a sigleton or not.
    41  	 */	
    42  	public function ObjectDefinition( type:String , singleton:Boolean ) 
    43  	{
    44  		if (type == null || type.length == 0)
    45  		{
    46  			throw new IllegalArgumentError( this + " constructor failed, the string 'type' passed in argument not must be empty or 'null' or 'undefined'.") ;	
    47  		}
    48  		_type = type ;
    49  		_singleton = singleton ;
    50  	}
    51  
    52  	/**
    53  	 * Returns the constructor arguments values of this object in a Array list.
    54  	 * @return the constructor arguments values of this object in a Array list.
    55  	 */	
    56  	public function getConstructorArguments():Array 
    57  	{
    58  		return _constructorArguments ;
    59  	}
    60  
    61  	/**
    62  	 * Returns the name of the method invoqued when the object is destroyed.
    63  	 * @return the name of the method invoqued when the object is destroyed.
    64  	 */	
    65  	public function getDestroyMethodName():String 
    66  	{
    67  		return _destroyMethodName;
    68  	}
    69  
    70  	/**
    71  	 * Returns the name of the method call when the object is instanciate.
    72  	 * @return the name of the method call when the object is instanciate.
    73  	 */	
    74  	public function getInitMethodName():String 
    75  	{
    76  		return _initMethodName;
    77  	}
    78  
    79  	/**
    80  	 * Returns the Map of all properties of this Definition.
    81  	 * @return the Map of all properties of this Definition.
    82  	 */	
    83  	public function getProperties():Map 
    84  	{
    85  		return _properties ;
    86  	}
    87  
    88  	/**
    89  	 * Returns the type of the object (the class name).
    90  	 * @return the type of the object (the class name).
    91  	 */	
    92  	public function getType():String 
    93  	{
    94  		return _type ;
    95  	}
    96  
    97  	/**
    98  	 * Returns {@code true} if the object in a Sigleton else the object is a prototype.
    99  	 * @return {@code true} if the object in a Sigleton else the object is a prototype.
   100  	 */		
   101  	public function isSingleton():Boolean 
   102  	{
   103  		return _singleton;
   104  	}
   105  
   106  	/**
   107  	 * Sets the constructor arguments values of this object.
   108  	 * @param value the array representation of all arguments in the constructor of the object instance.
   109  	 */	
   110  	public function setConstructorArguments( value:Array ):Void 
   111  	{
   112  		_constructorArguments = value ;
   113  	}
   114  
   115  	/**
   116  	 * Sets the name of the method invoqued when the object is destroyed.
   117  	 * @param value the name of the destroy method of the object.
   118  	 */	
   119  	public function setDestroyMethodName( value:String ):Void 
   120  	{
   121  		_destroyMethodName = value;
   122  	}
   123  
   124  	/**
   125  	 * Init the name of the method.
   126  	 * @param value the string 'init method' name.
   127  	 */		
   128  	public function setInitMethodName( value:String ):Void 
   129  	{
   130  		_initMethodName = value;
   131  	}
   132  
   133  	/**
   134  	 * Sets the map of all properties of this Definition.
   135  	 * @param value the Map of all properties of the object.
   136  	 */	
   137  	public function setProperties( value:Map ):Void 
   138  	{
   139  		_properties = value ;
   140  	}
   141  
   142  	/**
   143  	 * Sets the type of the object (the class name).
   144  	 * @param value the string representation of the type object.
   145  	 */	
   146  	public function setType( value:String ):Void 
   147  	{
   148  		_type = value ;
   149  	}
   150  
   151  	/**
   152  	 * The internal Map of all arguments use in the constructor of the object.
   153  	 */
   154  	private var _constructorArguments:Array ;
   155  	
   156  	/**
   157  	 * The name of the destroy method of the object.
   158  	 */
   159  	private var _destroyMethodName:String;
   160  
   161  	/**
   162  	 * The name of the init method of the object.
   163  	 */
   164  	private var _initMethodName:String;
   165  
   166  	/**
   167  	 * The internal Map of all properties of the object.
   168  	 */
   169  	private var _properties:Map ;
   170  	
   171  	/**
   172  	 * The internal Map of all singletons.
   173  	 */
   174  	private var _singleton:Boolean ;
   175  	
   176  	/**
   177  	 * The type of the IDefinition object.
   178  	 */
   179  	private var _type : String;
   180  
   181  }
