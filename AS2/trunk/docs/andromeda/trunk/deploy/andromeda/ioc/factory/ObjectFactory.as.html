    24  import andromeda.ioc.core.IObjectDefinition;
    25  import andromeda.ioc.core.ObjectDefinitionContainer;
    26  import andromeda.ioc.factory.IObjectFactory;
    27  
    28  import vegas.data.iterator.Iterator;
    29  import vegas.data.Map;
    30  import vegas.data.map.HashMap;
    31  import vegas.errors.NullPointerError;
    32  import vegas.util.ConstructorUtil;

    34  /**
    35   * The factory of all objects define with a IObjectDefinition reference.
    36   * @author eKameleon
    37   */
    38  class andromeda.ioc.factory.ObjectFactory extends ObjectDefinitionContainer implements IObjectFactory
    39  {
    40  	
    41  	/**
    42  	 * Creates a new LightContainer instance.
    43  	 */
    44  	function ObjectFactory() 
    45  	{
    46  		singletons = new HashMap();
    47  	}
    48  	
    49  	/**
    50  	 * The maps of all objects in the container.
    51  	 */
    52  	public var singletons:HashMap ;
    53  
    54  	/**
    55  	 * Returns {@code true} if the LightContainer contains the specified name.
    56  	 * @param name the name of the object in the container.
    57  	 * @return {@code true} if the LightContainer contains the specified name.
    58  	 */		
    59  	public function containsObject(name:String):Boolean 
    60  	{
    61  		return containsObjectDefinition(name);
    62  	}
    63  
    64  	/**
    65  	 * This method returns an object with the specified name in argument.
    66  	 * @param name The name of the object.
    67  	 * @return the instance of the object with the name passed in argument.
    68  	 */		
    69  	public function getObject( name:String ) 
    70  	{
    71  		var instance = _findInCache( name ) ;		
    72  		if ( instance == null )
    73  		{
    74  			var definition:IObjectDefinition = getObjectDefinition( name ) ;
    75  			if ( definition == null )
    76  			{
    77  				throw new NullPointerError( this +  " get( " + name + " ) method failed, the object isn't register in the container.") ; 
    78  			}
    79  			if ( definition.isSingleton())
    80  			{
    81  				instance = _createAndCacheSingletonInstance( name , definition ) ;
    82  			}
    83  			else
    84  			{
    85  				instance = _createObject( definition.getType() , definition ) ;
    86  			}
    87  		}
    88  		return instance || null ;
    89  	}
    90  
    91  	/**
    92  	 * This method defined if the object is a singleton or a prototype.
    93  	 * @param name The name of the object to find in the singleton map.
    94  	 * @return {@code true} if the object is a singleton or else if the object is a prototype. 
    95  	 */	
    96  	public function isSingleton( name:String):Boolean 
    97  	{
    98  		return singletons.containsKey( name ) ;
    99  	}
   100  
   101  	/**
   102  	 * Removes and destroy a singleton in the LightContainer. Invoke the 'destroy' method of this object is it's define in the IObjectDefinition of this singleton.
   103  	 * @param name The name of the singleton to remove.
   104  	 */
   105  	public function removeSingleton( name:String ):Void
   106  	{
   107  		if ( isSingleton(name) )
   108  		{
   109  			_invokeIDestroyMethod( singletons.get(name), getObjectDefinition(name) ) ;
   110  			singletons.remove( name ) ;	
   111  		}
   112  	}
   113  	
   114  	/**
   115  	 * Creates the arguments Array representation of the specified definition.
   116  	 * @return the arguments Array representation of the specified definition.
   117  	 */
   118  	private function _createArguments( argList:Array ):Array
   119  	{
   120  		var len:Number = argList.length ;
   121  		if ( len > 0 )
   122  		{
   123  			var stack:Array = [] ;
   124  			var item:Object ;
   125  			for (var i:Number = 0 ; i<len ; i++)
   126  			{
   127  				item = argList[i] ;
   128  				if (item.ref != null)
   129  				{
   130  					stack.push( getObject( item.ref) ) ;	
   131  				}
   132  				else if (item.value != null)
   133  				{
   134  					stack.push( item.value ) ;	
   135  				}
   136  				else
   137  				{
   138  					stack.push(null) ;
   139  				}
   140  			}
   141  			return stack ;		
   142  		}
   143  		else
   144  		{
   145  			return null ;
   146  		}
   147  	} 
   148  
   149  	/**
   150  	 * Creates and cache the singleton instance define with the specified name and IDefinition.
   151  	 * @param name the name of the class object.
   152  	 * @param definition the IDefinition to apply over the new instance.
   153  	 */
   154  	private function _createAndCacheSingletonInstance( name:String, definition:IObjectDefinition )
   155  	{
   156  		var instance = singletons[name] ;
   157  		if(!instance) 
   158  		{
   159  			instance = _createObject( definition.getType() , definition ) ;
   160  			singletons.put( name, instance ) ;
   161  		}
   162  		return instance;
   163  	}
   164  
   165  	/**
   166  	 * Creates a new Object with the specified name and the specified IObjectDefinition in argument.
   167  	 * @return a new Object with the specified name and the specified IObjectDefinition in argument.
   168  	 */
   169  	private function _createObject( name:String, definition:IObjectDefinition ) 
   170  	{
   171  		var c:Function = eval( "_global." + name ) ;
   172  		var instance = ConstructorUtil.createInstance( c , _createArguments( definition.getConstructorArguments() ) ) ;
   173  		_populateProperties( instance, definition.getProperties() );	
   174  		_invokeInitMethod( instance, definition ) ;
   175  		return instance ;
   176  	}
   177  
   178  	/**
   179  	 * Invoque the destroy method of the specified object, if the init method is define in the IDefinition object.
   180  	 */
   181  	private function _invokeIDestroyMethod( o , definition:IObjectDefinition ):Void
   182  	{
   183  		if( definition.getDestroyMethodName() ) 
   184  		{
   185  			o[ definition.getDestroyMethodName() ]() ;
   186  		}
   187  	}
   188  	
   189  	/**
   190  	 * Invoque the init method of the specified object, if the init method is define in the IDefinition object.
   191  	 */
   192  	private function _invokeInitMethod( o , definition:IObjectDefinition ):Void
   193  	{
   194  		if( definition.getInitMethodName() ) 
   195  		{
   196  			o[ definition.getInitMethodName() ]() ;
   197  		}
   198  	}
   199  
   200  	/**
   201  	 * Populates all properties in the Map passed in argument.
   202  	 */
   203  	private function _populateProperties( o, properties:Map ):Void 
   204  	{
   205  		if (properties != null && properties.size() > 0)
   206  		{
   207  			var it:Iterator = properties.iterator() ;
   208  			while(it.hasNext())
   209  			{
   210  				var value = it.next() ;
   211  				var key   = it.key()  ;
   212  				if (value instanceof Array) // method
   213  				{
   214  					o[key].apply(o, value) ;
   215  				}
   216  				else // property
   217  				{
   218  					if( containsObject( value ) ) 
   219  					{
   220  						value = getObject( value ) ;
   221          				properties.put( key , value ) ;
   222      				}
   223      				o[key] = value ;
   224  				}
   225  			}
   226  		} 
   227  	}
   228  
   229  	/**
   230  	 * Returns the object register in cache in this container.
   231  	 * @param the name of the object.
   232  	 * @return the object register in cache in this container.
   233  	 */
   234  	private function _findInCache( name:String ) 
   235  	{
   236  		return singletons.get(name) || null ;
   237  	}
   238  
   239  }
