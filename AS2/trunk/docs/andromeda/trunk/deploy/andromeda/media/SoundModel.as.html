    35  class andromeda.media.SoundModel extends AbstractModel

    37  	
    38  	/**
    39  	 * Creates a new SoundModel instance.
    40  	 * @param id the id of the model.
    41  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    42  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    43  	 */
    44  	function SoundModel( id , bGlobal:Boolean , sChannel:String ) 
    45  	{
    46  		super( id, bGlobal , sChannel ) ;
    47  		initEvent() ;
    48  		_bIsOn = true ;
    49  		_map = new HashMap() ;
    50  	}
    51  	
    52  	/**
    53  	 * Adds new sound in the model.
    54  	 * <p>If the id already exist in the model, the current sound is remove in the model before the new sound must be inserted in the model</p>
    55  	 * <p>Example
    56  	 * {@code
    57  	 *   var sModel:SoundModel = new SoundModel();
    58  	 *   sModel.addSound( "sound_1", new Sound(mc1) );
    59  	 *   sModel.addSound( "sound_2", new Sound(mc2) );
    60  	 * }
    61  	 * </p>
    62  	 * @param id Sound identifier in swf file.
    63  	 * @param sound Sound reference.
    64  	 * 
    65  	 * @return {@code String} sound identifier 
    66  	 * @throws ArgumentsError if the specified id and sound are 'null' or 'undefined'.
    67  	 */
    68  	public function addSound( id:String, sound:Sound )
    69  	{
    70  		if (id != null && sound != null)
    71  		{
    72  			
    73  			if (_map.containsKey(id))
    74  			{
    75  				removeSound(id) ;
    76  			}
    77  			
    78  			_map.put(id, sound) ;
    79  			// Reflexion of the sound object.
    80  			sound.toString = function():String
    81  			{
    82  				return "[Sound:'" + id + "']" ;
    83  			} ;
    84  			notifyAddSound(id, sound) ;
    85  			return id  ;
    86  		}
    87  		else
    88  		{
    89  			throw new ArgumentsError ( this + " addSound method failed, the specified id and sound not must be 'null' or 'undefined'." ) ;
    90  			return null ;
    91  		}
    92  	}

    94  	/**
    95  	 * Clear all sounds in the model.
    96  	 */
    97  	public function clear() : Void
    98  	{
    99  		_map.clear() ;
   100  		notifyClearSound() ;
   101  	}

   121  	/**
   122  	 * Disabled sound playing mode. (stop all sounds in the model if the model isn't empty)
   123  	 */
   124  	public function disable():Void 
   125  	{ 
   126  		_bIsOn = false;
   127  		var aSounds:Array = toArray() ;
   128  		var size:Number = aSounds.length ;
   129  		if (size > 0)
   130  		{
   131  			while (--size > -1)
   132  			{
   133  				aSounds[ size ].stop();
   134  			}
   135  		}
   136  		notifyDisableSound() ;
   137  	}

   139  	/**
   140  	 * Enabled sound playing mode.
   141  	 */
   142  	public function enable() : Void 
   143  	{ 
   144  		_bIsOn = true;
   145  		notifyEnableSound() ; 
   146  	}

   193  	/**
   194  	 * Returns {@code Sound} instance register in the model.
   195  	 * 
   196  	 * id the identifier of the sound in the model.
   197  	 * 
   198  	 * <p>Example
   199  	 * {@code
   200  	 *   var sModel:SoundModel = new SoundModel();
   201  	 *   sModel.addSound( "sound_1", new Sound(mc1) );
   202  	 *   
   203  	 *   var currentSound:Sound = sModel.getSound( "sound_1" );
   204  	 * }
   205  	 * 
   206  	 * @return {@code Sound} instance. If no sound is found, an empty 
   207  	 * {@code Sound} is returned.
   208  	 */
   209  	public function getSound( id:String ):Sound 
   210  	{ 
   211  		return _bIsOn ? _map.get(id) : null ;
   212  	}

   222  		_eEnabled  = new SoundModelEvent( SoundModelEvent.ENABLE_SOUNDS , this ) ;
   223  		_eRemove   = new SoundModelEvent( SoundModelEvent.REMOVE_SOUND , this ) ;
   224  	}

   289  		dispatchEvent( _eRemove ) ;	
   290  	}

   292  	/**
   293  	 * Removes a sound in the model.
   294  	 * 
   295  	 * <p>Example
   296  	 * {@code
   297  	 *   var sModel:SoundModel = new SoundModel();
   298  	 *   sModel.addSound( "sound_1", new Sound(mc1) );
   299  	 *   sModel.removeSound( "sound_1" );
   300  	 * }
   301  	 * 
   302  	 * @param id Sound identifier in swf file.
   303  	 * 
   304  	 * @return {@code null} if the model don't contains the specified id. 
   305  	 * @throws ArgumentsError if the specified id is 'null' or 'undefined'.
   306  	 */
   307  	public function removeSound( id:String )
   308  	{
   309  		if (id == null)
   310  		{
   311  			throw new ArgumentsError( this + " addSound method failed, the specified id and sound not must be 'null' or 'undefined'." ) ;
   312  		}
   313  		else if (_map.containsKey(id) )
   314  		{
   315  			var sound:Sound = _map.get(id) ;
   316  			var r = _map.remove(id) ;
   317  			notifyRemoveSound(id, sound) ;
   318  			return r ;
   319  		}
   320  		else
   321  		{
   322  			return null ;
   323  		}
   324  	}
