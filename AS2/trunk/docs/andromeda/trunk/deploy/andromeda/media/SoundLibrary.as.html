    35  /**
    36   * Sound library to register all sounds in an application.
    37   * <p>To load external "mp3" files using {@code loadSound} method.</p>
    38   * <p><b>Example :</b></p>
    39   * <p>
    40   * {@code
    41   * import andromeda.events.SoundModelEvent ;
    42   * import andromeda.media.SoundLibrary ;
    43   * import vegas.events.Delegate ;
    44   * 
    45   * var debug:Function = function ( e:SoundModelEvent ):Void
    46   * {
    47   *     var type:String = e.getType() ;
    48   *	   var id:String = e.getID() ;
    49   *	   var library:SoundLibrary = SoundLibrary(e.getModel()) ;
    50   *	   var sound:Sound = e.getSound() ;
    51   *	   
    52   *	   var message:String = "> " + library ;
    53   *	   message += " " + type ;
    54   *	   if (id != null)
    55   *	   {
    56   *		    message += ", id:" + id ;
    57   *	   }
    58   *	   if (sound != null)
    59   *	   {
    60   *		    message += ", sound:" + sound ;
    61   *	   }
    62   *	   trace( message ) ;
    63   * }
    64   *
    65   * var lib:SoundLibrary = new SoundLibrary() ;
    66   * lib.addGlobalEventListener( new Delegate(this, debug) ) ;
    67   *
    68   * lib.addSound( "sound_1" ) ;
    69   * lib.addSound( "sound_2" ) ;
    70   *
    71   * lib.addSounds( ["sound_1", "sound_2"] ) ;
    72   *
    73   * trace("> " + lib + " size : " + lib.size()) ;
    74   *
    75   * Key.addListener(this) ;
    76   *
    77   * onKeyDown = function()
    78   * {
    79   *     var code:Number = Key.getCode() ;
    80   *     switch ( code )
    81   *     {
    82   *	        case Key.UP : 
    83   *		    {
    84   *   			lib.getSound( "sound_1" ).start() ;
    85   *			    break ;
    86   *		    }
    87   *		    case Key.DOWN : 
    88   *		    {
    89   *			    lib.getSound( "sound_2" ).start() ;
    90   *			    break ;
    91   *		    }
    92   *		    case Key.SPACE :
    93   *		    {
    94   *			    lib.toggle() ;
    95   *			    break ;
    96   *		    }
    97   *	    }
    98   * }
    99   *
   100   * trace("> Press Key.SPACE to disable/enable the library.") ;
   101   * trace("> Press Key.UP to start the sound_1 sound.") ;
   102   * trace("> Press Key.DOWN to start the sound_2 sound.") ;
   103   * }
   104   * </p>
   105   * @author eKameleon
   106   */

   170  	/**
   171  	 * Adds new sound in the model.
   172  	 * <p>If the id already exist in the model, the current sound is remove in the model before the new sound must be inserted in the model</p>
   173  	 * <p>Example
   174  	 * {@code
   175  	 *   var sLib:SoundLibrary = new SoundLibrary();
   176  	 *   sLib.addSound( "sound_1" );
   177  	 *   sLib.addSound( "sound_2" );
   178  	 * }
   179  	 * </p>
   180  	 * @param id Sound identifier in swf file.
   181  	 * @param sound Sound reference.
   182  	 * 
   183  	 * @return {@code String} sound identifier 
   184  	 * @throws ArgumentsError if the specified id and sound are 'null' or 'undefined'.
   185  	 */
   186  	public function addSound( id:String )
   187  	{
   188  		if (!isInit())
   189  		{
   190  			initialize() ;
   191  		}
   192  		var sound:Sound = _createSound(id) ;
   193  		_mGain.put(id, DEFAULT_GAIN) ;
   194  		_setVolume( id ) ;
   195  		return super.addSound( id , sound ) ;
   196  	}

   198  	/**
   199  	 * Adds sounds id list to the the library.
   200  	 * <p>Example
   201  	 * {@code
   202  	 *   var sLib:SoundLibrary = new SoundLibrary();
   203  	 *   var aSoundList:Array = ["sound_1", "sound_2"] ;
   204  	 *   sLib.addSounds( aSoundList );
   205  	 * }
   206  	 * </p>
   207  	 * 
   208  	 * @param list an {@code Array} Sounds ID list.
   209  	 */
   210  	public function addSounds( list:Array) : Void
   211  	{
   212  		var l:Number = list.length;
   213  		for (var i:Number=0; i<l ; i++) 
   214  		{
   215  			addSound( list[i] );
   216  		}
   217  	}

   313  	/**
   314  	 * Loads and attach in the library a new external Sound.
   315  	 * @param sURL the string representation of the url of the external sound.
   316  	 * @param id the identifier of the sound, if this parameter is null the loadSound method try to use the name of the external file to defined the id value.
   317  	 * @return {@code String} sound identifier 
   318  	 */
   319  	public function loadSound(sURL:String, id:String):String
   320  	{
   321  		if (!isInit())
   322  		{
   323  			initialize() ;
   324  		}
   325  		
   326  		if ( id == null )
   327  		{
   328  			id = sURL.substring( sURL.lastIndexOf("/") + 1 , sURL.lastIndexOf(".") ) ;
   329  		}
   330  		
   331  		if (id != null && id.length > 0)
   332  		{
   333  			var sound:Sound = _createSound( id, sURL);
   334  			super.addSound( id , sound ) ;
   335  			return id ;
   336  		}
   337  		else
   338  		{
   339  			throw new Warning( this + " loadSound method failed, the specified id not must be 'null' or 'undefined' or a empty string." ) ;
   340  		}
   341  	}

   392  	/**
   393  	 * Removes a sound in the model.
   394  	 * <p>Example</p>
   395  	 * {@code
   396  	 *   var sLib:SoundLibrary = new SoundLibrary();
   397  	 *   sLib.addSound( "sound_1" );
   398  	 *   sLib.removeSound( "sound_1" );
   399  	 * }
   400  	 * @param id Sound identifier in swf file.
   401  	 * @return {@code null} if the model don't contains the specified id. 
   402  	 * @throws ArgumentsError if the specified id is 'null' or 'undefined'.
   403  	 */
   404  	public function removeSound( id:String ):Void
   405  	{
   406  		if ( containsByID(id) )
   407  		{
   408  			_removeSound(id) ;
   409  			_mGain.remove(id) ;
   410  			super.removeSound( id ) ;
   411  		}
   412  	}

   414  	/**
   415  	 * Defines {@code gain} value for all registered sounds.
   416  	 * @param nGain a number between 0 and 127.
   417  	 */
   418  	public function setAllGain(nGain:Number) : Void
   419  	{
   420  		nGain = RANGE_GAIN.clamp(nGain) ;
   421  		var ar:Array = _mGain.getKeys() ;
   422  		var len:Number = ar.size() ;
   423  		while( --len > -1 )
   424  		{
   425  			var id:String = ar[len] ;
   426  			_mGain.put( id , nGain ) ;
   427  			_setVolume( id ) ;
   428  		}
   429  	}

   475  	{

   478  		var len:Number = ar.size() ;

   490  	{
   491  		value = Range.PERCENT_RANGE.clamp(value) ;
   492  		_nVolume = value ;
   493  		var ar:Array = _mGain.getKeys() ;
   494  		var len:Number = ar.size() ;
   495  		while( --len > -1 )
   496  		{
   497  			_setVolume( ar[len] );
   498  		}
   499  	}

   555  	/**
   556  	 * Creates a new sound in the library.
   557  	 * @param id the ID of the sound in the library.
   558  	 * @param sURL (optional) the url of the external sound to load. If this argument is null the id property is the link in the library of the sound object.
   559  	 */
   560  	private function _createSound( id:String, sURL:String):Sound
   561  	{
   562  		
   563  		var target:MovieClip = _container.createEmptyMovieClip("sound_" + id, _nDepth++);
   564  		var sound:Sound = new Sound( target ) ;
   565  		// Reflexion of the sound object.
   566  		sound.toString = function():String
   567  		{
   568  			return "[Sound:'" + id + "']" ;
   569  		} ;
   570  		sound.onSoundComplete = Delegate.create(this, notifyComplete, id, sound) ;
   571  		sound.onID3 = Delegate.create(this, notifyID3, id, sound) ;
   572  		sound.onLoad = Delegate.create(this, notifyLoad, id, sound, sURL) ;
   573  		
   574  		if (sURL)
   575  		{
   576  			sound.loadSound(sURL, false);
   577  
   578  		} else
   579  		{
   580  			sound.attachSound( id );
   581  		}
   582  		
   583  		return sound ;
   584  	}
