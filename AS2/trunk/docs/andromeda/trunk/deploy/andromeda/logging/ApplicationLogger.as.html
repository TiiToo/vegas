    24  import vegas.logging.ILogger;
    25  import vegas.logging.Log;
    26  import vegas.logging.LogEventLevel;
    27  import vegas.logging.targets.LuminicTarget;
    28  import vegas.logging.targets.SOSTarget;
    29  import vegas.util.ResolverProxy;
    30  
    31  /**
    32   * This static tool class initialize the global ILogger singleton of the application and create a link with all the ILogable CoreObject.
    33   * @author eKameleon
    34   */
    35  class andromeda.logging.ApplicationLogger 
    36  {
    37  	
    38  	/**
    39  	 * The application global logger id.
    40  	 */
    41  	public static var APPLICATION_CHANNEL:String = "application" ;
    42  
    43  	/**
    44  	 * The vegas errors logger id.
    45  	 */
    46  	public static var VEGAS_ERRORS_CHANNEL:String = "vegas.errors.*" ;
    47  
    48  	/**
    49  	 * Logs the specified data using the LogEventLevel.DEBUG level.
    50  	 * @param message:String The information to log. This string can contain special marker characters of the form {x}, where x is a zero based index that will be replaced with the additional parameters found at that index if specified.
    51   	 * @param ... rest Additional parameters that can be subsituted in the str parameter at each "{x}" location, where x is an integer (zero based) index value into the Array of values specified.
    52  	 */
    53  	static var debug:Function ; 
    54  	
    55  	/**
    56  	 * Logs the specified data using the LogEventLevel.ERROR level.
    57  	 * @param message:String The information to log. This string can contain special marker characters of the form {x}, where x is a zero based index that will be replaced with the additional parameters found at that index if specified.
    58   	 * @param ... rest Additional parameters that can be subsituted in the str parameter at each "{x}" location, where x is an integer (zero based) index value into the Array of values specified.
    59  	 */
    60  	static var error:Function ; 
    61  	
    62  	/**
    63  	 * Logs the specified data using the LogEventLevel.FATAL level.
    64  	 * @param message:String The information to log. This string can contain special marker characters of the form {x}, where x is a zero based index that will be replaced with the additional parameters found at that index if specified.
    65   	 * @param ... rest Additional parameters that can be subsituted in the str parameter at each "{x}" location, where x is an integer (zero based) index value into the Array of values specified.	 
    66   	 */
    67  	static var fatal:Function ; 
    68  	
    69  	/**
    70  	 * Logs the specified data using the LogEvent.INFO level.
    71   	 * @param message:String The information to log. This string can contain special marker characters of the form {x}, where x is a zero based index that will be replaced with the additional parameters found at that index if specified.
    72   	 * @param ... rest Additional parameters that can be subsituted in the str parameter at each "{x}" location, where x is an integer (zero based) index value into the Array of values specified.
    73  	 */
    74  	static var info:Function ;
    75  
    76  	/**
    77  	 * Logs the specified data at the given level.
    78  	 * @param level:int The level this information should be logged at. Valid values are:
    79  	 * <li>LogEventLevel.FATAL designates events that are very harmful and will eventually lead to application failure</li>
    80  	 * <li>LogEventLevel.ERROR designates error events that might still allow the application to continue running.</li>
    81  	 * <li>LogEventLevel.WARN designates events that could be harmful to the application operation</li>
    82  	 * <li>LogEventLevel.INFO designates informational messages that highlight the progress of the application at coarse-grained level.</li>
    83  	 * <li>LogEventLevel.DEBUG designates informational level messages that are fine grained and most helpful when debugging an application.</li>
    84  	 * <li>LogEventLevel.ALL intended to force a target to process all messages.</li>
    85  	 * @param message:String The information to log. This string can contain special marker characters of the form {x}, where x is a zero based index that will be replaced with the additional parameters found at that index if specified.
    86   	 * @param ... rest Additional parameters that can be subsituted in the str parameter at each "{x}" location, where x is an integer (zero based) index value into the Array of values specified.
    87   	 */
    88  	static var log:Function ; 
    89  	
    90  	/**
    91  	 * Logs the specified data using the LogEventLevel.WARN level.
    92   	 * @param message:String The information to log. This string can contain special marker characters of the form {x}, where x is a zero based index that will be replaced with the additional parameters found at that index if specified.
    93   	 * @param ... rest Additional parameters that can be subsituted in the str parameter at each "{x}" location, where x is an integer (zero based) index value into the Array of values specified.
    94  	 */
    95  	static var warn:Function ; 
    96  
    97  	/**
    98  	 * Returns the ILogger of the application.
    99  	 * @return the ILogger of the application.
   100  	 */
   101  	public static function getLogger():ILogger
   102  	{
   103  		if( _logger == null )
   104  		{
   105  			_logger = Log.getLogger(APPLICATION_CHANNEL)  ;
   106  		}
   107  		return _logger ;	
   108  	}
   109  
   110  	/**
   111  	 * Initialize the global Logger singleton.
   112  	 */
   113  	public static function initialize():Void
   114  	{
   115  		
   116  		// initialize the ITargets of the application.
   117  		
   118  		var filters:Array = [ APPLICATION_CHANNEL , VEGAS_ERRORS_CHANNEL ] ;
   119  		
   120  		var sosTarget:SOSTarget = new SOSTarget() ;
   121  		sosTarget.filters = filters ;
   122  		sosTarget.includeLines = true ;
   123  		sosTarget.level = LogEventLevel.ALL ;
   124  		
   125  		Log.addTarget(sosTarget) ;
   126  
   127  		var luminicTarget:LuminicTarget = new LuminicTarget() ;
   128  		luminicTarget.filters = filters ;
   129  		luminicTarget.isCollapse = false ;
   130  		luminicTarget.includeTime = true ;
   131  		luminicTarget.level = LogEventLevel.ALL ;
   132  		
   133  		Log.addTarget(luminicTarget) ;
   134  
   135  		// Initialize the proxy between the static class and the internal getLogger() reference.
   136  		
   137  		initProxy() ;
   138  
   139  		// This hack change the getLogger method in all CoreObject objects. (All object in VEGAS extends the CoreObject class)		
   140  		
   141  		vegas.core.CoreObject.prototype["getLogger"] = getLogger ;
   142  		
   143  		ApplicationLogger.debug( "[ApplicationLogger] initialize." ) ;
   144  		
   145  	}
   146  	
   147  	/**
   148  	 * Creates a proxy link between the static class ApplicationLogger and the getLogger() reference.
   149  	 * For example, when we call the method {@code Logger.debug()} the internal resolver proxy call the {@code Logger.getLogger().debug()} method.
   150  	 * @see ResolverProxy
   151  	 * @see __resolve  
   152  	 */
   153  	public static function initProxy():Void
   154  	{
   155  		ResolverProxy.initialize( ApplicationLogger ) ;
   156  		ApplicationLogger["setProxy"]( getLogger() ) ;	
   157  	}
   158  
   159  	/**
   160  	 * The internal global logger of the application.
   161  	 */
   162  	private static var _logger:ILogger ;
   163  
   164  }
