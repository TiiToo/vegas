    24  import andromeda.display.abstract.ILoaderDisplay;
    25  
    26  import asgard.display.ConfigurableDisplayObject;
    27  
    28  import pegas.maths.Range;
    29  
    30  /**
    31   * The abstract class of all loader displays in the application.
    32   * @author eKameleon
    33   */
    34  class andromeda.display.abstract.AbstractLoaderDisplay extends ConfigurableDisplayObject implements ILoaderDisplay 
    35  {
    36  	
    37  	/**
    38  	 * Abstract constructor, this constructor must be override.
    39  	 * @param sName the name of the display.
    40  	 * @param target the DisplayObject instance control this target.
    41  	 * @param depth the depth of the view reference of this display.
    42  	 */
    43  	public function AbstractLoaderDisplay( sName:String , target , depth:Number ) 
    44  	{
    45  		super( sName , target.createEmptyMovieClip( sName, depth )  ) ;
    46  	}
    47  
    48  	/**
    49  	 * Returns the label value.
    50  	 * @return the label value.
    51  	 */
    52  	public function get label():String
    53  	{
    54  		return getLabel() ;
    55  	}
    56  	
    57  	/**
    58  	 * Set the label value.
    59  	 */
    60  	public function set label(s:String):Void
    61  	{
    62  		setLabel(s) ;
    63  	}
    64  
    65  	/**
    66  	 * Returns the percent of the loader in percent.
    67  	 * @return the percent of the loader in percent.
    68  	 */
    69  	public function get percent():Number
    70  	{
    71  		return getPercent() ;
    72  	}
    73  	
    74  	/**
    75  	 * Sets the percent of the loader in percent.
    76  	 */
    77  	public function set percent(n:Number):Void
    78  	{
    79  		setPercent(n) ;
    80  	}
    81  
    82  	/**
    83  	 * Returns the label value.
    84  	 * @return the label value.
    85  	 */
    86  	public function getLabel():String
    87  	{
    88  		return _label ;
    89  	}
    90  	
    91  	/**
    92  	 * Returns the percent of the loader in percent.
    93  	 * @return the percent of the loader in percent.
    94  	 */
    95  	public function getPercent():Number
    96  	{
    97  		return _percent ;
    98  	}
    99  
   100  	/**
   101  	 * Resets the loader.
   102  	 */
   103  	public function reset():Void
   104  	{
   105  		_label    = null ;
   106  		_percent = 0 ;
   107  		update() ;	
   108  	}
   109  	
   110  	/**
   111  	 * Set the label value.
   112  	 */
   113  	public function setLabel(s:String):Void
   114  	{
   115  		_label = s ;
   116  		update() ;	
   117  	}
   118  	
   119  	/**
   120  	 * Sets the label and the percent values of the loader display.
   121  	 * @param message the value of the message to show.
   122  	 * @param percent the value of the percent position of this loader.
   123  	 */
   124  	public function setLoader( message, percent:Number ):Void
   125  	{
   126  		_label   = message.toString() || null ;
   127  		_percent = Range.PERCENT_RANGE.clamp( (percent>0) ? percent : 0 ) ;
   128  		update() ;
   129  	}
   130  	
   131  	/**
   132  	 * Sets the percent of the loader in percent.
   133  	 */
   134  	public function setPercent(n:Number):Void
   135  	{
   136  		_percent = Range.PERCENT_RANGE.clamp( (n>0) ? n : 0 ) ;
   137  		update() ;
   138  	}
   139  
   140  	/**
   141  	 * Update the display.
   142  	 */	
   143  	public function update():Void
   144  	{
   145  		// overrides this method.
   146  	}
   147  	
   148  	/**
   149  	 * Show the display.
   150  	 */
   151  	/*override*/ public function show():Void
   152  	{
   153  		reset() ;
   154  		super.show() ;
   155  	}
   156  
   157  	/**
   158  	 * The internal label value of this loader display.
   159  	 */
   160  	private var _label:String ;
   161  	
   162  	/**
   163  	 * The internal position value of this loader display.
   164  	 */
   165  	private var _percent:Number = 0 ;
   166  	
   167  }
