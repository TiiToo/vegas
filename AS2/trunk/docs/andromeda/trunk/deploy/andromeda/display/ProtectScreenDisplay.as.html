    24  import andromeda.core.ApplicationDepthList;
    25  import andromeda.core.ApplicationList;
    26  
    27  import asgard.display.BackgroundDisplay;
    28  
    29  import vegas.events.Delegate;
    30  
    31  /**
    32   * This display protect the view of the application of all mouse activity.
    33   * @author eKameleon
    34   */
    35  class andromeda.display.ProtectScreenDisplay extends BackgroundDisplay
    36  {
    37  
    38  	/**
    39  	 * Creates a new ProtectScreenDisplay instance.
    40  	 * @param target the DisplayObject instance control this target.
    41  	 * @param nDepth the depth of the display target reference.
    42  	 */
    43  	function ProtectScreenDisplay( target:MovieClip ) 
    44  	{
    45  		
    46  		super( ApplicationList.PROTECT_SCREEN, target.createEmptyMovieClip( ApplicationList.PROTECT_SCREEN, ApplicationDepthList.PROTECT_DEPTH )  ) ;
    47          
    48  		Stage.addListener(this) ;
    49  		
    50  		background.onPress = Delegate.create(this, doNothing) ;
    51  		background.useHandCursor = false ;
    52  		
    53  		themeAlpha = 0 ;
    54  		themeColor = 0 ;
    55  		
    56  		hide() ;
    57  						
    58  		isFull = true ;
    59  	}
    60  
    61  	/**
    62  	 * This Method do nothing when the use click on the ProtectScreen
    63  	 */	
    64  	 public function doNothing():Void
    65  	 {
    66  	 	getLogger().warn( this + " do nothing ..." ) ;
    67  	 }
    68  
    69  	/**
    70  	 * Invoqued when the Stage size change.
    71  	 */
    72  	public function onResize():Void
    73  	{
    74  		update() ;
    75  	}
    76  
    77  }
