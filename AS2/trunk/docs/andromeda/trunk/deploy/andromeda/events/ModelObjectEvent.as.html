    24  import andromeda.model.IModelObject;
    25  import andromeda.model.IValueObject;
    26  
    27  import vegas.events.BasicEvent;
    28  
    29  /**
    30   * The ModelObjectEvent is the basic event use in a IModelObject to notify changed.
    31   * @author eKameleon
    32   */
    33  class andromeda.events.ModelObjectEvent extends BasicEvent 
    34  {
    35  	
    36  	/**
    37  	 * Creates a new ModelObjectEvent instance.
    38  	 * @param type the type of the event.
    39  	 * @param target the IModelObject reference of this event.
    40  	 * @param (optional) the IValueObject of this event.
    41  	 */	
    42  	public function ModelObjectEvent( type:String, target:IModelObject , vo:IValueObject ) 
    43  	{
    44  		super( type, target ) ;
    45  		setVO( vo ) ;	
    46  	}
    47  
    48  	/**
    49  	 * Default event type when the addVO method is invoqued.
    50  	 */
    51  	static public var ADD_VO:String = "onAddVO" ;
    52  
    53  	/**
    54  	 * Default event type when the clear method is invoqued.
    55  	 */
    56  	static public var CLEAR_VO:String = "onClearVO" ;
    57  
    58  	/**
    59  	 * Default event type when the setVO method is invoqued.
    60  	 */
    61  	static public var CHANGE_CURRENT_VO:String = "onChangeCurrentVO" ;
    62  
    63  	/**
    64  	 * Default event type when the removeVO method is invoqued.
    65  	 */
    66  	static public var REMOVE_VO:String = "onRemoveVO" ;
    67  
    68  	/**
    69  	 * Default event type when the removeVO method is invoqued.
    70  	 */
    71  	static public var UPDATE_VO:String = "onUpdateVO" ;
    72  	
    73  	/**
    74  	 * Returns the target reference of this event.
    75  	 * @return the target reference of this event.
    76  	 */
    77  	public function getTarget():IModelObject
    78  	{
    79  		return IModelObject( super.getTarget() ) ;	
    80  	}
    81  	
    82  	/**
    83  	 * Returns a IValueObject reference.
    84  	 * @return a IValueObject reference.
    85  	 */
    86  	public function getVO():IValueObject
    87  	{
    88  		return _vo ;	
    89  	}
    90  
    91  	/**
    92  	 * Sets a IValueObject reference.
    93  	 */
    94  	public function setVO( vo:IValueObject ):Void
    95  	{
    96  		_vo = vo || null ;	
    97  	}
    98  
    99  	private var _vo:IValueObject ;
   100  
   101  }
