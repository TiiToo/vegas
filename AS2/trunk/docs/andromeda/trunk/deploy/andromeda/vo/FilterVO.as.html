    24  import andromeda.model.SimpleValueObject;
    25  
    26  import vegas.util.ConstructorUtil;
    27  import vegas.util.serialize.Serializer;
    28  
    29  /**
    30   * This class provides a binary filter value object.
    31   * <p><b>Example :</b></p>
    32   * {@code
    33   * 
    34   * var VIDEO:Number = 1 ;
    35   * var MP3:Number = 2 ; 
    36   * 
    37   * var filter:FilterVO = new FilterVO() ;
    38   * 
    39   * trace( "filter : " +  filter ) ;
    40   * 
    41   * filter.toggleFilter( VIDEO , true ) ;
    42   * 
    43   * trace( "filter : " +  filter ) ;
    44   * 
    45   * trace( "filter.toggleFilter( MP3, true ) : " + filter.toggleFilter( ProductFilter.TRAINING, true ) ) ;
    46   * trace( "filter.toggleFilter( MP3, true ) : " + filter.toggleFilter( ProductFilter.TRAINING, true ) ) ;
    47   * 
    48   * trace("filter : " +  filter ) ;
    49   * 
    50   * filter.toggleFilter( VIDEO , false ) ;
    51   * 
    52   * trace( "filter : " +  filter ) ;
    53   * 
    54   * trace( "filter.contains( VIDEO ) : " + filter.contains( VIDEO ) ) ;
    55   * trace( "filter.contains( MP3 ) : " + filter.contains( MP3 ) ) ;
    56   * }
    57   * @author eKameleon
    58   */
    59  class andromeda.vo.FilterVO extends SimpleValueObject 
    60  {
    61  	
    62  	/**
    63  	 * Creates a new FilterVO instance. 
    64  	 * @param filter (optional) The default filter value of this value object.
    65  	 * @param init A generic object containing properties with which to populate the newly instance. If this argument is null, it is ignored
    66  	 */	
    67  	public function FilterVO( filter:Number , init:Object ) 
    68  	{
    69  		super(init) ;
    70  		setFilter(filter) ;	
    71  	}
    72  	
    73  	/**
    74  	 * The default filter value of the filter value objects.
    75  	 */
    76  	public static var NONE:Number = 0 ;
    77  	
    78  	/**
    79  	 * The filter value of this object.
    80  	 */
    81  	public var filter:Number ;
    82  	
    83  	/**
    84  	 * Clear the filter value.
    85  	 */
    86  	public function clear():Void
    87  	{
    88  		filter = NONE ;
    89  	}
    90  	
    91  	/**
    92  	 * Returns {@code true} if the filter number value contains the option number value.
    93  	 * @return {@code true} if the filter number value contains the option number value.
    94  	 */
    95  	public function contains( value:Number ) : Boolean
    96  	{
    97  		return Boolean( value & getFilter() ) ;
    98  	}
    99  	
   100  	/**
   101  	 * Returns the current filter value of this object.
   102  	 * @return the current filter value of this object.
   103  	 */
   104  	public function getFilter():Number
   105  	{
   106  		return isNaN( filter ) ? NONE : filter ;
   107  	}
   108  
   109  	/**
   110  	 * Returns {@code true} if the filter is NONE.
   111  	 * @return {@code true} if the filter is NONE.
   112  	 */
   113  	public function isNone():Boolean
   114  	{
   115  		return getFilter() == NONE ;
   116  	}
   117  
   118  	/**
   119  	 * Sets the current filter value of this object.
   120  	 */
   121  	public function setFilter( n:Number ):Void
   122  	{
   123  		filter = isNaN(n) ? NONE : n ;
   124  	}
   125  
   126  	/**
   127  	 * Toggle a filter value in this filter object.
   128  	 */
   129  	public function toggleFilter( value:Number, b:Boolean ):Boolean
   130  	{
   131  		var old:Number = filter ;
   132  		filter = b ? ( filter | value ) : ( filter & ~value ) ;
   133  		return old != filter ;
   134  	}
   135  	
   136  	/**
   137  	 * Returns the Object representation of this object.
   138  	 * @return the Object representation of this object.
   139  	 */
   140  	public function toObject():Object
   141  	{
   142  		return { filter:filter } ;
   143  	}
   144  	
   145  	/**
   146  	 * Returns the eden source representation of this object.
   147  	 * @return the eden source representation of this object.
   148  	 */
   149  	public function toSource():Object
   150  	{
   151  		return Serializer.getSourceOf( this, [ Serializer.toSource( filter ) ]  ) ;
   152  	}
   153  	
   154  	/**
   155  	 * Returns the String representation of this object.
   156  	 * @return the String representation of this object.
   157  	 */
   158  	public function toString():Object
   159  	{
   160  		return "[" + ConstructorUtil.getName(this) + ":" + filter + "]" ;
   161  	}
   162  	
   163  }
