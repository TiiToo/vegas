    24  import vegas.core.CoreObject;
    25  import vegas.core.ICloneable;

    29  /**
    30   * A binary tree node from which you can build a binary tree.
    31   * <p>A Binary Tree is a simplified tree structure in which every node is only allowed to have up 
    32   * to two children nodes, which are called the left and right child.</p>
    33   */
    34  class andromeda.data.tree.BinaryTreeNode extends CoreObject implements ICloneable, ICopyable, ISerializable 
    35  {
    36  
    37  	/**
    38  	 * Creates a BinaryTreeNode instance.
    39  	 * @param data The data to register inside the node.
    40  	 */
    41  	public function BinaryTreeNode( data ) 
    42  	{
    43  		this.left   = null ;
    44  		this.right  = null ;
    45  		this.parent = null ;
    46  		this.data  = data ;
    47  	}
    48  
    49  	/**
    50  	 * The data of this node.
    51  	 */
    52  	public var data ;
    53  
    54  	/**
    55  	 * The left BinaryTreeNode of this node.
    56  	 */
    57  	public var left:BinaryTreeNode ;
    58  
    59  	/**
    60  	 * The parent BinaryTreeNode of this node.
    61  	 */
    62  	public var parent:BinaryTreeNode ;
    63  	
    64  	/**
    65  	 * The right BinaryTreeNode of this node.
    66  	 */
    67  	public var right:BinaryTreeNode ;
    68  	
    69  	/**
    70  	 * Returns a shallow copy of this object.
    71  	 * @return a shallow copy of this object.
    72  	 */
    73  	public function clone() 
    74  	{
    75  		return new BinaryTreeNode(data) ;
    76  	}
    77  
    78  	/**
    79  	 * Returns a deep copy of this object.
    80  	 * @return a deep copy of this object.
    81  	 */
    82  	public function copy() 
    83  	{
    84  		return new BinaryTreeNode(data) ;
    85  	}
    86  	
    87  	/**
    88  	 * Recursively calculates and returns the depth of a tree.
    89  	 * @return The depth of the tree.
    90  	 */
    91  	public function getDepth( node:BinaryTreeNode ):Number
    92  	{
    93  		var nLeft:Number  = -1 ;
    94  		var nRight:Number = -1 ;
    95  		if (node == null) 
    96  		{
    97  			node = this ;
    98  		}
    99  		if (node.left)
   100  		{
   101  			nLeft = getDepth(node.left);
   102  		}
   103  		if (node.right)
   104  		{
   105  			nRight = getDepth(node.right);
   106  		}
   107  		return (Math.max(nLeft, nRight) + 1);
   108  	}
   109  	
   110  	/**
   111  	 * Returns {@code true} if this node is left of its parent.
   112  	 * @return {@code true} if this node is left, otherwise false.
   113  	 */
   114  	public function isLeft():Boolean
   115  	{
   116  		return this == parent.left ;
   117  	}
   118  		
   119  	/**
   120  	 * Returns {@code true} if this node is right of its parent.
   121  	 * @return {@code true} if this node is right, otherwise false.
   122  	 */
   123  	public function isRight():Boolean
   124  	{
   125  		return this == parent.right;
   126  	}
   127  
   128  	/**
   129  	 * Recursively clears the tree by deleting all child nodes underneath the node the method is called on.
   130  	 */
   131  	public function reset():Void
   132  	{
   133  		if (left)
   134  		{
   135  			left.reset();
   136  		}
   137  		left = null;
   138  		if (right)
   139  		{
   140  			right.reset();
   141  		}
   142  		right = null;
   143  	}
   144  
   145  	/**
   146  	 * Sets data into the left child.
   147  	 * @param data The data value into the left child.
   148  	 */
   149  	public function setLeft( data ):Void
   150  	{
   151  		if (!left)
   152  		{
   153  			left = new BinaryTreeNode(data);
   154  			left.parent = this;
   155  		}
   156  		else
   157  		{
   158  			left.data = data;
   159  		}
   160  	}
   161  		
   162  	/**
   163  	 * Sets data into the right child.
   164  	 * @param data The data value into the right child.
   165  	 */
   166  	public function setRight( data ):Void
   167  	{
   168  		if (! right )
   169  		{
   170  			right = new BinaryTreeNode(data) ;
   171  			right.parent = this ;
   172  		}
   173  		else
   174  		{
   175  			right.data = data ;
   176  		}
   177  	}
   178  	
   179  	/**
   180  	 * Recursively counts the total number of nodes including this node.
   181  	 * @return the total number of nodes including this node.
   182  	 */
   183  	public function size():Number
   184  	{
   185  		var c:Number = 1;
   186  		if (left)
   187  		{
   188  			c += left.size();
   189  		}
   190  		if (right)
   191  		{
   192  			c += right.size();
   193  		}
   194  		return c ;
   195  	}

   199  	 * @return a string representing the source code of the object.
   200  	 */
   201  	public function toSource(indent:Number, indentor:String):String 
   202  	{
   203  		return "new andromeda.data.tree.BinaryTreeNode(" + data + ")" ;
   204  	}
   205  
   206  	/**
   207  	 * Returns the string representation of this instance.
   208  	 * @return the string representation of this instance.
   209  	 */
   210  	public function toString():String 
   211  	{
   212  		return "[BinaryTreeNode data:" + data + "]" ;
   213  	}
