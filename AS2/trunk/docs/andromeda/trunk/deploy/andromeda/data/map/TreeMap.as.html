    42  /**
    43   * This class provides a red-black tree implementation of the SortedMap interface. 
    44   * Elements in the Map will be sorted by either a user-provided IComparator object, or by the natural ordering of the keys.
    45   * <p><b>Example :</b></p>
    46   * {@code
    47   * import andromeda.data.iterator.TreeMapIterator ;
    48   * import andromeda.data.map.TreeMap ;
    49   * 
    50   * import vegas.util.comparators.StringComparator ;
    51   * 
    52   * var tree:TreeMap = new TreeMap( StringComparator.getStringComparator() ) ;
    53   * 
    54   * trace ( "put key1/value1 : " + tree.put("key1", "value1") ) ;
    55   * trace ( "put key2/value2 : " +  tree.put("key2", "value2") ) ;
    56   * //tree.put("key3", "value3") ;
    57   * 
    58   * trace( "size : " + tree.size() ) ;
    59   * trace( "tree : " + tree) ;
    60   * tree.put("Adobe", "Mountain View, CA");
    61   * tree.put("IBM", "White Plains, NY");
    62   * tree.put("Learning Tree", "Los Angeles, CA");
    63   * tree.put("Microsoft", "Redmond, WA");
    64   * tree.put("Netscape", "Mountain View, CA");
    65   * tree.put("O'Reilly", "Sebastopol, CA");
    66   * tree.put("Sun", "Mountain View, CA");
    67   * //var low = sortedMap.firstKey() ;
    68   * 
    69   * //var high = sortedMap.lastKey() ;
    70   * //trace("low : " + low);
    71   * //trace("high : " + high);
    72   * var it = tree.keySet().iterator();
    73   * for (int i = 0; i <= 6; i++) 
    74   * {
    75   *     if (i == 3)
    76   *     {
    77   *         low = it.next();
    78   *     }
    79   *     if (i == 6)
    80   *     {
    81   *         high = it.next();
    82   *     }
    83   *     else
    84   *     {
    85   *         it.next();
    86   *     }
    87   * }
    88   * trace(low);
    89   * trace(high);
    90   * trace(tree.subMap(low, high));
    91   * trace(tree.headMap(high));
    92   * trace(tree.tailMap(low));
    93   * }
    94   * @author eKameleon
    95   */

   186  	/**
   187  	 * Compares two elements by the set comparator, or by natural ordering.
   188  	 * @param o1 the first object.
   189  	 * @param o2 the second object.
   190  	 * @return <p>
   191  	 * <li>-1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
   192  	 * <li> 1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
   193  	 * <li> 0 if o1 and o2 are equal.</li>
   194  	 * </p>
   195  	 * @throws NullPointerError if o1 and o2 is null with natural ordering.
   196  	 */
   197  	public function compare( o1, o2 ):Number
   198  	{
   199  		if ( o1 == null || o2 == null )
   200  		{
   201  			throw new NullPointerError(this + " ") ;	
   202  		}
   203  		if ( _comparator == null )
   204  		{
   205  			if ( o1 instanceof IComparable )
   206  			{
   207  				return IComparable(o1).compareTo(o2) ;
   208  			}
   209  			else
   210  			{
   211  				return HashCode.compare( o1, o2 ) ;
   212  			} 
   213  		}
   214  		else
   215  		{
   216  			return _comparator.compare(o1, o2) ;	
   217  		}
   218  	}

   320  	 * Returns the TreeMapNode associated with key, or the NIL node if no such node exists in the tree.
   321  	 * @param key the key to search for.
   322  	 * @return the node where the key is found, or NIL.
   323  	 */
   324  	public function getNode( key )
   325  	{
   326  		var current:TreeMapNode = _rootEntry ;
   327  		while ( current != NIL )
   328  		{
   329  			var comparison = compare(key, current.key ) ;
   330  			if (comparison > 0)
   331  			{
   332  				current = current.right ;
   333  			}
   334  			else if (comparison < 0 )
   335  			{
   336  				current = current.left ;
   337  			}
   338  			else
   339  			{
   340  				return current ;	
   341  			}	
   342  		}
   343  		return current ;
   344  	}

   364  	 * @param key the upper bound, exclusive.
   365  	 * @return the previous node.
   366  	 */
   367  	public function highestLessThan( key ):TreeMapNode
   368  	{
   369  		if (key  == NIL)
   370  		{
   371  			return lastNode() ;	
   372  		}
   373  		
   374  		var last:TreeMapNode = NIL ;
   375  		var current:TreeMapNode = _rootEntry ;
   376  		var comparison:Number = 0 ;
   377  		
   378  		while(current != NIL)
   379  		{
   380  			last = current ;
   381  			comparison = compare(key, current.key) ;
   382  			if (comparison > 0)
   383  			{
   384  				current = current.right ;	
   385  			}	
   386  			else if (comparison < 0)
   387  			{
   388  				current = current.left ;	
   389  			}
   390  			else
   391  			{
   392  				return predecessor( last ) ;

   394  		}
   395  		return comparison <= 0 ? predecessor(last) : last ;

   488    	 * @return the prior node in sorted order.
   489    	 */
   490  	public function predecessor( node:TreeMapNode ):TreeMapNode
   491  	{
   492  		if ( node.left != NIL )
   493  		{
   494  			node = node.left ;
   495  			while( node.right != NIL )
   496  			{
   497  				node = node.right ;	
   498  			}	
   499  			return node ;
   500  		}
   501  		
   502  		var parent:TreeMapNode = node.parent ;
   503  		
   504  		// Exploit fact that NIL.left == NIL and node is non-nil.
   505  		while( node == parent.left )
   506  		{
   507  			node = parent ;
   508  			parent = node.parent ;	
   509  		}
   510  		return parent ;
   511  	}

   513  	/**

   517  	 * @return the prior mapping of the key, or {@code null} if there was none.

   519  	public function put(key, value) 
   520  	{
   521  		
   522  		var comparison:Number   = 0 ;
   523  		var current:TreeMapNode = _rootEntry ;
   524  		var parent:TreeMapNode  = NIL ;
   525  
   526  		while (current != NIL)
   527  		{
   528  			parent = current ;
   529  			comparison = compare(key, current.key) ;
   530  			if (comparison > 0)
   531  			{
   532  				current = current.right ;
   533  			}
   534  			else if (comparison <0)
   535  			{
   536  				current = current.left ;
   537  			}
   538  			else
   539  			{
   540  				current.setValue(value) ;
   541  				return current.getValue() ; 
   542  			}
   543  		}
   544  		
   545  		// creates new node
   546  		var n:TreeMapNode = new TreeMapNode(key, value, RED) ;
   547  		n.parent = parent ;
   548  		
   549  		// insert node in the tree
   550  		_modCount ++ ;
   551  		_size ++ ;
   552  		if (parent == NIL)
   553  		{
   554  			_rootEntry = n ;
   555  			return null ;	
   556  		}
   557  		if (comparison > 0)
   558  		{
   559  			parent.right = n ;	
   560  		}
   561  		else
   562  		{
   563  			parent.left = n ;
   564  		}
   565  		
   566  		_insertFixup( n ) ;
   567  		return null ;
   568  		
   569  	}

   616  	/**
   617  	 * Removes node from tree. This will increment modCount and decrement size.
   618  	 * The TreeMapNode must exist in the tree.
   619  	 * @param node the node to remove.
   620  	 */
   621  	public function removeNode( node:TreeMapNode ):Void
   622  	{
   623  		var child:TreeMapNode ;
   624  		var splice:TreeMapNode ;
   625  		
   626  		_modCount ++ ;
   627  		_size -- ;
   628  		
   629  		// Find splice, the node at the position to actually remove from the tree.
   630  		if (node.left == NIL)
   631  		{
   632  			// Node to be deleted has 0 or 1 children.
   633  			splice = node ;
   634  			child = node.right ;	
   635  		}
   636  		else if (node.right == NIL)
   637  		{
   638  			// Node to be deleted has 1 child.
   639  			splice = node ;
   640  			child = node.left ;	
   641  		}
   642  		else
   643  		{
   644  			// Node has 2 children. Splice is node's predecessor, and we swap its contents into node.
   645  			splice = node.left ;
   646  			while( splice.right != NIL)
   647  			{
   648  				splice = splice.right ;	
   649  			}
   650  			child = splice.left ;
   651  			node.key = splice.key ;
   652  			node.value = splice.value ;
   653  		}
   654  		
   655  		// Unlink splice from the tree.
   656  		var parent:TreeMapNode = splice.parent ;
   657  		if (child != NIL)
   658  		{
   659  			child.parent = parent ;	
   660  		}
   661  		if (parent == NIL)
   662  		{
   663  			// Special case for 0 or 1 node remaining.
   664  			_rootEntry = child ;
   665  			return ;	
   666  		}
   667  		if (splice == parent.left)
   668  		{

   801  	/**

   803  	 * @param node the child of the node just deleted, possibly NIL.
   804  	 * @param parent parent the parent of the node just deleted, never NIL.
   805  	 */
   806  	private function _deleteFixup( node:TreeMapNode , parent:TreeMapNode ):Void
   807  	{
   808  		// TODO if (parent == nil) throw new InternalError();
   809  
   810  	    // If a black node has been removed, we need to rebalance to avoid violating the "same number of black nodes on any path" rule.
   811  	    // If node is red, we can simply recolor it black and all is well.
   812  	    
   813  	    var sibling:TreeMapNode ;
   814  	    
   815  	    while( node != _rootEntry && node.color == BLACK )
   816  	    {
   817  	    	
   818  	    	if (node == parent.left)
   819  	    	{
   820  	    		// Rebalance left side.
   821  	    		sibling = parent.right ;
   822  	    		
   823  	    		// TODO if (sibling == NIL) throw new InternalError() ;
   824  	    		
   825  	    		if (sibling.color == RED)
   826  	    		{
   827  	    			// Case 1 : Sibling is red.
   828  	    			// Recolor sibling and parent, and rotate parent left. 	
   829  	    			sibling.color = BLACK ;
   830  	    			parent.color = RED ;
   831  	    			_rotateLeft(parent) ;
   832  	    			sibling = parent.right ;
   833  	    		}
   834  	    		
   835  	    		if ( sibling.left.color == BLACK && sibling.right.color == BLACK )
   836  	    		{
   837  	    			// Case 2 : Sibling has no red children.
   838  	    			// Recolor sibling and move top parent.
   839  	    			sibling.color = RED ;
   840  	    			node = parent ;
   841  	    			parent = parent.parent ;
   842  	    		}
   843  	    		else
   844  	    		{
   845  	    			if (sibling.right.color == BLACK)
   846  	    			{
   847  	    				// Case 3: Sibling has red left child.
   848  	    				// Recolor sibling and left child, rotate sibling right.
   849  	    				sibling.left.color = BLACK ;
   850  	    				sibling.color = RED ;
   851  	    				_rotateRight(sibling) ;
   852  	    				sibling = parent.right ;
   853  	    			}
   854  	    			// Case 4: Sibling has red right child. 
   855  	    			// Recolor sibling, right child, and parent, and rotate parent left.
   856  	    			sibling.color = parent.color ;
   857  	    			parent.color = BLACK ;
   858  	    			sibling.right.color = BLACK ;
   859  	    			_rotateLeft( parent ) ;
   860  	    			node = _rootEntry ;
   861  	    		}
   862  	    		
   863  	    	}
   864  	    	else
   865  	    	{
   866  	    		
   867  	    		// Symmetric "mirror" of left-side case.
   868  	    		sibling = parent.left ;
   869  	    		
   870  	    		// TODO if (sibling == NIL) throw new InternalError() ;
   871  	    		
   872  	    		if (sibling.color == RED)
   873  	    		{
   874  	    			// Case 1 : Sibling is red.
   875  	    			// Recolor sibling and parent, and rotate parent right.	
   876  	    			sibling.color = BLACK ;
   877  	    			parent.color = RED ;
   878  	    			_rotateRight(parent) ;
   879  	    			sibling = parent.left ;
   880  	    		}
   881  	    		
   882  	    		if ( sibling.right.color == BLACK && sibling.left.color == BLACK)
   883  	    		{
   884  	    			// Case 2: Sibling has no red children.
   885  	    			// Recolor sibling, and move to parent.
   886  	    			sibling.color = RED ;
   887  	    			node = parent ;
   888  	    			parent = parent.parent ;
   889  	    		}
   890  	    		else
   891  	    		{
   892  	    			if (sibling.left.color == BLACK)
   893  	    			{
   894  	    				// Case 3: Sibling has red right child.
   895  	    				// Recolor sibling and right child, rotate sibling left.
   896                      	sibling.right.color = BLACK;
   897                      	sibling.color = RED;
   898                      	_rotateLeft(sibling);
   899                      	sibling = parent.left;
   900  	    			}

   915  	/**
   916  	 * Construct a perfectly balanced tree consisting of n "blank" nodes. 
   917  	 * This permits a tree to be generated from pre-sorted input in linear time.
   918  	 * @param count the number of blank nodes, non-negative.
   919  	 */
   920  	private function _fabricateTree ( count:Number ):Void
   921  	{
   922  		
   923  		if (count == 0)
   924  		{
   925  			_rootEntry = NIL ;
   926  			_size = 0 ;	
   927  			return ;
   928  		}
   929  		
   930  		// Make the root node.
   931  		_rootEntry = new TreeMapNode(null, null, BLACK) ;
   932  		_size = count ;
   933  		
   934  		var row:TreeMapNode = _rootEntry ;
   935  		
   936  		var left:TreeMapNode ;
   937  		var last:TreeMapNode ;
   938  		var next:TreeMapNode ;
   939  		var parent:TreeMapNode ;
   940  		var right:TreeMapNode ;
   941  		
   942  		// Fill each row that is completely full of nodes.
   943  		for (var rowSize:Number = 2 ; (rowSize + rowSize) <= count ; rowSize <<=1 )
   944  		{
   945  			parent = row ;
   946  			last   = null ;
   947  			for ( var i:Number = 0 ; i < rowSize ; i += 2 )
   948  			{
   949  				left  = new TreeMapNode(null, null, BLACK) ;
   950  				right = new TreeMapNode(null, null, BLACK) ;
   951  				
   952  				left.parent = parent ;
   953  				left.right  = right ;
   954  				
   955  				right.parent = parent ;
   956  				parent.left  = left ;
   957  				
   958  				next = parent.right ; 
   959  				parent.right = right ;
   960  				
   961  				parent = next ;
   962  				
   963  				if (last != null)
   964  				{
   965  					last.right = left ;	
   966  				}
   967  				
   968  				last = right ;
   969  			}
   970  			row = row.left ;
   971  		}
   972  		
   973  		// Now do the partial final row in red.
   974  		var overflow:Number = count - rowSize ;
   975  		parent = row ;
   976  		var j:Number ;
   977  		for ( j = 0 ; j < overflow ; j += 2 )
   978  		{
   979  			
   980  			left  = new TreeMapNode(null, null, RED) ;
   981  			right = new TreeMapNode(null, null, RED) ;
   982  			
   983  			left.parent = parent ;
   984  			right.parent = parent ;
   985  			
   986  			parent.left = left ;
   987  			next = parent.right ;
   988  			
   989  			parent.right = right ;
   990  			parent = next ;
   991  			
   992  		}
   993  		
   994  		// Add a lone left node if necessary.
   995     	 	if ( (j - overflow) == 0 )
   996        	{
   997          	
   998          	left = new TreeMapNode(null, null, RED) ;
   999          	left.parent = parent ;
  1000          	parent.left = left ;
  1001          	parent = parent.right ;
  1002          	left.parent.right = NIL ;
  1003  		}
  1004  		
  1005  		// Unlink the remaining nodes of the previous row.
  1006      	while (parent != NIL)
  1007        	{
  1008          	next = parent.right ;
  1009          	parent.right = NIL ;
  1010          	parent = next;
  1011        	}
  1012  		
  1013  	}
