    24  import andromeda.data.map.TreeMap;
    25  import andromeda.data.map.TreeMapNode;
    26  
    27  import vegas.core.CoreObject;
    28  import vegas.data.iterator.Iterator;
    29  import vegas.errors.ConcurrentModificationError;
    30  import vegas.errors.IllegalStateError;
    31  import vegas.errors.NoSuchElementError;
    32  import vegas.errors.NullPointerError;
    33  
    34  /**
    35   * Iterate over TreeMap's entries. 
    36   * This implementation is parameterized  to give a sequential view of keys, values, or entries.
    37   * @author eKameleon
    38   */
    39  class andromeda.data.iterator.TreeMapIterator extends CoreObject implements Iterator 
    40  {
    41  	
    42  	/**
    43  	 * Creates a new TreeMapIterator instance.
    44  	 * @param tree the TreeMap reference of this iterator.
    45  	 * @param type the type of this iterator.
    46  	 */
    47  	public function TreeMapIterator( tree:TreeMap , type:Number , first:TreeMapNode, max:TreeMapNode ) 
    48  	{
    49  		
    50  		if ( tree == null )
    51  		{
    52  			throw new NullPointerError(this + " constructor failed, the TreeMap reference passed in argument not must be 'null' or 'undefined'.") ;	
    53  		}
    54  		_tree = tree ;
    55  		_knownMod = _tree.getModCount() ;
    56  		_type = type ;
    57  		_max  = max   || TreeMap.NIL ;
    58  		_next = first || tree.firstNode();
    59       	
    60  	}
    61  
    62  	/**
    63  	 * The iterator types for an ENTRIES value.
    64  	 */
    65  	public static var ENTRIES:Number = 2 ;
    66  
    67  	/**
    68  	 * The iterator types for an KEYS value.
    69  	 */
    70  	public static var KEYS:Number = 0 ;
    71  
    72  	/**
    73  	 * The iterator types for an VALUES value.
    74  	 */
    75  	public static var VALUES:Number = 1 ;
    76  
    77      /**
    78       * Returns {@code true} if the Iterator has more elements.
    79       * @return {@code true} if there are more elements.
    80       */
    81  	public function hasNext() : Boolean 
    82  	{
    83  		return _next != _max ;
    84  	}
    85  
    86  	public function key() 
    87  	{
    88  		// throw an error.
    89  	}
    90  
    91  	/**
    92  	 * Returns the next element in the Iterator's sequential view.
    93  	 * @return the next element.
    94  	 * @throws ConcurrentModificationError if the TreeMap was modified.
    95  	 * @throws NoSuchElementError if there is none.
    96  	 */
    97  	public function next() 
    98  	{
    99        	if (_knownMod != _tree.getModCount())
   100        	{
   101          	throw new ConcurrentModificationError(this + " key() method failed, the TreeMap was modified.");
   102        	}
   103  		if ( _next == _max )
   104  		{
   105          	throw new NoSuchElementError(this + " key() method failed, if there is none");
   106  		}
   107        	_last = _next ;
   108        	_next = _tree.successor(_last);
   109        	
   110        	// trace(this + " next : " + _next) ;
   111  		
   112  		if ( _type == VALUES )
   113  		{
   114          	return _last.value ;
   115  		}
   116  		else if ( _type == KEYS )
   117  		{
   118          	return _last.key;
   119  		}
   120  		return _last ;
   121  	}
   122      /**
   123       * Removes from the backing TreeMap the last element which was fetched with the {@code next()} method.
   124       * @throws ConcurrentModificationError if the TreeMap was modified.
   125       * @throws IllegalStateError if called when there is no last element.
   126       */
   127  	public function remove() 
   128  	{
   129        	if (_last == null)
   130        	{
   131          	throw new IllegalStateError(this + " remove() method failed if the TreeMap was modified." );
   132        	}
   133  		if ( _knownMod != _tree.getModCount() )
   134  		{
   135  			throw new ConcurrentModificationError( this + " remove() method failed if called when there is no last element." );
   136  		}
   137  		_tree.removeNode( _last ) ;
   138        	_last = null;
   139        	_knownMod ++ ;
   140  	}
   141  
   142  	public function reset() : Void 
   143  	{
   144  		// throw an error.
   145  	}
   146  
   147  	public function seek(n : Number) : Void 
   148  	{
   149  		// throw an error.	
   150  	}
   151  	
   152      /** 
   153       * The number of modifications to the backing Map that we know about.
   154       */
   155      private var _knownMod:Number ;
   156  
   157  	/**
   158  	 * The last Entry returned by a next() call.
   159  	 */
   160  	private var _last:TreeMapNode ;
   161  
   162      /**
   163       * The last node visible to this iterator. This is used when iterating on a SubMap.
   164       */
   165      private var _max:TreeMapNode ;
   166  
   167  	/**
   168  	 * The next entry that should be returned by next().
   169  	 */
   170  	private var _next:TreeMapNode ;
   171  
   172  	/**
   173  	 * The TreeMap reference of this SubTreeMap.
   174  	 */
   175  	private var _tree:TreeMap ;
   176  
   177  	/**
   178  	 * The type of this Iterator : KEYS or VALUES or ENTRIES.
   179  	 */
   180  	private var _type:Number ;
   181  
   182  }
