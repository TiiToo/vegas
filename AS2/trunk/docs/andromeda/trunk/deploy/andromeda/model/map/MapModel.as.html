    35  /**
    36   * This model use an internal Map to register value objects with a specific key.
    37   * @author eKameleon
    38   */
    39  class andromeda.model.map.MapModel extends AbstractModelObject implements Iterable
    40  {
    41  
    42  	/**
    43  	 * Creates a new MapModel instance.
    44  	 * @param id the id of this model.
    45  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    46  	 * @param sChannel (optional) the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    47  	 */	
    48  	function MapModel( id , bGlobal:Boolean , sChannel:String ) 
    49  	{
    50  		super( id , bGlobal, sChannel ) ;
    51  		_map = initializeMap() ;
    52  	}
    53  
    54  	/**
    55  	 * Inserts a value object in the model.
    56  	 * @throws IllegalArgumentError if the argument of this method is 'null' or 'undefined'. 
    57  	 * @throws Warning if the {@code IValueObject} passed in argument is already register in the model.
    58  	 */
    59  	public function addVO( vo:IValueObject ):Void
    60  	{
    61  		if (vo == null)
    62  		{
    63  			throw new IllegalArgumentError( this + " addVO method failed, the argument passed in argument not must be 'null' or 'undefined'.") ;	
    64  		}
    65  		validate(vo) ;
    66  		if ( !_map.containsKey( vo.getID() ) )
    67  		{
    68  			_map.put( vo.getID() , vo ) ;
    69  			notifyAdd(vo) ;
    70  		}
    71  		else
    72  		{
    73  			throw new Warning( this + " addVO method failed, the IValueObject passed in argument already register in the model, you must remove this 'id' key before add a new value object.") ;	
    74  		}
    75  	}
    76  	
    77  	/**
    78  	 * Removes all value objects in the model.
    79  	 */
    80  	public function clear():Void
    81  	{
    82  		_map.clear() ;
    83  		super.clear() ;
    84  	}
    85  
    86  	/**
    87  	 * Returns {@code true} if the model contains the specified IValueObject.
    88  	 * @return {@code true} if the model contains the specified IValueObject.
    89  	 */
    90  	public function contains( vo:IValueObject ):Boolean
    91  	{
    92  		return _map.containsValue( vo ) ;
    93  	}
    94  	
    95  	/**
    96  	 * Returns {@code true} if the model contains the specified id key in argument.
    97  	 * @return {@code true} if the model contains the specified id key in argument
    98  	 */
    99  	public function containsByID( id ):Boolean
   100  	{
   101  		return _map.containsKey( id ) ;
   102  	}
   103  
   104  	/**
   105  	 * Returns {@code true} if the model contains the specified attribute value.
   106  	 * @return {@code true} if the model contains the specified id key in argument
   107  	 */
   108  	public function containsByProperty( propName:String , value ):Boolean
   109  	{
   110  		if ( propName == null )
   111  		{
   112  			return false ;
   113  		}
   114  		var datas:Array = _map.getValues() ;
   115  		var size:Number = datas.length ;
   116  		if (size > 0)
   117  		{
   118  			while ( --size > -1 )
   119  			{
   120  				if ( datas[size][propName] == value )
   121  				{
   122  					return true ;
   123  				}
   124  			}
   125  			return false ;
   126  		}
   127  		else
   128  		{
   129  			return false ;
   130  		}
   131  	}

   158  	}

   167  	}

   171  	 * @return the IValueObject defined by the id passed in argument.

   175  		return _map.get( id ) ;

   178  	/**
   179  	 * This method is invoqued in the constructor of the class to initialize all events.
   180  	 */
   181  	/*override*/ public function initEvent():Void
   182  	{
   183  		super.initEvent() ;
   184  		_eAdd    = createNewModelObjectEvent( ModelObjectEvent.ADD_VO ) ;
   185  		_eRemove = createNewModelObjectEvent( ModelObjectEvent.REMOVE_VO ) ; 
   186  		_eUpdate = createNewModelObjectEvent( ModelObjectEvent.UPDATE_VO ) ;
   187  	}
   188  
   189  	/**
   190  	 * Initialize the internal Map instance in the constructor of the class.
   191  	 * You can overrides this method if you want change the default HashMap use in this model.
   192  	 */
   193  	public function initializeMap():Map
   194  	{
   195  		return new HashMap() ;	
   196  	}

   198  	/**
   199  	 * Returns {@code true} if the model is empty.
   200  	 * @return {@code true} if the model is empty.
   201  	 */
   202  	public function isEmpty():Boolean 
   203  	{
   204  		return _map.isEmpty() ;	
   205  	}

   208  	 * Returns the iterator of this model.
   209  	 * @return the iterator of this model.
   210  	 */
   211  	public function iterator():Iterator
   212  	{
   213  		return _map.iterator() ;	
   214  	}

   216      /**
   217       * Notify a {@code ModelObjectEvent} when a {@code IValueObject} is inserted in the model.
   218       */ 
   219      public function notifyAdd( vo:IValueObject ):Void
   220      {
   221          if ( isLocked() )
   222          {
   223              return ;
   224          }
   225          _eAdd.setVO( vo ) ;
   226          dispatchEvent( _eAdd  ) ;
   227      }
   228  
   229      /**
   230       * Notify a {@code ModelObjectEvent} when a {@code IValueObject} is removed in the model.
   231       */ 
   232      public function notifyRemove( vo:IValueObject ):Void
   233      {
   234          if ( isLocked() )
   235          {
   236              return ;
   237          }
   238          _eRemove.setVO( vo ) ;
   239          dispatchEvent( _eRemove  ) ;
   240      }
   241  
   242      /**
   243       * Notify a {@code ModelObjectEvent} when a {@code IValueObject} is updated in the model.
   244       */ 
   245  	public function notifyUpdate( vo:IValueObject ):Void
   246  	{
   247          if ( isLocked() )
   248          {
   249              return ;
   250          }
   251  		_eUpdate.setVO( vo ) ;
   252  		dispatchEvent( _eUpdate  ) ;
   253  	}
   254  
   255  	/**
   256  	 * Removes a value object in the model.
   257  	 */
   258  	public function removeVO( vo:IValueObject ):Void
   259  	{
   260  		if (vo == null)
   261  		{
   262  			throw new IllegalArgumentError( this + " removeVO method failed, the IValueObject passed in argument not must be 'null' or 'undefined'.") ;	
   263  		}
   264  		validate(vo) ;
   265  		if ( _map.containsKey( vo.getID() ) )
   266  		{
   267  			_map.remove( vo.getID() ) ;
   268  			notifyRemove( vo ) ;
   269  		}
   270  		else
   271  		{
   272  			throw new Warning( this + " removeVO method failed, the id passed in argument allready register in the model, you must remove this 'id' key before add a noew value object.") ;	
   273  		}
   274  	}

   277  	 * Sets the event name use in the {@code addVO} method.

   279  	public function setEventTypeADD( type:String ):Void
   280  	{

   282  	}
   283

   286  	 */
   287  	public function setEventTypeREMOVE( type:String ):Void
   288  	{
   289  		_eRemove.setType( type ) ;
   290  	}

   293  	 * Sets the event name use in the {@code addVO} method.

   295  	public function setEventTypeUPDATE( type:String ):Void
   296  	{

   298  	}

   302  	 */
   303  	public function setMap( m:Map ):Void
   304  	{
   305  		_map = m || initializeMap() ;	
   306  	}

   308  	/**
   309  	 * Returns the number of IValueObject in this model.
   310  	 * @return the number of IValueObject in this model.
   311  	 */
   312  	public function size():Number
   313  	{
   314  		return _map.size() ;
   315  	}
   316  
   317  	/**
   318  	 * Update a value object in the model.
   319  	 * @throws Warning if the value object passed-in argument don't exist.
   320  	 */
   321  	public function updateVO( vo:IValueObject ):Void
   322  	{
   323  		if ( _map.containsKey( vo.getID() ) )
   324  		{
   325  			_map.put( vo.getID() , vo ) ;
   326  			notifyUpdate(vo) ;
   327  		}
   328  		else
   329  		{
   330  			throw new Warning( this + " updateVO method failed, the value object passed in argument don't exist in the model.") ;
   331  		}
   332  	}

   342  	private var _eRemove:ModelObjectEvent ;

   347  	private var _eUpdate:ModelObjectEvent ;
