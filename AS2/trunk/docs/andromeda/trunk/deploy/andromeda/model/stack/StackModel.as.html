    24  import andromeda.events.ModelObjectEvent;
    25  import andromeda.model.AbstractModelObject;
    26  import andromeda.model.IValueObject;
    27  
    28  import vegas.data.iterator.Iterator;
    29  import vegas.data.Stack;
    30  import vegas.data.stack.SimpleStack;
    31  import vegas.errors.IllegalArgumentError;
    32  
    33  /**
    34   * This model use an internal {@code Stack} to register value objects.
    35   * @author eKameleon
    36   */
    37  class andromeda.model.stack.StackModel extends AbstractModelObject 
    38  {
    39  	
    40  	/**
    41  	 * Creates a new StackModel instance.
    42  	 * @param id the id of this model.
    43  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    44  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    45  	 */
    46  	function StackModel( id , bGlobal:Boolean , sChannel:String ) 
    47  	{
    48  		super(id, bGlobal, sChannel);
    49  		_stack = initializeStack() ;
    50  	}
    51  
    52  	/**
    53  	 * Default event type when the pop method is invoqued.
    54  	 */
    55  	public static var POP_VO:String = "onPopVO" ;
    56  	
    57  	/**
    58  	 * Default event type when the push method is invoqued.
    59  	 */
    60  	public static var PUSH_VO:String = "onPushVO" ;
    61  
    62  	/**
    63  	 * Removes all value objects in the model.
    64  	 */
    65  	public function clear():Void
    66  	{
    67  		_stack.clear() ;
    68  		super.clear() ;
    69  	}
    70  
    71  	/**
    72  	 * Returns the event name use in the {@code pop} method.
    73  	 * @return the event name use in the {@code pop} method.
    74  	 */
    75  	public function getEventTypePOP():String
    76  	{
    77  		return _ePop.getType() ;
    78  	}
    79  	
    80  	/**
    81  	 * Returns the event name use in the {@code push} method.
    82  	 * @return the event name use in the {@code push} method.
    83  	 */
    84  	public function getEventTypePUSH():String
    85  	{
    86  		return _ePush.getType() ;
    87  	}
    88  
    89  	/**
    90  	 * Returns the internal {@code Stack} reference of this model.
    91  	 * @return the internal {@code Stack} reference of this model.
    92  	 */
    93  	public function getStack():Stack
    94  	{
    95  		return _stack ;	
    96  	}
    97  
    98  	/**
    99  	 * This method is invoqued in the constructor of the class to initialize all events.
   100  	 */
   101  	/*override*/ public function initEvent():Void
   102  	{
   103  		super.initEvent() ;
   104  		_ePop  = createNewModelObjectEvent( StackModel.POP_VO ) ;
   105  		_ePush = createNewModelObjectEvent( StackModel.PUSH_VO ) ; 
   106  	}
   107  
   108  	/**
   109  	 * Initialize the internal Stack instance in the constructor of the class.
   110  	 * You can overrides this method if you want change the default {@code SimpleStack} use in this model.
   111  	 */
   112  	public function initializeStack():Stack
   113  	{
   114  		return new SimpleStack() ;	
   115  	}
   116  
   117  	/**
   118  	 * Returns {@code true} if this model is empty.
   119  	 * @return {@code true} if this model is empty.
   120  	 */
   121  	public function isEmpty():Boolean 
   122  	{ 
   123  		return _stack.isEmpty() ;
   124  	}
   125  
   126  	/**
   127  	 * Returns the iterator of this model.
   128  	 * @return the iterator of this model.
   129  	 */
   130  	public function iterator():Iterator
   131  	{
   132  		return _stack.iterator() ;	
   133  	}
   134  
   135      /**
   136       * Notify a {@code ModelObjectEvent} when a {@code IValueObject} is poped in the model.
   137       */ 
   138      public function notifyPop( vo:IValueObject ):Void
   139      {
   140          if ( isLocked() )
   141          {
   142              return ;
   143          }
   144  		_ePop.setVO( vo ) ;
   145  		dispatchEvent( _ePop ) ;
   146      }
   147  
   148      /**
   149       * Notify a {@code ModelObjectEvent} when a {@code IValueObject} is pushed in the model.
   150       */ 
   151      public function notifyPush( vo:IValueObject ):Void
   152      {
   153          if ( isLocked() )
   154          {
   155              return ;
   156          }
   157  		_ePush.setVO( vo ) ;
   158  		dispatchEvent( _ePush ) ;
   159      }
   160  
   161  	/**
   162  	 * Removes and returns the lastly pushed value object in the model.
   163  	 * @return the lastly pushed value object in the model.
   164  	 */
   165  	public function pop():IValueObject
   166  	{
   167  		var vo:IValueObject = _stack.pop() ;
   168  		_ePop.setVO( vo ) ;
   169  		dispatchEvent( _ePop ) ;
   170  		return vo ;
   171  	}
   172  
   173  	/**
   174  	 * Pushes the passed-in value object to this stack.
   175  	 * @param vo The IValueObject to push.
   176  	 */
   177  	public function push( vo:IValueObject ):Void
   178  	{
   179  		if (vo == null)
   180  		{
   181  			throw new IllegalArgumentError( this + " enqueue method failed, the IValueObject passed in argument not must be 'null' or 'undefined'.") ;	
   182  		}
   183  		validate(vo) ;
   184  		_stack.push( vo ) ;
   185  		notifyPush( vo ) ;
   186  	}
   187  
   188  	/**
   189  	 * Sets the event name use in the {@code pop} method.
   190  	 */
   191  	public function setEventTypePOP( type:String ):Void
   192  	{
   193  		_ePop.setType( type ) ;
   194  	}
   195  	
   196  	/**
   197  	 * Sets the event name use in the {@code push} method.
   198  	 */
   199  	public function setEventTypePUSH( type:String ):Void
   200  	{
   201  		_ePush.setType( type ) ;
   202  	}
   203  
   204  	/**
   205  	 * Sets the internal Stack of this model. 
   206  	 * By default the initializeStack() method is used if the passed-in argument is null.
   207  	 */
   208  	public function setStack( s:Stack ):Void
   209  	{
   210  		_stack = s || initializeStack() ;	
   211  	}
   212  
   213  	/**
   214  	 * Returns the number of IValueObject in this model.
   215  	 * @return the number of IValueObject in this model.
   216  	 */
   217  	public function size():Number
   218  	{
   219  		return _stack.size() ;
   220  	}
   221  
   222  	/**
   223  	 * Returns the {@code Array} representation of this object.
   224  	 * @return the {@code Array} representation of this object.
   225  	 */
   226  	public function toArray():Array
   227  	{
   228  		return _stack.toArray() ;	
   229  	}
   230  
   231  	/**
   232  	 * The internal ModelObjectEvent use in the {@code pop} method.
   233  	 */
   234  	private var _ePop:ModelObjectEvent ;
   235  
   236  	/**
   237  	 * The internal ModelObjectEvent use in the {@code push} method.
   238  	 */
   239  	private var _ePush:ModelObjectEvent ;
   240  
   241  	/**
   242  	 * The internal {@code Stack} reference.
   243  	 */
   244  	private var _stack:Stack ;
   245  
   246  }
