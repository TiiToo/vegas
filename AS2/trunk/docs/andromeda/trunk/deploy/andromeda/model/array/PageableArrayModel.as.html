    24  import andromeda.events.ModelObjectEvent;
    25  import andromeda.model.AbstractModelObject;
    26  import andromeda.vo.IValueObject;
    27  
    28  import vegas.core.IRunnable;
    29  import vegas.data.iterator.Iterator;
    30  import vegas.data.iterator.PageByPageIterator;
    31  import vegas.events.ArrayEvent;
    32  import vegas.events.Event;

    34  /**
    35   * Defines an array model with a 'page by page' iterator.
    36   * @author eKameleon
    37   */
    38  class andromeda.model.array.PageableArrayModel extends AbstractModelObject implements IRunnable
    39  {
    40  	
    41  	/**
    42  	 * Creates a new PageableArrayModel instance.
    43  	 * @param id the id of this model.
    44  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    45  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    46  	 */
    47  	function PageableArrayModel( id , bGlobal:Boolean , sChannel:String ) 
    48  	{
    49  		super( id , bGlobal, sChannel) ;
    50  		_a = new Array() ;
    51  	}
    52  
    53  	/**
    54  	 * Inserts all IValueObject in the array passed in argument.
    55  	 * @param datas The array of all value objects to insert in the model.
    56  	 * @param noClear (optional) If this argument is {@code true} the clear method isn't called when this process begin.
    57  	 * @param noRefresh (optional) If this argument is {@code true} the refresh method isn't called when this process is finish.
    58  	 */
    59  	public function addAllVO( datas:Array , noClear:Boolean , noRefresh:Boolean ):Void
    60  	{
    61  		if ( !noClear )
    62  		{
    63  			clear() ;
    64  		}
    65  		var len:Number = datas.length ;
    66  		for (var i:Number = 0 ; i<len ; i++)
    67  		{
    68  			var vo:IValueObject = IValueObject( datas[i] ) ;
    69  			if ( supports( vo ) )
    70  			{
    71  				_a.push( vo ) ;
    72  			}	
    73  		}
    74  		if ( noRefresh == true )
    75  		{
    76  			return ;	
    77  		}
    78  		refresh() ;
    79  	}
    80  
    81  	/**
    82  	 * Clear the model.
    83  	 */	
    84  	public function clear():Void
    85  	{
    86  		_a = []  ;
    87  		super.clear() ;
    88  	}
    89  
    90  	/**
    91  	 * Returns the count of the IValueObject in a page of this model.
    92  	 * @return the count of the IValueObject in a page of this model.
    93  	 */
    94  	public function getCountVO():Number
    95  	{
    96  		return _voCount ;
    97  	}
    98  
    99  	/**
   100  	 * Returns the event name use in the {@code addVO} method.
   101  	 * @return the event name use in the {@code addVO} method.
   102  	 */
   103  	public function getEventTypeUPDATE():String
   104  	{
   105  		return _sUpdateType || ModelObjectEvent.UPDATE_VO ;
   106  	}
   107  
   108  	/**
   109  	 * Returns the current page selected in the model.
   110  	 * @return the current page selected in the model.
   111  	 */
   112  	public function getCurrentPage():Number
   113  	{
   114  		return _itPage.currentPage() ;	
   115  	}
   116  
   117  	/**
   118  	 * Returns the numbers of page with the current model.
   119  	 * @return the numbers of page with the current model.
   120  	 */
   121  	public function getPageCount():Number
   122  	{
   123  		return _itPage.pageCount() ;	
   124  	}
   125  
   126  	/**
   127  	 * Returns {@code true} if the list has a next page.
   128  	 * @return {@code true} if the list has a next page.
   129  	 */
   130  	public function hasNext():Boolean
   131  	{
   132  		return 	_itPage.hasNext() ;
   133  	}
   134  	
   135  	/**
   136  	 * Returns {@code true} if the list has a previous page.
   137  	 * @return {@code true} if the list has a previous page.
   138  	 */
   139  	public function hasPrevious():Boolean
   140  	{
   141  		return _itPage.hasPrevious() ;
   142  	}
   143  	
   144  	/**
   145  	 * Returns a PageByPageIterator of this model.
   146  	 * @return a PageByPageIterator of this model.
   147  	 */
   148  	public function iterator():Iterator
   149  	{
   150  		return new PageByPageIterator( _voCount , _a ) ;	
   151  	}
   152  	
   153  	/**
   154  	 * Seek the iterator in the last page of this object.
   155  	 */
   156  	public function lastPage():Void
   157  	{
   158  		if ( _itPage != null )
   159  		{
   160  			_itPage.lastPage() ;
   161  		}
   162  	}
   163  
   164      /**
   165       * Notify an {@code Event} when a {@code IValueObject} is inserted in the model. 
   166       * If the model countVO value is > 1 notify an ArrayEvent else if the coutVO value is 1 notify a ModelObjectEvent. 
   167       */ 
   168      public function notifyUpdate( value )
   169      {
   170          if ( isLocked() == false )
   171          {
   172          	if ( getCountVO() > 1 )
   173          	{
   174          		_eUpdate = new ArrayEvent( getEventTypeUPDATE() ) ;
   175          		ArrayEvent(_eUpdate).setArray( value ) ;
   176          	}
   177          	else
   178          	{
   179          		_eUpdate = new ModelObjectEvent( getEventTypeUPDATE() , this , value ) ;
   180          	}
   181  			dispatchEvent( _eUpdate  ) ;    
   182          }
   183          return value ;
   184      }
   185  
   186  	/**
   187  	 * Show the next page of the model.
   188  	 */
   189  	public function next()
   190  	{
   191  		if ( hasNext() )
   192  		{
   193  			return notifyUpdate( _itPage.next() ) ;
   194  		}
   195  	}
   196  
   197  	/**
   198  	 * Show in the previous page in the list or previous screen.
   199  	 */
   200  	public function previous()
   201  	{
   202  		if ( hasPrevious() )
   203  		{
   204  			return notifyUpdate( _itPage.previous() ) ;
   205  		}
   206  	}
   207  
   208  	/**
   209  	 * Refresh the model.
   210  	 */
   211  	public function refresh():Void
   212  	{
   213  		if (size() > 0)
   214  		{
   215  			_itPage = new PageByPageIterator ( _voCount , _a ) ;
   216  		}
   217  		else
   218  		{
   219  			_itPage = null ;	
   220  		}
   221  		run() ;
   222  	}
   223  	
   224  	/**
   225  	 * Resets the key pointer of the iterator.
   226  	 */
   227  	public function reset():Void 
   228  	{
   229  		if ( _itPage != null )
   230  		{
   231  			_itPage.reset() ;
   232  		}
   233  	}
   234  
   235  	/**
   236  	 * Run the model when is initialize.
   237  	 */
   238  	public function run():Void
   239  	{
   240  		next() ;
   241  	}
   242  
   243  	/**
   244  	 * Set the count of the IValueObject in a page of this model. 
   245  	 * This value must be >=1.
   246  	 */
   247  	public function setCountVO( n:Number ):Void
   248  	{
   249  		_voCount = n > 1 ? n : 1 ;
   250  		refresh() ;
   251  	}
   252  
   253  	/**
   254  	 * Sets the event name use in the {@code addVO} method.
   255  	 */
   256  	public function setEventTypeUPDATE( type:String ):Void
   257  	{
   258  		_sUpdateType = type ;
   259  	}
   260  
   261  	/**
   262  	 * Returns the number of elements in the model.
   263  	 * @return the number of elements in the model.
   264  	 */
   265  	public function size():Number
   266  	{
   267  		return _a.length ;	
   268  	}
   269  	
   270  	/**
   271  	 * Returns the array representation of all vos in this model.
   272  	 * @return the array representation of all vos in this model.
   273  	 */
   274  	public function toArray():Array
   275  	{
   276  		return [].contact(_a) ;	
   277  	}
   278  
   279  	/**
   280  	 * The internal array of all elements.
   281  	 */
   282  	private var _a:Array ; 
   283  
   284  	/**
   285  	 * The internal ArrayEvent when the update event type is use.
   286  	 */
   287  	private var _eUpdate:Event ;
   288  
   289  	/**
   290  	 * The current PageByPageIterator instance.
   291  	 */
   292  	private var _itPage:PageByPageIterator ;
   293  	
   294  	/**
   295  	 * The type of the update event.
   296  	 */
   297  	private var _sUpdateType:String ;
   298  	
   299  	/**
   300  	 * The numbers of items in the model.
   301  	 */
   302  	private var _voCount:Number = 1 ;
   303  
   304  
   305  }
