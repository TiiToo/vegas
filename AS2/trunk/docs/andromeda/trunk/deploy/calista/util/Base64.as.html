    24  /**
    25   * The Base64 algorithm encoding tool class.
    26   * @author eKameleon
    27   */
    28  class calista.util.Base64 
    29  {
    30  
    31  	/**
    32  	 * Encodes a base64 string.
    33   	 * <p><b>Example :</b></p>
    34   	 * {@code
    35   	 * import calista.util.Base64 ;
    36   	 * var source:String = "hello world with a base 64 algorithm" ;
    37   	 * 
    38   	 * var encode:String = Base64.encode( source ) ;
    39   	 * trace("encode : " + encode) ;
    40   	 * }
    41  	 */
    42  	public static function encode( str:String ):String
    43  	{
    44  		str = new String(str) ; // optimize
    45  		var size:Number   = str.length ;
    46  		var i:Number      = 0 ;
    47  		var output:String = "" ;
    48  		var chr1:Number, chr2:Number, chr3:Number;
    49  		var enc1:Number, enc2:Number, enc3:Number, enc4:Number;
    50  		while (i < size) 
    51  		{
    52  			chr1 = str.charCodeAt(i++) ;
    53  			chr2 = str.charCodeAt(i++) ;
    54  			chr3 = str.charCodeAt(i++) ;
    55  			enc1 = chr1 >> 2;
    56  			enc2 = ((chr1 & 3) << 4) | (chr2 >> 4);
    57  			enc3 = ((chr2 & 15) << 2) | (chr3 >> 6);
    58  			enc4 = chr3 & 63;
    59  			if(isNaN(chr2)) 
    60  			{
    61  				enc3 = enc4 = 64;
    62  			}
    63  			else if(isNaN(chr3)) 
    64  			{
    65  				enc4 = 64;
    66  			}
    67  			output += base64chars.charAt(enc1) + base64chars.charAt(enc2) ;
    68  			output += base64chars.charAt(enc3) + base64chars.charAt(enc4) ;
    69  		}
    70  		return output;
    71  	}
    72  
    73  	/**
    74  	 * Decodes a base64 string.
    75   	 * <p><b>Example :</b></p>
    76   	 * {@code
    77   	 * import calista.util.Base64 ;
    78   	 * 
    79   	 * var source:String = "hello world with a base 64 algorithm" ;
    80   	 * 
    81   	 * var encode:String = Base64.encode( source ) ;
    82   	 * trace("encode : " + encode) ;
    83   	 * 
    84   	 * var decode:String = Base64.decode( encode ) ;
    85   	 * trace("decode : " + decode) ;
    86   	 * }
    87  	 */
    88  	public static function decode( str:String ):String
    89  	{
    90  		str = new String(str) ; // optimize
    91  		var size:Number   = str.length ;
    92  		var i:Number      = 0;
    93  		var output:String = "" ;
    94  		var chr1:Number, chr2:Number, chr3:Number;
    95  		var enc1:Number, enc2:Number, enc3:Number, enc4:Number;
    96  		while (i < size) 
    97  		{
    98  			enc1 = base64chars.indexOf(str.charAt(i++)) ;
    99  			enc2 = base64chars.indexOf(str.charAt(i++)) ;
   100  			enc3 = base64chars.indexOf(str.charAt(i++)) ;
   101  			enc4 = base64chars.indexOf(str.charAt(i++)) ;
   102  			chr1 = (enc1 << 2) | (enc2 >> 4);
   103  			chr2 = ((enc2 & 15) << 4) | (enc3 >> 2);
   104  			chr3 = ((enc3 & 3) << 6) | enc4 ;
   105  			output += String.fromCharCode(chr1);
   106  			if (enc3 != 64) 
   107  			{
   108  				output = output + String.fromCharCode(chr2);
   109  			}
   110  			if (enc4 != 64) 
   111  			{
   112  				output = output + String.fromCharCode(chr3);
   113  			}
   114  		}
   115  		return output;
   116  	}
   117  
   118  	/**
   119  	 * Internal base 64 characters.
   120  	 */
   121  	private static var base64chars:String = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";
   122  	
   123  }
