    32  /**
    33   * A Binary tree {@code BinaryTree} register datas in a recursive manner so that you can access it quickly 
    34   * by using a key. 
    35   * <p>Therefore, a {@code BinaryTree} automatically sorts data as it is inserted.</p>
    36   * <p>For a {@code BinaryTree} to be valid, every node has to follow two rules:</p>
    37   * <p>
    38   * <li>The data value in the left subtree must be less than the data value in the current node.</li>
    39   * <li>The data value in the right subtree must be greater than the data value in the current node.</li>
    40   * </p> 
    41   * @author eKameleon
    42   * @see IComparator
    43   */
    44  class asgard.data.tree.BinaryTree extends CoreObject implements ICloneable, ISerializable

    52  	{
    53  		root = null ;
    54  		setComparator( comp ) ;
    55  	}

   211  	/**
   212  	 * Returns {@code true} if the specified node contains the specified data.
   213  	 * @return {@code true} if the specified node contains the specified data.
   214  	 */
   215  	private function _contains(node:BinaryTreeNode, data):Boolean 
   216  	{
   217          if (BinaryTreeNode == null) 
   218          {
   219          	return false ;
   220          }
   221          var result:Number = _comparator.compare(data, node.data) ;
   222          if (result == 0 )
   223  		{
   224          	return true ;
   225          }
   226          else if ( result == -1 ) 
   227          {
   228          	return _contains( node.left, data ) ;
   229          }
   230          else 
   231          {
   232          	return _contains( node.right, data ) ;
   233          }
   234      }

   236      private function _getDoubleTree( node:BinaryTreeNode ):String
   237      {
   238          var s:String = "doubleTreeRecursr" ;
   239          var oldLeft:BinaryTreeNode ;
   240  
   241          if (node==null) 
   242          {
   243          	return ;
   244          }
   245  
   246          // do the subtrees
   247          
   248          s += "doubleTreeRecurs.node.leftr";
   249          
   250          s += _getDoubleTree( node.left ) ;
   251          
   252          s += "doubleTreeRecurs.node.rightr";
   253          
   254          s += _getDoubleTree( node.right );
   255  
   256          oldLeft = node.left ;
   257          node.setLeft( new BinaryTreeNode( node.data ) ) ;
   258          node.left.setLeft( oldLeft ) ;
   259          
   260          return s ;
   261      }
   262  	/**
   263  	 * Recursive insert, given a BinaryTreeNode pointer, recursive down and insert the given data into the tree.
   264  	 * @param BinaryTreeNode a BinaryTreeNode instance.
   265  	 * @param data a real number.
   266  	 * @return the new BinaryTreeNode pointer (the standard way to communicate a changed pointer back to the caller).
   267  	 */
   268  	private function _insert(node:BinaryTreeNode, data ):BinaryTreeNode 
   269  	{
   270  		if (node == null) 
   271  		{
   272  			node = new BinaryTreeNode(data) ;
   273  		}
   274          else 
   275          {
   276          	var result:Number = _comparator.compare(data, node.data) ;
   277              if ( result > -1 ) 
   278              {
   279              	node.setRight( _insert(node.right, data) ) ;
   280              }
   281              else 
   282              {
   283              	node.setLeft( _insert(node.left, data) ) ;
   284              }
   285          }
   286          return node ;
   287      }

   289  	/**
   290       * Recursive lookup, given a BinaryTreeNode, recur down searching for the given data.
   291       * @param BinaryTreeNode a BinaryTreeNode instance ($root).
   292       * @param data The data in the node.
   293       * @return a boolean if the data is find.
   294  	 */
   295      private function _lookup( node:BinaryTreeNode, data ):Boolean
   296      {
   297          if ( BinaryTreeNode == null || _comparator == null )
   298          {
   299          	return false ;
   300          }
   301  		var result:Number = _comparator.compare(data, node.data) ;
   302          if (result == 0)
   303          {
   304          	return true;
   305          }
   306          else if ( result == -1 )
   307          {
   308          	return _lookup( node.left,data)  ;
   309          }
   310          else 
   311          {
   312          	return _lookup( node.right, data );
   313          }
   314      }
   315  	
   316  	private function _maxDepth( node:BinaryTreeNode):Number 
   317  	{
   318          if ( node == null)
   319          {
   320          	return 0 ;
   321          }
   322          else 
   323          {
   324              var l:Number = _maxDepth(node.left);
   325              var r:Number = _maxDepth(node.right);
   326              return Math.max(l, r) + 1 ;
   327          }
   328      }
   329  
   330  	private function _max(node:BinaryTreeNode):Number 
   331  	{
   332          var cur:BinaryTreeNode = node ;
   333  		while (cur.right !=null ) 
   334  		{
   335  			cur = cur.right ;
   336  		}
   337          return cur.data ;
   338      }
   339  
   340  	private function _min(node:BinaryTreeNode):Number 
   341  	{
   342          var cur:BinaryTreeNode = node ;
   343  		while (cur.left !=null )
   344  		{
   345  			cur = cur.left ;
   346  		}
   347          return cur.data ;
   348      }

   350     	private function _printArray( ar:Array , len:Number ):String
   351  	{
   352  		var s = "Print Array r" ;
   353  		for ( var i:Number = 0; i<len ; i++)
   354  		{
   355  			s += "# ints["+i+"] : " + ar[i] + "r" ;
   356  		}
   357  		return s ;
   358  	}

   360      private function _printPathsRecurs(node:BinaryTreeNode, path:Array, pathLen:Number):String
   361      {
   362          
   363          var s:String = "# printPathsRecursr";
   364  
   365          if (node == null) 
   366          {
   367          	return "" ;
   368          }
   369  
   370          // append this BinaryTreeNode to the path array
   371          
   372          trace("???? : " + node.data) ;
   373          
   374          path[pathLen] = node.data ;
   375          
   376          pathLen++ ;
   377  
   378          // it's a leaf, so print the path that led to here
   379          if (node.left == null && node.right == null )
   380          {
   381              s += _printArray( path , pathLen );
   382          }
   383          else
   384          {
   385              s += "# printPathsRecurs.node.left r";
   386              s += _printPathsRecurs( node.left, path, pathLen ) || "" ;
   387              s += "# printPathsRecurs.node.right r";
   388              s += _printPathsRecurs( node.right, path, pathLen ) || "";
   389          }
   390          return s ;
   391      }

   393  	private function _toString( node:BinaryTreeNode ):String 
   394  	{
   395  		if ( node==null ) 
   396  		{
   397  			return "" ;
   398  		}
   399          // left, BinaryTreeNode itself, right
   400  		var txt:String = "" ;
   401          txt += "L --- " + _toString( node.left) + " : " + node.data + "r" ; 
   402          txt += "R --- " + _toString( node.right) ;
   403  		return txt ;
   404      }
