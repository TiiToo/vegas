    24  import asgard.events.TextEvent;
    25  
    26  /**
    27   * The {@code ErrorEvent} are dispatched when an error causes a network operation to fail. There is only one type of error event: {@code ErrorEvent.ERROR}.
    28   * The {@code ErrorEvent} class also serves as the base class for the {@code IOErrorEvent} and {@code SecurityErrorEvent} classes.
    29   * @author eKameleon
    30   */
    31  class asgard.events.ErrorEvent extends TextEvent 
    32  {
    33  	
    34  	/**
    35  	 * Creates an Event object that contains information about error events. 
    36  	 * Event objects are passed as parameters to event listeners.
    37  	 * @param type The type of the event. Event listeners can access this information through the inherited type property. There is only one type of error event: ErrorEvent.ERROR.
    38  	 * @param type Text to be displayed as an error message. Event listeners can access this information through the text property.
    39  	 * @param id A reference number to associate with the specific error (default 0).
    40  	 * @param target the target of the event.
    41  	 * @param context the optional context object of the event.
    42  	 * @param bubbles indicates if the event is a bubbling event.
    43  	 * @param eventPhase the current EventPhase of the event.
    44  	 * @param time this parameter is used in the Eden deserialization.
    45  	 * @param stop this parameter is used in the Eden deserialization.
    46  	 */
    47  	public function ErrorEvent(type : String, text:String, id:Number, target, context, bubbles : Boolean, eventPhase : Number, time : Number, stop : Number) 
    48  	{
    49  		super(type, text, target, context, bubbles, eventPhase, time, stop);
    50  		errorID = isNaN(id) ? 0 : id ;
    51  	}
    52  
    53  	/**
    54  	 * Defines the value of the type property of an error event object.
    55  	 */
    56  	public static var ERROR:String = "error" ; 
    57  
    58  	/**
    59  	 * Contains the reference number associated with the specific error. 
    60  	 * For a custom ErrorEvent object, this number is the value from the id  parameter supplied in the constructor.
    61  	 */
    62  	public var errorID:Number ;
    63  
    64  	/**
    65  	 * Returns the shallow copy of this event.
    66  	 * @return the shallow copy of this event.
    67  	 */
    68  	public function clone() 
    69  	{
    70  		return new ErrorEvent(getType(), text, errorID, getTarget(), getContext()) ;
    71  	}
    72  
    73  }
