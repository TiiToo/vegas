    24  import vegas.events.BasicEvent;
    25  
    26  /**
    27   * The ProgressEvent are dispatched when a load operation has begun or a socket has received data. 
    28   * These events are usually generated when SWF files, images or data are loaded into Flash Player. 
    29   * There are two types of progress events: ProgressEvent.PROGRESS and ProgressEvent.SOCKET_DATA.
    30   * @author eKameleon
    31   */
    32  class asgard.events.ProgressEvent extends BasicEvent 
    33  {
    34  	
    35  	/**
    36  	 * Creates a new {@code ProgressEvent} instance.
    37  	 * @param type the string type of the instance.
    38  	 * @param bytesLoaded The number of items or bytes loaded at the time the listener processes the event (default 0).
    39  	 * @param bytesTotal The total number of items or bytes that will be loaded if the loading process succeeds (default 0). 
    40  	 * @param target the target of the event.
    41  	 * @param context the optional context object of the event.
    42  	 * @param bubbles indicates if the event is a bubbling event.
    43  	 * @param eventPhase the current EventPhase of the event.
    44  	 * @param time this parameter is used in the Eden deserialization.
    45  	 * @param stop this parameter is used in the Eden deserialization.
    46  	 */
    47  	public function ProgressEvent(type : String, bytesLoaded:Number, bytesTotal:Number, target, context, bubbles : Boolean, eventPhase : Number, time : Number, stop : Number) 
    48  	{
    49  		super(type, target, context, bubbles, eventPhase, time, stop);
    50  		this.bytesLoaded = isNaN(bytesLoaded) ? 0 : bytesLoaded ;
    51  		this.bytesTotal  = isNaN(bytesTotal)  ? 0 : bytesTotal  ;
    52  	}
    53  
    54  	/**
    55  	 * Defines the value of the type property of a progress event object.
    56  	 */
    57  	public static var PROGRESS:String = "progress" ; 
    58  
    59  	/**
    60  	 * Defines the value of the type property of a socketData event object.
    61  	 */
    62  	public static var SOCKET_DATA:String = "socketData" ; 
    63  
    64  	/**
    65  	 * The number of items or bytes loaded when the listener processes the event.
    66  	 */
    67  	public var bytesLoaded:Number ;
    68  	
    69  	/**
    70  	 * The total number of items or bytes that will be loaded if the loading process succeeds.
    71  	 */
    72  	public var bytesTotal:Number ;
    73  
    74  	/**
    75  	 * The percentage of the process.
    76  	 */
    77  	public var percent:Number ;
    78  
    79  	/**
    80  	 * Returns the shallow copy of this event.
    81  	 * @return the shallow copy of this event.
    82  	 */
    83  	public function clone() 
    84  	{
    85  		var e:ProgressEvent = new ProgressEvent(getType(), getTarget(), bytesLoaded, bytesTotal, getContext()) ;
    86  		e.percent = this.percent ;
    87  		return e ;
    88  	}
    89  
    90  }
