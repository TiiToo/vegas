    24  import asgard.events.LoaderEvent;
    25  import asgard.media.IMediaLoader;
    26  
    27  /**
    28   * The MediaEvent class.
    29   * @author eKameleon
    30   * @version 1.0.0.0
    31   */
    32  class asgard.events.MediaEvent extends LoaderEvent 
    33  {
    34  
    35  	/**
    36  	 * Creates a new MediaEvent instance.
    37  	 */
    38  	public function MediaEvent(type:String, loader:IMediaLoader, p_code:Number, p_error:String, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number) 
    39  	{
    40  		super(type, loader, p_code, p_error, context, bubbles, eventPhase, time, stop) ;
    41  	}
    42  	
    43  	/**
    44  	 * The name of the MediaEvent when the media is cleared.
    45  	 */
    46  	public static var MEDIA_CLEAR:String = "onMediaClear" ;
    47  
    48  	/**
    49  	 * The name of the MediaEvent when the media is finished.
    50  	 */
    51  	public static var MEDIA_FINISH:String = "onMediaFinished" ;
    52  
    53  	/**
    54  	 * The name of the MediaEvent when the media progress.
    55  	 */
    56  	public static var MEDIA_PROGRESS:String = "onMediaProgress" ;
    57  	
    58  	/**
    59  	 * The name of the MediaEvent when the media is resumed.
    60  	 */
    61  	public static var MEDIA_RESUME:String = "onMediaResumed" ;
    62  
    63  	/**
    64  	 * The name of the MediaEvent when the media is started.
    65  	 */
    66  	public static var MEDIA_START:String = "onMediaStarted" ;
    67  	
    68  	/**
    69  	 * The name of the MediaEvent when the media is stopped.
    70  	 */
    71  	public static var MEDIA_STOP:String = "onMediaStopped" ;
    72  
    73  	/**
    74  	 * Returns a shallow copy of this object.
    75  	 * @return a shallow copy of this object.
    76  	 */
    77  	public function clone() 
    78  	{
    79  		return new MediaEvent(getType(), getTarget()) ;
    80  	}
    81  
    82  	/**
    83  	 * Returns the duration of the media.
    84  	 * @return the duration of the media.
    85  	 */
    86  	public function getDuration():Number 
    87  	{
    88  		return getLoader().getDuration() ;	
    89  	}
    90  
    91  	/**
    92  	 * Returns the load of the media.
    93  	 * @return the load of the media.
    94  	 */
    95  	public function getLoader():IMediaLoader 
    96  	{
    97  		return _oLoader ;
    98  	}
    99  
   100  	/**
   101  	 * Returns the position of the media.
   102  	 * @return the position of the media.
   103  	 */
   104  	public function getPosition():Number 
   105  	{
   106  		return getLoader().getPosition() ;	
   107  	}
   108  	
   109  	/**
   110  	 * Returns the volume of the media.
   111  	 * @return the volume of the media.
   112  	 */
   113  	public function getVolume():Number 
   114  	{
   115  		return getLoader().getVolume() ;	
   116  	}
   117  
   118  	/**
   119  	 * Internal IMediaLoader reference.
   120  	 */	
   121  	private var _oLoader:IMediaLoader ;
   122  
   123  }
