    24  import asgard.media.SoundModel;
    25  
    26  import vegas.core.Identifiable;
    27  import vegas.events.BasicEvent;
    28  
    29  /**
    30   * The SoundModelEvent class.
    31   * @author eKameleon
    32   */
    33  class asgard.events.SoundModelEvent extends BasicEvent implements Identifiable
    34  {
    35  
    36  	/**
    37  	 * Creates a new SoundModelEvent instance.
    38  	 * @param type the string type of the instance. 
    39  	 * @param model The SoundModel reference of this event.
    40  	 * @param id The sound id of this event.
    41  	 * @param sound The {@code Sound} reference of this event.
    42  	 * @param url th 
    43  	 * @param context the optional context object of the event.
    44  	 * @param bubbles indicates if the event is a bubbling event.
    45  	 * @param eventPhase the current EventPhase of the event.
    46  	 * @param time this parameter is used in the Eden deserialization.
    47  	 * @param stop this parameter is used in the Eden deserialization.
    48  	 * 
    49  	 */	
    50  	public function SoundModelEvent(type:String, model:SoundModel, id:String, sound:Sound, url:String, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number) 
    51  	{
    52  		super(type, model ,context, bubbles, eventPhase, time, stop );
    53  		_id    = id ;
    54  		_model = model ;
    55  		_sound = sound ;
    56  		_url   = url ; 
    57  	}
    58  
    59  	/**
    60  	 * The name of the SoundModelEvent when a new Sound is added in a SoundModel instance.
    61  	 */
    62  	public static var ADD_SOUND:String = "onAddSound" ;
    63  
    64  	/**
    65  	 * The name of the SoundModelEvent when all the sounds in the SoundModel are removed.
    66  	 */
    67  	public static var CLEAR_SOUND:String = "onClearSound" ;
    68  
    69  	/**
    70  	 * The type of the event when the sound is complete.
    71  	 */
    72  	public static var COMPLETE_SOUND:String = "onCompleteSound" ;
    73  
    74  	/**
    75  	 * The type of the event when all sounds are enabled in the model.
    76  	 */
    77  	public static var ENABLE_SOUNDS:String = "onEnabledSounds" ;
    78  
    79  	/**
    80  	 * The type of the event when all sounds are disabled in the model.
    81  	 */
    82  	public static var DISABLE_SOUNDS:String = "onDisabledSounds" ;
    83  	
    84  	/**
    85  	 * The type of the event when a sound id3 is notify.
    86  	 */
    87  	public static var ID3_SOUND:String = "onID3Sound" ; 
    88  	
    89  	/**
    90  	 * The type of the event when a sound is loading.
    91  	 */
    92  	public static var LOAD_SOUND:String = "onLoadSound" ; 
    93  
    94  	/**
    95  	 * The name of the SoundModelEvent when a Sound is removed in a SoundModel instance.
    96  	 */
    97  	public static var REMOVE_SOUND:String = "onRemoveSound" ; 
    98  
    99  	/**
   100  	 * The success boolean value if the SoundModelEvent type is 'LOAD_SOUND'.
   101  	 */
   102  	public var success:Boolean = null ;
   103  
   104  	/**
   105  	 * Returns the id of the sound reference in the model.
   106  	 * @return the id of the sound reference in the model.
   107  	 */
   108  	public function getID()
   109  	{
   110  		return _id ;	
   111  	}
   112  
   113  	/**
   114  	 * Returns the model reference of this event.
   115  	 * @return the model reference of this event.
   116  	 */
   117  	public function getModel():SoundModel
   118  	{
   119  		return _model ;	
   120  	}
   121  	
   122  	/**
   123  	 * Returns the sound reference.
   124  	 * @return the sound reference.
   125  	 */
   126  	public function getSound():Sound
   127  	{
   128  		return _sound ;	
   129  	}
   130  	
   131  	/**
   132  	 * Returns the url of the sound reference if this sound is an external sound.
   133  	 * @return the url of the sound reference if this sound is an external sound.
   134  	 */
   135  	public function getUrl():String
   136  	{
   137  		return _url ;	
   138  	}
   139  
   140  	/**
   141  	 * Sets the id of the sound reference.
   142  	 */
   143  	public function setID( id ):Void
   144  	{
   145  		_id = id ;	
   146  	}
   147  
   148  	/**
   149  	 * Sets the model reference of this event.
   150  	 */
   151  	public function setModel( model:SoundModel ):Void
   152  	{
   153  		_model = model ;	
   154  	}
   155  
   156  	/**
   157  	 * Sets the sound reference.
   158  	 */
   159  	public function setSound( s:Sound ):Void
   160  	{
   161  		_sound = s ;	
   162  	}
   163  
   164  	/**
   165  	 * Sets the url of the sound reference if this sound is an external sound.
   166  	 */
   167  	public function setUrl(sURL:String):Void
   168  	{
   169  		_url = sURL ;	
   170  	}
   171  
   172  	/**
   173  	 * Id of the sound in the model.
   174  	 */
   175  	private var _id:String ;
   176  	
   177  	/**
   178  	 * The model reference of this event.
   179  	 */
   180  	private var _model:SoundModel ;
   181  	
   182  	/**
   183  	 * Internal Sound reference.
   184  	 */
   185  	private var _sound:Sound ;
   186  	
   187  	/**
   188  	 * Internal url of the Sound reference.
   189  	 */
   190  	private var _url:String ;
   191  
   192  }
