    24  import vegas.events.DynamicEvent;
    25  import vegas.util.serialize.Serializer;
    26  
    27  /**
    28   * The {@code TextEvent} to dispatch an event when a user enters text in a text field or clicks a hyperlink in an HTML-enabled text field. 
    29   * There are two types of text events : {@code TextEvent.LINK} and {@code TextEvent.TEXT_INPUT}.
    30   * @author eKameleon
    31   */
    32  class asgard.events.TextEvent extends DynamicEvent 
    33  {
    34  
    35  	/**
    36  	 * Creates an Event object that contains information about error events. 
    37  	 * Event objects are passed as parameters to event listeners.
    38  	 * @param type The type of the event. Event listeners can access this information through the inherited type property. There is only one type of error event: ErrorEvent.ERROR.
    39  	 * @param type Text to be displayed as an error message. Event listeners can access this information through the text property.
    40  	 * @param target the target of the event.
    41  	 * @param context the optional context object of the event.
    42  	 * @param bubbles indicates if the event is a bubbling event.
    43  	 * @param eventPhase the current EventPhase of the event.
    44  	 * @param time this parameter is used in the Eden deserialization.
    45  	 * @param stop this parameter is used in the Eden deserialization.
    46  	 */
    47  	public function TextEvent(type:String, text:String, target, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number)
    48  	{
    49  		super(type, target, context, bubbles, eventPhase, time, stop) ;
    50  		this.text = text ;
    51  	}
    52  
    53  	/**
    54  	 * The string text reference.
    55  	 */
    56  	public var text:String ;
    57  
    58  	/**
    59  	 * Returns the shallow copy of this event.
    60  	 * @return the shallow copy of this event.
    61  	 */
    62  	public function clone() 
    63  	{
    64  		return new TextEvent(getType(), text, getTarget(), getContext()) ;
    65  	}
    66  
    67  	/*protected*/ private function _getParams():Array 
    68  	{
    69  		var ar:Array = super._getParams() ;
    70  		ar.splice(1, null, Serializer.toSource(text)) ;
    71  		return ar ;
    72  	}
    73  
    74  
    75  }
