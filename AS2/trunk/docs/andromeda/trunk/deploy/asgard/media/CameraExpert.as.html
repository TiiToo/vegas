    34  /**
    35   * This singleton expert to control a Camera instance.
    36   * @author eKameleon
    37   */
    38  class asgard.media.CameraExpert extends AbstractCoreEventDispatcher 
    39  {
    40  	
    41  	/**
    42  	 * Creates a new CameraExpert instance.
    43  	 * @param cameraID the id name of the Camera to control.
    44  	 * @param cameraIndex (optional) this index defines a zero-based integer that specifies which camera to get, as determined from the array returned by the Camera.names property.
    45  	 * This argument is used only the first time when the specifiec CameraExpert is created. To get the default camera (which is recommended for most applications), omit this parameter.
    46  	 */
    47  	function CameraExpert( cameraID:String , cameraIndex:Number ) 
    48  	{
    49  		_cameraID = cameraID ;
    50  		initEvent() ;
    51  		_cam = Camera.get( cameraIndex ) ;
    52  		_cam.onActivity = Delegate.create(this, _onCameraActivity ) ;
    53  		_cam.onStatus   = Delegate.create(this, _onCameraStatus ) ;
    54  		updateCamera() ;
    55  	}
    56  	
    57  	/**
    58  	 * The name of the event when the Camera activity change.
    59  	 */
    60  	static public var CAMERA_ACTIVITY_CHANGE:String = "onCameraActivityChange" ;
    61  
    62  	/**
    63  	 * The name of the event when the Camera is muted.
    64  	 */
    65  	static public var CAMERA_IS_MUTED:String = "onCameraIsMuted" ;
    66  
    67  	/**
    68  	 * The name of the event when the Camera is unmuted.
    69  	 */
    70  	static public var CAMERA_IS_UNMUTED:String = "onCameraIsUnMuted" ;
    71  
    72  	/**
    73  	 * The default camera mode favor area value.
    74  	 */
    75  	static public var DEFAULT_CAMERA_MODE_FAVOR_AREA:Boolean = true ;
    76  	
    77  	/**
    78  	 * The default camera mode fps value.
    79  	 */
    80  	static public var DEFAULT_CAMERA_MODE_FPS:Number = 24 ;
    81  	
    82  	/**
    83  	 * The default camera mode height value.
    84  	 */
    85  	static public var DEFAULT_CAMERA_MODE_HEIGHT:Number = 100 ;
    86  	
    87  	/**
    88  	 * The default camera mode width value.
    89  	 */
    90  	static public var DEFAULT_CAMERA_MODE_WIDTH:Number = 140 ;
    91  	
    92  	/**
    93  	 * The default camera motion level.
    94  	 */
    95  	static public var DEFAULT_CAMERA_MOTION_LEVEL:Number = 50 ;
    96  	
    97  	/**
    98  	 * The default camera motion timeout.
    99  	 */
   100  	static public var DEFAULT_CAMERA_MOTION_TIMEOUT:Number = 2000 ;
   101  	
   102  	/**
   103  	 * The default camera quality bandwidth.
   104  	 */
   105  	static public var DEFAULT_CAMERA_QUALITY_BANDWIDTH:Number = 6144 ;
   106  	
   107  	/**
   108  	 * The default camera quality level.
   109  	 */
   110  	static public var DEFAULT_CAMERA_QUALITY_LEVEL:Number = 0 ;

   112  	/**
   113  	 * Returns {@code true} if the CameraExpert contains the specified camera id.
   114  	 * @return {@code true} if the CameraExpert contains the specified camera id.

   116  	static public function contains( cameraID:String  ):Boolean
   117  	{
   118  		return _map.containsKey( cameraID ) ;	
   119  	}

   166  	/**
   167  	 * Returns a CameraExpert singleton.
   168  	 * @param cameraID the camera id of the CameraExpert singleton.
   169  	 * @param cameraIndex (optional) this index defines a zero-based integer that specifies which camera to get, as determined from the array returned by the Camera.names property.
   170  	 * This argument is used only the first time when the specifiec CameraExpert is created. To get the default camera (which is recommended for most applications), omit this parameter.
   171  	 * @return a CameraExpert singleton or null .
   172  	 */
   173  	static public function getInstance( cameraID:String , cameraIndex:Number ):CameraExpert
   174  	{
   175  		if ( _map == null)
   176  		{
   177  			_map = new HashMap() ;	
   178  		}
   179  		if ( !_map.containsKey( cameraID ) )
   180  		{
   181  			_map.put( cameraID, new CameraExpert( cameraID , cameraIndex ) ) ;
   182  		}
   183  		return _map.get( cameraID ) ;
   184  	}
   185  
   186  	/**
   187  	 * This method is invoqued in the constructor of the class to initialize all events.
   188  	 * Overrides this method.
   189  	 */
   190  	public function initEvent():Void
   191  	{
   192  		_eActivity   = new BooleanEvent( CAMERA_ACTIVITY_CHANGE , null, this) ;
   193  		_eMuted      = new BasicEvent( CAMERA_IS_MUTED, this ) ;
   194  		_eUnMuted    = new BasicEvent( CAMERA_IS_UNMUTED, this ) ;
   195  	}
   196  
   197  	/**
   198  	 * Release the specified CameraExpert singleton.
   199  	 * @param cameraID the camera id of the CameraExpert singleton.
   200  	 * @return the removed CameraExpert singleton or null .
   201  	 */
   202  	static public function release( cameraID:String ):CameraExpert
   203  	{
   204  		return _map.remove( cameraID ) ;
   205  	}

   207  	/**
   208  	 * Sets the camera capture mode to the native mode that best meets the specified requirements.
   209  	 */
   210  	public function setCameraMode( init:Object ):Void
   211  	{
   212  		
   213  		var width:Number = init.width ;
   214  		if (isNaN( width )) 
   215  		{
   216  			width = DEFAULT_CAMERA_MODE_WIDTH ;
   217  		}
   218  		
   219  		getLogger().info( this + " setCameraMode width : " + width ) ;
   220  
   221  		var height:Number = init.height ;
   222  		if (isNaN( height )) 
   223  		{
   224  			height = DEFAULT_CAMERA_MODE_HEIGHT ;
   225  		}
   226  		
   227  		getLogger().info( this + " setCameraMode height : " + height ) ;
   228  		
   229  		var fps:Number = init.fps ;
   230  		if (isNaN( fps ))
   231  		{
   232  			fps = DEFAULT_CAMERA_MODE_FPS ;	
   233  		}
   234  		
   235  		getLogger().info( this + " setCameraMode fps : " + fps ) ;
   236  		
   237  		_cam.setMode( width , height, fps ) ;
   238  	
   239  		var favorarea:Boolean = init.favorarea ;
   240  		if ( favorarea == null )
   241  		{
   242  			favorarea = DEFAULT_CAMERA_MODE_FAVOR_AREA ;	
   243  		}
   244  		
   245  		getLogger().info( this + " setCameraMode favorarea : " + favorarea ) ;
   246  		
   247  		_cam.setMode( width , height, fps , favorarea ) ;
   248  		
   249  	}
   250  	
   251  	/**
   252  	 * Specifies how much motion is required to invoke Camera.onActivity(true).
   253  	 */
   254  	public function setCameraMotionLevel( init:Object ):Void
   255  	{
   256  		
   257  		var level:Number = init.level ;
   258  		if (isNaN( level )) 
   259  		{
   260  			level = DEFAULT_CAMERA_MOTION_LEVEL ;
   261  		}
   262  		getLogger().info( this + " setCameraMotionLevel level : " + level ) ;
   263  
   264  		var timeout:Number = init.timeout ;
   265  		if (isNaN( timeout )) 
   266  		{
   267  			timeout = DEFAULT_CAMERA_MOTION_TIMEOUT ;
   268  		}
   269  		getLogger().info( this + " setCameraMotionLevel timeout : " + timeout ) ;
   270  
   271  		_cam.setMotionLevel( level, timeout ) ;
   272  		
   273  	}
   274  
   275  	/**
   276  	 * Sets the maximum amount of bandwidth per second or the required picture quality of the current outgoing video feed.
   277  	 */
   278  	public function setCameraQualityLevel( init:Object ):Void
   279  	{
   280  		
   281  		var bandwidth:Number = init.bandwidth ;
   282  		if (isNaN( bandwidth )) 
   283  		{
   284  			bandwidth = DEFAULT_CAMERA_QUALITY_BANDWIDTH ;
   285  		}
   286  		getLogger().info( this + " setQualityLevel bandwidth : " + bandwidth ) ;
   287  		
   288  		var level:Number = init.level ;
   289  		if (isNaN( level )) 
   290  		{
   291  			level = DEFAULT_CAMERA_QUALITY_LEVEL ;
   292  		}
   293  		
   294  		getLogger().info( this + " setQualityLevel level : " + level ) ;
   295  		
   296  		_cam.setQuality( bandwidth, level ) ;
   297  		
   298  	}

   334  	/**
   335  	 * Update the Camera setting.
   336  	 */
   337  	public function updateCamera( init:Object ):Void
   338  	{
   339  		setCameraQualityLevel( init.quality ) ;
   340  		setCameraMode( init.mode ) ;
   341  		setCameraMotionLevel( init.motionlevel) ;
   342  	}

   352  	private var _cameraID:String ;
   353

   358  	private var _eUnMuted:Event ;

   361  
   362  	/**
   363  	 * Invoked when the camera starts or stops detecting sound.
   364  	 */
   365  	public function _onCameraActivity( activity:Boolean ):Void
   366  	{
   367  		BooleanEvent(_eActivity).setBoolean( activity ) ;
   368  		dispatchEvent( _eActivity ) ;
   369  	}
   370  
   371  	/**
   372  	 * Invoqued when the camera status change.
   373  	 */
   374  	private function _onCameraStatus( oInfo:Object ):Void
   375  	{
   376  		var code:String = oInfo.code ;
   377  		getLogger().fatal(this + " camera status : " + code) ;
   378  		switch( code )
   379  		{
   380  			case CameraStatus.MUTED :
   381  			{
   382  				dispatchEvent( _eMuted ) ;
   383  				break ;
   384  			}	
   385  			case CameraStatus.UNMUTED :
   386  			{
   387  				dispatchEvent( _eUnMuted ) ;
   388  				break ;	
   389  			}
   390  		}
   391  	}
