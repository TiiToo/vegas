    24  import vegas.core.CoreObject;
    25  
    26  /**
    27   * The FLVMetaData instances contains all datas values injected in the external FLV video.
    28   * <p>More informations in the <a href='http://www.buraks.com/flvmdi/'>flvmdi page</a> write by Burak.</p>
    29   * @author eKameleon
    30   */
    31  class asgard.media.FLVMetaData extends CoreObject 
    32  {
    33  	
    34  	/**
    35  	 * Creates a new FLVMetaData object.
    36  	 * @param o An object with all MetaData properties to fill this instance.
    37  	 */
    38  	public function FLVMetaData( o ) 
    39  	{
    40  		for (var prop:String in o)
    41  		{
    42  			this[prop] = o[prop] ;	
    43  		}
    44  	}
    45  	
    46  	/**
    47  	 * Audio codec ID number used in the FLV. (Uncompressed = 0, ADPCM = 1, MP3 = 2, NellyMoser = 5 and 6).
    48  	 * @see asgard.media.AudioCodec
    49  	 */
    50  	public var audiocodecid:Number ;
    51  	
    52  	/**
    53  	 * The audio data rate (Defaults to 0).
    54  	 */
    55  	public var audiodatarate:Number ;
    56  	
    57  	/**
    58  	 * Audio delay in seconds. Flash 8 encoder delays the video for better synch with audio 
    59  	 * (Audio and video does not start both at time 0, Video starts a bit later). 
    60  	 * This value is also important for Flash 8 Video Encoder injected Cue Points, because logical time of the cue points does not correspond to physical time they are inserted in the FLV. 
    61  	 * (Cue points are injected before encoding, when the video is shifted by 'audio delay' seconds, cue points are also shifted and their physical time in the FLV changes).
    62  	 */
    63  	public var audiodelay:Number ;
    64  	
    65  	/**
    66  	 * The audio data size.
    67  	 */
    68  	public var audiosize:Number ;
    69  	
    70  	/**
    71  	 * Is {@code true} if the last video tag is a key frame and hence can be 'seek'ed.
    72  	 */
    73  	public var canSeekToEnd:Boolean ;
    74  	
    75  	/**
    76  	 * The date of creation of the video.
    77  	 */
    78  	public var creationdate:String ;
    79  
    80  	/**
    81  	 * The data size of the video.
    82  	 */
    83  	public var datasize:Number ;
    84  	
    85  	/**
    86  	 * Length of the FLV in seconds. FLVMDI computes this value.
    87  	 */
    88  	public var duration:Number ;
    89  
    90  	/**
    91  	 * The size of the file.
    92  	 */
    93  	public var filesize:Number ;
    94  
    95  	/**
    96  	 * The framerate of the video.
    97  	 */
    98  	public var framerate:Number ;
    99  
   100  	/**
   101  	 * Height of the video in pixels. (Flash exporter 1.1 sets this to 0).
   102  	 */
   103  	public var height:Number ;
   104  
   105  	/**
   106  	 * This object is added only if you specify the /k switch. 
   107  	 * 'keyframes' is known to FLVMDI and if /k switch is not specified, 'keyframes' object will be deleted.
   108  	 * 'keyframes' object has 2 arrays : 'filepositions' and 'times'. 
   109  	 * Both arrays have the same number of elements, which is equal to the number of key frames in the FLV.
   110  	 * Values in times array are in 'seconds'. Each correspond to the timestamp of the n'th key frame.
   111  	 * Values in filepositions array are in 'bytes'. Each correspond to the fileposition of the nth key frame video tag (which starts with byte tag type 9).
   112  	 */
   113  	public var keyframes:Object ;
   114  
   115  	/**
   116  	 * TimeStamp of the last video tag which is a key frame. 
   117  	 * This info might be needed because seeking a frame after this time usually does not work.
   118  	 */
   119  	public var lastkeyframetimestamp:Number ;
   120  
   121  	/**
   122  	 * TimeStamp of the last tag in the FLV file. 
   123  	 */
   124  	public var lasttimestamp:Number ;
   125  
   126  	/**
   127  	 * Date and time metadata added. (Note that this is not of type string like 'creationdate').
   128  	 */
   129  	public var metadatacreator:Date ;
   130  
   131  	/**
   132  	 * The metadata date of the video.
   133  	 */
   134  	public var metadatadate:Date ;
   135  
   136  	/**
   137  	 * Video codec ID number used in the FLV. (Sorenson H.263 =2, Screen Video =3, On2 VP6 = 4 and 5, Screen Video V2 = 6).
   138  	 */
   139  	public var videocodecid:Number ;
   140  
   141  	/**
   142  	 * The video data rate (Defaults to 0).
   143  	 */
   144  	public var videodatarate:Number ;
   145  
   146  	/**
   147  	 * The video data size.
   148  	 */
   149  	public var videosize:Number ;
   150  	
   151  	/**
   152  	 * Width of the video in pixels. (Flash exporter 1.1 sets this to 0).
   153  	 */
   154  	public var width:Number ;
   155  
   156  	/**
   157  	 * Additional string data if specified.
   158  	 */
   159  	public var xtradata:String ;
   160  	
   161  
   162  }
