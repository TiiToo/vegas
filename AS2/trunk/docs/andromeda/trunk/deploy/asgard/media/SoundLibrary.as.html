    36  /**
    37   * Sound library to register all sounds in an application.
    38   * <p>To load external "mp3" files using {@code loadSound} method.</p>
    39   * <p><b>Example :</b></p>
    40   * <p>
    41   * {@code
    42   * import asgard.events.SoundModelEvent ;
    43   * import asgard.media.SoundLibrary ;
    44   * import vegas.events.Delegate ;
    45   * 
    46   * var debug:Function = function ( e:SoundModelEvent ):Void
    47   * {
    48   *     var type:String = e.getType() ;
    49   *	   var id:String = e.getID() ;
    50   *	   var library:SoundLibrary = SoundLibrary(e.getModel()) ;
    51   *	   var sound:Sound = e.getSound() ;
    52   *	   
    53   *	   var message:String = "> " + library ;
    54   *	   message += " " + type ;
    55   *	   if (id != null)
    56   *	   {
    57   *		    message += ", id:" + id ;
    58   *	   }
    59   *	   if (sound != null)
    60   *	   {
    61   *		    message += ", sound:" + sound ;
    62   *	   }
    63   *	   trace( message ) ;
    64   * }
    65   *
    66   * var lib:SoundLibrary = new SoundLibrary() ;
    67   * lib.addGlobalEventListener( new Delegate(this, debug) ) ;
    68   *
    69   * lib.addSound( "sound_1" ) ;
    70   * lib.addSound( "sound_2" ) ;
    71   *
    72   * lib.addSounds( ["sound_1", "sound_2"] ) ;
    73   *
    74   * trace("> " + lib + " size : " + lib.size()) ;
    75   *
    76   * Key.addListener(this) ;
    77   *
    78   * onKeyDown = function()
    79   * {
    80   *     var code:Number = Key.getCode() ;
    81   *     switch ( code )
    82   *     {
    83   *	        case Key.UP : 
    84   *		    {
    85   *   			lib.getSound( "sound_1" ).start() ;
    86   *			    break ;
    87   *		    }
    88   *		    case Key.DOWN : 
    89   *		    {
    90   *			    lib.getSound( "sound_2" ).start() ;
    91   *			    break ;
    92   *		    }
    93   *		    case Key.SPACE :
    94   *		    {
    95   *			    lib.toggle() ;
    96   *			    break ;
    97   *		    }
    98   *	    }
    99   * }
   100   *
   101   * trace("> Press Key.SPACE to disable/enable the library.") ;
   102   * trace("> Press Key.UP to start the sound_1 sound.") ;
   103   * trace("> Press Key.DOWN to start the sound_2 sound.") ;
   104   * }
   105   * </p>
   106   * @author eKameleon
   107   */
   108  class asgard.media.SoundLibrary extends SoundModel 
   109  {
   110  	
   111  	/**
   112  	 * Creates a new SoundLibrary 
   113  	 */
   114  	function SoundLibrary() 
   115  	{
   116  		super();
   117  		_mGain = new HashMap() ;
   118  	}
   119  
   120  	/**
   121  	 * The default gain of all sounds the first time in the library.
   122  	 */
   123  	static public var DEFAULT_GAIN:Number = 90 ;
   124  
   125  	/**
   126  	 * The range of all gains in the sound library.
   127  	 */
   128  	static public var RANGE_GAIN:Range = new Range(0, 127) ;
   129  
   130  	/**
   131  	 * The range of all pans in the sound library.
   132  	 */
   133  	static public var RANGE_PAN:Range = new Range(-100, 100) ;
   134  
   135  	/**
   136  	 * Adds new sound in the model.
   137  	 * <p>If the id already exist in the model, the current sound is remove in the model before the new sound must be inserted in the model</p>
   138  	 * <p>Example
   139  	 * {@code
   140  	 *   var sLib:SoundLibrary = new SoundLibrary();
   141  	 *   sLib.addSound( "sound_1" );
   142  	 *   sLib.addSound( "sound_2" );
   143  	 * }
   144  	 * </p>
   145  	 * @param id Sound identifier in swf file.
   146  	 * @param sound Sound reference.
   147  	 * 
   148  	 * @return {@code String} sound identifier 
   149  	 * @throws ArgumentsError if the specified id and sound are 'null' or 'undefined'.
   150  	 */
   151  	public function addSound( id:String )
   152  	{
   153  		if (!isInit())
   154  		{
   155  			initialize() ;
   156  		}
   157  		var sound:Sound = _createSound(id) ;
   158  		_mGain.put(id, DEFAULT_GAIN) ;
   159  		_setVolume( id ) ;
   160  		return super.addSound( id , sound ) ;
   161  	}
   162  
   163  	/**
   164  	 * Adds sounds id list to the the library.
   165  	 * <p>Example
   166  	 * {@code
   167  	 *   var sLib:SoundLibrary = new SoundLibrary();
   168  	 *   var aSoundList:Array = ["sound_1", "sound_2"] ;
   169  	 *   sLib.addSounds( aSoundList );
   170  	 * }
   171  	 * </p>
   172  	 * 
   173  	 * @param list an {@code Array} Sounds ID list.
   174  	 */
   175  	public function addSounds( list:Array) : Void
   176  	{
   177  		var l:Number = list.length;
   178  		for (var i:Number=0; i<l ; i++) 
   179  		{
   180  			addSound( list[i] );
   181  		}
   182  	}
   183  
   184  	/**
   185  	 * Clear all sounds in the model.
   186  	 */
   187  	public function clear() : Void
   188  	{
   189  		release() ;
   190  		_mGain.clear() ;
   191  		super.clear() ;
   192  	}
   193  	
   194  	/**
   195  	 * Returns defined gain for passed-in 'id' sound of this library.
   196  	 * @param id the Sound 'id'.
   197  	 * @return the value of the gain of the specific sound.
   198  	 */
   199  	public function getGain( id:String ):Number
   200  	{
   201  		return _mGain.get(id) ;
   202  	}
   203  
   204  	/**
   205  	 * Returns the value of the pan of all sounds in the library.
   206  	 * @return the value of the pan of all sounds in the library.
   207  	 */
   208  	public function getPan() : Number
   209  	{
   210  		return _nPan;
   211  	}
   212  
   213  	/**
   214  	 * Returns the volume percent value of this library.
   215  	 * @return the volume percent value of this library.
   216  	 */
   217  	public function getVolume():Number
   218  	{
   219  		return _nVolume;
   220  	}	
   221  
   222  	/**
   223  	 * Initialize the library.
   224  	 */
   225  	public function initialize( target:MovieClip ):Void
   226  	{
   227  		_nDepth = 0 ;
   228  		_isInit = true ;
   229  		_container = (target == null) ? _level0.createEmptyMovieClip("__mcSound__", 65535) : target ;
   230  	}
   231  	
   232  	/**
   233  	 * Returns {@code true} if the library is initialized.
   234  	 */
   235  	public function isInit():Boolean
   236  	{
   237  		return _isInit ;	
   238  	}
   239  
   240  	/**
   241  	 * Loads and attach in the library a new external Sound.
   242  	 * @param sURL the string representation of the url of the external sound.
   243  	 * @param id the identifier of the sound, if this parameter is null the loadSound method try to use the name of the external file to defined the id value.
   244  	 * @return {@code String} sound identifier 
   245  	 */
   246  	public function loadSound(sURL:String, id:String):String
   247  	{
   248  		if (!isInit())
   249  		{
   250  			initialize() ;
   251  		}
   252  		
   253  		if ( id == null )
   254  		{
   255  			id = sURL.substring( sURL.lastIndexOf("/") + 1 , sURL.lastIndexOf(".") ) ;
   256  		}
   257  		
   258  		if (id != null && id.length > 0)
   259  		{
   260  			var sound:Sound = _createSound( id, sURL);
   261  			super.addSound( id , sound ) ;
   262  			return id ;
   263  		}
   264  		else
   265  		{
   266  			throw new Warning( this + " loadSound method failed, the specified id not must be 'null' or 'undefined' or a empty string." ) ;
   267  		}
   268  	}
   269  
   270  	/**
   271  	 * Release the library and remove the sound container.
   272  	 */
   273  	public function release():Void
   274  	{
   275  		_isInit = false ;
   276  		_container.removeMovieClip() ;
   277  	}
   278  
   279  	/**
   280  	 * Removes a sound in the model.
   281  	 * 
   282  	 * <p>Example
   283  	 * {@code
   284  	 *   var sLib:SoundLibrary = new SoundLibrary();
   285  	 *   sLib.addSound( "sound_1" );
   286  	 *   sLib.removeSound( "sound_1" );
   287  	 * }
   288  	 * 
   289  	 * @param id Sound identifier in swf file.
   290  	 * 
   291  	 * @return {@code null} if the model don't contains the specified id. 
   292  	 * @throws ArgumentsError if the specified id is 'null' or 'undefined'.
   293  	 */
   294  	public function removeSound( id:String ):Void
   295  	{
   296  		if ( containsByID(id) )
   297  		{
   298  			_removeSound(id) ;
   299  			_mGain.remove(id) ;
   300  			super.removeSound( id ) ;
   301  		}
   302  	}
   303  
   304  	/**
   305  	 * Defines {@code gain} value for all registered sounds.
   306  	 * @param nGain a number between 0 and 127.
   307  	 */
   308  	public function setAllGain(nGain:Number) : Void
   309  	{
   310  		nGain = RANGE_GAIN.clamp(nGain) ;
   311  		var ar:Array = _mGain.getKeys() ;
   312  		var len:Number = ar.size() ;
   313  		while( --len > -1 )
   314  		{
   315  			var id:String = ar[len] ;
   316  			_mGain.put( id , nGain ) ;
   317  			_setVolume( id ) ;
   318  		}
   319  	}
   320  
   321  	/**
   322  	 * Sets the gain value for passed-in id sound.
   323  	 * @param id the identifier of the sound in the library if exist.
   324  	 * @param nGain a number between 0 and 127.
   325  	 */
   326  	public function setGain( id:String, nGain:Number ):Void
   327  	{
   328  		if (containsByID(id))
   329  		{
   330  			nGain = RANGE_GAIN.clamp(nGain) ;
   331  			_map.put(id, nGain) ;
   332  			_setVolume( id );
   333  		}
   334  	}
   335  
   336  	/**
   337  	 * Sets the pan value for all sounds list.
   338  	 * @param nPan a number between -100 and 100.
   339  	 */
   340  	public function setPan(nPan:Number) : Void
   341  	{
   342  		_nPan = RANGE_PAN.clamp(nPan) ;
   343  		var ar:Array = toArray() ;
   344  		var len:Number = ar.size() ;
   345  		while( --len > -1 )
   346  		{
   347  			Sound(ar[len]).setPan(_nPan) ;
   348  		}
   349  	}
   350  
   351  	/**
   352  	 * Sets the volume value for all sounds in the library.
   353  	 * @param value a number between 0 and 100.
   354  	 */
   355  	public function setVolume( value:Number) : Void
   356  	{
   357  		value = Range.PERCENT_RANGE.clamp(value) ;
   358  		_nVolume = value ;
   359  		var ar:Array = _mGain.getKeys() ;
   360  		var len:Number = ar.size() ;
   361  		while( --len > -1 )
   362  		{
   363  			_setVolume( ar[len] );
   364  		}
   365  	}
   366  	
   367  	/**
   368  	 * The internal movieclip reference to control all sounds in this library.
   369  	 */
   370  	private var _container:MovieClip ;
   371  
   372  	/**
   373  	 * Internal boolean to defined if the library is init or not.
   374  	 */
   375  	private var _isInit:Boolean = false ;
   376  
   377  	/**
   378  	 * The internal HashMap to control all gains in the sound library.
   379  	 */	
   380  	private var _mGain:HashMap ;
   381  
   382  	/**
   383  	 * Internal depth value to creates all movieclip containers for all sounds in the library.
   384  	 */
   385  	private var _nDepth:Number ;
   386  
   387  	/**
   388  	 * Internal pan value of all sounds.
   389  	 */
   390  	private var _nPan:Number = 0 ;
   391  
   392  	/**
   393  	 * Internal volume value of all sounds.
   394  	 */
   395  	private var _nVolume:Number = 127 ;
   396  
   397  	/**
   398  	 * Calculate the volume value of a sound with a specific gain.
   399  	 */
   400  	private function _calculateGainVolume( nGain:Number ):Number
   401  	{
   402  		return ( nGain / 100 ) * _nVolume ;
   403  	}
   404  
   405  	/**
   406  	 * Creates a new sound in the library.
   407  	 */
   408  	private function _createSound( id:String, sURL:String):Sound
   409  	{
   410  		var target:MovieClip = _container.createEmptyMovieClip("sound_" + id, _nDepth++);
   411  		var sound:Sound = new Sound( target ) ;
   412  		
   413  		var eComplete = new SoundModelEvent( SoundModelEvent.COMPLETE_SOUND , this, id, sound ) ;
   414  		sound.onSoundComplete = Delegate.create(this, notifyChanged, eComplete) ;
   415  		
   416  		var eID3      = new SoundModelEvent( SoundModelEvent.ID3_SOUND , this, id, sound ) ;
   417  		sound.onID3 = Delegate.create(this, notifyChanged, eID3) ;
   418  		
   419  		var eLoad     = new SoundModelEvent( SoundModelEvent.LOAD_SOUND, this, id, sound , sURL) ;
   420  		sound.onLoad = Delegate.create(this, _fireLoadEvent, eLoad) ;
   421  		
   422  		if (sURL)
   423  		{
   424  			sound.loadSound(sURL, false);
   425  
   426  		} else
   427  		{
   428  			sound.attachSound( id );
   429  		}
   430  		
   431  		return sound ;
   432  	}
   433  	
   434  	/**
   435  	 * Dispatch the SoundModelEvent with the type LOAD_SOUND
   436  	 */
   437  	private function _fireLoadEvent( success:Boolean , e:SoundModelEvent ):Void
   438  	{
   439  		e.success = success ;
   440  		notifyChanged( e ) ;
   441  	}
   442  
   443  	/**
   444  	 * Removes a sound in the library.
   445  	 */
   446  	private function _removeSound( id:String ):Void
   447  	{
   448  		MovieClip(_container["sound_" + id]).removeMovieClip() ;
   449  	}
   450  	
   451  	/**
   452  	 * Set the volume of a specific sound with the id key in the library.
   453  	 */
   454  	private function _setVolume( id:String ):Void
   455  	{
   456  		getSound( id ).setVolume( _calculateGainVolume( getGain( id ) ) );
   457  	}
