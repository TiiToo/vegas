    24  import asgard.display.ConfigurableDisplayObject;
    25  
    26  import pegas.maths.Range;
    27  
    28  import vegas.errors.IllegalArgumentError;
    29  import vegas.events.NumberEvent;
    30  
    31  /**
    32   * This display control a Video object and this attached Sound object.
    33   * @author eKameleon
    34   */
    35  class asgard.media.AbstractVideoDisplay extends ConfigurableDisplayObject
    36  {
    37  	
    38  	/**
    39  	 * Creates a new AbstractVideoDisplay instance.
    40  	 * If the target reference contains a {@ode Video} instance with the name "video", this object is used in the display.
    41  	 * @param sName the name of the display.
    42  	 * @param target the target of this display.
    43  	 * @param (optional) the {@code Video} object of this display. 
    44  	 */	
    45  	public function AbstractVideoDisplay( sName:String, target:MovieClip, video:Video) 
    46  	{
    47  		
    48  		super(sName, target);
    49  		
    50  		_oVideo = video ? video : view.video ;
    51  		
    52  		_oVideo._width  = view._width ;
    53  		_oVideo._height = view._height ;
    54  		_oVideo.toString = function()
    55  		{
    56  			return "[Video]" ;	
    57  		} ;
    58  		
    59  		_oSound = new Sound( view ) ;
    60  		_oSound.setVolume( VOLUME_DEFAULT ) ;
    61  		_oSound.toString = function()
    62  		{
    63  			return "[Sound]" ;	
    64  		} ;
    65  
    66  		if ( !_oVideo ) 
    67  		{
    68  			throw new IllegalArgumentError( this + " constructor failed. Invalid video object was passed in the constructor.");
    69  		}
    70  		
    71  	}
    72  
    73  	/**
    74  	 * The default value of the volume in the display.
    75  	 */
    76  	static public var VOLUME_DEFAULT:Number = 80 ;
    77  
    78  	/**
    79  	 * const Defined the event name of the event 
    80  	 */
    81  	static public var SOUND_VOLUME_CHANGE:String = "onVolumeChange" ;
    82  
    83  	/**
    84  	 * Returns the sound's volume value of the UI.
    85  	 * @return the sound's volume value of the UI.
    86  	 */
    87  	public function get volume():Number
    88  	{
    89  		return _oSound.getVolume() ;	
    90  	}
    91  	
    92  	/**
    93  	 * Sets the sound's volume value of the display.
    94  	 * This property dispatch a {@code NumberEvent} object with the volume value returns by the getNumber() method of the event.
    95  	 * To handle this event you can use the {@code SOUND_VOLUME_CHANGE} static property.
    96  	 * @param value the Number value of the volume of this display.
    97  	 */
    98  	public function set volume( value:Number ):Void
    99  	{
   100  		var n:Number = Range.PERCENT_RANGE.clamp(value) ;
   101  		_oSound.setVolume( n ) ;
   102  		dispatchEvent( new NumberEvent( SOUND_VOLUME_CHANGE , _oSound.getVolume() , this ) ) ;
   103  	}
   104  
   105  	/**
   106  	 * Attaching an audio NetStream or a Microphone source to play.
   107  	 * @param oSource the audio source.  
   108  	 */
   109  	public function attachAudio( oSource ):Void
   110  	{
   111  		view.attachAudio( oSource ) ;	
   112  	}
   113  	
   114  	/**
   115  	 * Attaching the client camera
   116  	 * @param cam The specified {@code Camera} object.
   117  	 */
   118  	public function attachCamera( cam:Camera ):Void
   119  	{
   120  		_oVideo.attachVideo( cam ) ;	
   121  	}
   122  	
   123  	/**
   124  	 * Attaching a NetStream object.
   125  	 * @param stream The specified {@code NetStream} object.
   126  	 */
   127  	public function attachStream( stream:NetStream ):Void
   128  	{
   129  		_oVideo.attachVideo( stream ) ;	
   130  	}
   131  	
   132  	/**
   133  	 * Clear the video in the display.
   134  	 */
   135  	public function clear():Void
   136  	{
   137  		view.attachAudio(null) ;
   138  		_oVideo.attachVideo(null) ;
   139  		_oVideo.clear() ;
   140  	}
   141  
   142  	/**
   143  	 * Returns the sound reference of this display.
   144  	 * @return the sound reference of this display.
   145  	 */
   146  	public function getSound():Sound
   147  	{
   148  		return _oSound ;	
   149  	}
   150  	
   151  	/**
   152  	 * Returns the video reference of this display.
   153  	 * @return the video reference of this display.
   154  	 */
   155  	public function getVideo():Video
   156  	{
   157  		return _oVideo ;
   158  	}
   159  
   160  	/**
   161  	 * Sets the virtual width and height values of the component.
   162  	 */
   163  	public function setSize( w:Number, h:Number ) : Void 
   164  	{
   165  		view._width = w ;
   166  		view._height = h ;
   167  		update() ;
   168  	}
   169  
   170  	/**
   171  	 * Internal sound object.
   172  	 */
   173  	private var _oSound : Sound;
   174  	
   175  	/**
   176  	 * Internal video object.
   177  	 */
   178  	private var _oVideo : Video;
   179  
   180  
   181  }
