    24  import asgard.media.MicrophoneStatus;
    25  
    26  import vegas.data.map.HashMap;
    27  import vegas.events.AbstractCoreEventDispatcher;
    28  import vegas.events.BasicEvent;
    29  import vegas.events.BooleanEvent;
    30  import vegas.events.Delegate;
    31  import vegas.events.Event;
    32  import vegas.util.ConstructorUtil;
    33  
    34  /**
    35   * This singleton expert to control a Microphone instance.
    36   * @author eKameleon
    37   */
    38  class asgard.media.MicrophoneExpert extends AbstractCoreEventDispatcher 
    39  {
    40  	
    41  	/**
    42  	 * Creates a new MicrophoneExpert instance.
    43  	 * @param MicrophoneID the id name of the Microphone to control.
    44  	 * @param MicrophoneIndex (optional) this index defines a zero-based integer that specifies which Microphone to get, as determined from the array returned by the Microphone.names property.
    45  	 * This argument is used only the first time when the specifiec MicrophoneExpert is created. To get the default Microphone (which is recommended for most applications), omit this parameter.
    46  	 */
    47  	function MicrophoneExpert( microID:String , microIndex:Number ) 
    48  	{
    49  		_microID = microID ;
    50  		initEvent() ;
    51  		_mic = Microphone.get( microIndex ) ;
    52  		_mic.onActivity = Delegate.create(this, _onMicroActivity ) ;
    53  		_mic.onStatus   = Delegate.create(this, _onMicroStatus ) ;
    54  		updateMicrophone() ;
    55  	}
    56  	
    57  	/**
    58  	 * The name of the event when the Microphone activity change.
    59  	 */
    60  	static public var MICROPHONE_ACTIVITY_CHANGE:String = "onMicrophoneActivityChange" ;
    61  
    62  	/**
    63  	 * The name of the event when the Microphone is muted.
    64  	 */
    65  	static public var MICROPHONE_IS_MUTED:String = "onMicrophoneIsMuted" ;
    66  
    67  	/**
    68  	 * The name of the event when the Microphone is unmuted.
    69  	 */
    70  	static public var MICROPHONE_IS_UNMUTED:String = "onMicrophoneIsUnMuted" ;
    71  
    72  	/**
    73  	 * The default microphone echo suppression value.
    74  	 */
    75  	static public var DEFAULT_MICRO_ECHO_SUPPRESSION:Boolean = false ;
    76  
    77  	/**
    78  	 * The default microphone rate value.
    79  	 */
    80  	static public var DEFAULT_MICRO_RATE:Number = 11 ;
    81  	
    82  	/**
    83  	 * The default microphone silence level value.
    84  	 */
    85  	static public var DEFAULT_MICRO_SILENCE_LEVEL:Number = 10 ;
    86  
    87  	/**
    88  	 * The default microphone silence timeout value.
    89  	 */
    90  	static public var DEFAULT_MICRO_SILENCE_TIMEOUT:Number = 1000 ;
    91  
    92  	/**
    93  	 * Returns {@code true} if the MicrophoneExpert contains the specified camera id.
    94  	 * @return {@code true} if the CameraExpert contains the specified camera id.
    95  	 */
    96  	static public function contains( microID:String  ):Boolean
    97  	{
    98  		return _map.containsKey( microID ) ;	
    99  	}
   100  
   101  	/**
   102  	 * Returns the Microphone reference of this object.
   103  	 * @return the Microphone reference of this object.
   104  	 */
   105  	public function getMicrophone():Microphone
   106  	{
   107  		return _mic ;
   108  	}
   109  	
   110  	/**
   111  	 * Returns the id of this expert.
   112  	 * @return the id of this expert.
   113  	 */
   114  	public function getMicrophoneID():String
   115  	{
   116  		return _microID ;	
   117  	}
   118  
   119  	/**
   120  	 * Returns the event name when the Microphone activity change.
   121  	 * @return the event name when the Microphone activity change.
   122  	 */
   123  	public function getEventTypeACTIVITY():String
   124  	{
   125  		return _eActivity.getType() ;		
   126  	}
   127  
   128  	/**
   129  	 * Returns the event name when the Microphone is muted.
   130  	 * @return the event name when the Microphone is muted.
   131  	 */
   132  	public function getEventTypeMUTED():String
   133  	{
   134  		return _eMuted.getType() ;		
   135  	}
   136  	
   137  	/**
   138  	 * Returns the event name when the Microphone is unmuted.
   139  	 * @return the event name when the Microphone is unmuted.
   140  	 */
   141  	public function getEventTypeUNMUTED():String
   142  	{
   143  		return _eUnMuted.getType() ;		
   144  	}
   145  
   146  	/**
   147  	 * Returns a MicrophoneExpert singleton.
   148  	 * @param microID the Microphone id of the MicrophoneExpert singleton.
   149  	 * @param microIndex (optional) this index defines a zero-based integer that specifies which Microphone to get, as determined from the array returned by the Microphone.names property.
   150  	 * This argument is used only the first time when the specifiec MicrophoneExpert is created. To get the default Microphone (which is recommended for most applications), omit this parameter.
   151  	 * @return a MicrophoneExpert singleton or null .
   152  	 */
   153  	static public function getInstance( microID:String , microIndex:Number ):MicrophoneExpert
   154  	{
   155  		if ( _map == null)
   156  		{
   157  			_map = new HashMap() ;	
   158  		}
   159  		if ( !_map.containsKey( microID ) )
   160  		{
   161  			_map.put( microID, new MicrophoneExpert( microID , microIndex ) ) ;
   162  		}
   163  		return _map.get( microID ) ;
   164  	}
   165  
   166  	/**
   167  	 * This method is invoqued in the constructor of the class to initialize all events.
   168  	 */
   169  	public function initEvent():Void
   170  	{
   171  		_eActivity   = new BooleanEvent( MICROPHONE_ACTIVITY_CHANGE , null, this) ;
   172  		_eMuted      = new BasicEvent( MICROPHONE_IS_MUTED, this ) ;
   173  		_eUnMuted    = new BasicEvent( MICROPHONE_IS_UNMUTED, this ) ;
   174  	}
   175  
   176  	/**
   177  	 * Release the specified MicrophoneExpert singleton.
   178  	 * @param microID the microphone id of the MicrophoneExpert singleton.
   179  	 * @return the removed MicrophoneExpert singleton or null .
   180  	 */
   181  	static public function release( microID:String ):MicrophoneExpert
   182  	{
   183  		return _map.remove( microID ) ;
   184  	}
   185  
   186  	/**
   187  	 * Sets the Microphone rate value.
   188  	 * @param rate the microphone rate value.
   189  	 */
   190  	public function setMicrophoneRate( rate:Number ):Void
   191  	{
   192  		if ( isNaN( rate ) ) 
   193  		{
   194  			rate = DEFAULT_MICRO_RATE ;
   195  		}
   196  		getLogger().info( this + " setRate : " + rate ) ;
   197  		_mic.setRate( rate ) ;
   198  	}
   199  	
   200  	/**
   201  	 * Sets the Microphone silence level.
   202  	 * @param silencelevel the setting object.
   203  	 */
   204  	public function setMicrophoneSilenceLevel( silencelevel:Object )
   205  	{
   206  		var level:Number = silencelevel.level ;
   207  		if ( isNaN( level ) ) 
   208  		{
   209  			level = DEFAULT_MICRO_SILENCE_LEVEL ;
   210  		}
   211  		getLogger().info( this + " setSilenceLevel level : " + level ) ;
   212  		
   213  		var timeout:Number = silencelevel.timeout ;
   214  		if ( isNaN( timeout ) ) 
   215  		{
   216  			timeout = DEFAULT_MICRO_SILENCE_TIMEOUT ;
   217  		}
   218  		getLogger().info( this + " setSilenceLevel timeout : " + timeout ) ;
   219  		
   220  		_mic.setSilenceLevel( level , timeout ) ;
   221  	}
   222  
   223  	/**
   224  	 * Sets the Microphone echo suppression flag.
   225  	 * @param echosuppression the flag to indicates if the echosuppression is on or off.
   226  	 */
   227  	public function setMicrophoneUseEchoSuppression( echosuppression:Boolean ):Void
   228  	{
   229  		if ( echosuppression == null ) 
   230  		{
   231  			echosuppression = DEFAULT_MICRO_ECHO_SUPPRESSION ;
   232  		}
   233  		getLogger().info( this + " setEchoSuppression : " + echosuppression ) ;
   234  		_mic.setUseEchoSuppression( echosuppression  ) ;
   235  	}
   236  	
   237  	/**
   238  	 * Sets the event name when the Microphone activity change.
   239  	 */
   240  	public function setEventTypeACTIVITY(type:String ):Void
   241  	{
   242  		_eActivity.setType( type ) ;		
   243  	}
   244  
   245  	/**
   246  	 * Sets the event name when the Microphone is muted.
   247  	 */
   248  	public function setEventTypeMUTED(type:String ):Void
   249  	{
   250  		_eMuted.setType( type ) ;		
   251  	}
   252  	
   253  	/**
   254  	 * Sets the event name when the Microphone is unmuted.
   255  	 */
   256  	public function setEventTypeUNMUTED(type:String ):Void
   257  	{
   258  		_eUnMuted.setType( type ) ;		
   259  	}
   260  
   261  	/**
   262  	 * Returns the String representation of the object.
   263  	 * @return the String representation of the object.
   264  	 */
   265  	public function toString():String
   266  	{
   267  		return "[" + ConstructorUtil.getName(this) + ( getMicrophoneID() != null ? (":" + getMicrophoneID() ) : "" ) + "]" ;	
   268  	}
   269  
   270  	/**
   271  	 * Update the Microphone setting.
   272  	 */
   273  	public function updateMicrophone( init:Object ):Void
   274  	{
   275  		setMicrophoneUseEchoSuppression( init.echosuppression ) ;
   276  		setMicrophoneRate( init.rate ) ;
   277  		setMicrophoneSilenceLevel( init.silencelevel ) ;
   278  	}
   279  
   280  	/**
   281  	 * The Microphone singleton reference.
   282  	 */
   283  	private var _mic:Microphone ;
   284  
   285  	/**
   286  	 * The MicrophoneExpert id.
   287  	 */
   288  	private var _microID:String ;
   289  
   290  	private var _eActivity:Event ;
   291  	
   292  	private var _eMuted:Event ;
   293  	
   294  	private var _eUnMuted:Event ;
   295  
   296  	static private var _map:HashMap  ;
   297  
   298  	/**
   299  	 * Invoked when the Microphone starts or stops detecting sound.
   300  	 */
   301  	public function _onMicroActivity( activity:Boolean ):Void
   302  	{
   303  		BooleanEvent(_eActivity).setBoolean( activity ) ;
   304  		dispatchEvent( _eActivity ) ;
   305  	}
   306  
   307  	/**
   308  	 * Invoqued when the Microphone status change.
   309  	 */
   310  	private function _onMicroStatus( oInfo:Object ):Void
   311  	{
   312  		var code:String = oInfo.code ;
   313  		switch( code )
   314  		{
   315  			case MicrophoneStatus.MUTED :
   316  			{
   317  				dispatchEvent( _eMuted ) ;
   318  				break ;
   319  			}	
   320  			case MicrophoneStatus.UNMUTED :
   321  			{
   322  				dispatchEvent( _eUnMuted ) ;
   323  				break ;	
   324  			}
   325  		}
   326  	}
