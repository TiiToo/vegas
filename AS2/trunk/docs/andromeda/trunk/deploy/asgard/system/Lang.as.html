   128  	public static var LANGS:HashMap = new HashMap () ;
   129  	
   130  	/**
   131  	 * Returns the name of the Lang define with the specified id in argument.
   132  	 * @return the name of the Lang define with the specified id in argument.
   133  	 */
   134  	public static function get(id:String):String 
   135  	{
   136  		if (LANGS.size() == 0) 
   137  		{
   138  			Lang.initialize() ;
   139  		}
   140  		return LANGS.get(id) ;			
   141  	}
   142  
   143  	/**
   144  	 * Initialize the model of this class.
   145  	 */
   146  	public static function initialize():Void 
   147  	{
   148  		LANGS.clear() ;
   149  		LANGS.put(CS, CS_FULL) ;
   150  		LANGS.put(DA, DA_FULL) ;

   152  		LANGS.put(EN, EN_FULL) ;

   155  		LANGS.put(DE, DE_FULL) ;
   156  		LANGS.put(HU, HU_FULL) ;
   157  		LANGS.put(IT, IT_FULL) ;

   159  		LANGS.put(KO, KO_FULL) ;
   160  		LANGS.put(NO, NO_FULL) ;
   161  		LANGS.put(XU, XU_FULL) ;

   170  	}

   172  	/**
   173  	 * Inserts a new Lang in the Lang model.
   174  	 */
   175  	public static function put( lang:Lang , label:String ) 
   176  	{
   177  		return LANGS.put(lang, label) ;
   178  	}
   179  
   180  	/**
   181  	 * Removes a new Lang in the Lang model.
   182  	 */
   183  	public static function remove( lang:Lang ) 
   184  	{
   185  		return LANGS.remove(lang) ;
   186  	}
   187  
   188  	/**
   189  	 * Validate the specified lang in argument.
   190  	 * @return {@code true} if the lang passed in argument is registered in the model.
   191  	 */
   192  	public static function validate( lang:String ):Boolean 
   193  	{
   194  		
   195  		var langs:Array = 
   196  		[
   197  			CS, DA, NL, EN, FI, FR,
   198  			DE, HU, IT, JA, KO, NO,
   199  			XU, PL, PT, RU, ZH_CN,
   200  			ES, SV, ZH_TW, TR
   201  		] ;
   202  		
   203  		return ArrayUtil.contains(langs, lang) ;
   204  	}
