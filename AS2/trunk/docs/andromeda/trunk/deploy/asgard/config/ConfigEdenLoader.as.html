    24  import asgard.config.Config;
    25  import asgard.events.ConfigLoaderEvent;
    26  import asgard.events.LoaderEvent;
    27  import asgard.net.EdenLoader;
    28  import asgard.net.URLRequest;
    29  
    30  import vegas.errors.Warning;
    31  
    32  /**
    33   * The ConfigEdenLoader class based on the Eden notation.
    34   * @author eKameleon
    35   */
    36  class asgard.config.ConfigEdenLoader extends EdenLoader 
    37  {
    38  	
    39  	/**
    40  	 * Creates a new ConfigEdentLoader instance.
    41  	 */
    42  	function ConfigEdenLoader( config ) 
    43  	{
    44  		_oConfig = config || Config.getInstance() ;
    45  	}
    46  
    47  	/**
    48  	 * The default config file name.
    49  	 */
    50  	public var default_file_name:String = "config" ;
    51  	
    52  	/**
    53  	 * The default suffix of config file name.
    54  	 */
    55  	public var default_file_suffix:String = ".eden" ;
    56  	
    57  	/**
    58  	 * (read-write) Returns the config file name.
    59  	 * @return the config file name.
    60  	 */
    61  	public function get fileName():String 
    62  	{
    63  		return getFileName() ;	
    64  	}
    65  	
    66  	/**
    67  	 * (read-write) Sets the config file name.
    68  	 */
    69  	public function set fileName(s:String):Void 
    70  	{
    71  		setFileName(s) ;	
    72  	}
    73  
    74  	/**
    75  	 * (read-write) Returns the prefix path of the config file name.
    76  	 * @return the prefix path of the config file name.
    77  	 */
    78  	public function get path():String 
    79  	{
    80  		return getPath() ;	
    81  	}
    82  
    83  	/**
    84  	 * (read-write) Sets the prefix path of the config file name.
    85  	 */
    86  	public function set path(s:String):Void 
    87  	{
    88  		setPath(s) ;	
    89  	}
    90  
    91  	/**
    92  	 * (read-write) Returns the suffix of the config file name.
    93  	 * @return the suffix of the config file name.
    94  	 */
    95  	public function get suffix():String 
    96  	{
    97  		return getSuffix() ;	
    98  	}
    99  	
   100  	/**
   101  	 * (read-write) Sets the suffix of the config file name.
   102  	 */
   103  	public function set suffix(s:String):Void 
   104  	{
   105  		setSuffix(s) ;	
   106  	}
   107  
   108  	/**
   109  	 * This method launch the deserialization of the external loaded data in the current Config object. 
   110  	 */
   111  	public function deserializeData():Void 
   112  	{
   113  		super.deserializeData() ;
   114  		var data = getData() ;
   115  		for (var each:String in data) 
   116  		{
   117  			_oConfig[each] = data[each] ;
   118  		}
   119  	}
   120  	
   121  	/**
   122  	 * Returns the Config object reference.
   123  	 * @return the Config object reference.
   124  	 */
   125  	public function getConfig():Config 
   126  	{
   127  		return _oConfig ;	
   128  	}
   129  
   130  	/**
   131  	 * Returns the config file name.
   132  	 * @return the config file name.
   133  	 */
   134  	public function getFileName():String 
   135  	{
   136  		return (_fileName == null) ? default_file_name : _fileName ;	
   137  	}
   138  
   139  	/**
   140  	 * Returns the prefix path of the config file name.
   141  	 * @return the prefix path of the config file name.
   142  	 */
   143  	public function getPath():String 
   144  	{
   145  		return _path || "" ;
   146  
   147  	}
   148  
   149  	/**
   150  	 * Returns the suffix of the config file name.
   151  	 * @return the suffix of the config file name.
   152  	 */
   153  	public function getSuffix():String 
   154  	{
   155  		return (_suffix == null) ? default_file_suffix : _suffix ;
   156  	}
   157  
   158  	public function initEvent():Void 
   159  	{
   160  		_e = new ConfigLoaderEvent( null, this ) ;
   161  	}
   162  	
   163  	/**
   164  	 * Loads the specified config file.
   165  	 * @param fileName the name of the file to load to create the config object of the application.
   166  	 */
   167  	public function load( fileName:String ):Void 
   168  	{
   169  		if (fileName) 
   170  		{
   171  			setFileName(fileName) ;
   172  		}
   173  		var uri:String = getPath() + getFileName() + getSuffix() ;
   174  		super.load(new URLRequest(uri) ) ;
   175  	}
   176  	
   177  	/**
   178  	 * Invoqued when the loader is initialize.
   179  	 * Overrides this method.
   180  	 */
   181    	/*override*/ public function onLoadInit(e:LoaderEvent):Void 
   182    	{
   183  		// 
   184  	}
   185  
   186  	/**
   187  	 * Protects a config object with a __resolve method and a Warning exception. 
   188  	 * This protection is important to test if the external properties exist in the external config.
   189  	 */
   190  	static public function protectConfig( oConfig ):Void 
   191  	{
   192  		oConfig.__resolve = function( p:String ):String 
   193  		{
   194  			try
   195  			{
   196  				throw new Warning("Config object : '" + p + "' property is undefined") ;
   197  			}
   198  			catch(e)
   199  			{
   200  					
   201  			}
   202  			return "" ;
   203  		} ;
   204  		_global.ASSetPropFlags(oConfig, "__resolve", 7, 1) ;
   205  	}
   206  
   207  	/**
   208  	 * Sets the config file name.
   209  	 */
   210  	public function setFileName(sFileName:String):Void 
   211  	{
   212  		_fileName = sFileName ;	
   213  	}
   214  
   215  	/**
   216  	 * Sets the prefix path of the config file name.
   217  	 */
   218  	public function setPath( sPath:String ):Void 
   219  	{
   220  		_path = sPath || "" ;
   221  	}
   222  
   223  	/**
   224  	 * Sets the suffix of the config file name.
   225  	 */
   226  	public function setSuffix( sSuffix:String ):Void 
   227  	{
   228  		_suffix = sSuffix || null ;
   229  	}
   230  
   231  	private var _fileName:String = null ;
   232  
   233  	private var _oConfig:Config ;
   234  
   235  	private var _path:String = null ;
   236  
   237  	private var _suffix:String = null ;
   238  
   239  }
