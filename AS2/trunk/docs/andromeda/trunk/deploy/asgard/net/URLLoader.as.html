   141  	/**
   142  	 * Load the external content in the loader.
   143  	 */
   144  	/*override*/ public function load(request:URLRequest):Void 
   145  	{
   146  
   147  		_oData = null ;
   148  		
   149  		if (request) 
   150  		{
   151  		
   152  			System.useCodepage = (request.getUseCodePage() || false) ;
   153  			
   154  			if (request.getUrl() != undefined) 
   155  			{
   156  				setUrl( request.getUrl() ) ;
   157  			}
   158  			
   159  			_sContentType = request.getContentType() || URLRequest.DEFAULT_CONTENT_TYPE ;
   160  			
   161  			_sMethod = request.getMethod() ;
   162  			
   163  			_aHeaders = request.getRequestHeaders() ;
   164  			
   165  		}
   166  		
   167  		var sender:LoadVars = new LoadVars() ;
   168  		//sender.contentType = _sContentType ;
   169  		
   170  		for (var prop:String in request.data)
   171  		{
   172  			sender[prop] = request.data[prop] ;	
   173  		}
   174  		
   175  		var len:Number = _aHeaders.length ;
   176  		if (len > 0) 
   177  		{
   178  			while(--len > -1) 
   179  			{
   180  				var header:URLRequestHeader = _aHeaders[len] ;
   181  				sender.addRequestHeader( header.name , header.value ) ;
   182  			} 
   183  		}
   184  		
   185  		var receive:LoadVars = new LoadVars() ;
   186  		receive["onHTTPStatus"] = Delegate.create(this, _onHTTPStatus, receive) ;
   187  		receive.onData = Delegate.create(this, _onData) ;
   188  
   189  		setContent( receive ) ;
   190  				
   191  		sender.sendAndLoad(this.getUrl(), receive, _sMethod ) ;
   192  		
   193    		super.load() ;
   194    		
   195    	}

   240

   242

   245  	private var _sMethod:String = "POST" ;
   246  
   247  	private var _oData = null ;
   248  	
   249  	private function _onData( source:String )
   250  	{
   251  
   252  		if (source == undefined) 
   253  		{
   254  			
   255  			_onLoad(false) ;
   256  			
   257  		}
   258  		else 
   259  		{
   260  		
   261  			switch (getDataFormat()) {
   262  			
   263  				case DataFormat.TEXT :
   264  					_oData = new String(source) ;
   265  					break ;
   266  				
   267  				case DataFormat.VARIABLES :
   268  					_oData = new URLVariables(source) ;
   269  					break ;
   270  
   271  				case DataFormat.BINARY :
   272  				default :
   273  					_oData = source ;
   274  					break ;
   275  
   276  			}
   277  
   278  			_onLoad(true) ;
   279  					
   280  		}
   281  	}
   282  	
   283  	/**
   284  	 * Invoqued when the source is loading or not.
   285  	 */
   286  	private function _onLoad(success:Boolean):Void 
   287  	{
   288  		
   289  		if (success) 
   290  		{
   291  			checkData() ;
   292  		}
   293  		else 
   294  		{
   295  			notifyError(this + " : Unload to load/parse URL file : " + this.getUrl() , null ) ;
   296  		}
   297  	}
   298  	
   299    	private function _onHTTPStatus( httpStatus:Number , loader)
   300    	{
   301    		var httpStatusType:String = "unknow httpstatus" ;
   302       	
   303       	if(httpStatus < 100) 
   304       	{
   305            	httpStatusType = "flashError";
   306       	}
   307       	else if(httpStatus < 200) 
   308       	{
   309           	httpStatusType = "informational";
   310       	}
   311       	else if(httpStatus < 300) 
   312       	{
   313          	httpStatusType = "successful";
   314       	}
   315       	else if(httpStatus < 400) 
   316       	{
   317            	httpStatusType = "redirection";
   318       	}
   319       	else if(httpStatus < 500) 
   320       	{
   321           	 httpStatusType = "clientError";
   322       	}
   323       	else if(httpStatus < 600) 
   324       	{
   325            	httpStatusType = "serverError";
   326       	}
   327       	
   328       	getLogger().warn( this + " HTTP status:" + httpStatus + ", type:" + httpStatusType ) ;
   329    		
   330    	}
