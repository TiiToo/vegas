    44  /**
    45   * The Stream class opens a one-way streaming connection between Flash Player and RTMP server through a connection made available by a NetConnection object.
    46   * A Stream object is like a channel inside a NetConnection object; this channel can either publish audio 
    47   * and/or video data, using NetStream.publish(), or subscribe to a published stream and receive data, 
    48   * using NetStream.play(). You can publish or play live (real-time) data and previously recorded data.
    49   * @author eKameleon
    50   */

   476  	 * Invoqued when the onMetaData changed.
   477  	 */
   478  	private function onMetaData( info:Object ):Void 
   479  	{
   480  
   481  		for (var props in info) 
   482  		{
   483  			getLogger().info( this + " onMetaData, " + props + " : " + info[props]) ;
   484  		}
   485  		
   486  		_oMetaData = new FLVMetaData(info) ;
   487  		
   488  		setDuration( isNaN(info.duration) ? 0 : parseInt(info.duration) ) ;
   489  		
   490  	}

   510  	/**
   511  	 * Internal method to register the Stream in the StreamCollector with the specified id in argument.
   512  	 * @see StreamCollector.
   513  	 */
   514  	private function _setID( id ):Void 
   515  	{
   516  		if ( StreamCollector.contains( this._id ) )
   517  		{
   518  			StreamCollector.remove( this._id ) ;
   519  		}
   520  		this._id = id ;
   521  		if (this._id != null)
   522  		{
   523  			StreamCollector.insert ( this._id, this ) ;
   524  		}
   525  	}
