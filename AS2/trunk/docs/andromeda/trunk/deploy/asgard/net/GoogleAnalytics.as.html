     8  /**
     9   * Track application/site internal navigation throw Google Analytic system.
    10   * <p>More informations on http://www.google.com/analytics/</p>
    11   * <p>Link passed to tracker is check before processing :<br />
    12   * At the end, link must start with a {@code /} char with no space chars at the end.
    13   * </p>
    14   * <p>Analytics tracking code must be placed in your HTML code above any of these calls.<br />
    15   * <b>Example</b>
    16   * {@code
    17   *    
    18   *    //<![CDATA[
    19   *    <script type="text/javascript" src="http://www.google-analytics.com/urchin.js"></script>
    20   *    _uacct = "YOUR_GOOGLE_TRACK_ID";
    21   *    urchinTracker();	
    22   *    //]]>
    23   *    
    24   * }
    25   * </p>
    26   * <p>Check if {@code ExternalInterface.available}.<br />
    27   * if {@code true}, use {@code ExternalInterface.call}, otherwise use {@code getURL( "javascript:"" )}.
    28   * </p>
    29   * Thanks Romain Ecarnot with this Fever version of this class who inspire me this version.
    30   * @see asgard.system.Application
    31   * @see vegas.util.StringUtil
    32   * @author eKameleon
    33   */
    34  class asgard.net.GoogleAnalytics 
    35  {
    36  
    37  	/** 
    38  	 * Default logical directory structure for outbound links.
    39  	 */
    40  	public static var OUTGOING_TRACK:String = "outgoing";
    41  	
    42  	/** 
    43  	 * Default logical directory structure for downloads links.
    44  	 */
    45  	public static var DOWNLOAD_TRACK:String = "downloads";
    46  	
    47  	/**
    48  	 * The name of the application.
    49  	 */
    50  	public static var NAME:String = "" ;
    51  
    52  	/**
    53  	 * Track the passed-in {@code link}.
    54  	 */
    55  	public static function track( link : String ) : Void
    56  	{
    57  		_track( _getCorrectLink( link ) );	
    58  	}
    59  	
    60  	/**
    61  	 * Track the passed-in {@code link} as an outbound link.
    62  	 * <p>Logical link structure can me modify with {@link #OUTGOING_TRACK} property.</p>
    63  	 * @param link the link to be track
    64  	 */
    65  	public static function trackOutBound( link : String ):Void
    66  	{
    67  		_track( _getCorrectLink( link, OUTGOING_TRACK ) );
    68  	}
    69  	
    70  	/**
    71  	 * Track passed-in {@code link} as a download link.
    72  	 * <p>Logical link structure can me modify with {@link #DOWNLOAD_TRACK} property.</p>
    73  	 * @param link the link to be track
    74  	 */
    75  	public static function trackDownload( link : String ):Void
    76  	{
    77  		_track( _getCorrectLink( link, DOWNLOAD_TRACK ) );		
    78  	}
    79  
    80  	/**
    81  	 * Creates a correct Google Analytic track link.
    82  	 */
    83  	private static function _getCorrectLink( link : String, baseLink : String ) : String
    84  	{
    85  		if( link )
    86  		{
    87  			link = "/" + StringUtil.trim( link , String(EMPTY_CHARS + "/").split("") );
    88  			link = StringUtil.replace( link, " ", "_" );
    89  			link = StringUtil.replace( link, "http://", "" );
    90  			link = StringUtil.replace( link, "https://", "" );
    91  			if( baseLink ) 
    92  			{
    93  				link = "/" + baseLink + link;
    94  			}
    95  		}
    96  		else 
    97  		{
    98  			link = "";
    99  		}
   100  		
   101  		var appName:String = (NAME.length > 0) ? NAME : "" ;
   102  		appName = ( appName ) ? ( "/" + StringUtil.trim( appName, String(EMPTY_CHARS + "/").split("") ) ) : "" ;	
   103  		
   104  		var result : String = appName.toLowerCase() + link; 
   105  		return result;
   106  	}
   107  
   108  	/**
   109  	 * Call {@code urchinTracker()} javascript method with correct link.
   110  	 * <p>Check if {@code ExternalInterface.available}.<br />
   111  	 * if {@code true}, use {@code ExternalInterface.call}, otherwise use {@code getURL( "javascript:"" )}.
   112  	 * </p>
   113  	 * @param link Correct link built with {@link #_buildCorrectLink()}
   114  	 * @throws Warning if the application isn't online.
   115  	 */
   116  	private static function _track( link : String ) : Void
   117  	{
   118  		if( Application.isOnline() && ( link.length > 2 ) )
   119  		{
   120  			if( ExternalInterface.available ) 
   121  			{
   122  				ExternalInterface.call( "urchinTracker", link);
   123  			}
   124  			else 
   125  			{
   126  				getURL("javascript:urchinTracker('"+ link +"');");
   127  			}
   128  		}
   129  		else
   130  		{
   131  			throw new Warning( "GoogleAnalytic track failed, your application isn't online : " + link ) ;	
   132  		}
   133  	}
   134  
   135  	private static var EMPTY_CHARS:String = "\n\tr " + chr(13) + chr(10);
   136  
   137  }
