    24  import vegas.core.CoreObject;
    25  
    26  /**
    27   * The FileFilter class is used to indicate what files on the user's system are shown in the file-browsing dialog box that 
    28   * is displayed when {@code FileReference.browse()} or {@code FileReferenceList.browse()} is called. 
    29   * FileFilter instances are passed to a browse() function. 
    30   * If you use a FileFilter instance, extensions and file types that aren't specified in the FileFilter instance are filtered out; that is, they are not available to the user to select for uploading. If no FileFilter object is passed to browse(), all files are shown in the dialog box.
    31   * 
    32   * <p>You can use FileFilter instances in one of two ways :
    33   * <ul>
    34   * <li>A description with Windows file extensions only</li>
    35   * <li>A description with Windows file extensions and Macintosh file types</li>
    36   * </ul>
    37   * </p>
    38   * <p>The two formats are not interchangeable within a single browse() call. You must use one or the other.</p>
    39   * <p>You can pass one or more FileFilter instances to {@code FileReference.browse()} or {@code FileReferenceList.browse()}. 
    40   * The following examples show different ways to create and pass {@code FileFilter} instances to a {@code browse()} call (for Windows only). 
    41   * The first example creates FileFilter instances outside of the browse() call:</p>
    42   * {@code
    43   * var imagesFilter:FileFilter = new FileFilter("Images", "*.jpg;*.gif;*.png") ;
    44   * var docFilter:FileFilter    = new FileFilter("Documents", "*.pdf;*.doc;*.txt") ;
    45   * var myFileReference:FileReference = new FileReference() ;
    46   * myFileReference.browse([imagesFilter, docFilter]) ;
    47   * }
    48   * <p>The second example creates FileFilter instances within the browse() call:</p>
    49   * {@code
    50   * myFileReference.browse( [ new FileFilter("Images", "*.jpg;*.gif;*.png"), new FileFilter("Flash Movies", "*.swf") ] );
    51   * }
    52   * The list of extensions in the FileFilter.extension property is used to filter the files in Windows, 
    53   * depending on the file selected by the user. 
    54   * It is not actually displayed in the dialog box; to display the file types for users, you must list the file 
    55   * types in the description string as well as in the extension list. 
    56   * The description string is displayed in the dialog box in Windows. 
    57   * (It is not used on the Macintosh.) On the Macintosh, if you supply a list of Macintosh file types, that list is used to filter the files. 
    58   * If not, the list of Windows extensions is used.
    59   * @author eKameleon
    60   */
    61  class asgard.net.FileFilter extends CoreObject 
    62  {
    63  
    64  	/**
    65  	 * Creates a new FileFilter instance.
    66  	 * @param description The description string that is visible to users when they select files for uploading.
    67  	 * @param extension A list of file extensions that indicate which Windows file formats are visible to users when they select files for uploading.
    68  	 * @param macType A list of Macintosh file types that indicate which file types are visible to users when they select files for uploading.
    69  	 */	
    70  	public function FileFilter( description:String, extension:String, macType:String ) 
    71  	{
    72  		this.description = description || "" ;
    73  		this.extension   = extension   || "" ;
    74  		this.macType     = macType     || "" ;
    75  	}
    76  	
    77  	/**
    78  	 * The description string for the filter.
    79  	 * The description is visible to the user in the dialog box that opens when 
    80  	 * {@code FileReference.browse()} or {@code FileReferenceList.browse()} is called. 
    81  	 * The description string contains a string, such as "Images (*.gif, *.jpg, *.png)", 
    82  	 * that can help instruct the user on what file types can be uploaded or downloaded. 
    83  	 * Note that the actual file types that are supported by this FileReference object are stored in the extension  property.
    84  	 */
    85  	public var description:String ;
    86  
    87  	/**
    88  	 * The list o file extensions.
    89  	 * This list indicates the types of files that you want to show in the file-browsing dialog box. 
    90  	 * (The list is not visible to the user; the user sees only the value of the description property.) 
    91  	 * The extension property contains a semicolon-delimited list of Windows file extensions, with a wildcard (*) preceding each extension, as shown in the following string: "*.jpg;*.gif;*.png".
    92  	 */
    93  	public var extension:String ;
    94  	
    95  	/**
    96  	 * The list of Macintosh file types.
    97  	 * This list indicates the types of files that you want to show in the file-browsing dialog box. 
    98  	 * (This list itself is not visible to the user; the user sees only the value of the description property.) 
    99  	 * The macType property contains a semicolon-delimited list of Macintosh file types, as shown in the following string: "JPEG;jp2_;GIFF".
   100  	 */
   101  	public var macType:String ;
   102  
   103  }
