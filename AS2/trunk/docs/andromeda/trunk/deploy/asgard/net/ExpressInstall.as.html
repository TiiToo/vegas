     2  /**
     3   * Use this class to invoke the Macromedia Flash Player Express Install functionality
     4   * ExpressInstall class based on the SWFObject implementation. 
     5   * This file is intended for use with the FlashObject embed script. You can download FlashObject 
     6   * and this class here: http://blog.deconcept.com/flashobject/
     7   * <p>Usage :</p>
     8   * <p>Launch this code in the first frame of your Flash movie :</p>
     9   * {@code
    10   * asgard.net.ExpressInstall.init() ;
    11   * }
    12   * You should not place any other code on the first frame of your movie, as the first frame needs to be playable in the Flash player v. 6.0.65.
    13   */
    14  class asgard.net.ExpressInstall 
    15  {
    16  
    17  	/**
    18  	 * Creates a new ExpressInstall instance.
    19  	 */
    20  	public function ExpressInstall() 
    21  	{
    22  		//
    23  	}
    24  
    25  	/**
    26  	 * Returns the singleton instance of the ExpressInstall class.
    27  	 * @return the singleton instance of the ExpressInstall class.
    28  	 */
    29  	public static function getInstance():ExpressInstall 
    30  	{
    31  		if (_instance == null) _instance = new ExpressInstall() ;
    32  		return _instance ;
    33  	}
    34  
    35  	/**
    36  	 * Initialize the Express Install. 
    37  	 * if the flashvars 'MMplayerType' is not set, the var was not passed in from the SWFObject embed, so the upgrade is not needed.
    38  	 */
    39  	public static function init():Boolean 
    40  	{
    41  		if ( _root.MMplayerType == undefined ) 
    42  		{
    43  			return false ;
    44  		}
    45  		ExpressInstall.getInstance()._loadUpdater() ;
    46  		return true ;
    47  	}
    48  
    49  	/**
    50  	 * Start the loading.
    51  	 */
    52  	public function loadInit():Void 
    53  	{
    54  		_hold.redirectURL  = _root.MMredirectURL ;
    55  		_hold.MMplayerType = _root.MMplayerType  ;
    56  		_hold.MMdoctitle   = _root.MMdoctitle    ;
    57  		_hold.startUpdate() ;
    58  	}
    59  
    60  	/**
    61  	 * The allow domain of the Express install plugin.
    62  	 */
    63  	public static var ALLOW_DOMAIN:String = "fpdownload.macromedia.com" ;
    64  
    65  	/**
    66  	 * The end user chose "NO" when prompted to install the new player. 
    67  	 * By default no User Interface is presented in this case. It is left up to  
    68  	 * the developer to provide an alternate experience in this case 
    69  	 * feel free to change this to whatever you want, js errors are sufficient for this example.
    70  	 */ 
    71  	public static var DOWNLOAD_CANCEL:String = "Download.Cancelled" ;
    72  
    73  	/**
    74  	 * Installation is complete. 
    75  	 * In most cases the browser window that this SWF is hosted in will be closed 
    76  	 * by the installer or manually by the end user 
    77  	 */ 
    78  	public static var DOWNLOAD_COMPLETE:String = "Download.Complete" ;
    79  	
    80  	/**
    81  	 * The end user failed to download the installer due to a network failure 
    82  	 * by default no User Interface is presented in this case. It is left up to 
    83  	 * the developer to provide an alternate experience in this case 
    84  	 * feel free to change this to whatever you want, js errors are sufficient for this example.
    85  	 */
    86  	public static var DOWNLOAD_FAILED:String = "Download.Failed" ;
    87  	
    88  	/**
    89  	 * Invoqued when the status of the express install change.
    90  	 */
    91     	public function installStatus( statusValue ):Void 
    92     	{
    93  		switch(statusValue) 
    94  		{
    95  			case DOWNLOAD_COMPLETE :
    96  			{
    97  				getURL("javascript:window.opener=self;self.close()");
    98  				break ;
    99  			}
   100  			case DOWNLOAD_CANCEL :
   101  			{
   102  				getURL("javascript:alert('This content requires a more recent version of the Macromedia Flash Player.')") ;
   103  				break ;
   104  			}
   105  			case DOWNLOAD_FAILED :
   106  			{
   107  				getURL("javascript:alert('There was an error in downloading the Flash Player update. Please try again later, or visit macrmedia.com to download the latest version of the Flash plugin.')");
   108  				break ;
   109  			}
   110  		}
   111  	}
   112  
   113  	private var _hold:MovieClip ;
   114  	
   115  	private static var _instance:ExpressInstall ;
   116  	
   117  	private var _updater:MovieClip ;
   118  
   119  	/**
   120  	 * Load the updater.
   121  	 */
   122  	private function _loadUpdater():Void 
   123  	{
   124  		
   125  		System.security.allowDomain( ALLOW_DOMAIN ) ;
   126  		_updater = _root.createEmptyMovieClip("expressInstallHolder", 10000000);
   127  		// register the callback so we know if they cancel or there is an error
   128  		_updater.installStatus = this.installStatus ;
   129  		
   130  		_hold = _updater.createEmptyMovieClip("hold", 1) ;
   131  		
   132  		_updater.onEnterFrame = function() 
   133  		{
   134  			if (typeof ( this.hold.startUpdate ) == "function") 
   135  			{
   136  				ExpressInstall.getInstance().loadInit() ;
   137  				delete this.onEnterFrame ;
   138  			}
   139  		} ;
   140  		
   141  		var cache:Number = Math.random() ;
   142  		
   143  		_hold.loadMovie("http://fpdownload.macromedia.com/pub/flashplayer/update/current/swf/autoUpdater.swf?" + cache ) ;
   144  	
   145  	}
   146  
   147  }
