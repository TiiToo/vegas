    67  	/**
    68  	 * Decodes the specified string.
    69  	 * @return the decode string.
    70  	 */
    71  	static function decode( text:String, removeCRLF:Boolean ):String 
    72  	{
    73  		var i:Number ;
    74  		var ch:String ;
    75  		var entity:String ;
    76  		var len:Number = entities.length ;
    77  		
    78  		for( i=0; i<len ; i++ )
    79  		{
    80  			ch = specialchars[ i ];
    81  			entity = entities[ i ];
    82  			
    83  			// trace(ch + " : " + i) ;
    84  			
    85  			if( text.indexOf( entity ) > -1 )
    86  			{
    87  				text = StringUtil.replace( text, entity, ch );
    88  			}
    89  		}
    90  		
    91  		if( removeCRLF )
    92  		{
    93  			text = StringUtil.replace( text, "r\n", "" );
    94  		}
    95  		return text;
    96  	}

    98  	/**
    99  	 * Encodes the specified text passed in argument.
   100  	 * @return a string encode text.
   101  	 */
   102  	static function encode( text:String ):String
   103  	{
   104  		var i:Number ;
   105  		
   106  		var ch:String ;
   107  		
   108  		var entity:String ;
   109  		
   110  		var len:Number = entities.length ;
   111  		
   112  		for( i=0; i<len; i++ )
   113  		{
   114  			
   115  			ch = specialchars[i];
   116  			entity = entities[i];
   117  			
   118  			if( text.indexOf( ch ) > -1 )
   119  			{
   120  				text = StringUtil.replace( text, ch, entity );
   121  			}
   122  		}
   123  		return text.toString() ;
   124  	}
