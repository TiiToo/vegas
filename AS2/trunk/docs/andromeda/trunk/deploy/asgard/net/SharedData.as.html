    24  import asgard.events.SharedDataEvent;
    25  import asgard.events.SharedDataEventType;
    26  import asgard.net.NetServerConnection;
    27  import asgard.net.SharedDataStatus;

    29  import vegas.events.AbstractCoreEventDispatcher;

    32  /**
    33   * The {@code SharedData} class is used to read and store limited amounts of data on a user's computer or on a server.
    34   * @author eKameleon
    35   * @version 1.0.0.0
    36   */
    37  class asgard.net.SharedData extends AbstractCoreEventDispatcher
    38  {
    39  	
    40  	/**
    41  	 * Creates a new SharedData instance.
    42  	 * @param sName the name of the SharedData reference.
    43  	 * @param nc The NetServerConnection reference of this SharedData object.
    44  	 * @param persistant indicates if the internal SharedObject of this SharedData is persistant.
    45  	 * @param autoConnect Indicates if the SharedData is auto connect.
    46  	 */
    47  	function SharedData( sName:String, nc:NetServerConnection, persistant:Boolean, autoConnect:Boolean) 
    48  	{
    49  		
    50  		_isConnected = false ;
    51  		_isFirst = true ;
    52  	
    53  		initEvent() ;
    54  		
    55  		initialize(sName, nc, persistant, autoConnect) ;
    56  		
    57  	}
    58  
    59  	/**
    60  	 * [read-only] The collection of attributes assigned to the data property of the object; 
    61  	 * these attributes can be shared and stored.
    62  	 */
    63  	public function get data() 
    64  	{
    65  		return getData() ; 
    66  	}
    67  	
    68  	/**
    69  	 * [read-only] Indicates if the SharedData is connected.
    70  	 */
    71  	public function get isConnected():Boolean 
    72  	{
    73  		return getIsConnected() ;	
    74  	}
    75  	
    76  	/**
    77  	 * For local shared objects, purges all of the data and deletes the shared object from the disk.
    78  	 */
    79  	public function clear():Void 
    80  	{
    81  		_so.clear() ;
    82  	}
    83  	
    84  	/**
    85  	 * Closes the connection between a remote shared object and the server.
    86  	 */
    87   	public function close():Void 
    88   	{
    89  		if (_isConnected) 
    90  		{
    91  			_so.close() ;
    92  			_isConnected = false ;
    93  			dispatchEvent(_eClose) ;
    94  		}
    95  	}
    96  
    97  	/**
    98  	 * Connects to a remote shared object on the server through the specified connection.
    99  	 * @param nc A NetServerConnection object (such as one used to communicate with Flash Media Server) that is using the Real-Time Messaging Protocol (RTMP). 
   100  	 */
   101  	public function connect( nc:NetServerConnection ):Boolean 
   102  	{
   103  		close() ;
   104  		_isConnected = _so.connect(nc) ;
   105  		return _isConnected ;
   106  	}
   107  	
   108  	/**
   109  	 * Dispatches a context object to all the reference of this SharedData in all remote application.
   110  	 */
   111  	public function fireEvent( eContext ):Void 
   112  	{
   113    		_so.send(SharedDataEventType.FIRE, eContext );
   114    	}
   115  	
   116  	/**
   117  	 * Immediately writes a locally persistent shared object to a local file.
   118  	 */
   119  	public function flush():Void 
   120  	{
   121  		_so.flush() ;
   122  	}	
   123  
   124  	/**
   125  	 * Returns the collection of attributes assigned to the data property of the object; 
   126  	 * these attributes can be shared and stored.
   127  	 */
   128  	public function getData() 
   129  	{
   130  		return _so.data ;	
   131  	}
   132  
   133  	/**
   134  	 * Returns {@code true} if the SharedData is connected.
   135  	 * @return {@code true} if the SharedData is connected.
   136  	 */
   137  	public function getIsConnected():Boolean 
   138  	{
   139  		return _isConnected ;	
   140  	}
   141  	
   142  	/**
   143  	 * Returns the value of the specified property.
   144  	 * @return  the value of the specified property.
   145  	 */
   146  	public function getProperty(sName:String) 
   147  	{
   148    		return _so.data[sName];
   149    	}
   150    	
   151    	/**
   152    	 * Initialize all internal events of this SharedData instance.
   153    	 */
   154    	public function initEvent():Void
   155    	{
   156    		_eChange  = new SharedDataEvent( SharedDataEventType.CHANGE       , this ) ;
   157  		_eClear   = new SharedDataEvent( SharedDataEventType.CLEAR        , this ) ;
   158  		_eClose   = new SharedDataEvent( SharedDataEventType.CLOSE        , this ) ;
   159  		_eFire    = new SharedDataEvent( SharedDataEventType.FIRE         , this ) ;
   160  		_eDelete  = new SharedDataEvent( SharedDataEventType.DELETE       , this ) ;
   161  		_eReject  = new SharedDataEvent( SharedDataEventType.REJECT       , this ) ;
   162  		_eSuccess = new SharedDataEvent( SharedDataEventType.SUCCESS      , this ) ;
   163  		_eSync    = new SharedDataEvent( SharedDataEventType.SYNCHRONISED , this ) ;
   164    	}
   165    		
   166  	/**
   167  	 * Initialize the SharedData instance.
   168  	 * @param sName the name of the SharedData reference.
   169  	 * @param nc The NetServerConnection reference of this SharedData object.
   170  	 * @param persistant indicates if the internal SharedObject of this SharedData is persistant.
   171  	 * @param autoConnect Indicates if the SharedData is auto connect.
   172  	 */
   173  	public function initialize(sName:String, nc:NetServerConnection, persistant:Boolean, autoConnect:Boolean):Void 
   174  	{
   175  		if (_so ) 
   176  		{
   177  			_so.close() ;
   178  		}
   179  		_so = SharedObject.getRemote(sName, nc.uri, persistant) ;
   180  		_so.onSync = Delegate.create(this, _onSync) ;
   181  		_so[SharedDataEventType.FIRE] = Delegate.create(this, _onFired) ;	
   182  		if (autoConnect) 
   183  		{
   184  			connect(nc) ;
   185  		}
   186  	}
   187  
   188  	/**
   189  	 * Locks the object.
   190  	 */
   191  	public function lock():Void 
   192  	{
   193  		super.lock() ;
   194  		_so.setFps(0);
   195  	}
   196  	
   197  	/**
   198  	 * Resets all the properties of the SharedData.
   199  	 */
   200  	public function reset():Void 
   201  	{
   202  		var o:Object = getData() ;
   203  		for (var each:String in o) 
   204  		{
   205  			delete o[each] ;
   206  		}
   207  	}	
   208  	
   209  	/**
   210  	 * Specifies the number of times per second that a client's changes to a shared object are sent to the server.
   211  	 */
   212  	public function setFps(n:Number):Boolean 
   213  	{
   214    		return _so.setFps(n);
   215    	}
   216    	
   217    	/**
   218    	 * Updates the value of a property (defined with the data property) in a shared object and indicates to the server that the value of the property has changed.
   219    	 */
   220    	public function setProperty(sName:String, value ):Void 
   221    	{
   222    		_so.data[sName] = value ;
   223  		_so.flush() ;
   224    	}
   225  	
   226  	/**
   227  	 * Returns the current size of the shared object, in bytes.
   228  	 * @return the current size of the shared object, in bytes.
   229  	 */
   230  	public function size():Number 
   231  	{
   232  		return _so.getSize() ;	
   233  	}
   234  
   235  	/**
   236  	 * Unlock Method like server side SSAS
   237  	 */
   238   	public function unLock():Void 
   239   	{
   240   		super.unLock() ;
   241  		_so.setFps(0) ;
   242  		_so.setFps(-1) ;
   243  	}

   245  	private var _eChange:SharedDataEvent ;
   246  	private var _eClear:SharedDataEvent;
   247  	private var _eClose:SharedDataEvent ;	
   248  	private var _eDelete:SharedDataEvent ;	
   249  	private var _eFire:SharedDataEvent ;
   250  	private var _eReject:SharedDataEvent ;		
   251  	private var _eSuccess:SharedDataEvent ;		
   252  	private var _eSync:SharedDataEvent ;		
   253  
   254  	private var _isConnected:Boolean ;
   255  	private var _isFirst:Boolean ;
   256  	
   257  	private var _so:SharedObject ;
   258  		
   259  	private function _onSync( list ):Void 
   260  	{
   261  		var name:String ;
   262  		if (_isFirst) 
   263  		{
   264  			_isFirst = false ;
   265  			dispatchEvent(_eSync) ;
   266  			for ( name in _so.data ) 
   267  			{
   268  				_eChange.setProperty(name, _so.data[name]) ;
   269  				dispatchEvent( _eChange ) ;
   270  			}
   271  			return ;
   272  		}
   273  		
   274  		for (var prop:String in list) 
   275  		{
   276  			
   277  			var item:Object = list[prop] ;
   278  			var code:SharedDataStatus = SharedDataStatus.format( item.code )  ;
   279  			name = item.name ;
   280  			var value = _so.data[name] ;
   281  
   282  			//**** DEBUG
   283  			// var txt:String = "> " + this + "._onSync :: " + code ;
   284  			// if(name) txt += " , name:" + name ;
   285  			// if(value) txt += " , value:" + value ;
   286  			// trace (txt) ;
   287  			//***/
   288  			
   289  			switch (code) 
   290  			{
   291  				case SharedDataStatus.CLEAR :
   292  				{
   293  					dispatchEvent(_eClear);
   294  					break ;
   295  				}
   296  				case SharedDataStatus.CHANGE :
   297  				{
   298  					_eChange.setProperty(name, value) ;
   299  					dispatchEvent(_eChange);
   300  					break ;
   301  				}	
   302  				case SharedDataStatus.DELETE :
   303  				{
   304  					_eDelete.setProperty(name) ;
   305  					dispatchEvent(_eDelete );
   306  					break ;
   307  				}
   308  				case SharedDataStatus.REJECT :
   309  				{
   310  					_eReject.setProperty(name, value) ;
   311  					dispatchEvent(_eReject );
   312  					break ;
   313  				}
   314  				case SharedDataStatus.SUCCESS :
   315  				{
   316  					_eSuccess.setProperty(name, value) ;
   317  					dispatchEvent(_eSuccess );
   318  					break ;
   319  				}
   320  			}
   321     		}
   322  	}
   323  
   324  	private function _onFired( eventContext ):Void 
   325  	{
   326  		_eFire.setContext ( eventContext ) ;
   327     		dispatchEvent( _eFire ) ;
   328  	}
