    24  import asgard.events.SharedDataEvent;
    25  import asgard.events.SharedDataEventType;
    26  import asgard.net.NetServerConnection;
    27  import asgard.net.SharedDataStatus;

    29  import vegas.events.AbstractCoreEventDispatcher;
    30  import vegas.events.Delegate;

    32  /**
    33   * The {@code SharedData} class is used to read and store limited amounts of data on a user's computer or on a server.
    34   * @author eKameleon
    35   * @version 1.0.0.0
    36   */
    37  class asgard.net.SharedData extends AbstractCoreEventDispatcher 
    38  {
    39  	
    40  	/**
    41  	 * Creates a new SharedData instance.
    42  	 */
    43  	function SharedData( sName:String, nc:NetServerConnection, persistant:Boolean, autoConnect:Boolean) 
    44  	{
    45  		
    46  		_isConnected = false ;
    47  		_isFirst = true ;
    48  	
    49  		_eChange = new SharedDataEvent(SharedDataEventType.CHANGE, this) ;
    50  		_eClear = new SharedDataEvent(SharedDataEventType.CLEAR, this) ;
    51  		_eClose = new SharedDataEvent(SharedDataEventType.CLOSE, this) ;
    52  		_eFire = new SharedDataEvent( SharedDataEventType.FIRE , this ) ;
    53  		_eDelete = new SharedDataEvent(SharedDataEventType.DELETE, this) ;
    54  		_eReject = new SharedDataEvent(SharedDataEventType.REJECT, this) ;
    55  		_eSuccess = new SharedDataEvent(SharedDataEventType.SUCCESS, this) ;
    56  		_eSync = new SharedDataEvent(SharedDataEventType.SYNCHRONISED, this) ;
    57  		
    58  		initialize(sName, nc, persistant, autoConnect) ;
    59  		
    60  	}
    61  
    62  
    63  	public function get data() 
    64  	{
    65  		return getData() ; 
    66  	}
    67  	
    68  	public function get isConnected():Boolean 
    69  	{
    70  		return getIsConnected() ;	
    71  	}
    72  
    73  	public function clear():Void 
    74  	{
    75  		_so.clear() ;
    76  	}
    77  
    78   	public function close():Void 
    79   	{
    80  		if (_isConnected) 
    81  		{
    82  			_so.close() ;
    83  			_isConnected = false ;
    84  			dispatchEvent(_eClose) ;
    85  		}
    86  	}
    87  	
    88  	public function connect(nc:NetServerConnection):Boolean 
    89  	{
    90  		close() ;
    91  		_isConnected = _so.connect(nc) ;
    92  		return _isConnected ;
    93  	}
    94  
    95  	public function fireEvent( eContext ):Void 
    96  	{
    97    		_so.send(SharedDataEventType.FIRE, eContext );
    98    	}
    99  
   100  	public function flush():Void 
   101  	{
   102  		_so.flush() ;
   103  	}	
   104  	
   105  	public function getData() 
   106  	{
   107  		return _so.data ;	
   108  	}
   109  	
   110  	public function getIsConnected():Boolean 
   111  	{
   112  		return _isConnected ;	
   113  	}
   114  
   115  	public function getProperty(sName:String) 
   116  	{
   117    		return _so.data[sName];
   118    	}
   119  
   120  	public function initialize(sName:String, nc:NetServerConnection, persistant:Boolean, autoConnect:Boolean):Void 
   121  	{
   122  		if (_so ) 
   123  		{
   124  			_so.close() ;
   125  		}
   126  		_so = SharedObject.getRemote(sName, nc.uri, persistant) ;
   127  		_so.onSync = Delegate.create(this, _onSync) ;
   128  		_so[SharedDataEventType.FIRE] = Delegate.create(this, _onFired) ;	
   129  		if (autoConnect) 
   130  		{
   131  			connect(nc) ;
   132  		}
   133  	}
   134  
   135  	public function lock():Void 
   136  	{
   137  		_so.setFps(0);
   138  	}
   139  
   140  	public function reset():Void 
   141  	{
   142  		var o:Object = getData() ;
   143  		for (var each:String in o) 
   144  		{
   145  			delete o[each] ;
   146  		}
   147  	}	
   148  	
   149  	public function setFps(n:Number):Boolean 
   150  	{
   151    		return _so.setFps(n);
   152    	}
   153    	
   154    	public function setProperty(sName:String, value ):Void 
   155    	{
   156    		_so.data[sName] = value ;
   157  		_so.flush() ;
   158    	}
   159  
   160  	public function size():Number 
   161  	{
   162  		return _so.getSize() ;	
   163  	}
   164  
   165  	/**
   166  	 * Unlock Method like server side SSAS !! Test this method !!
   167  	 */
   168   	public function unlock():Void 
   169   	{
   170  		_so.setFps(0) ;
   171  		_so.setFps(-1) ;
   172  	}
   173  
   174  	private var _eChange:SharedDataEvent ;
   175  	private var _eClear:SharedDataEvent;
   176  	private var _eClose:SharedDataEvent ;	
   177  	private var _eDelete:SharedDataEvent ;	
   178  	private var _eFire:SharedDataEvent ;
   179  	private var _eReject:SharedDataEvent ;		
   180  	private var _eSuccess:SharedDataEvent ;		
   181  	private var _eSync:SharedDataEvent ;		
   182  
   183  	private var _isConnected:Boolean ;
   184  	private var _isFirst:Boolean ;
   185  	
   186  	private var _so:SharedObject ;
   187  		
   188  	private function _onSync( list ):Void 
   189  	{
   190  		
   191  		if (_isFirst) 
   192  		{
   193  			_isFirst = false ;
   194  			dispatchEvent(_eSync) ;
   195  			for (var name:String in _so.data ) 
   196  			{
   197  				_eChange.setProperty(name, _so.data[name]) ;
   198  				dispatchEvent( _eChange ) ;
   199  			}
   200  			return ;
   201  		}
   202  		
   203  		for (var prop:String in list) 
   204  		{
   205  			
   206  			var item:Object = list[prop] ;
   207  			var code:SharedDataStatus = SharedDataStatus.format( item.code )  ;
   208  			var name:String = item.name ;
   209  			var value = _so.data[name] ;
   210  
   211  			//**** DEBUG
   212  			// var txt:String = "> " + this + "._onSync :: " + code ;
   213  			// if(name) txt += " , name:" + name ;
   214  			// if(value) txt += " , value:" + value ;
   215  			// trace (txt) ;
   216  			//***/
   217  			
   218  			switch (code) 
   219  			{
   220  				case SharedDataStatus.CLEAR :
   221  				{
   222  					dispatchEvent(_eClear);
   223  					break ;
   224  				}
   225  				case SharedDataStatus.CHANGE :
   226  				{
   227  					_eChange.setProperty(name, value) ;
   228  					dispatchEvent(_eChange);
   229  					break ;
   230  				}	
   231  				case SharedDataStatus.DELETE :
   232  				{
   233  					_eDelete.setProperty(name) ;
   234  					dispatchEvent(_eDelete );
   235  					break ;
   236  				}
   237  				case SharedDataStatus.REJECT :
   238  				{
   239  					_eReject.setProperty(name, value) ;
   240  					dispatchEvent(_eReject );
   241  					break ;
   242  				}
   243  				case SharedDataStatus.SUCCESS :
   244  				{
   245  					_eSuccess.setProperty(name, value) ;
   246  					dispatchEvent(_eSuccess );
   247  					break ;
   248  				}
   249  			}
   250     		}
   251  	}
   252  
   253  	private function _onFired( eventContext ):Void 
   254  	{
   255  		_eFire.setContext ( eventContext ) ;
   256     		dispatchEvent( _eFire ) ;
   257  	}
   258  	
   259  }
