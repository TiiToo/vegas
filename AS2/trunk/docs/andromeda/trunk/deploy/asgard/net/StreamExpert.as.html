   298  	/**
   299  	 * Returns the formatted string representation of the time position of the stream passed in argument.
   300  	 * @param pattern the optional string pattern to format the hours, the minutes and the seconds of the passed-in time. 
   301  	 * By default the method use the static constant DEFAULT_TIME_PATTERN with the value "{0}:{1}:{2}".
   302  	 * <li>{0} is the representation of the hours value.</li>
   303  	 * <li>{1} is the representation of the minutes value.</li>
   304  	 * <li>{2} is the representation of the seconds value.</li>
   305  	 * @return the formatted string representation of the time position of the stream passed in argument.
   306  	 * @see StringFormatter
   307  	 */
   308  	public function getTimeString( pattern:String ):String
   309  	{
   310  	
   311  		var time:Number ;
   312  
   313  		if ( StreamCollector.contains( getStreamID() ) )
   314  		{
   315  			time = getStream().time ;
   316  		}
   317  	
   318  		var t:Time = new Time( isNaN(time)? 0 : time , "s" ) ;
   319  		
   320  		_formatter.pattern = pattern || DEFAULT_TIME_PATTERN ;
   321  		
   322  		var hours:Number   = t.getHours(0) ;
   323  		var minutes:Number = t.getMinutes(0) ;
   324  		var seconds:Number = t.getSeconds(0) ;
   325  
   326  		return _formatter.format
   327  		(
   328  			( (hours < 10)   ? "0" : "" ) +  hours.toString() ,
   329  			( (minutes < 10) ? "0" : "" ) +  minutes.toString() ,
   330  			( (seconds < 10) ? "0" : "" ) +  seconds.toString() 
   331  		) ;	
   332  		
   333  	}

   336  	 * This method is invoqued in the constructor of the class to initialize all events.
   337  	 * Overrides this method.
   338  	 */
   339  	public function initEvent():Void
   340  	{
   341  		_ePlayFinish   = new StringEvent   ( STREAM_PLAY_FINISH , getStreamID() ) ;
   342  		_ePlayLoop     = new StringEvent   ( STREAM_PLAY_LOOP , getStreamID() ) ;
   343  		_ePlayPause    = new StringEvent   ( STREAM_PLAY_PAUSE , getStreamID() ) ;
   344  		_ePlayProgress = new ProgressEvent ( STREAM_PLAY_PROGRESS ) ;
   345  		_ePlayResume   = new StringEvent   ( STREAM_PLAY_RESUME , getStreamID() ) ;
   346  		_ePlayStart    = new StringEvent   ( STREAM_PLAY_START , getStreamID() ) ;
   347  		_ePlayStop     = new StringEvent   ( STREAM_PLAY_STOP , getStreamID() ) ;
   348  	}

   351  	 * Pause the Stream.
   352  	 */
   353  	public function pause():Void
   354  	{
   355  		if ( StreamCollector.contains( getStreamID() ) )
   356  		{
   357  			getStream().pause( true ) ;
   358  			dispatchEvent( _ePlayPause ) ;

   364  		}

   368  	 * This method start the stream.
   369  	 */
   370  	public function play( name:Object, start:Number, len:Number, reset:Object ):Void
   371  	{
   372  		
   373  		if ( StreamCollector.contains( getStreamID() ) )
   374  		{
   375  			var s:Stream = getStream() ;
   376  			s.play.apply( s, arguments ) ;
   377  			dispatchEvent( _ePlayStart ) ;	
   378  			_tProgress.start() ;
   379  		}
   380  		else
   381  		{

   383  		}
