    24  import asgard.net.URLRequestMethod;
    25  
    26  import vegas.core.CoreObject;
    27  import vegas.core.ICloneable;
    28  import vegas.errors.ArgumentsError;
    29  
    30  /**
    31   * The URLRequest class captures all of the information in a single HTTP request.
    32   * @author eKameleon
    33   * @version 1.0.0.0
    34   */
    35  class asgard.net.URLRequest extends CoreObject implements ICloneable 
    36  {
    37  	
    38  	/**
    39  	 * Creates a new URLRequest instance.
    40  	 * @param url the uri of the da
    41  	 */
    42  	public function URLRequest(url:String) {
    43  		super() ;
    44  		_sURL = url ;
    45  	}
    46  
    47  	/**
    48  	 * The default content type value.
    49  	 */
    50  	static public var DEFAULT_CONTENT_TYPE:String = "application/x-www-form-urlencoded" ;
    51  	
    52  	static private var __ASPF__ = _global.ASSetPropFlags(URLRequest, null , 7, 7) ;
    53  
    54  	/**
    55  	 * Returns the MIME content type of any POST data.
    56  	 * @return the MIME content type of any POST data.
    57  	 */
    58  	public function get contentType():String 
    59  	{
    60  		return getContentType() ;
    61  	}
    62  	
    63  	/**
    64  	 * Sets the MIME content type of any POST data.
    65  	 */
    66  	public function set contentType(s:String):Void 
    67  	{
    68  		setContentType(s) ;	
    69  	}
    70  
    71  	/**
    72  	 * Returns an object containing data to be transmitted with the URL request.
    73  	 * @return an object containing data to be transmitted with the URL request.
    74  	 */
    75  	public function get data() 
    76  	{
    77  		return getData() ;
    78  	}
    79  	
    80  	/**
    81  	 * Sets an object containing data to be transmitted with the URL request.
    82  	 */
    83  	public function set data(o):Void 
    84  	{
    85  		setData(o) ;	
    86  	}
    87  
    88  	/**
    89  	 * Returns the HTTP form submission method, this method must be a GET or POST operation.
    90  	 * @return the HTTP form submission method, this method must be a GET or POST operation.
    91  	 * @see URLRequestMethod
    92  	 */
    93  	public function get method():String 
    94  	{
    95  		return getMethod() ;
    96  	}
    97  	
    98  	/**
    99  	 * Controls whether the HTTP form submission method is a GET or POST operation.
   100  	 * @see URLRequestMethod
   101  	 */
   102  	public function set method(s:String):Void 
   103  	{
   104  		setMethod(s) ;	
   105  	}
   106  
   107  	/**
   108  	 * Returns the array of HTTP request headers to be appended to the HTTP request.
   109  	 * @return the array of HTTP request headers to be appended to the HTTP request.
   110  	 */
   111  	public function get requestHeaders():Array 
   112  	{
   113  		return getRequestHeaders() ;
   114  	}
   115  	
   116  	/**
   117  	 * Sets the array of HTTP request headers to be appended to the HTTP request.
   118  	 */
   119  	public function set requestHeaders(ar:Array):Void 
   120  	{
   121  		setRequestHeaders(ar) ;	
   122  	}
   123  
   124  	/**
   125  	 * Returns the URL to be requested.
   126  	 * @return the URL to be requested.
   127  	 */
   128  	public function get url():String 
   129  	{
   130  		return this.getUrl() ;
   131  	}
   132  	
   133  	/**
   134  	 * Sets the URL to be requested.
   135  	 */
   136  	public function set url(sURL:String):Void 
   137  	{
   138  		setUrl(sURL) ;	
   139  	}
   140  
   141  	/**
   142  	 * Returns {@code true} if the request is encoded using the system code page, rather than Unicode.
   143  	 * @return {@code true} if the request is encoded using the system code page, rather than Unicode.
   144  	 */
   145  	public function get useCodePage():Boolean 
   146  	{
   147  		return getUseCodePage() ;
   148  	}
   149  	
   150  	/**
   151  	 * Sets {@code true} if the request is encoded using the system code page, rather than Unicode.
   152  	 */
   153  	public function set useCodePage(b:Boolean):Void 
   154  	{
   155  		setUseCodePage(b) ;	
   156  	}
   157  
   158  	/**
   159  	 * Returns a shallow copy of this instance.
   160  	 * @return a shallow copy of this instance.
   161  	 */
   162  	public function clone() 
   163  	{
   164  		var request:URLRequest = new URLRequest(this.getUrl()) ;
   165  		request.setData( getData() ) ;
   166  		request.setContentType( getContentType() ) ;
   167  		request.setMethod( getMethod() ) ;
   168  		request.setRequestHeaders( getRequestHeaders() ) ;
   169  		request.setUseCodePage( getUseCodePage() ) ;
   170  		return request ;
   171  	}
   172  	
   173  	/**
   174  	 * Returns the MIME content type of any POST data.
   175  	 * @return the MIME content type of any POST data.
   176  	 */
   177  	public function getContentType():String 
   178  	{
   179  		return _sContentType || DEFAULT_CONTENT_TYPE ;	
   180  	}
   181  
   182  	/**
   183  	 * Returns An object containing data to be transmitted with the URL request.
   184  	 * @return An object containing data to be transmitted with the URL request.
   185  	 */
   186  	public function getData() 
   187  	{
   188  		return _oData ;	
   189  	}
   190  
   191  	/**
   192  	 * Controls whether the HTTP form submission method is a GET  or POST operation.
   193  	 */
   194  	public function getMethod():String 
   195  	{
   196  		return _sMethod || URLRequestMethod.POST ;
   197  	}
   198  
   199  	/**
   200  	 * Retreive the array of HTTP request headers to be appended to the HTTP request.
   201  	 * @return an array of HTTP request headers.
   202  	 */
   203  	public function getRequestHeaders():Array 
   204  	{
   205  		return _aRequestHeaders ;
   206  	}
   207  
   208  	/**
   209  	 * Returns The URL to be requested.
   210  	 * @return The URL to be requested.
   211  	 */
   212  	public function getUrl():String 
   213  	{
   214  		return _sURL ;	
   215  	}
   216  
   217  	/**
   218  	 * Returns {@code true} if the request is encoded using the system code page, rather than Unicode.
   219  	 * @return {@code true} if the request is encoded using the system code page, rather than Unicode.
   220  	 */
   221  	public function getUseCodePage():Boolean 
   222  	{
   223  		return _bUseCodePage ;
   224  	}
   225  
   226  	/**
   227  	 * Sets the MIME content type of any POST data.
   228  	 */
   229  	public function setContentType(sType:String):Void 
   230  	{
   231  		_sContentType = sType ;
   232  	}
   233  
   234  	/**
   235  	 * Sets an object containing data to be transmitted with the URL request.
   236  	 */
   237  	public function setData( oData ) 
   238  	{
   239  		_oData = oData ;	
   240  	}
   241  
   242  	/**
   243  	 * Controls whether the HTTP form submission method is a GET or POST operation.
   244  	 * @see URLRequestMethod
   245  	 */
   246  	public function setMethod(sMethod:String):Void 
   247  	{
   248  		if (URLRequestMethod.validate(sMethod)) 
   249  		{
   250  			_sMethod = sMethod ;
   251  		}
   252  		else 
   253  		{
   254  			throw new ArgumentsError(this + ".setMethod("+ sMethod + ") argument is not a URLRequestHeader value" ) ;	
   255  		}
   256  	}
   257  
   258  	/**
   259  	 * Sets the array of HTTP request headers to be appended to the HTTP request.
   260  	 */
   261  	public function setRequestHeaders(ar:Array):Void 
   262  	{
   263  		_aRequestHeaders = ar ;
   264  	}
   265  
   266  	/**
   267  	 * Sets the URL to be requested.
   268  	 */
   269  	public function setUrl(sURL:String):Void 
   270  	{
   271  		_sURL = sURL ;	
   272  	}
   273  
   274  	/**
   275  	 * Sets the code page value, if this value is {@code true} the request is encoded using the system code page, rather than Unicode.
   276  	 */
   277  	public function setUseCodePage(b:Boolean):Void 
   278  	{
   279  		_bUseCodePage = b ;
   280  	}
   281  
   282  	private var _aRequestHeaders:Array ;
   283  	private var _bUseCodePage:Boolean ;	
   284  	private var _oData ;
   285  	private var _sContentType:String ;
   286  	private var _sMethod:String ;
   287  	private var _sURL:String ;
   288  	
   289  }
