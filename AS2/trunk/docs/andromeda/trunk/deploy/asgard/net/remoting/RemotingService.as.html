   303  	/**
   304  	 * Run the process of the remoting service.
   305  	 */
   306  	public function run():Void 
   307  	{
   308  		
   309  		_rc = RemotingConnection.getConnection( _gatewayUrl ) ;
   310  		
   311  		if (_rc == null ) 
   312  		{
   313  			// ici notifier qu'il est impossible de lancer la connection.	
   314  		}
   315  		
   316  		if (_authentification != null)
   317  		{
   318  			_rc.setCredentials(_authentification) ;
   319  		}
   320  		
   321  		if ( getRunning() && multipleSimultaneousAllowed == false)  
   322  		{
   323  			notifyProgress() ;
   324  		}
   325  		else 
   326  		{
   327  			notifyStarted() ;
   328  			_result = null ;
   329  			_setRunning(true) ;
   330  			_timer.start() ;
   331  			var arg:Array = [_serviceName + "." + _methodName , getResponder()].concat(_args) ;
   332  			_rc.call.apply( _rc, arg );
   333  			
   334  		} 
   335  	}

   491  	/**
   492  	 * Resolve an unknow method over the RemotingService instance.
   493  	 */
   494  	private function __resolve__( methodName:String ):Function 
   495  	{
   496  		
   497  		if (_rc == null ) 
   498  		{
   499  			getLogger().warn( this + " resolve failed with no connection." ) ;	
   500  			return null ;
   501  		}
   502  		
   503  		if(methodName == "__path__") 
   504  		{
   505  			return null ; // hack pour le toString avec ConstructorUtil.getName !!
   506  		}
   507  		
   508  		var responder:RemotingServiceResponder = getResponder() ;
   509  		responder.setMethodName(methodName) ;
   510  
   511  		var rc:RemotingConnection = _rc ;
   512  		var serviceName:String = _serviceName ;
   513  
   514  		return function () 
   515  		{
   516  			var args:Array = [ serviceName + "." + methodName , responder ].concat(arguments) ;
   517  			return rc.call.apply( rc, args ); 
   518  		} ;
   519  		
   520  		
   521  	}

   523  	private function _onStatus ( ev:Object ):Void 
   524  	{
   525  		_timer.stop() ; // stop timeout interval
   526  		if (ev.level == RemotingService.LEVEL_ERROR) 
   527  		{
   528  			notifyError(ev.code) ;
   529  		}
   530  	}

   532  	private function _onTimeOut(e:TimerEvent):Void 
   533  	{
   534  		_timer.stop() ;
   535  		_rc.close() ;
   536  		_setRunning(false) ;
   537  		notifyTimeOut() ;
   538  		notifyFinished() ;
   539  	}
