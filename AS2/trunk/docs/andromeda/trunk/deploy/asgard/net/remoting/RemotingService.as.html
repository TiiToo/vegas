    48  	/**
    49  	 * Creates a new RemotingService instance.
    50  	 * @param gatewayUrl the url of the gateway of the remoting service.
    51  	 * @param serviceName the name of the service in the server.
    52  	 * @param responder (optional) The RemotingServiceResponder use to receive data from the server.
    53  	 */
    54  	function RemotingService( gatewayUrl:String , serviceName:String , responder:RemotingServiceResponder ) 
    55  	{
    56  		
    57  		super() ;
    58  		
    59  		setGatewayUrl( gatewayUrl );
    60  		setServiceName( serviceName ) ;
    61  		setResponder( responder ) ;
    62  		
    63  		_eFinish = new RemotingEvent(RemotingEvent.FINISH, this) ;
    64  		_eProgress = new RemotingEvent(RemotingEvent.PROGRESS, this) ;
    65  		_eStart = new RemotingEvent(RemotingEvent.START, this) ;
    66  		_eError = new RemotingEvent(RemotingEvent.ERROR, this)  ;
    67  		
    68  		_global.System.onStatus = Delegate.create (this, _onStatus) ;
    69  		
    70  		_timerListener = new Delegate(this, _onTimeOut) ;
    71  		_timer = new Timer(DEFAULT_DELAY, 1) ;
    72  		setTimeoutPolicy(TimeoutPolicy.LIMIT) ;
    73  		
    74  	}

   191  	public function notifyError( code:String ):Void 
   192  	{
   193  		_timer.stop() ;
   194  		_setRunning(false) ;
   195  		_eError.code = code || RemotingEvent.ERROR ;
   196  		dispatchEvent( _eError ) ;
   197  		notifyFinished() ;
   198  	}

   200  	public function onFault( e:RemotingEvent ) : Void 
   201  	{
   202  		_timer.stop() ;
   203  		_setRunning(false) ;
   204  		e.setTarget(this) ;
   205  		dispatchEvent( e ) ;
   206  		notifyFinished() ;
   207  	}

   209  	public function onResult( e:RemotingEvent ):Void 
   210  	{
   211  		_timer.stop() ;
   212  		_setRunning(false) ;
   213  		e.setTarget(this) ;
   214  		_result = e.getResult() ;
   215  		dispatchEvent( e ) ;
   216  		notifyFinished() ;
   217  	}

   219  	/**
   220  	 * Run the process of the remoting service.
   221  	 */
   222  	public function run():Void 
   223  	{
   224  		
   225  		_rc = RemotingConnection.getConnection( _gatewayUrl ) ;
   226  		
   227  		if (_rc == null ) 
   228  		{
   229  			// ici notifier qu'il est impossible de lancer la connection.	
   230  		}
   231  		
   232  		if (_authentification != null)
   233  		{
   234  			_rc.setCredentials(_authentification) ;
   235  		}
   236  		
   237  		if ( getRunning() && multipleSimultaneousAllowed == false)  
   238  		{
   239  			notifyProgress() ;
   240  		}
   241  		else 
   242  		{
   243  			notifyStarted() ;
   244  			_result = null ;
   245  			_setRunning(true) ;
   246  			_timer.start() ;
   247  			var arg:Array = [_serviceName + "." + _methodName , getResponder()].concat(_args) ;
   248  			_rc.call.apply( _rc, arg );
   249  			
   250  		} 
   251  	}

   272  			_gatewayUrl = url ;
   273  			_rc = RemotingConnection.getConnection( _gatewayUrl ) ;
   274  		} else {
   275  			_rc = null ;	
   276  		}
   277  	}

   297  	public function setResponder( responder:RemotingServiceResponder ):Void 
   298  	{
   299  		if (_responder) 
   300  		{
   301  			_responder.setService(null) ;
   302  		}
   303  		_responder = responder || new RemotingServiceResponder(this, onResult, onFault) ;
   304  		_responder.setService(this) ;
   305  	}

   312  	/**
   313  	 * Use limit timeout interval.
   314  	 * @see TimeoutPolicy
   315  	 */
   316  	public function setTimeoutPolicy( policy:TimeoutPolicy ):Void 
   317  	{
   318  		_policy = policy ;
   319  		
   320  		if (_policy == TimeoutPolicy.LIMIT) 
   321  		{
   322  			_timer.addEventListener( TimerEvent.TIMER, _timerListener) ;
   323  		}
   324  		else 
   325  		{
   326  			_timer.removeEventListener(TimerEvent.TIMER, _timerListener) ;
   327  		}
   328  	}

   378  	/**
   379  	 * Resolve an unknow method over the RemotingService instance.
   380  	 */
   381  	private function __resolve__( methodName:String ):Function 
   382  	{
   383  		
   384  		if (_rc == null ) {
   385  			// ici notifier qu'il est impossible de lancer la connection.	
   386  			return null ;
   387  		}
   388  		
   389  		if(methodName == "__path__") 
   390  		{
   391  			return null ; // hack pour le toString avec ConstructorUtil.getName !!
   392  		}
   393  		
   394  		var responder:RemotingServiceResponder = getResponder() ;
   395  		responder.setMethodName(methodName) ;
   396  
   397  		var rc:RemotingConnection = _rc ;
   398  		var serviceName:String = _serviceName ;
   399  
   400  		return function () 
   401  		{
   402  			var args:Array = [ serviceName + "." + methodName , responder ].concat(arguments) ;
   403  			return rc.call.apply( rc, args ); 
   404  		} ;
   405  		
   406  		
   407  	}
   408  	
   409  	private function _onStatus ( ev:Object ):Void 
   410  	{
   411  		
   412  		_timer.stop() ; // stop timeout interval
   413  		
   414  		if (ev.level == RemotingService.LEVEL_ERROR) 
   415  		{
   416  			notifyError(ev.code) ;
   417  		}
   418  	}
   419  
   420  	private function _onTimeOut(e:TimerEvent):Void 
   421  	{
   422  		
   423  		_timer.stop() ;
   424  		
   425  		_rc.close() ;
   426  		
   427  		_setRunning(false) ;
   428  		
   429  		notifyTimeOut() ;
   430  		notifyFinished() ;
   431  			
   432  	}
