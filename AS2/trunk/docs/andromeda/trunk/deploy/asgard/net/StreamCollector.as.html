    24  import asgard.net.Stream;
    25  import asgard.net.StreamExpert;
    26  
    27  import vegas.data.map.HashMap;
    28  import vegas.errors.Warning;
    29  
    30  /**
    31   * This collector use a Map to register all {@code Stream} in the application.
    32   * @author eKameleon
    33   */
    34  class asgard.net.StreamCollector 
    35  {
    36  
    37  	/**
    38  	 * Removes all {@code Stream} references in the collector.
    39  	 */
    40  	static public function clear():Void 
    41  	{
    42  		_map.clear() ;	
    43  	}
    44  	
    45  	/**
    46  	 * Returns {@code true} if the collector contains the {@code Stream} register with the id passed in argument.
    47  	 * @param id the id of the {@code Stream} register in the collector.
    48  	 * @return {@code true} if the collector contains the {@code Stream} register with the id passed in argument.
    49  	 */
    50  	static public function contains( id ):Boolean 
    51  	{
    52  		return _map.containsKey( id ) ;	
    53  	}
    54  	
    55  	/**
    56  	 * Returns {@code true} if the collector contains the {@code Stream} passed in argument.
    57  	 * @param stream the Stream to search in the collector.
    58  	 * @return {@code true} if the collector contains the {@code Stream} passed in argument.
    59  	 */
    60  	static public function containsStream( stream:Stream ):Boolean
    61  	{
    62  		return _map.containsValue( stream ) ;	
    63  	}
    64  	
    65  	/**
    66  	 * Returns the {@code Stream} reference with the name passed in argument.
    67  	 * @return the {@code Stream} reference with the name passed in argument.
    68  	 * @throws Warning if the the specified id isn't register in the collector.
    69  	 */
    70  	static public function get( id ):Stream 
    71  	{
    72  
    73  		if (!contains( id ) ) 
    74  		{
    75  			throw new Warning("[StreamCollector] get(" + id + ") failed. The specified id isn't register in the collector." ) ;
    76  		} ;
    77  		return Stream( _map.get( id ) ) ;	
    78  	}
    79  	
    80  	/**
    81  	 * Insert a {@code Stream} in the collector and indexed it with the string name in the first parameter.
    82  	 * @param id the id key of the {@code Stream} in the collector.
    83  	 * @param stream the {@code Stream} reference register in the map of the collector.
    84  	 * @return {@code true} if the specified {@code Stream} is inserted in the collector.
    85  	 * @throws Warning if the specified id is already registered in the collector.
    86  	 */
    87  	static public function insert( id , stream:Stream):Boolean 
    88  	{
    89  		if ( contains(id) ) 
    90  		{
    91  			throw new Warning( "[StreamCollector] insert method failed. A Stream instance is already registered with '" + id + "' id." ) ;
    92  		} ;
    93  		return _map.put(id, stream) == null ;	
    94  	}
    95  	
    96  	/**
    97  	 * Returns {@code true} if the collector is empty.
    98  	 * @return {@code true} if the collector is empty.
    99  	 */
   100  	static public function isEmpty():Boolean 
   101  	{
   102  		return _map.isEmpty() ;	
   103  	}
   104  
   105  	/**
   106  	 * Removes the {@code Stream} in the collector specified by the argument {@code id}. 
   107  	 * @param id the id of the {@code Stream} to unregister in the collector.
   108  	 */
   109  	static public function remove( id ):Void 
   110  	{
   111  		if (StreamExpert.contains( id ) )
   112  		{
   113  			StreamExpert.release( id ) ;
   114  		}
   115  		_map.remove( id ) ;
   116  	}
   117  
   118  	/**
   119  	 * Returns the number of elements in the collector.
   120  	 * @return the number of elements in the collector.
   121  	 */
   122  	static public function size():Number 
   123  	{
   124  		return _map.size() ;	
   125  	}
   126  
   127  	/**
   128  	 * Internal HashMap of all {@code Stream} in the application.
   129  	 */	
   130  	static private var _map:HashMap = new HashMap() ;
   131  	
   132  }
