    41  /**
    42   * Enqueue ILoader in a buffer and run all ILoader one by one.
    43   * @author eKameleon
    44   */
    45  class asgard.net.MassiveLoader extends AbstractLoader implements LoaderListener 
    46  {
    47  
    48  	/**
    49  	 * Create a new MassiveLoader.
    50  	 */ 
    51  	public function MassiveLoader() 
    52  	{
    53  		
    54  		super() ;
    55  		
    56  		_timer = new Timer(120, 1) ;
    57  		_timer.addEventListener( TimerEvent.TIMER, new Delegate(this, _next) ) ;
    58  			
    59  		_timerComplete = new Timer(120, 1) ;
    60  		_timerComplete.addEventListener(TimerEvent.TIMER, new Delegate(this, _onLoadComplete)) ;
    61  		
    62  		_listenerComplete = new Delegate(this, onLoadComplete) ;
    63  		_listenerError    = new Delegate(this, onLoadError) ;
    64  		_listenerInit     = new Delegate(this, onLoadInit) ;
    65  		_listenerProgress = new Delegate(this, onLoadProgress) ;
    66  		_listenerStart    = new Delegate(this, onLoadStart) ;
    67  		_listenerTimeout  = new Delegate(this, onLoadTimeOut) ;
    68  		
    69  		_mListeners = new HashMap() ;
    70  		
    71  		_qBuffer = new LinearQueue() ;
    72  		
    73  	}
    74  
    75  	/**
    76  	 * Clear the buffer.
    77  	 */
    78  	public function clear():Void 
    79  	{
    80  		_qBuffer.clear() ;	
    81  	}
    82  	
    83  	/**
    84  	 * Enqueue a new loader in the massive loader buffer.
    85  	 */
    86  	public function enqueue( loader:ILoader, sName:String, sURL:String ):String 
    87  	{
    88  
    89  		try 
    90  		{
    91  			if (sName) 
    92  			{
    93  				loader.setName(sName) ;
    94  			}
    95  			if (loader.getName() == null) 
    96  			{
    97  				throw new Warning("You passed ILoader object without any name property in " + this + ".enqueue()." ) ;
    98  			}
    99  		}
   100  		catch(e:Warning) 
   101  		{
   102  			e.toString() ;
   103  		}
   104  		catch(e:Error)
   105  		{
   106  			e.toString() ;	
   107  		}
   108  
   109  		try 
   110  		{
   111  			if (sURL) 
   112  			{
   113  				loader.setUrl(sURL) ;
   114  			}
   115  			if (loader.getUrl() == undefined) 
   116  			{
   117  				throw new Warning("You passed ILoader object without any url property in " + this + ".enqueue().") ;
   118  			}
   119  		} 
   120  		catch(e:Warning) 
   121  		{
   122  			e.toString() ;					
   123  		}
   124  		
   125  		if (loader.getName() == null) 
   126  		{
   127  			loader.setName( "loader_library" + HashCode.next() ) ;
   128  		} 
   129  		_qBuffer.enqueue(loader) ;
   130  		
   131  		return loader.getName() ;
   132  		
   133  	}
   134  
   135  	/**
   136  	 * Returns the delay between 2 loader when the MassiveLoader is running.
   137  	 * @return the delay between 2 loader when the MassiveLoader is running.
   138  	 */
   139  	public function getDelay():Number 
   140  	{
   141  		return _timer.getDelay() ;	
   142  	}
   143  
   144  	/**
   145  	 * Returns {@code true} if the massive loader is in progress.
   146  	 * @return {@code true} if the massive loader is in progress.
   147  	 */
   148  	public function getRunning():Boolean 
   149  	{
   150  		return _isRunning ;	
   151  	}
   152  
   153  	/**
   154  	 * Returns {@code true} is the buffer is empty.
   155  	 * @return {@code true} is the buffer is empty.
   156  	 */
   157  	public function isEmpty():Boolean 
   158  	{
   159  		return _qBuffer.isEmpty() ;	
   160  	}
   161  
   162  	/**
   163  	 * Load all the loaders in the massive loader.
   164  	 */
   165  	public function load():Void 
   166  	{
   167  		if ( !_isRunning && !isEmpty() ) 
   168  		{
   169  			var ar:Array = toArray() ;
   170  			var len:Number = ar.length ;
   171  			while (--len > -1) 
   172  			{
   173  				var oLoader:ILoader = ar[len] ;
   174  				if (oLoader.getUrl() == undefined) 
   175  				{
   176  					notifyError( this + ".run() encounters ILoader object without url property, load fails." ) ;
   177  					return ;
   178  				}
   179  			}
   180  			notifyEvent( getEventTypeSTART() ) ;
   181  			_setRunning(true) ;
   182  			_next() ;
   183  		}
   184  	}
   185  
   186  	/**
   187  	 * Invoqued when a loader is complete.
   188  	 */
   189  	public function onLoadComplete(e:LoaderEvent):Void 
   190  	{
   191  		// overrides this method.
   192  	}
   193  
   194  	/**
   195  	 * Invoqued when the loading failed.
   196  	 */
   197  	public function onLoadError(e:LoaderEvent):Void 
   198  	{
   199  		if (isEmpty()) 
   200  		{
   201  			_timerComplete.start() ;
   202  		} 
   203  		else 
   204  		{
   205  			_timer.start() ;
   206  		}
   207  	}
   208  
   209  	/**
   210  	 * Invoqued when the loading is complete and the content initialize.
   211  	 */
   212  	public function onLoadInit( e:LoaderEvent ):Void 
   213  	{
   214  		if (isEmpty()) 
   215  		{
   216  			_timerComplete.start() ;
   217  		}
   218  		else 
   219  		{
   220  			_timer.start() ;
   221  		}
   222  	}
   223  
   224  	/**
   225  	 * Invoqued during the load progress.
   226  	 */
   227  	public function onLoadProgress(e : LoaderEvent) : Void 
   228  	{
   229  		// internal method - override this method
   230  	}
   231  
   232  	/**
   233  	 * Invoqued when the loading start.
   234  	 */
   235  	public function onLoadStart(e : LoaderEvent) : Void 
   236  	{
   237  		// internal method - override this method
   238  	}
   239  
   240  	/**
   241  	 * Invoqued if the loading is out of time.
   242  	 */
   243  	public function onLoadTimeOut(e : LoaderEvent) : Void 
   244  	{
   245  		if (isEmpty()) 
   246  		{
   247  			_timerComplete.start() ;
   248  		} 
   249  		else 
   250  		{
   251  			_timer.start() ;
   252  		}
   253  	}

   256  	 * Run the MassiveLoader.

   258  	public function run():Void

   260  		super.run() ;
   261  	}

   264  	 * Defines the delay between 2 loadings in the internal queue.

   266  	public function setDelay(n:Number):Void 
   267  	{
   268  		_timer.setDelay(n) ;
   269  		_timerComplete.setDelay(n) ;
   270  	}

   275  	 */
   276  	 public function size():Number 
   277  	 {
   278  		return _qBuffer.size() ;	
   279  	}

   282  	 * Returns the array representation of the MassiveLoader.
   283  	 * @return the array representation of the MassiveLoader.
   284  	 */
   285  	public function toArray():Array 
   286  	{
   287  		return _qBuffer.toArray() ;	
   288  	}

   291  	
   292  	private var _listenerError:EventListener ;
   293  	
   294  	private var _listenerInit:EventListener ;
   295  	
   296  	private var _listenerProgress:EventListener ;
   297  	
   298  	private var _listenerStart:EventListener ;
   299  	
   300  	private var _listenerTimeout:EventListener ;
   301  
   302  	private var _qBuffer:LinearQueue ;
   303  
   304  	private var _timer:Timer ;
   305  
   306  	private var _timerComplete:Timer ;
   307  
   308  	private var _mListeners:HashMap ;
   309  
   310  	private var _oCurrentLoader:ILoader ;
   311  
   312  	/**
   313  	 * Launch the next loader.
   314  	 */	
   315  	private function _next():Void 
   316  	{
   317  		if (_oCurrentLoader != undefined) 
   318  		{
   319  			_unRegisterCurrentLoader() ;
   320  		}
   321  		_oCurrentLoader = ILoader( _qBuffer.poll() );
   322  		AbstractLoader(_oCurrentLoader).setParent( getEventDispatcher() ) ;
   323  		_registerCurrentLoader() ;
   324  		_oCurrentLoader.load() ;
   325  	}
   326  
   327  	/**
   328  	 * Invoqued when the massiveloader is complete.
   329  	 */
   330  	private function _onLoadComplete():Void 
   331  	{
   332  		_unRegisterCurrentLoader() ;
   333  		notifyEvent( LoaderEvent.FINISH ) ;
   334  	}
   335  
   336  	/**
   337  	 * Register the listeners of the curent loader.
   338  	 */
   339  	private function _registerCurrentLoader():Void 
   340  	{
   341  		EventTarget(_oCurrentLoader).addEventListener(LoaderEvent.COMPLETE, _listenerComplete) ;
   342  		EventTarget(_oCurrentLoader).addEventListener(LoaderEvent.IO_ERROR, _listenerError) ;
   343  		EventTarget(_oCurrentLoader).addEventListener(LoaderEvent.INIT, _listenerInit) ;
   344  		EventTarget(_oCurrentLoader).addEventListener(LoaderEvent.PROGRESS, _listenerProgress) ;
   345  		EventTarget(_oCurrentLoader).addEventListener(LoaderEvent.START, _listenerStart) ;
   346  		EventTarget(_oCurrentLoader).addEventListener(LoaderEvent.TIMEOUT, _listenerTimeout) ;
   347  	}
   348  
   349  	/**
   350  	 * Unregister the listeners of the curent loader.
   351  	 */
   352  	private function _unRegisterCurrentLoader():Void 
   353  	{
   354  		EventTarget(_oCurrentLoader).removeEventListener(LoaderEvent.COMPLETE, _listenerComplete) ;
   355  		EventTarget(_oCurrentLoader).removeEventListener(LoaderEvent.IO_ERROR, _listenerError) ;
   356  		EventTarget(_oCurrentLoader).removeEventListener(LoaderEvent.INIT, _listenerInit) ;
   357  		EventTarget(_oCurrentLoader).removeEventListener(LoaderEvent.PROGRESS, _listenerProgress) ;
   358  		EventTarget(_oCurrentLoader).removeEventListener(LoaderEvent.START, _listenerStart) ;
   359  		EventTarget(_oCurrentLoader).removeEventListener(LoaderEvent.TIMEOUT, _listenerTimeout) ;
   360  	}
