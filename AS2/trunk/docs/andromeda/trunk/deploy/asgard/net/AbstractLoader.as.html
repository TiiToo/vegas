   241  	/**
   242  	 * Load the external content in the loader.
   243  	 */
   244  	public function load():Void 
   245  	{
   246  		if (_tLoadProgress.running) 
   247  		{
   248  			_stopLoadProgress() ;
   249  		}
   250  		if ( this.getUrl() != null ) 
   251  		{
   252  			notifyStart() ;
   253  			_nLastBytesLoaded = 0;
   254  			_nTime = getTimer();
   255  			_setRunning(true) ;
   256  			_startLoadProgress() ;
   257  		} 
   258  		else 
   259  		{
   260  			
   261  			notifyError(this + ".load() can't retrieve file url : " + this.getUrl() , null ) ;
   262  		}
   263  	}

   493  	/**
   494  	 * Check the current timeout during the loading.
   495  	 */
   496  	/*protected*/ private function _checkTimeOut( nLastBytesLoaded:Number, nTime:Number ) : Void 
   497  	{
   498  		if ( nLastBytesLoaded != _nLastBytesLoaded) 
   499  		{
   500  			_nLastBytesLoaded = nLastBytesLoaded ;
   501  			_nTime = nTime ;
   502  		}
   503  		else if ( (nTime - _nTime)  > _nTimeOut) 
   504  		{
   505  			notifyEvent( getEventTypeTIMEOUT() );
   506  			release() ;
   507  			notifyError(this + " load timeout with url : '" + this.getUrl() + "'." , null ) ;
   508  		}
   509  	}
   510  	
   511  	/**
   512  	 * Notify if the loading is in progress.
   513  	 */
   514  	/*protected*/ private function _onLoadProgress():Void 
   515  	{
   516  		_checkTimeOut( getBytesLoaded(), getTimer() ) ;
   517  		if ( getBytesLoaded() > 4 && getBytesLoaded() == getBytesTotal()) 
   518  		{
   519  			_stopLoadProgress() ;
   520  			notifyEvent( getEventTypePROGRESS() );
   521  			notifyEvent( getEventTypeCOMPLETE() ) ;
   522  			_startInitTimer() ;
   523  		} 
   524  		else 
   525  		{
   526  			notifyEvent( getEventTypePROGRESS() );
   527  		}
   528  	}
   529  	
   530  	/*protected*/ private function _setInitTimer( f:Function ) 
   531  	{
   532  		_tInit = new Timer(150, 1) ;	
   533  		_tInit.addEventListener(TimerEvent.TIMER, new Delegate(this, f || onLoadInit)) ;
   534  	}
   535  
   536  	/*protected*/ private function _setProgressTimer( f:Function ) 
   537  	{
   538  		_tLoadProgress = new Timer(50) ;
   539  		_tLoadProgress.addEventListener(TimerEvent.TIMER, new Delegate(this, f || _onLoadProgress)) ;
   540  	}
