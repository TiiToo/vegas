   551  	/**
   552  	 * Invoqued when the onStatus event is invoqued.
   553  	 * @see NetServerStatus
   554  	 */	
   555  	private function onStatus( oInfo ):Void 
   556  	{
   557  		
   558  		_timer.stop() ;
   559  
   560  		var code:NetServerStatus = NetServerStatus.format(oInfo.code) ;
   561  		
   562  		getLogger().info(this + " status : " + code) ;
   563  		
   564  		switch (code) 
   565  		{
   566  
   567  			case NetServerStatus.SUCCESS :
   568  			{
   569  				notifyStatus( NetServerStatus.SUCCESS, oInfo ) ;
   570  				notifyFinished() ;
   571  				break ;
   572  			}
   573  			case NetServerStatus.BAD_VERSION :
   574  			{
   575  				notifyStatus(NetServerStatus.BAD_VERSION, oInfo) ;
   576  				notifyFinished() ;
   577  				break ;
   578  			}
   579  			case NetServerStatus.CLOSED :
   580  			{
   581  				notifyStatus(NetServerStatus.CLOSED, oInfo) ;
   582  				break ;
   583  			}
   584  			case NetServerStatus.FAILED :
   585  			{
   586  				notifyStatus(NetServerStatus.FAILED, oInfo) ;
   587  				notifyFinished() ;
   588  				break ;
   589  			}
   590  			case NetServerStatus.INVALID :
   591  			{
   592  				notifyStatus(NetServerStatus.INVALID, oInfo) ;
   593  				notifyFinished() ;
   594  				break ;
   595  			}	
   596  			case NetServerStatus.REJECTED :
   597  			{
   598  				notifyStatus(NetServerStatus.REJECTED, oInfo) ;
   599  				notifyFinished() ;
   600  				break ;
   601  			}
   602  			case NetServerStatus.SHUTDOWN :
   603  			{
   604  				notifyStatus(NetServerStatus.SHUTDOWN, oInfo) ;
   605  				notifyFinished() ;
   606  				break ;
   607  			}
   608  
   609  		}
   610  		
   611  	}

   619  		notifyFinished() ;
   620  		close() ;
   621  	}
