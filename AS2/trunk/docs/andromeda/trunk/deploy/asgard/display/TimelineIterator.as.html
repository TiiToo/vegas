    24  import vegas.core.CoreObject;
    25  import vegas.data.iterator.OrderedIterator;
    26  import vegas.errors.IllegalArgumentError;
    27  import vegas.errors.UnsupportedOperation;
    28  import vegas.util.MathsUtil;
    29  
    30  /**
    31   * This iterator control the timeline in a MovieClip target.
    32   * @author eKameleon
    33   */
    34  class asgard.display.TimelineIterator extends CoreObject implements OrderedIterator 
    35  {
    36  	
    37  	/**
    38  	 * Creates a new TimelineIterator instance.
    39  	 * @param target The MovieClip reference of this iterator.
    40  	 * @param framePosition the default framePosition of the specified MovieClip target (default frame 1).
    41  	 * @param stepSize (optional) the step between two frames returns by the iterator (default 1).
    42  	 * @throws IllegalArgumentError if the {@code target} argument of this constructor is empty.
    43  	 */
    44  	public function TimelineIterator( target:MovieClip , framePosition:Number, stepSize:Number )
    45  	{
    46  		super();
    47  		
    48  		if (target == null)
    49  		{
    50  			throw new IllegalArgumentError( this + " can't be instanciate with an empty MovieClip reference in the first argument of the constructor.") ;	
    51  		}
    52  		
    53  		this._target = target ;
    54  		
    55  		this._step = (stepSize > 1) ? stepSize : DEFAULT_STEP ; 
    56  		if ( !isNaN(framePosition) )
    57  		{
    58  			this.seek(framePosition) ;	
    59  		} 
    60  	}
    61  
    62  	/**
    63  	 * The default step value in all the PageByPageIterators.
    64  	 */
    65  	static public var DEFAULT_STEP:Number = 1 ;
    66  
    67  	/**
    68  	 * (read-only) The current frame of the iterator.
    69  	 */
    70  	public function get currentFrame():Number
    71  	{
    72  		return _target._currentframe ;	
    73  	}
    74  
    75  	/**
    76  	 * (read-only) Returns the target reference of this iterator.
    77  	 */
    78  	public function get target():MovieClip
    79  	{
    80  		return _target ;	
    81  	}
    82  
    83  	/**
    84  	 * (read-only) The current frame of the iterator.
    85  	 */
    86  	public function get totalFrames():Number
    87  	{
    88  		return _target._totalFrames ;	
    89  	}
    90  	
    91  	/**
    92  	 * Returns the step size of this PageByPageIterator.
    93  	 * @return the step size of this PageByPageIterator.
    94  	 */
    95  	public function getStepSize():Number
    96  	{
    97  		return _step ;	
    98  	}
    99  
   100  	/**
   101  	 * Checks to see if there is a previous element that can be iterated to.
   102  	 * @return {@code true} if the iterator has more elements.
   103  	 */
   104  	public function hasPrevious() : Boolean 
   105  	{
   106  		return _target._currentFrame > 1 ;
   107  	}
   108  
   109  	/**
   110  	 * Returns {@code true} if the iteration has more elements.
   111  	 * @return {@code true} if the iterator has more elements.
   112  	 */
   113  	public function hasNext() : Boolean 
   114  	{
   115  		return _target._currentFrame < _target._totalFrames ;
   116  	}
   117  
   118  	/**
   119  	 * Returns the current page number.
   120  	 * @return the current page number.
   121  	 */
   122  	public function key() 
   123  	{
   124  		return _target._currentFrame ;
   125  	}
   126  
   127  	/**
   128  	 * Seek the key pointer of the iterator over the last frame of the timeline.
   129  	 */
   130  	public function last():Void 
   131  	{
   132  		_target.gotoAndStop(_target._totalFrames) ;
   133  	}
   134  
   135  	/**
   136  	 * Returns the next Array page of elements or the next element in the Array if the getStepSize() value is 1.
   137  	 * @return the next Array page of elements or the next element in the Array if the getStepSize() value is 1.
   138  	 */
   139  	public function next() 
   140  	{
   141  		var k:Number = _target._currentFrame ;
   142  		k += _step ;
   143  		k = (k < _target._totalFrames) ? k : _target._totalFrames ;
   144  		_target.gotoAndStop(k) ;
   145  		return k ;
   146  	}
   147  
   148  	/**
   149  	 * Returns the previous Array page of elements or the previous element in the Array if the getStepSize() value is 1.
   150  	 * @return the previous element from the collection.
   151  	 */
   152  	public function previous() 
   153  	{
   154  		var k:Number = _target._currentFrame ;
   155  		k -= _step ;
   156  		k = (k > 1) ? k : 1 ;
   157  		_target.gotoAndStop(k) ;
   158  		return k ;
   159  	}
   160  
   161  	/**
   162  	 * Unsupported operation in this iterator.
   163  	 * @throws UnsupportedOperation the method remove() in this iterator is unsupported. 
   164  	 */
   165  	public function remove() 
   166  	{
   167  		throw new UnsupportedOperation(this + " remove method is unsupported by this instance.") ;
   168  	}
   169  
   170  	/**
   171  	 * Resets the key pointer of the iterator.
   172  	 */
   173  	public function reset():Void 
   174  	{
   175  		_target.gotoAndStop(1) ;
   176  	}
   177  
   178  	/**
   179  	 * Seek the key pointer of the iterator.
   180  	 */
   181  	public function seek(n : Number):Void 
   182  	{
   183  		_target.gotoAndStop(MathsUtil.clamp( n, 1, _target._totalFrames )) ;
   184  	}
   185  	
   186  	/**
   187  	 * The numbers of frame to step with the iterator.
   188  	 */
   189  	private var _step:Number ;
   190  	
   191  	/**
   192  	 * The MovieClip target of this iterator.
   193  	 */
   194  	private var _target:MovieClip ;
   195  
   196  }
