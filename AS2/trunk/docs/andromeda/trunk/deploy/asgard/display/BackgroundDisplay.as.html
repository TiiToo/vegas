    24  import asgard.display.ConfigurableDisplayObject;
    25  
    26  import pegas.draw.IPen;
    27  import pegas.draw.RectanglePen;
    28  import pegas.events.UIEvent;
    29  import pegas.events.UIEventType;
    30  
    31  import vegas.util.MathsUtil;
    32  
    33  /**
    34   * This display is used to create a background in your application or in an other display of the application.
    35   * @author eKameleon
    36   */
    37  class asgard.display.BackgroundDisplay extends ConfigurableDisplayObject 
    38  {
    39  
    40  	/**
    41  	 * Creates a new BackgroundDisplay instance.
    42  	 * @param sName the name of the display.
    43  	 * @param target the DisplayObject instance control this target.
    44  	 */
    45  	public function BackgroundDisplay(sName : String, target) 
    46  	{
    47  		
    48  		super(sName, target);
    49  	
    50  		background = view.createEmptyMovieClip( "background", 0 ) ;
    51  
    52  		_bgDraw  = new RectanglePen(background) ;
    53  		_eResize = new UIEvent( UIEventType.RESIZE , this) ;
    54  		
    55  	}
    56  	
    57  	/**
    58  	 * The background reference of this display.
    59  	 */
    60  	public var background:MovieClip ;
    61  
    62  	/**
    63  	 * (read-only) Returns the virtual height value of this component.
    64  	 * @return the virtual height value of this component.
    65  	 */
    66  	public function get h():Number 
    67  	{
    68  		return getH() ;	
    69  	}
    70  
    71  	/**
    72  	 * (read-only) Sets the virtual width value of this component.
    73  	 */
    74  	public function set h( n:Number ):Void 
    75  	{
    76  		setH( n ) ;	
    77  	}
    78  
    79  	/**
    80  	 * (read-write) Returns {@code true} if the background use full size (Stage.width and Stage.height).
    81  	 * @return {@code true} if the background use full size (Stage.width and Stage.height).
    82  	 */
    83  	public function get isFull():Boolean
    84  	{
    85  		return getIsFull() ;
    86  	}
    87  
    88  	/**
    89  	 * Sets if the background use full size (Stage.width and Stage.height).
    90  	 * @param b A boolean flag to indicates if the display use full size or not.
    91  	 */
    92  	public function set isFull(b:Boolean):Void
    93  	{
    94  		setIsFull(b) ;
    95  	}
    96  
    97  	/**
    98  	 * This property defined the mimimun height of this component.
    99  	 */
   100  	public var minHeight:Number ;
   101  
   102  	/**
   103  	 * This property defined the mimimun width of this component.
   104  	 */
   105  	public var minWidth:Number ;
   106  
   107  	/**
   108  	 * This property defined the maximum width of this component.
   109  	 */
   110  	public var maxWidth:Number ;
   111  	
   112  	/**
   113  	 * This property defined the maximum height of this component.
   114  	 */
   115  	public var maxHeight:Number ;
   116  
   117  	/**
   118  	 * The alpha value of the screen.
   119  	 */
   120  	public var themeAlpha:Number = 100 ;
   121  
   122  	/**
   123  	 * The alpha value of the border of this display.
   124  	 */
   125  	public var themeBorderAlpha:Number = null ;
   126  
   127  	/**
   128  	 * The color value of the border of this display.
   129  	 */
   130  	public var themeBorderColor:Number = null ;
   131  
   132  	/**
   133  	 * The thickness of the border of this display.
   134  	 */
   135  	public var themeBorderThickness:Number = null ;
   136  
   137  	/**
   138  	 * The color value of the screen.
   139  	 */
   140  	public var themeColor:Number = 0x000000 ;
   141  
   142  	/**
   143  	 * (read-only) Returns the virtual width value of this component.
   144  	 * @return the virtual width value of this component.
   145  	 */
   146  	public function get w():Number 
   147  	{ 
   148  		return getW() ;	
   149  	}
   150  	
   151  	/**
   152  	 * (read-only) Sets the virtual width value of this component.
   153  	 */
   154  	public function set w( n:Number ):Void 
   155  	{
   156  		setW( n ) ;	
   157  	}
   158  
   159  	 /**
   160  	  * Draw the display.
   161  	  */
   162  	 public function draw( w:Number , h:Number ):Void
   163  	 {
   164  	 	var $w:Number = isNaN(w) ? getW() : w ;
   165  	 	var $h:Number = isNaN(h) ? getH() : h ;
   166  	 	_bgDraw.clear() ;
   167  	 	_bgDraw.lineStyle( themeBorderThickness, themeBorderColor, themeBorderAlpha ) ;
   168  		_bgDraw.beginFill( themeColor, themeAlpha ) ;
   169  		_bgDraw.draw( $w, $h ) ;
   170  		_bgDraw.endFill() ;
   171  	 }
   172  
   173  	/**
   174  	 * Returns {@code true} if the background use full size (Stage.width and Stage.height).
   175  	 * @return {@code true} if the background use full size (Stage.width and Stage.height).
   176  	 */
   177  	public function getIsFull():Boolean
   178  	{
   179  		return _isFull ;
   180  	}
   181  
   182  	/**
   183  	 * (read-only) Returns the virtual height value of this component.
   184  	 * @return the virtual height value of this component.
   185  	 */
   186  	public function getH():Number 
   187  	{ 
   188  		return isFull ? Stage.height : _h ;
   189  	}
   190  
   191  	/**
   192  	 * (read-only) Returns the virtual width value of this component.
   193  	 * @return the virtual width value of this component.
   194  	 */
   195  	public function getW():Number 
   196  	{ 
   197  		return isFull ? Stage.width : _w ;
   198  	}
   199  	
   200  	/**
   201  	 * Notify an event when you resize the component.
   202  	 */
   203  	public function notifyResized():Void 
   204  	{
   205  		dispatchEvent(_eResize) ;
   206  	}
   207  
   208  	/**
   209  	 * Sets if the background use full size (Stage.width and Stage.height).
   210  	 * @param b A boolean flag to indicates if the display use full size or not.
   211  	 */
   212  	public function setIsFull(b:Boolean):Void
   213  	{
   214  		_isFull = b ;
   215  		update() ;
   216  	}
   217  
   218  	/**
   219  	 * Sets the virtual height value of the component.
   220  	 */
   221  	public function setH( n:Number ) : Void 
   222  	{
   223  		_h = MathsUtil.clamp(n, minHeight, maxHeight) ;
   224  		notifyResized() ;
   225  		update() ;
   226  	}
   227  
   228  
   229  	/**
   230  	 * Sets the virtual width and height values of the component.
   231  	 */
   232  	public function setSize( w:Number, h:Number ) : Void 
   233  	{
   234  		_w = MathsUtil.clamp( w , minWidth, maxWidth) ;
   235  		_h = MathsUtil.clamp( h , minHeight, maxHeight) ;
   236  		notifyResized() ;
   237  		update() ;
   238  	}
   239  	
   240  	/**
   241  	 * Sets the virtual width value of the component.
   242  	 */
   243  	public function setW( n:Number ):Void 
   244  	{
   245  		_w = MathsUtil.clamp(n, minWidth, maxWidth) ; 
   246  		notifyResized() ;
   247  		update() ;
   248  	}
   249  
   250  	/**
   251  	 * Update the display.
   252  	 */	
   253  	/*override*/ public function update():Void 
   254  	{
   255  		if ( isLocked() ) 
   256  		{
   257  			return ;
   258  		}
   259  		draw() ;
   260  		viewChanged() ;
   261  	}
   262  	
   263  	/**
   264  	 * This method is invoqued after the draw() method in the update() method.
   265  	 * Overrides this method.
   266  	 */
   267  	public function viewChanged():Void
   268  	{
   269  		// overrides
   270  	}
   271  	
   272  	/**
   273  	 * The internal pen to draw the background of the display.
   274  	 */
   275  	private var _bgDraw:IPen ;
   276  
   277  	private var _eResize:UIEvent ;
   278  
   279  	private var _h:Number ;
   280  
   281  	private var _isFull:Boolean = false ;
   282  	
   283  	private var _w:Number ;
   284  
   285  }
