    24  import asgard.display.DisplayLoaderCollector;
    25  import asgard.events.DisplayLoaderEvent;

    29  import vegas.core.HashCode;

    32  /**
    33   * The DisplayLoader class is used to load SWF files or image (JPG, PNG, or GIF) files. 
    34   * <p>Use the load() method to initiate loading.</p>
    35   * <p><b>Example :</b></p>
    36   * {@code
    37   * import asgard.display.DisplayLoader ;
    38   * import asgard.events.LoaderEvent ;
    39   * import asgard.events.LoaderEventType ;
    40   * import asgard.net.URLRequest ;
    41   * 
    42   * import vegas.events.Delegate ;
    43   * 
    44   * var onLoadError = function(ev:LoaderEvent):Void
    45   * {
    46   *     trace("> " + ev.getType() + " : " + ev.error ) ;
    47   * }
    48   * 
    49   * var onLoadComplete = function(ev:LoaderEvent):Void
    50   * {
    51   *     trace(ev.getTarget() + " > " + ev.getType() ) ;
    52   * }
    53   * 
    54   * var onLoadInit = function(ev:LoaderEvent):Void
    55   * {
    56   *     trace(ev.getTarget() + " > " + ev.getType() ) ;
    57   * }
    58   * 
    59   * var onLoadProgress = function (ev:LoaderEvent):Void
    60   * {
    61   *     trace(ev.getTarget() + " > " + ev.getType() + " : " + ev.getPercent() + "%") ;
    62   * }
    63   * 
    64   * var onLoadStart = function (ev:LoaderEvent):Void
    65   * {
    66   *     trace(ev.getTarget() + " > " + ev.getType() ) ;
    67   * }
    68   * 
    69   * var url:String = "pic/picture.jpg" ;
    70   * var mc:MovieClip = createEmptyMovieClip("container", 1) ;
    71   * 
    72   * mc._x = 25 ;
    73   * mc._y = 25 ;
    74   * 
    75   * var loader:DisplayLoader = new DisplayLoader(mc) ;
    76   * loader.addEventListener(LoaderEventType.COMPLETE, new Delegate(this, onLoadComplete)) ;
    77   * loader.addEventListener(LoaderEventType.IO_ERROR, new Delegate(this, onLoadError)) ;
    78   * loader.addEventListener(LoaderEventType.INIT, new Delegate(this, onLoadInit)) ;
    79   * loader.addEventListener(LoaderEventType.PROGRESS, new Delegate(this, onLoadProgress)) ;
    80   * loader.addEventListener(LoaderEventType.START, new Delegate(this, onLoadStart)) ;
    81   * 
    82   * var request:URLRequest = new URLRequest(url);
    83   * loader.load(request) ;
    84   * 
    85   * }
    86   * @author eKameleon
    87   */
    88  class asgard.display.DisplayLoader extends AbstractLoader 
    89  {
    90  	
    91  	/**
    92  	 * Creates a new DisplayLoader instance.
    93  	 * @param mcTarget the target of this loader.
    94  	 * @param nDepth the depth of the internal movieclip use to load the external content.
    95  	 * @param bAutoShow indicates if the external content (picture, swf) is show when is loading.
    96  	 */
    97  	function DisplayLoader( mcTarget:MovieClip, nDepth:Number, bAutoShow:Boolean ) 
    98  	{
    99  		
   100  		super() ;
   101  		
   102  		if (! mcTarget) 
   103  		{
   104  			throw new IllegalArgumentError(this + " mcTarget argument not must be 'undefined' or 'null'.") ;
   105  		}
   106  		
   107  		nDepth = (isNaN(nDepth)) ? _uniqueID++ : nDepth ;
   108  		
   109  		HashCode.identify( mcTarget ) ;
   110  		
   111  		_container = mcTarget.createEmptyMovieClip("__container__" +  HashCode.nextName(), nDepth ) ;
   112  		
   113  		HashCode.identify( _container ) ;
   114  		
   115  		_isAutoShow = (bAutoShow == null) ? true : bAutoShow ;
   116  		_isCollected = false ;
   117  		
   118  	}
   119  
   120  	/**
   121  	 * Returns the absolute url of the root swf object.
   122  	 * @return the absolute url of the root swf object.
   123  	 */
   124  	static public function getLoaderUrl():String 
   125  	{
   126  		return _root._url ;
   127  	}
   128  
   129  	/**
   130  	 * Returns the view of this loader.
   131  	 * @return the view of this loader.
   132  	 */
   133  	public function getView():MovieClip 
   134  	{
   135  		return super.getContent() ;
   136  	}
   137  	
   138  	/**
   139  	 * Hide the view of this loader.
   140  	 */
   141  	public function hide():Void 
   142  	{
   143  		_container._visible = false ;
   144  	}
   145  
   146  	/**
   147  	 * Init the internal event of this loader.
   148  	 */
   149  	/*override*/ public function initEvent():Void 
   150  	{
   151  		_e = new DisplayLoaderEvent(null, this) ;
   152  	}
   153  
   154  	/**
   155  	 * Return {@code true} if the view of this loader is visible when is loading.
   156  	 * @return {@code true} if the view of this loader is visible when is loading
   157  	 */
   158  	public function isAutoShow():Boolean 
   159  	{
   160  		return _isAutoShow ;	
   161  	}
   162  	
   163  	/**
   164  	 * Return {@code true} if the view of this loader is visible.
   165  	 * @return {@code true} if the view of this loader is visible.
   166  	 */
   167  	public function isVisible():Boolean 
   168  	{
   169  		return _container._visible ;	
   170  	}
   171  
   172  	/**
   173  	 * Load the current URLRequest.
   174  	 */
   175  	/*override*/ public function load(request:URLRequest):Void 
   176  	{
   177  		if (request.getUrl() != undefined) 
   178  		{
   179  			setUrl( request.getUrl() );
   180  		}
   181  		release() ;
   182  		setContent( _container.createEmptyMovieClip("__mcExternal", 1) ) ;
   183  		getContent().loadMovie( super.getUrl() );
   184  		hide();
   185  		super.load() ;
   186  	}
   187  
   188  	/**
   189  	 * Invoqued when the loading is finish and the loader is init.
   190  	 */
   191  	/*override*/ public function onLoadInit():Void 
   192  	{
   193  		if (_sName) 
   194  		{
   195  			
   196  			if (!(DisplayLoaderCollector.contains(_sName))) 
   197  			{
   198  				
   199  				_isCollected = true ;
   200  				DisplayLoaderCollector.insert( _sName, this ) ;
   201  				
   202  			} 
   203  			else 
   204  			{
   205  				_isCollected = false ;
   206  				// TODO créer une erreur.
   207  				notifyError( this + " can't be collected to [DisplayCollector] with '" + _sName + "' name. This name already exists." ) ;
   208  			}
   209  		}
   210  		
   211  		super.onLoadInit();
   212  		
   213  		if (_isAutoShow)
   214  		{
   215  			show() ;
   216  		}
   217  		
   218  	}
   219  	
   220  	/**
   221  	 * Release the loader.
   222  	 */
   223  	/*override*/ public function release():Void 
   224  	{
   225  		super.release() ;
   226  		getContent().removeMovieClip() ;
   227  		_container._visible = _isAutoShow ;
   228  		if (_isCollected) 
   229  		{
   230  			DisplayLoaderCollector.remove(_sName) ;
   231  			_isCollected = false ;	
   232  		}	
   233  	}
   234  	
   235  	/**
   236  	 * Sets if the view is show when the content is loading.
   237  	 */
   238  	public function setAutoShow(b:Boolean):Void 
   239  	{
   240  		_isAutoShow = b ;
   241  	}
   242  
   243  	/**
   244  	 * Show the view of the loader.
   245  	 */
   246  	public function show():Void 
   247  	{
   248  		_container._visible = true ;
   249  	}
   250  
   251  	/**
   252  	 * The internal container of this loader.
   253  	 */
   254  	private var _container:MovieClip ;
   255  
   256  	/**
   257  	 * The boolean value to indicated if the view is auto show when the content is loading.
   258  	 */
   259  	private var _isAutoShow:Boolean ;
   260  
   261  	/**
   262  	 * The boolean to indicated if the loader is collected.
   263  	 */
   264  	private var _isCollected:Boolean ;
   265  
   266  	/**
   267  	 * The internal uniqueID to creates a new container.
   268  	 */
   269  	static private var _uniqueID:Number = 0 ;
   270  
   271  
   272  }
