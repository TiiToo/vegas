    26  import asgard.config.Config;
    27  import asgard.config.ConfigCollector;
    28  import asgard.config.IConfigurable;
    29  import asgard.display.DisplayObject;
    30  
    31  /**
    32   * This display is IConfigurable and is ready to uses the Config model of ASGard.
    33   * @author eKameleon
    34   */
    35  class asgard.display.ConfigurableDisplayObject extends DisplayObject implements IConfigurable 
    36  {
    37  	
    38  	/**
    39  	 * Creates a new ConfigurableDisplayObject instance.
    40  	 * @param sName the name of the display.
    41  	 * @param target the DisplayObject instance control this target.
    42  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    43  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    44  	 */
    45  	public function ConfigurableDisplayObject( sName:String, target , bGlobal:Boolean , sChannel:String ) 
    46  	{
    47  		super( sName, target, bGlobal, sChannel );
    48  		isConfigurable = true ;
    49  	}
    50  
    51  	/**
    52  	 * Returns {@code true} if the display is configurable.
    53  	 * @return {@code true} if the display is configurable.
    54  	 */
    55  	public function get isConfigurable():Boolean
    56  	{
    57  		return _isConfigurable ;
    58  	}
    59  	
    60  	/**
    61  	 * Sets if the display is configurable.
    62  	 * @param b a boolean flag to determinaites if the display is configurable.
    63  	 */
    64  	public function set isConfigurable(b:Boolean):Void
    65  	{
    66  		_isConfigurable = (b == true) ;
    67  		if (_isConfigurable)
    68  		{
    69  			ConfigCollector.insert(this) ;	
    70  		}
    71  		else
    72  		{
    73  			ConfigCollector.remove(this) ;
    74  		}
    75  	}
    76  
    77  	/**
    78  	 * Invoqued by the ConfigCollector when the ConfigCollector is running.*
    79  	 * By default this method launch a search in the Config object with the name of the display and initialize the display.
    80  	 * If you want use your custom setup method override this method.
    81  	 */
    82  	public function setup():Void 
    83  	{
    84  		var config:Object = Config.getInstance()[getName()] ; 
    85  		if ( config != null ) 
    86  		{
    87  			for (var prop:String in config)
    88  			{
    89  				this[prop] = config[prop] ; 
    90  			}
    91  		}
    92  		update() ;	
    93  	}	 
    94  	
    95  	/**
    96  	 * Update the display.
    97  	 * You must override this method. This method is launch by the setup() method when the Config is checked.
    98  	 */	
    99  	public function update() : Void 
   100  	{
   101  		// overrides this method.
   102  	}
   103  	
   104  	private var _isConfigurable:Boolean ;
   105  }
