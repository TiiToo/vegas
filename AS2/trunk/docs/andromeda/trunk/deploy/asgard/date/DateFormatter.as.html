    31  /**
    32   * DateFormatter formats a given date with a specified pattern.
    33   * <p>Use the declared constants as placeholders for specific parts of the date-time.</p>
    34   * <p>All characters from 'A' to 'Z' and from 'a' to 'z' are reserved, although not all of these characters are interpreted right now.</p> 
    35   * <p>If you want to include plain text in the pattern put it into quotes (') to avoid interpretation.</p>
    36   * <p>If you want a quote in the formatted date-time, put two quotes directly after one another. For example: {@code "hh 'o''clock'"}.</p>
    37   * <p><b>Example :</b>
    38   * {@code
    39   * import asgard.date.DateFormatter ;
    40   * 
    41   * var f:DateFormatter = new DateFormatter() ;
    42   * 
    43   * f.pattern = "yyyy DDDD d MMMM - hh 'h' nn 'mn' ss 's'" ;
    44   * var result:String = f.format() ;
    45   * trace("pattern : " + f.pattern) ;
    46   * trace("result  : " + result) ;
    47   * 
    48   * trace("----") ;
    49   * 
    50   * f.pattern = "DDDD d MMMM yyyy" ;
    51   * var result:String = f.format(new Date(2005, 10, 22)) ;
    52   * trace("pattern : " + f.pattern) ;
    53   * trace("result  : " + result) ;
    54   * }
    55   * @author eKameleon
    56   */
    57  class asgard.date.DateFormatter extends AbstractFormatter

   160  	/**
   161  	 * Format a date based on the current style set for the Date Formatter.
   162  	 * <p>If the passed-in {@code date} is {@code null} or {@code undefined}, the current date-time will be used instead.
   163  	 * @param o the date-time value to format into a date-time string.
   164  	 * @return The formatted date-time string representation.
   165  	 */
   166  	public function format(o):String 
   167  	{
   168  		if (pattern == null) 
   169  		{
   170  			return "" ;
   171  		}
   172  		var date:Date = (o instanceof Date) ? o : new Date() ;
   173  		var p:String = getPattern() ;
   174  		var a:Array = p.split("") ;
   175  		var l:Number = a.length ;
   176  		var cpt:Number ;
   177  		var ch:String ; // current character
   178  		var i:Number = -1 ;
   179  		var r:String = "" ;
   180  		while (++i < l) 
   181  		{
   182  			ch = a[i] ;
   183  			if (ch == DateFormatter.QUOTE) 
   184  			{
   185  				if (a[i + 1] == DateFormatter.QUOTE) 
   186  				{
   187  					r += "'" ; 
   188  					i++ ;
   189  				}
   190  				var next:Number = i ;
   191  				var prev:Number ;
   192  				while (true) 
   193  				{
   194  					prev = next ;
   195  					next = p.indexOf("'", next + 1) ;
   196  					if (a[next + 1] != QUOTE) break ;
   197  					r += p.substring(prev+1, next+1) ;
   198  					next++;
   199  				}
   200  				r += p.substring(prev+1, next) ;
   201  				i = next ;
   202  			} 
   203  			else if (ch == YEAR) 
   204  			{
   205  				cpt = _count(ch, a.slice(i));
   206  				r += formatYear( date.getFullYear(), cpt );
   207  				i += cpt - 1 ;
   208  			}
   209  			else if (ch == MONTH_AS_NUMBER) 
   210  			{
   211  				cpt = _count(ch, a.slice(i));
   212  				r += formatMonthAsNumber(date.getMonth(), cpt);
   213  				i += cpt - 1 ;
   214  			}
   215  			else if (ch == MONTH_AS_TEXT) 
   216  			{
   217  				cpt = _count(ch, a.slice(i));
   218  				r += formatMonthAsText(date.getMonth(), cpt) ;
   219  				i += cpt - 1 ;
   220  			} 
   221  			else if (ch == DAY_AS_NUMBER) 
   222  			{
   223  				cpt = _count(ch, a.slice(i)) ;
   224  				r += formatDayAsNumber(date.getDate(), cpt) ;
   225  				i += cpt - 1 ;
   226  			}
   227  			else if (ch == DAY_AS_TEXT) 
   228  			{
   229  				cpt = _count(ch, a.slice(i)) ;
   230  				r += formatDayAsText(date.getDay(), cpt) ;
   231  				i += cpt - 1 ;
   232  			} 
   233  			else if (ch == HOUR_IN_AM_PM) 
   234  			{
   235  				cpt = _count(ch, a.slice(i));
   236  				r += formatHourInAmPm(date.getHours(), cpt) ;
   237  				i += cpt - 1 ;
   238  			} 
   239  			else if (ch == HOUR_IN_DAY) 
   240  			{
   241  				cpt = _count(ch, a.slice(i));
   242  				r += formatHourInDay(date.getHours(), cpt) ;
   243  				i += cpt - 1 ;
   244  			} 
   245  			else if (ch == MINUTE) 
   246  			{
   247  				cpt = _count(ch, a.slice(i));
   248  				r += formatMinute(date.getMinutes(), cpt) ;
   249  				i += cpt - 1 ;
   250  			}
   251  			else if (ch == SECOND) 
   252  			{
   253  				cpt = _count(ch, a.slice(i));
   254  				r += formatSecond( date.getSeconds(), cpt ) ;
   255  				i += cpt - 1 ;
   256  			}
   257  			else if (ch == MILLISECOND) 
   258  			{
   259  				cpt = _count(ch, a.slice(i));
   260  				r += formatMillisecond(date.getMilliseconds(), cpt);
   261  				i += cpt - 1 ;
   262  			} 
   263  			else 
   264  			{
   265  				r += ch;
   266  			}
   267  		} 
   268  		return r ;
   269  	}

   271  	public function formatDayAsNumber(day:Number, cpt:Number):String 
   272  	{
   273  		if (isNaN(cpt)) 
   274  		{
   275  			cpt = 0 ;
   276  		}
   277  		var string:String = day.toString();
   278  		return (getZeros(cpt - string.length) + string);
   279  	}

   281  	public function formatDayAsText(day:Number, cpt:Number):String 
   282  	{
   283  		if (RANGE_DAY_AS_TEXT.isOutOfRange(day)) 
   284  		{
   285  			throw new IllegalArgumentError(this + " formatDayAsText method failed, the day value is out of range.") ;
   286  		}
   287  		if (isNaN(cpt)) 
   288  		{
   289  			cpt = 0 ;
   290  		}
   291  		var days:Array = LocalDate.getDays() ;
   292  		var r:String = days[day] ;
   293  		if (cpt < 4) return r.substr(0, 2);
   294  		return r ;
   295  	}

   297  	/**
   298  	 * Formats the specified hour value in a string representation with the am-pm notation.
   299  	 * @return the specified hour value in a string representation with the am-pm notation.
   300  	 */
   301  	public function formatHourInAmPm(hour:Number, cpt:Number):String 
   302  	{
   303  		if (RANGE_HOUR.isOutOfRange(hour)) 
   304  		{
   305  			throw new IllegalArgumentError(this + " formatHourInAmPm method failed, the hour value is out of range.") ;
   306  		}
   307  		if (isNaN(cpt)) 
   308  		{
   309  			cpt = 0 ;
   310  		}
   311  		var s:String ;
   312  		if (hour == 0) 
   313  		{
   314  			s = (12).toString() ;
   315  		}
   316  		else if (hour > 12) 
   317  		{
   318  			s = (hour - 12).toString() ;
   319  		}
   320  		else 
   321  		{
   322  			s = hour.toString();
   323  		}
   324  		return (getZeros(cpt - s.length) + s);
   325  	}

   412  	/**
   413  	 * Format the year value passed in argument.
   414  	 * @return the year string representation of this DateFormatter.
   415  	 */
   416  	public function formatYear( year:Number, cpt:Number ):String 
   417  	{
   418  		if ( isNaN(year) ) 
   419  		{
   420  			throw new IllegalArgumentError(this + " formatYear method failed, the year value must be a Number.") ;
   421  		}
   422  		if (isNaN(cpt)) 
   423  		{
   424  			cpt = 0 ;
   425  		}
   426  		if (cpt < 4) 
   427  		{
   428  			return year.toString().substr(2) ;
   429  		}
   430  		return (getZeros(cpt - 4) + year.toString());
   431  	}

   433  	/**
   434  	 * Returns a string representation fill by 0 values or an empty string if the cpt value is NaN or <1.
   435  	 * @return a string representation fill by 0 values or an empty string if the cpt value is NaN or <1.
   436  	 */
   437  	public function getZeros(cpt:Number):String 
   438  	{
   439  		if (cpt < 1 || isNaN(cpt)) 
   440  		{
   441  			return "" ;
   442  		}
   443  		if (cpt < 2) 
   444  		{
   445  			return "0" ;
   446  		}
   447  		var r:String = "00";
   448  		cpt -= 2;
   449  		while (cpt) 
   450  		{
   451  			r += "0" ;
   452  			cpt-- ;
   453  		}
   454  		return r ;
   455  	}
   456  	
   457  	private function _count(char:String, a:Array):Number 
   458  	{
   459  		if (!a) return 0 ;
   460  		var r:Number = 0 ;
   461  		var i:Number = -1 ;
   462  		var l:Number = a.length ;
   463  		while (++i < l && a[r] == char) 
   464  		{
   465  			r++ ;
   466  		}
   467  		return r ;
   468  	}
