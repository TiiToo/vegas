    50   * trace ("-- c2 : " + c2) ;
    51   * 
    52   * trace ("> c1 after c2 : " + c1.after(c2)) ;
    53   * trace ("> c2 after c1 : " + c2.after(c1)) ;
    54   * trace ("> c1 before c2 : " + c1.before(c2)) ;

   141  	/**
   142   	 * Adds the specified amount of time to the this instance.
   143   	 * <p>Examples :</p>
   144   	 * {@code
   145   	 * import asgard.date.Calendar ;
   146   	 *
   147   	 * var begin:Calendar = new Calendar( 2007, 5 , 14, 11, 30, 0 ) ;
   148   	 * var duration:Number = 50 ;
   149   	 * 
   150   	 * var end:Calendar = Calendar.add( begin, Calendar.MINUTE , 50 ) ;
   151   	 *
   152   	 * var sBegin:String = begin.format("MM dd yyyy hh:nn:ss") ;
   153   	 * var sFinish:String   = end.format("MM dd yyyy hh:nn:ss") ;
   154   	 * 
   155   	 * trace( "start  : " + sBegin ) ;
   156   	 * trace( "finish : " + sFinish ) ;
   157   	 * }
   158  	 * @param date The Date object to perform addition on
   159  	 * @param field The this field constant to be used for performing addition.
   160  	 * @param amount The number of units (measured in the field constant) to add to the date.
   161  	 * @return the new {@code Calendar} object.
   162  	 */
   163  	public static function add( date:Date, field:String, amount:Number ):Calendar 
   164  	{
   165  		var c:Calendar = new Calendar( (date == null) ? (new Date()).valueOf() : date.valueOf() );
   166  		switch (field) 
   167  		{
   168  			case Calendar.MILLISECOND :
   169  			{
   170  				c.setTime( c.getTime() + amount ) ; 
   171  				break ;
   172  			}
   173  			case Calendar.SECOND :
   174  			{
   175  				c.setTime( c.getTime() + (amount * Time.SECOND) ) ;
   176  				break ;	
   177  			}
   178  			case Calendar.MINUTE :
   179  			{
   180  				c.setTime( c.valueOf() + (amount * Time.MINUTE) ) ;
   181  				break ;
   182  			}
   183  			case Calendar.HOUR :
   184  			{
   185  				c.setTime( c.getTime() + (amount * Time.HOUR) ) ;
   186  				break ;
   187  			}
   188  			case Calendar.DAY :
   189  			{
   190  				c.setDate( date.getDate() + amount ) ;
   191  				break;
   192  			}
   193  			case Calendar.WEEK :
   194  			{
   195  				c.setDate( date.getDate() + 7 ) ;
   196  				break;
   197  			}
   198  			case Calendar.MONTH :
   199  			{
   200  				var newMonth:Number = date.getMonth() + amount;
   201  				var years:Number = 0;
   202  				if (newMonth < 0) 
   203  				{
   204  					while (newMonth < 0) 
   205  					{
   206  						newMonth += 12;
   207  						years -= 1 ;
   208  					}
   209  				}
   210  				else if (newMonth > 11) 
   211  				{
   212  					while (newMonth > 11) 
   213  					{
   214  						newMonth -= 12;
   215  						years += 1 ;
   216  					}
   217  				}
   218  				c.setMonth(newMonth) ;
   219  				c.setFullYear(date.getFullYear() + years) ;
   220  				break ;
   221  			}
   222  			case Calendar.YEAR :
   223  			{
   224  				c.setFullYear(date.getFullYear() + amount) ;
   225  				break;
   226  			}
   227  		}
   228  		return c ;
   229  	}

   345  	/**
   346  	 * Returns an array representation of all days in a full month. The array can begin with null values if the first day in the first week are previous days of the previous month.
   347  	 * <p><b>Example :</b></p>
   348  	 * {@code
   349  	 * import asgard.date.Calendar ;
   350  	 * var ar:Array = Calendar.getFullMonthCalendar( new Date(2007,03) , 1 ) ;
   351  	 * trace(ar) ;
   352  	 * }
   353  	 * @param d (optional) The specified {@code Date} to return the full month calendar. The default value is the current {@code Date} object.
   354  	 * @param offset the day offset value between 0 and 6 to fill the calendar. The default value is 0 (Sunday). 
   355  	 * @return an array representation of a full month.
   356  	 */
   357  	public static function getFullMonthCalendar( d:Date , offset:Number ):Array 
   358  	{
   359  		
   360  		if (d == null)
   361  		{
   362  			d = new Date() ;	
   363  		}
   364  		
   365  		if (isNaN(offset))
   366  		{
   367  			offset = DEFAULT_FULL_MONTH_OFFSET ;
   368  		}
   369  
   370  		offset = MathsUtil.clamp(offset, 0, 6) ; 
   371  		
   372  		var y:Number = d.getFullYear() ;
   373  		var m:Number = d.getMonth() ;
   374  		var min:Number = getFirstDay ( d ) - offset ;
   375  		
   376  		if (min < 0)
   377  		{
   378  			min += 7 ;
   379  		}
   380  		
   381  		var max:Number = min + getDaysInMonth ( d ) ;
   382  		var ar:Array = new Array () ;
   383  		for (var i:Number = 0 ; i < max ; i++) 
   384  		{
   385  			var date:Number = (i - min + 1)  ;
   386  			ar[i] = (i < min) ? null : new Calendar(y, m, date) ;
   387  		}
   388  		return ar ;
   389  	}

   441  	/**
   442  	 * Calculates the week number for the given date. This function assumes that week 1 is the 
   443  	 * week in which January 1 appears, regardless of whether the week consists of a full 7 days.
   444  	 * The calendar year can be specified to help find what a the week number would be for a given date if the date overlaps years. 
   445  	 * For instance, a week may be considered week 1 of 2005, or week 53 of 2004. 
   446  	 * Specifying the optional calendarYear allows one to make this distinction easily.
   447   	 * @param date The date for which to find the week number
   448  	 * @param calendarYear (optional) The calendar year to use for determining the week number. Default is the calendar year of parameter "date".
   449  	 * @param weekStartsOn (optional) The integer (0-6) representing which day a week begins on. Default is 0 (for Sunday).
   450  	 * @return The week number of the given date.
   451  	 */
   452  	public static function getWeekNumber(date:Date, calendarYear:Number, weekStartsOn:Number):Number 
   453  	{
   454  		if (isNaN(weekStartsOn)) weekStartsOn = 0 ;
   455  		if (isNaN(calendarYear)) calendarYear = date.getFullYear();
   456  		
   457  		var weekNum:Number = -1 ;
   458  		
   459  		var jan1:Date = Calendar.getJan1(calendarYear);
   460  		var jan1DayOfWeek:Number = jan1.getDay() ;
   461  		
   462  		var dayOffset:Number = Calendar.getDayOffset(date, calendarYear); // Days since Jan 1, Calendar Year
   463  		
   464  		weekNum = 1 ;
   465  			
   466  		if (dayOffset < 0 && dayOffset >= (-1 * jan1DayOfWeek)) 
   467  		{
   468  			//			
   469  		}
   470  		else
   471  		{
   472  			var testDate:Date = Calendar.getJan1(calendarYear);
   473  			while (testDate.getTime() < date.getTime() && testDate.getFullYear() == calendarYear) 
   474  			{
   475  				weekNum += 1;
   476  				testDate = Calendar.add(testDate, Calendar.WEEK, 1);
   477  			}
   478  		}
   479  		
   480  		return weekNum;
   481  	};
