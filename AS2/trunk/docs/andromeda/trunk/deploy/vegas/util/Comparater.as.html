    24  import vegas.core.IEquality;
    25  import vegas.util.StringUtil;
    26  import vegas.util.TypeUtil;

    28  /**
    29   * The {@code Comparater} utility class is an all-static class with a method to returns {@true} if two object are equals.
    30   * @author eKameleon
    31   */
    32  class vegas.util.Comparater
    33  {
    34  
    35  	/**
    36  	 * Compares if two objects are equal by value. 
    37  	 * @param o1 the first object to compare.
    38  	 * @param o2 the second object to compare.
    39  	 * @return {@code true} if the 2 objects are equals.
    40  	 */	
    41  	static public function compare( o1, o2 ) 
    42  	{
    43  		if (o1 === undefined && o2 === undefined) 
    44  		{
    45  			return true ;
    46  		}
    47  		if (o1 === null && o2 === null)
    48  		{
    49  			return true ;
    50  		}
    51  		if (o1 instanceof IEquality) 
    52  		{
    53  			return o1.equals(o2) ;
    54  		}
    55  		else if (TypeUtil.typesMatch(o1, Array)) 
    56  		{
    57  			return arrayCompare(o1, o2) ;
    58  		}
    59  		else if (TypeUtil.typesMatch(o1, Boolean)) 
    60  		{
    61  			return booleanCompare(o1, o2);
    62  		}
    63  		else if (TypeUtil.typesMatch(o1, Date))
    64  		{
    65  			return dateCompare(o1, o2) ;
    66  		}
    67  		else if (TypeUtil.typesMatch(o1, Error))
    68  		{
    69  			return errorCompare(o1, o2) ;	
    70  		}
    71  		else if (TypeUtil.typesMatch(o1, Function))
    72  		{
    73  			return functionCompare(o1, o2) ;
    74  		}
    75  		else if (TypeUtil.typesMatch(o1, Number))
    76  		{
    77  			return numberCompare(o1, o2) ;
    78  		}
    79  		else if (TypeUtil.typesMatch(o1, String))
    80  		{
    81  			return stringCompare(o1, o2) ;
    82  		}
    83  		else if (typeof(o1) === "object")
    84  		{
    85  			return objectCompare(o1, o2) ;
    86  		}
    87  		else
    88  		{
    89  			return o1 == o2 ;
    90  		}
    91  	}	
    92  
    93  	/**
    94  	 * Compares if two Arrays are equal by value. 
    95  	 * @return {@code true} if the 2 objects are equals.
    96  	 */
    97  	static public function arrayCompare( ar1, ar2 ):Boolean
    98      {
    99  	    
   100  	    if(ar1 == null || !TypeUtil.typesMatch(ar1, Array))
   101  	    {
   102  	    	return false ;	
   103  	    }
   104  	    
   105      	if(ar2 == null || !TypeUtil.typesMatch(ar2, Array)) 
   106          {
   107          	return false ;
   108          }
   109      
   110      	if( ar1 == ar2 )
   111          {
   112          	return true ;
   113          }
   114      
   115      	if( ar1.length != ar2.length )
   116          {
   117          	return false ;
   118          }
   119      
   120      	var len:Number = ar1.length ;
   121      	for( var i:Number = 0 ; i < len ; i++ )
   122          {
   123          	if( ar1[i] == null )
   124              {
   125              	if( ar1[i] != ar2[i] )
   126                  {
   127                  	return false;
   128                  }
   129              	continue ;
   130              }
   131          
   132          	if( ! compare(ar1[i], ar2[i] ) )
   133              {
   134              	return false ;
   135              }
   136          }
   137      
   138      	return true ;
   139      }

   141  	/**
   142  	 * Compares if two Booleans are equal by value. 
   143  	 * @return {@code true} if the 2 objects are equals.
   144  	 */
   145  	static public function booleanCompare( b1 , b2 ):Boolean
   146      {
   147      	if(b1 == null || !TypeUtil.typesMatch(b1, Boolean))
   148          {
   149          	return false ;
   150          }
   151      	if(b2 == null || !TypeUtil.typesMatch(b2, Boolean))
   152          {
   153          	return false ;
   154          }
   155      	return b1.valueOf() == b2.valueOf() ;
   156      }
   157  
   158  	/**
   159  	 * Compares if two Dates are equal by value.
   160  	 * @return {@code true} if the 2 objects are equals.
   161  	 */
   162  	static public function dateCompare( d1 , d2 ):Boolean
   163  	{
   164      	if(d1 == null || !TypeUtil.typesMatch(d1, Date))
   165          {
   166          	return false ;
   167          }
   168      	if(d2 == null || !TypeUtil.typesMatch(d2, Date))
   169          {
   170          	return false ;
   171          }
   172  		return d1.toString() == d2.toString() ;
   173  	}
   174  	
   175  	/**
   176  	 * Compares if two Errors are equal by value.
   177  	 * @return {@code true} if the 2 objects are equals.
   178  	 */
   179  	static public function errorCompare( e1 , e2 ):Boolean
   180  	{
   181      	if(e1 == null || !TypeUtil.typesMatch(e1, Error))
   182          {
   183          	return false ;
   184          }
   185      	if(e2 == null || !TypeUtil.typesMatch(e2, Error))
   186          {
   187          	return false ;
   188          }
   189  		return e1.toString() == e2.toString() ;
   190  	}
   191  	
   192  	/**
   193  	 * Compares if two Functions are equal by value.
   194  	 * @return {@code true} if the 2 objects are equals.
   195  	 */
   196  	static public function functionCompare( f1 , f2:Function ):Boolean
   197  	{
   198      	if(f1 == null || !TypeUtil.typesMatch(f1, Function))
   199          {
   200          	return false ;
   201          }
   202      	if(f2 == null || !TypeUtil.typesMatch(f2, Function))
   203          {
   204          	return false ;
   205          }
   206  		return f1.valueOf() == f2.valueOf() ;
   207  	}
   208  
   209  	/**
   210  	 * Compares if two Numbers are equal by value.
   211  	 * @return {@code true} if the 2 objects are equals.
   212  	 */
   213  	static public function numberCompare( n1 , n2 ):Boolean
   214  	{
   215      	
   216      	if(n1 == null || !TypeUtil.typesMatch(n1, Number))
   217          {
   218          	return false ;
   219          }
   220      	if(n2 == null || !TypeUtil.typesMatch(n2, Number))
   221          {
   222          	return false ;
   223          }
   224  
   225  		if ( n1.valueOf() == n2.valueOf() )
   226  		{
   227  			return true ;
   228  		}
   229  
   230  		// Test if the 2 objects are NaN
   231          if ( isNaN(n1) && isNaN(n2) )
   232          {
   233          	return true ;	
   234          }
   235  
   236  		if ( n1.toString() == n2.toString() ) 
   237  		{
   238  			return true ;	
   239  		}
   240  
   241  		return false ;
   242  	}
   243  	
   244  	/**
   245  	 * Compares if two Objects are equal by value.
   246  	 * @return {@code true} if the 2 objects are equals.
   247  	 */
   248  	static public function objectCompare( o1 , o2 ):Boolean
   249  	{
   250  		if ( TypeUtil.typesMatch(o2, Function ) )
   251  		{
   252  			return functionCompare( o2, o1 ) ;	
   253  		}
   254  		if ( o1.valueOf() == null && o2 == null )
   255  		{
   256  			return true ;
   257  		}
   258  		if (o2 == null || !TypeUtil.typesMatch(o2, Object))
   259  		{
   260  			return false ;	
   261  		}
   262  		if ( o1 == o2 )
   263  		{
   264  			return true ;
   265  		}
   266  		
   267  		for (var member:String in o1)
   268  		{
   269  			/* Note: by convention we consider members starting with __
   270             		to be internal properties which should not be compared.
   271          	*/
   272  			if ( StringUtil.startsWith(member, "__") ) 
   273  			{
   274  				continue ;	
   275  			}
   276  			if( !Comparater.compare(o1[member], o2[member] ) )
   277              {
   278              	return false;
   279              }
   280          }	
   281          
   282  		return true ;		
   283  	}
   284  	
   285  	/**
   286  	 * Compares if two Strings are equal by value.
   287  	 * @return {@code true} if the 2 objects are equals.
   288  	 */
   289  	static public function stringCompare( s1 , s2 ):Boolean
   290      {
   291      	if( (s1 == null) || !TypeUtil.typesMatch( s2 , String ) )
   292          {
   293          	return false ;
   294          }
   295          
   296      	if( (s2 == null) || !TypeUtil.typesMatch( s2 , String ) )
   297          {
   298          	return false ;
   299          }
   300      
   301      	if( s1.toString() == s2.toString() )
   302          {
   303          	return true;
   304          }
   305      
   306  	    return false;
   307  
   308      }
