    24  import vegas.core.CoreObject;
    25  import vegas.core.IComparator;
    26  import vegas.core.ISerializable;
    27  import vegas.util.serialize.Serializer;
    28  
    29  /**
    30   * Reverse an IComparator object. For example if the comparator must return 1 the reverse comparator return -1.
    31   * <p><b>Example :</b></p>
    32   * {@code
    33   * import vegas.util.comparators.ReverseComparator ;
    34   * import vegas.util.comparators.StringComparator ;
    35   * 
    36   * var c:StringComparator = new StringComparator() ;
    37   * var s:ReverseComparator = new ReverseComparator( c ) ;
    38   * 
    39   * trace( c.compare( "hello", "world" ) ) ; // -1
    40   * trace( s.compare( "hello", "world" ) ) ; // 1
    41   * }
    42   * @author eKameleon
    43   */
    44  class vegas.util.comparators.ReverseComparator extends CoreObject implements IComparator
    45  {
    46  
    47  	/**
    48  	 * Creates a new ReverseComparator instance.
    49  	 * @param c the {@code IComparator} to be reverse.
    50  	 */
    51  	public function ReverseComparator( c:IComparator ) 
    52  	{
    53  		comparator = c ;
    54  	}
    55  	
    56  	/**
    57  	 * The {@code IComparator} instance to reverse.
    58  	 */	
    59  	public var comparator:IComparator ;
    60  	
    61  	/**
    62  	 * Returns an integer value to compare two objects but the value is reverse.
    63  	 * The compare method use the internal compare method of the {@code IComparator} defined in the comparator property or in the constructor of this object.
    64  	 * @param o1 the first object to compare.
    65  	 * @param o2 the second object to compare.
    66  	 * @return <p>
    67  	 * <li> 1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
    68  	 * <li> -1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
    69  	 * <li> 0 if o1 and o2 are equal.</li>
    70  	 * </p>
    71  	 */
    72  	public function compare(o1, o2):Number 
    73  	{
    74  		return comparator.compare(o2, o1) ;
    75  	}
    76  
    77  	/**
    78  	 * Returns the singleton instance of a ReverseComparator.
    79  	 * Developers are encouraged to use the comparator returned from this method instead of constructing a new instance to reduce allocation and GC overhead when multiple comparable comparators may be used in the same application.
    80   	 * @return the singleton instance of a ReverseComparator.
    81  	 */
    82  	public static function getInstance():ReverseComparator
    83  	{
    84  		if (_instance == null)
    85  		{
    86  			_instance = new ReverseComparator() ;
    87  		}
    88  		return _instance ;	
    89  	}
    90  
    91  	/**
    92  	 * Returns a Eden reprensation of the object.
    93  	 * @return a string representing the source code of the object.
    94  	 */
    95  	public function toSource(indent : Number, indentor : String):String 
    96  	{
    97  		return Serializer.getSourceOf(this,[ ISerializable(comparator).toSource() || "null" ] ) ;
    98  	}
    99  
   100  	/**
   101    	 * The internal static singleton of this class.
   102   	 */
   103  	private static var _instance:ReverseComparator ;
   104  
   105  }
