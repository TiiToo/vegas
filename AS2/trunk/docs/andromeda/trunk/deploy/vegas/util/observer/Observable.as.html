    24  import vegas.core.CoreObject;
    25  import vegas.data.iterator.Iterator;
    26  import vegas.data.list.ArrayList;
    27  import vegas.errors.NullPointerError;
    28  import vegas.util.observer.IObserver;
    29  
    30  /**
    31   * This class represents an observable object, or "data" in the model-view paradigm. It can be subclassed to represent an object that the application wants to have observed.
    32   * @author eKameleon
    33   */
    34  class vegas.util.observer.Observable extends CoreObject
    35  {
    36  
    37  	/**
    38  	 * Creates an Observable instance with zero Observers.
    39  	 */
    40  	public function Observable() 
    41  	{
    42  		_obs = new ArrayList() ;
    43  	}
    44  
    45  	/**
    46  	 *  Adds an observer to the set of observers for this object, provided that it is not the same as some observer already in the set.
    47   	 */
    48  	public function addObserver(o:IObserver):Boolean 
    49  	{
    50  		if (o == null) 
    51  		{
    52  			throw new NullPointerError(this + " the passed object in argument not must be 'null' or 'undefined'.") ;
    53  		}
    54  		if (!_obs.contains(o)) 
    55  		{
    56  			_obs.insert(o) ;
    57  			return true ;
    58  		}
    59  		return false ;
    60  	}
    61  
    62  	/**
    63  	 * Indicates that this object has no longer changed, or that it has already notified all of its observers of its most recent change, so that the hasChanged method will now return false.
    64  	 */
    65  	public function clearChanged():Void 
    66  	{
    67  		_changed = false ;
    68  	}
    69  	
    70  	/**
    71  	 * Tests if this object has changed.
    72  	 */
    73  	public function hasChanged():Boolean 
    74  	{
    75  		return _changed ;
    76  	}
    77  
    78  	/**
    79  	 * If this object has changed, as indicated by the hasChanged method, then notify all of its observers and then call the clearChanged method to indicate that this object has no longer changed.
    80  	 */
    81  	public function notifyObservers( arg ):Void 
    82  	{
    83  		if (arg == undefined) 
    84  		{
    85  			arg = null ;
    86  		}
    87  		if (!_changed) 
    88  		{
    89  			return ;
    90  		}
    91  		clearChanged() ;
    92  		var _obsMemory:ArrayList = _obs.clone() ;
    93  		var it:Iterator = _obsMemory.iterator() ;
    94  		while(it.hasNext()) 
    95  		{
    96  			IObserver(it.next()).update(this, arg) ;
    97  		}
    98  	}
    99  
   100  	/**
   101  	 * Removes an observer from the set of observers of this object.
   102  	 */
   103  	public function removeObservers(o:IObserver):Void 
   104  	{
   105  		if (o == undefined) 
   106  		{
   107  			_obs.clear() ;
   108  		}
   109  		else 
   110  		{
   111  			_obs.remove(o) ;
   112  		}
   113  	}
   114  	
   115  	/**
   116  	 * Marks this Observable object as having been changed; the hasChanged method will now return true
   117  	 */
   118  	public function setChanged():Void 
   119  	{
   120  		_changed = true ;
   121  	}
   122  
   123  	/**
   124  	 * Returns the number of observers of this Observable object.
   125  	 * @return the number of observers of this Observable object.
   126  	 */
   127  	public function size():Number 
   128  	{
   129  		return _obs.size() ;
   130  	}
   131  
   132  	/**
   133  	 * The internal ArrayList.
   134  	 */		
   135  	private var _obs:ArrayList ;
   136  	
   137  	/**
   138  	 * The internal value to notify if this Observable object is changed or not.
   139  	 */
   140  	private var _changed:Boolean ;
   141  	
   142  }
