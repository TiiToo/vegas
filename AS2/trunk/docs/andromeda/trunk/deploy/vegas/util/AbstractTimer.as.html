    24  import vegas.core.ICloneable;
    25  import vegas.core.ICopyable;
    26  import vegas.core.IFormattable;
    27  import vegas.core.IRunnable;
    28  import vegas.core.ITimer;
    29  import vegas.events.EventDispatcher;
    30  import vegas.events.TimerEvent;
    31  
    32  /**
    33   * This abstract class is used to create concrete {@code ITimer} implementations.
    34   * @author eKameleon
    35   * @see EventDispatcher
    36   * @see TimerEvent
    37   */
    38  class vegas.util.AbstractTimer extends EventDispatcher implements ICloneable, ICopyable, ITimer, IFormattable, IRunnable 
    39  {
    40  
    41  	/**
    42  	 * Creates a new ITimer instance if you extend this class.
    43  	 * @param delay the delay between two intervals in this timer.
    44  	 * @param count the number of repetitions of this timer.
    45  	 */
    46  	private function AbstractTimer( nDelay :Number, nCount:Number) 
    47  	{
    48  		setDelay( nDelay ) ;
    49  		setRepeatCount( nCount ) ;
    50  	}
    51  
    52  	/**
    53  	 * (read-write) Returns the delay between timer events, in milliseconds.
    54  	 * @return the delay between timer events, in milliseconds.
    55  	 */
    56  	public function  get delay():Number 
    57  	{
    58  		return getDelay() ;
    59  	}
    60  
    61  	/**
    62  	 * (read-write) Sets the delay between timer events, in milliseconds.
    63  	 */
    64  	public function set delay( n:Number ):Void 
    65  	{
    66  		setDelay(n) ;	
    67  	}
    68  
    69  	/**
    70  	 * (read-write) Returns the number of repetitions. If zero, the timer repeats infinitely. If nonzero, the timer runs the specified number of times and then stops.
    71  	 * @return  the number of repetitions. If zero, the timer repeats infinitely. If nonzero, the timer runs the specified number of times and then stops.
    72  	 */
    73  	public function  get repeatCount():Number 
    74  	{
    75  		return getRepeatCount() ;
    76  	}
    77  
    78  	/**
    79  	 * (read-write) Sets the number of repetitions. If zero, the timer repeats infinitely. If nonzero, the timer runs the specified number of times and then stops.
    80  	 */
    81  	public function set repeatCount ( n:Number ):Void 
    82  	{
    83  		setRepeatCount(n) ;	
    84  	}
    85  	
    86  	/**
    87  	 * (read-only) Returns {@code true} if the timer is in progress.
    88  	 * @return {@code true} if the timer is in progress.
    89  	 */
    90  	public function get running():Boolean 
    91  	{
    92  		return getRunning() ;
    93  	}	
    94  
    95  	/**
    96  	 * Clear the timer.
    97  	 */
    98  	public function clear():Void 
    99  	{
   100  		// override this method
   101  	}
   102  
   103  	/**
   104  	 * Returns a shallow copy of this object.
   105  	 * @return a shallow copy of this object.
   106  	 */
   107  	public function clone() 
   108  	{
   109  		return null ;
   110  	}
   111  	
   112  	/**
   113  	 * Returns a deep copy of this object.
   114  	 * @return a deep copy of this object.
   115  	 */
   116  	/*override*/ public function copy() 
   117  	{
   118  		return null ;
   119  	}
   120  
   121  	/**
   122  	 * Returns the delay between timer events, in milliseconds.
   123  	 * @return the delay between timer events, in milliseconds.
   124  	 */
   125  	public function getDelay():Number 
   126  	{
   127  		return _delay ;
   128  	}
   129  
   130  	/**
   131  	 * Returns the current counter value of the timer.
   132  	 * @return the current counter value of the timer.
   133  	 */
   134  	public function getCount():Number
   135  	{
   136  		return _count ;	
   137  	}
   138  
   139  	/**
   140  	 * Returns the number of repetitions. If zero, the timer repeats infinitely. If nonzero, the timer runs the specified number of times and then stops.
   141  	 * @return the number of repetitions. If zero, the timer repeats infinitely. If nonzero, the timer runs the specified number of times and then stops.
   142  	 */
   143  	public function getRepeatCount():Number 
   144  	{
   145  		return _repeatCount ;
   146  	}
   147  
   148  	/**
   149  	 * Returns {@code true} if the timer is in progress.
   150  	 * @return {@code true} if the timer is in progress.
   151  	 */
   152  	public function getRunning():Boolean 
   153  	{
   154  		return _running ;	
   155  	}
   156  
   157  	/**
   158  	 * Restarts the timer. The timer is stopped, and then started.
   159  	 */
   160  	public function restart(noEvent:Boolean):Void 
   161  	{
   162  		if (getRunning()) this.stop() ;
   163  		_setRunning(true) ;
   164  		run() ;
   165  		if (!noEvent) 
   166  		{
   167  			dispatchEvent( new TimerEvent( TimerEvent.RESTART, this) ) ;
   168  		}
   169  	}
   170  
   171  	/**
   172  	 * Run the timer. Overrides this method.
   173  	 * @see IRunnable
   174  	 */
   175  	public function run():Void 
   176  	{
   177  		// override this method
   178  	}
   179  	
   180  	/**
   181  	 * Sets the delay between timer events, in milliseconds.
   182  	 */
   183  	public function setDelay(n:Number):Void 
   184  	{
   185  		_delay = (n > 0) ? n : 0 ;
   186  		if (getRunning()) 
   187  		{
   188  			restart() ;
   189  		}
   190  	}
   191  
   192  	/**
   193  	 * Sets the number of repetitions. If zero, the timer repeats infinitely. If nonzero, the timer runs the specified number of times and then stops.
   194  	 */
   195  	public function setRepeatCount(n:Number):Void
   196  	{
   197  		_repeatCount = (n > 0) ? n : 0 ;
   198  	}
   199  
   200  	/**
   201  	 * Starts the timer, if it is not already running.
   202  	 */
   203  	public function start():Void 
   204  	{
   205  		if (getRunning()) return ;
   206  		_count = 0 ;
   207  		dispatchEvent( new TimerEvent(TimerEvent.START, this) ) ;
   208  		restart(true) ;
   209  	}
   210  	
   211  	/**
   212  	 * Stops the timer.
   213  	 */
   214  	public function stop():Void 
   215  	{
   216  		_setRunning(false) ;
   217  		clear() ;
   218  		dispatchEvent( new TimerEvent(TimerEvent.STOP, this) ) ;
   219  	}
   220  	
   221  	/**
   222  	 * The internal counter of the timer.
   223  	 */
   224  	private var _count:Number ;
   225  	
   226  	/**
   227  	 * The internal delay of the timer.
   228  	 */
   229  	private var _delay:Number ;
   230  	
   231  	/**
   232  	 * The internal repeat counter of the timer.
   233  	 */
   234  	private var _repeatCount:Number ;
   235  	
   236  	/**
   237  	 * Determintates if the timer is in progress or not.
   238  	 */
   239  	private var _running:Boolean ;
   240  	
   241  	/**
   242  	 * Protected methods to defined the running property when the timer is in progress.
   243  	 */
   244  	private function _setRunning(b:Boolean):Void 
   245  	{
   246  		_running = b ;
   247  	}
   248  
   249  }
