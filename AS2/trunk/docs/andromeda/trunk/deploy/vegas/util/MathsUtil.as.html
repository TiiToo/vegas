    26  /**
    27   * The {@code MathUtil} utility class is an all-static class with methods for working with numbers.

    30  class vegas.util.MathsUtil

    33  	/**
    34  	 * Rounds and returns the ceiling of the specified number or expression. 
    35  	 * The ceiling of a number is the closest integer that is greater than or equal to the number.
    36  	 * <p><b>Example :</b></p>
    37  	 * {@code
    38  	 * import vegas.util.MathsUtil ;
    39  	 * var n = MathsUtil.ceil(4.572525153, 2) ;
    40  	 * trace ("n : " + n) ; // n : 4.58
    41  	 * 
    42  	 * var n = MathsUtil.ceil(4.572525153, -1) ;
    43  	 * trace ("n : " + n) ; // n : 5
    44  	 * }
    45  	 * @param n the number to round.
    46  	 * @param floatCount the count of number after the point.
    47  	 * @return the ceil value of a number by a count of floating points.
    48  	 */
    49  	public static function ceil(n:Number, floatCount:Number):Number 
    50  	{
    51  		if (isNaN(n)) 
    52  		{
    53  			throw new IllegalArgumentError("MathsUtil.floor, Argument 'n' must not be 'null' or 'undefined'.") ;
    54  		}
    55  		var r:Number = 1 ;
    56  		var i:Number = -1 ;
    57  		while (++i < floatCount) 
    58  		{
    59  			r *= 10 ;
    60  		}
    61  		return Math.floor(n*r) / r  ;
    62  	}

    65  	/**
    66  	 * Bounds a numeric value between 2 numbers.
    67  	 * <p><b>Example :</b></p>
    68  	 * {@code
    69  	 * import vegas.util.MathsUtil ;
    70  	 * 
    71  	 * var n = MathsUtil.clamp(4, 5, 10) ;
    72  	 * trace ("n : " + n) ;
    73  	 * 
    74  	 * var n = MathsUtil.clamp(12, 5, 10) ;
    75  	 * trace ("n : " + n) ;
    76  	 * 
    77  	 * var n = MathsUtil.clamp(6, 5, 10) ;
    78  	 * trace ("n : " + n) ;
    79  	 * 
    80  	 * var n = MathsUtil.clamp(null, 5, 10) ;
    81  	 * trace ("n : " + n) ;
    82  	 * }
    83  	 * @param value the value to clamp.
    84  	 * @param min the min value of the range.
    85  	 * @param max the max value of the range.
    86  	 * @return a bound numeric value between 2 numbers.
    87  	 */
    88  	public static function clamp(value:Number, min:Number, max:Number):Number 
    89  	{
    90  		if (isNaN(value)) 
    91  		{
    92  			throw new IllegalArgumentError("MathsUtils.clamp, argument 'value' must not be 'null' or 'undefined' or 'NaN'") ;
    93  		}
    94  		if (isNaN(min)) 
    95  		{
    96  			min = value ;
    97  		}
    98  		if (isNaN(max)) 
    99  		{
   100  			max = value ;
   101  		}
   102  		return Math.max(Math.min(value, max), min) ;
   103  	}

   105  	/**
   106  	 * Rounds and returns the number or expression specified in the parameter n. 
   107  	 * The floor is the closest integer that is less than or equal to the specified number or expression.
   108  	 * <p><b>Example :</b></p>
   109  	 * {@code
   110  	 * import vegas.util.MathsUtil ;
   111  	 * 
   112  	 * var n = MathsUtil.floor(4.572525153, 2) ;
   113  	 * trace ("n : " + n) ; // n : 4.57
   114  	 * 
   115  	 * var n = MathsUtil.floor(4.572525153, -1) ;
   116  	 * trace ("n : " + n) ; // n : 4
   117  	 * }
   118  	 * @param n the number to round.
   119  	 * @param floatCount the count of number after the point.
   120  	 * @return the floor value of a number by a count of floating points.
   121  	 */
   122  	public static function floor(n:Number, floatCount:Number):Number 
   123  	{
   124  		if (isNaN(n)) 
   125  		{
   126  			throw new IllegalArgumentError("MathsUtil.floor, Argument 'n' must not be 'null' or 'undefined'.") ;
   127  		}
   128  		var r:Number = 1 ;
   129  		var i:Number = -1 ;
   130  		while (++i < floatCount) 
   131  		{
   132  			r *= 10 ;
   133  		}
   134  		return Math.floor(n*r) / r  ;
   135  	}

   137  	/**
   138  	 * Returns a percentage value or null.
   139  	 * <p><b>Example :</b></p>
   140  	 * {@code
   141  	 * import vegas.util.MathsUtil ;
   142  	 * trace( MathsUtil.getPercent( 50 , 100 ) + "%" ) ; // 50%
   143  	 * trace( MathsUtil.getPercent( 68 , 425 ) + "%" ) ; // 16% 
   144  	 * }
   145  	 * @param nValue the current value.
   146  	 * @param nMax the max value.
   147  	 * @return a percentage value or null.
   148  	 */
   149  	public static function getPercent(nValue:Number, nMax:Number):Number 
   150  	{
   151  		var nP:Number = (nValue / nMax) * 100 ;
   152  		return (isNaN(nP) || !isFinite(nP)) ? null : nP ;
   153  	}
   154  	
   155  	/**
   156  	 * Rounds and returns a number by a count of floating points.
   157  	 * <p><b>Example :</b></p>
   158  	 * {@code
   159  	 * import vegas.util.MathsUtil ;
   160  	 * 
   161  	 * var n = MathsUtil.round(4.572525153, 2) ;
   162  	 * trace ("n : " + n) ;
   163  	 * 
   164  	 * var n = MathsUtil.round(4.572525153, -1) ;
   165  	 * trace ("n : " + n) ;
   166  	 * }
   167  	 * @param n the number to round.
   168  	 * @param floatCount the count of number after the point.
   169  	 * @return the round of a number by a count of floating points.
   170  	 */
   171  	public static function round(n:Number, floatCount:Number):Number 
   172  	{
   173  		if (isNaN(n)) 
   174  		{
   175  			throw new IllegalArgumentError("MathsUtil.round, Argument 'n' must not be 'null' or 'undefined'.") ;
   176  		}
   177  		var r:Number = 1 ;
   178  		var i:Number = -1 ;
   179  		while (++i < floatCount) 
   180  		{
   181  			r *= 10 ;
   182  		}
   183  		return Math.round(n*r) / r  ;
   184  	}

   186  	/**
   187  	 * Returns 1 if the value is positive or -1.
   188  	 * <p><b>Example :</b></p>
   189  	 * {@code
   190  	 * import vegas.util.MathsUtil ;
   191  	 * 
   192  	 * n = MathsUtil.sign(-150) ;
   193  	 * trace ("n : " + n) ;
   194  	 * 
   195  	 * n = MathsUtil.sign(200) ;
   196  	 * trace ("n : " + n) ;
   197  	 * 
   198  	 * n = MathsUtil.sign(0) ;
   199  	 * trace ("n : " + n) ;
   200  	 * }
   201  	 * @param n the number to defined this sign.
   202  	 * @return 1 if the value is positive or -1.
   203  	 */
   204  	public static function sign( n:Number ):Number 
   205  	{
   206  		if (isNaN(n)) 
   207  		{
   208  			throw new IllegalArgumentError("MathsUtil.sign, Argument 'n' must not be 'null' or 'undefined'.") ;
   209  		}
   210  		return n<0 ? -1 : 1 ;
   211  	}
