    24  import vegas.events.BasicEvent;
    25  
    26  /**
    27   * {@code BasicEvent} is the dynamic event structure to work with {@link EventDispatcher} and {@link FastDispatcher}.
    28   * <p><b>Example</b></p>
    29   * {@code var e:BasicEvent = new DynamicEvent(type, target, context) ; } 
    30   * @author  eKameleon
    31   * @see     CoreObject	
    32   * @see     Event	
    33   */
    34  dynamic class vegas.events.DynamicEvent extends BasicEvent 
    35  {
    36  
    37  	/**
    38  	 * Constructs a new {@code DynamicEvent} instance.
    39  	 * <p>
    40  	 * {@code
    41  	 *     var e:DynamicEvent = new DynamicEvent(type, target, context, [bubbles:Boolean, [eventPhase:Number, [time:Number, [stop:Boolean]]]]) ;
    42  	 * }
    43  	 * </p>
    44  	 * @param type the string type of the instance. 
    45  	 * @param target the target of the event.
    46  	 * @param context the optional context object of the event.
    47  	 * @param bubbles indicates if the event is a bubbling event.
    48  	 * @param eventPhase the current EventPhase of the event.
    49  	 * @param time this parameter is used in the Eden deserialization.
    50  	 * @param stop this parameter is used in the Eden deserialization.
    51  	 */
    52  	public function DynamicEvent( type:String, target, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number) 
    53  	{
    54  		super(type, target, context, bubbles, eventPhase, time, stop) ;
    55  	}
    56  
    57  	/**
    58  	 * Returns {@code true} if the event is bubbling.
    59  	 * @return {@code true} if the event is bubbling.
    60  	 */
    61  	public function get bubbles():Boolean 
    62  	{
    63  		return getBubbles() ;	
    64  	}
    65  
    66  	/**
    67  	 * Sets {@code true} if the event is bubbling.
    68  	 */
    69  	public function set bubbles(b:Boolean):Void 
    70  	{
    71  		setBubbles(b) ;	
    72  	}
    73  
    74  	/**
    75  	 * Returns the optional context of this event.
    76  	 * @return an object, corresponding the optional context of this event.
    77  	 */
    78  	public function get context() 
    79  	{
    80  		return getContext() ;	
    81  	}
    82  
    83  	/**
    84  	 * Sets the optional context of this event.
    85  	 */
    86  	public function set context(o):Void 
    87  	{
    88  		setContext(o) ;	
    89  	}
    90  
    91  	/**
    92  	 * Returns the object that is actively processing the Event object with an event listener.
    93  	 * @return the object that is actively processing the Event object with an event listener.
    94  	 */
    95  	public function get currentTarget() 
    96  	{
    97  		return getCurrentTarget() ;	
    98  	}
    99  
   100  	/**
   101  	 * Sets the object that is actively processing the Event object with an event listener.
   102  	 */
   103  	public function set currentTarget(o):Void 
   104  	{
   105  		setCurrentTarget(o) ;	
   106  	}
   107  
   108  	/**
   109  	 * Returns the current phase in the event flow.
   110  	 * @return the current phase in the event flow.
   111  	 * @see EventPhase
   112  	 */
   113  	public function get eventPhase():Number 
   114  	{
   115  		return getEventPhase() ;	
   116  	}
   117  
   118  	/**
   119  	 * Sets the current phase in the event flow.
   120  	 * @see EventPhase
   121  	 */
   122  	public function set eventPhase(n:Number):Void 
   123  	{
   124  		setEventPhase(n) ;	
   125  	}
   126  
   127  	/**
   128  	 * Returns the event target.
   129  	 * @return the event target.
   130  	 */
   131  	public function get target() 
   132  	{
   133  		return getTarget() ;	
   134  	}
   135  
   136  	/**
   137  	 * Sets the event target.
   138  	 */
   139  	public function set target(o):Void 
   140  	{
   141  		setTarget(o) ;	
   142  	}
   143  	
   144  	/**
   145  	 * Returns the timestamp of the event.
   146  	 * @return the timestamp of the event.
   147  	 */
   148  	public function get timeStamp():Number 
   149  	{
   150  		return getTimeStamp() ;	
   151  	}
   152  
   153  	/**
   154  	 * Returns the type of event.
   155  	 * @return the type of event.
   156  	 */
   157  	public function get type():String 
   158  	{
   159  		return getType() ;	
   160  	}
   161  
   162  	/**
   163  	 * Sets the type of event.
   164  	 */
   165  	public function set type(s:String):Void 
   166  	{
   167  		setType(s) ;	
   168  	}
   169  
   170  	/**
   171  	 * Returns the shallow copy of this event.
   172  	 * @return the shallow copy of this event.
   173  	 */
   174  	/*override*/ public function clone() 
   175  	{
   176  		return new DynamicEvent(getType(), getTarget(), getContext()) ;
   177  	}
   178  
   179  }
