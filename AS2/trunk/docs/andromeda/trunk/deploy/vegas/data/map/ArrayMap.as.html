    26  /**
    27   * ArrayMap is an Map implementation based on two arrays to defines the collections of the keys and the values. 
    28   * @author eKameleon
    29   */
    30  class vegas.data.map.ArrayMap extends HashMap 
    31  {
    32  
    33  	/**
    34  	 * Creates a new ArrayMap instance.
    35  	 */
    36  	public function ArrayMap() 
    37  	{
    38  		if ( arguments.caller == null )
    39  		{
    40  			Function( HashMap ).apply(this, [].concat(arguments) ) ;
    41  		} 
    42  	}
    43  
    44  	// TODO setKeyAt : verify if the key exist or not in the Map and notify an Error ?
    45  
    46  	/**
    47  	 * Sets the value of the "key" in the HashMap (ArrayMap) with the specified index.
    48  	 * @return the old 'key' value in the map if exist.
    49  	 */
    50  	public function setKeyAt( index:Number, key ) 
    51  	{
    52  		var old = _keys[index] ;
    53  		if (old === undefined)
    54  		{
    55  			return null ;	
    56  		}
    57  		_keys[index] = key ;
    58  		return old ;
    59  	}
    60  
    61  	/**
    62  	 * Sets the value of the "value" in the HashMap (ArrayMap) with the specified index.
    63  	 * @return the old value in the map if exist.
    64  	 */
    65  	public function setValueAt( index:Number, value ) 
    66  	{
    67  		var old = _values[index] ;
    68  		if (old === undefined)
    69  		{
    70  			return null ;	
    71  		}
    72  		_values[index] = value ;
    73  		return old ;
    74  	}
    75  
    76  }
