    24  import vegas.data.Enumeration;
    25  
    26  /**
    27   * The Dictionary is an interface such as Map, which maps keys to values. Every key and every value is an object. In any one Dictionary object, every key is associated with at most one value.
    28   * @author eKameleon
    29   */
    30  interface vegas.data.Dictionnary 
    31  {
    32  	
    33  	/**
    34  	 * Returns an enumeration of the keys in this dictionary.
    35  	 */
    36  	function getKeys():Enumeration ;
    37  
    38  	/**
    39  	 * Returns the value to which the key is mapped in this dictionary.
    40  	 */
    41  	function get(key) ;
    42  
    43  	/**
    44  	 * Returns an enumeration of the values in this dictionary.
    45  	 */
    46  	function getValues():Enumeration ;
    47  
    48  	/**
    49  	 * Returns {@code true} if this dictionnary contains no elements.
    50  	 * @return {@code true} if this dictionnary is empty else {@code false}.
    51  	 */
    52  	function isEmpty():Boolean ;
    53  	
    54  	/**
    55  	 * Maps the specified key to the specified value in this dictionary.
    56  	 */
    57  	function put( key , value ) ;
    58  	
    59  	/**
    60  	 * Removes the key (and its corresponding value) from this dictionary.
    61  	 */
    62  	function remove(key) ;
    63  
    64  	/**
    65  	 * Returns the number of entries (dinstint keys) in this dictionary.
    66  	 * @return the number of entries.
    67  	 */
    68  	function size():Number ;
    69  	
    70  }
