   178  	 * Sorts the elements in Map by key or value with the IComparator of this ArrayMap.
   179  	 * <p><b>Example :</b></p>
   180  	 * {@code
   181  	 * var map:SortedArrayMap = new SortedArrayMap() ;
   182  	 * 
   183   	 * map.put( "key1", "value4" ) ;
   184  	 * map.put( "key3", "value2" ) ;
   185   	 * map.put( "key4", "value3" ) ;
   186  	 * map.put( "key2", "value1" ) ;
   187  	 * 
   188  	 * trace(map) ; // {key1:value4,key3:value2,key4:value3,key2:value1}
   189  	 * 
   190   	 * map.comparator = StringComparator.getStringComparator() ;
   191  	 * 
   192  	 * map.options = SortedArrayMap.DESCENDING ;
   193  	 * map.sort() ;
   194  	 * 
   195  	 * trace(map) ; // {key4:value3,key3:value2,key2:value1,key1:value4}
   196  	 * 
   197   	 * map.options = SortedArrayMap.NONE ;
   198  	 * map.sort() ;
   199  	 * 
   200   	 * trace(map) ; // {key1:value4,key2:value1,key3:value2,key4:value3}
   201   	 * 
   202  	 * map.sortBy = SortedArrayMap.VALUE ;
   203  	 * 
   204  	 * map.options = SortedArrayMap.DESCENDING ;
   205  	 * map.sort() ;
   206  	 * 
   207   	 * trace(map) ; // {key1:value4,key4:value3,key3:value2,key2:value1}
   208  	 * 
   209   	 * map.options = SortedArrayMap.NONE ;
   210   	 * map.sort() ;
   211   	 * 
   212  	 * trace(map) ; // {key2:value1,key3:value2,key4:value3,key1:value4}
   213  	 * } 
   214   	 * @see SortedArrayMap#sortBy
   215   	 */
   216  	public function sort():Void  
   217  	{
   218  		
   219  		var compare:Function = _comparator.compare ;
   220  		
   221  		var max:Number = size() ;
   222  		
   223  		if (compare == null && !(max > 0) )
   224  		{
   225  			return ;	
   226  		}
   227  
   228  		var result:Array ;
   229  		var clone:Array ;
   230  		if ( sortBy == KEY )
   231  		{
   232  			result = _keys.sort( compare , _options | Array.RETURNINDEXEDARRAY ) ;
   233  			_keys.sort( compare , _options ) ;
   234  			clone = getValues() ;
   235  			while ( --max > -1 )
   236  			{
   237  				_values[max] = clone[ result[max] ] ;
   238  			}
   239  		}
   240  		else
   241  		{
   242  			result = _values.sort( compare , _options | Array.RETURNINDEXEDARRAY ) ;
   243  			_values.sort( compare , _options ) ;
   244  			clone = getKeys() ;
   245  			while ( --max > -1 )
   246  			{
   247  				_keys[max] = clone[ result[max] ] ;
   248  			}
   249  		}
   250  		
   251  	}

   320  	public function sortOn( fieldName , opts:Number ):Void 
   321  	{
   322  		
   323  		var max:Number = size() ;
   324  		
   325  		if (fieldName == null && !(max > 0) )
   326  		{
   327  			return ;	
   328  		}
   329  		
   330  		if ( !isNaN(opts) )
   331  		{
   332  			_options = opts ;	
   333  		}
   334  		
   335  		var result:Array ;
   336  		var clone:Array ;
   337  		if ( sortBy == KEY )
   338  		{
   339  			result = _keys.sortOn( fieldName , _options | Array.RETURNINDEXEDARRAY ) ;
   340  			_keys.sortOn( fieldName , _options ) ;
   341  			clone = getValues() ;
   342  			while ( --max > -1 )
   343  			{
   344  				_values[max] = clone[ result[max] ] ;
   345  			}
   346  		}
   347  		else
   348  		{
   349  			result = _values.sortOn( fieldName , _options | Array.RETURNINDEXEDARRAY ) ;
   350  			_values.sortOn( fieldName  , _options ) ;
   351  			clone = getKeys() ;
   352  			while ( --max > -1 )
   353  			{
   354  				_keys[max] = clone[ result[max] ] ;
   355  			}
   356  		}
   357  		
   358  	}
