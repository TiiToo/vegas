    35  /**
    36   * Converts a {@code LinkedList} to a specific {@code ListIterator}.
    37   * <p><b>Example :</b></p>
    38   * {@code
    39   * import vegas.data.iterator.LinkedListIterator ;
    40   * import vegas.data.list.LinkedList ;
    41   * 
    42   * var list:LinkedList = new LinkedList() ;
    43   * list.insert("item1") ;
    44   * list.insert("item2") ;
    45   * list.insert("item3") ;
    46   * var it:LinkedListIterator = LinkedListIterator( list.listIterator(2) ) ;
    47   * trace(it.hasPrevious() + " : " + it.previous()) ; // true : item2
    48   * }
    49   * @author eKameleon
    50   */
    51  class vegas.data.iterator.LinkedListIterator extends CoreObject implements ListIterator
    52  {
    53  
    54  	/**
    55  	 * Creates a new LinkedListIterator instance.
    56  	 */
    57  	function LinkedListIterator( index:Number , list:LinkedList )
    58  	{
    59  		
    60  		if (list == null)
    61  		{
    62  			throw new IllegalArgumentError(this + " constructor failed, the list passed in argument not must be 'null' or 'undefined'.") ;	
    63  		}
    64  		
    65  		this._list = list ;
    66  		
    67  		seek(index) ;
    68  		
    69  	}
    70  
    71  	/**
    72  	 * This method check the comodification of the LinkedList and test if the iterator is valid and can continue to process.
    73  	 */
    74  	public function checkForComodification() 
    75  	{
    76  	    if ( this._list.getModCount() != this._expectedModCount )
    77  	    {
    78  			throw new ConcurrentModificationError(this + " check for comodification failed with a LinkedList." ) ;
    79  		}
    80      }
    81  
    82  	/**
    83  	 * Returns {@code true} if the iteration has more elements.
    84  	 * @return {@code true} if the iteration has more elements.
    85  	 */	
    86  	public function hasNext():Boolean 
    87  	{
    88  		return this._nextIndex != this._list.size() ;
    89  	}
    90  
    91  	/**
    92  	 * Checks to see if there is a previous element that can be iterated to.
    93  	 * <p><b>Example :</b></p>
    94  	 * {@code
    95  	 * var list:LinkedList = new LinkedList() ;
    96  	 * list.insert("item1") ;
    97  	 * list.insert("item2") ;
    98  	 * list.insert("item3") ;
    99  	 * var it:ListIterator = list.listIterator(2) ;
   100  	 * trace( it.hasPrevious() ) ;
   101  	 * }
   102  	 */
   103  	public function hasPrevious():Boolean 
   104  	{
   105  		 return this._nextIndex != 0 ;
   106  	}
   107  
   108  	/**
   109  	 * Inserts the specified element into the list (optional operation).
   110  	 * <p><b>Example :</b></p>
   111  	 * {@code
   112  	 * var list:LinkedList = new LinkedList() ;
   113  	 * list.insert("item1") ;
   114  	 * list.insert("item2") ;
   115  	 * list.insert("item3") ;
   116  	 * 
   117  	 * var it:ListIterator = list.listIterator(1) ;
   118  	 * it.insert("item0") ;
   119  	 * 
   120  	 * trace(list) ;
   121  	 * }
   122  	 * <p><b>Attention :</b> Dont' use this method in a loop with hasPrevious() and previous() method.</p> 
   123  	 */
   124  	public function insert( o ):Void 
   125  	{
   126  	    this.checkForComodification();
   127  	    this._lastReturned = this._list.getHeader() ;
   128  	    this._list._insertBefore(o, this._next) ;
   129  	    this._nextIndex ++ ;
   130  	    this._expectedModCount ++ ;
   131  	}
   132  
   133  	/**
   134  	 * Unsupported operation in this iterator, uses nextIndex() and previousIndex() method.
   135  	 * @throws UnsupportedOperation if the user call this method, this method is unsupported.
   136  	 */
   137  	public function key() 
   138  	{
   139  		throw new UnsupportedOperation(this + " 'key' method is unsupported.") ;
   140  		return null ;
   141  	}
   142  
   143  	/**
   144  	 * Returns the next element in the iteration.
   145  	 * @return the next element in the iteration.
   146  	 */
   147  	public function next() 
   148  	{
   149  		this.checkForComodification();
   150  		
   151  	    if ( this._nextIndex == this._list.size() )
   152  	    {
   153  			throw new NoSuchElementError(this + " 'next' method failed.");
   154  	    }
   155  
   156  	    this._lastReturned = this._next ;
   157  	   this. _next = this._next.next ;
   158  	    this._nextIndex ++ ;
   159  	    
   160  	    return this._lastReturned.element ;
   161  	}
   162  
   163  	/**
   164  	 * Returns the index of the element that would be returned by a subsequent call to next.
   165  	 * @return the index of the element that would be returned by a subsequent call to next.
   166  	 */
   167  	public function nextIndex() : Number 
   168  	{
   169  		return this._nextIndex ; 
   170  	}
   171  
   172  	/**
   173  	 * Returns the previous element in the collection.
   174  	 * @return the previous element in the collection.
   175  	 */
   176  	public function previous() 
   177  	{
   178  	    if (this._nextIndex == 0)
   179  	    {
   180  			throw new NoSuchElementError(this + " 'previous' method failed.");
   181  	    }
   182  
   183  	   	this._lastReturned = this._next = this._next.previous ;
   184  	    this._nextIndex-- ;
   185  	    this.checkForComodification();
   186  	    return this._lastReturned.element ;
   187  	}
   188  
   189  	/**
   190  	 * Returns the index of the element that would be returned by a subsequent call to previous.
   191  	 * @return the index of the element that would be returned by a subsequent call to previous.
   192  	 */
   193  	public function previousIndex() : Number 
   194  	{
   195  		return this._nextIndex - 1 ;
   196  	}
   197  
   198  	/**
   199  	 * Removes from the underlying collection the last element returned by the iterator (optional operation).
   200  	 */
   201  	public function remove() 
   202  	{
   203  		this.checkForComodification() ;
   204          var lastNext:LinkedListEntry = this._lastReturned.next;
   205  		try 
   206  		{
   207             this._list._removeEntry( this._lastReturned ) ;
   208          }
   209          catch ( e:NoSuchElementError ) 
   210          {
   211  			throw new IllegalStateError( this + " 'remove' method failed.") ;
   212  		}
   213  	    if ( this._next == this._lastReturned)
   214  	    {
   215  			this._next = lastNext ;
   216  	    }
   217  		else
   218  		{
   219  			this._nextIndex-- ;
   220  		}
   221  	    this._lastReturned = this._list.getHeader() ;
   222  	    this._expectedModCount++;
   223  	}
   224  
   225  	/**
   226  	 * Reset the internal pointer of the iterator (optional operation).
   227  	 */
   228  	public function reset() : Void 
   229  	{
   230  		seek(0) ;
   231  	}
   232  
   233  	/**
   234  	 * Change the position of the internal pointer of the iterator (optional operation).
   235  	 */
   236  	public function seek(n:Number) : Void 
   237  	{
   238  
   239  		this._lastReturned = this._list.getHeader() ;
   240  		
   241  		this._expectedModCount = this._list.getModCount() ;
   242  
   243  		var size:Number = _list.size() ;
   244  		
   245  		if (n < 0 || n > size)
   246  	    {
   247  	    	throw new IndexOutOfBoundsError(this + " seek failed, index:" + n + ", size:" + size + "." ) ;
   248  	    }
   249  		
   250  		if ( n < ( size >> 1 ) ) 
   251  	    {
   252  			
   253  			_next = _list.getHeader().next ;
   254  			
   255  			for ( _nextIndex = 0 ; _nextIndex < n ; _nextIndex ++ )
   256  			{
   257  			    _next = _next.next ;
   258  			}
   259  	    } 
   260  	    else 
   261  	    {
   262  			_next = _list.getHeader()  ;
   263  			for ( _nextIndex = size ; _nextIndex > n ; _nextIndex-- )
   264  			{
   265  				_next = _next.previous ;
   266  			}
   267  		}
   268  	}
   269  
   270  	/**
   271  	 * Replaces the last element returned by next or previous with the specified element (optional operation).
   272  	 */
   273  	public function set( o ):Void 
   274  	{
   275  		
   276  		if (_lastReturned == _list.getHeader())
   277  		{
   278  			throw new IllegalStateError(this + " 'set' method failed.");
   279  		}
   280  	    
   281  	    this.checkForComodification();
   282  	    this._lastReturned.element = o ;
   283  	    
   284  	}
   285  
   286  	/**
   287  	 * The last list entry returned.
   288  	 */
   289  	private var _lastReturned:LinkedListEntry = null ;
   290  	
   291  	/**
   292  	 * The list reference of this iterator.
   293  	 */
   294  	private var _list:LinkedList = null ;
   295  	
   296  	/**
   297  	 * The next entry.
   298  	 */
   299  	private var _next:LinkedListEntry = null ;
   300  	
   301  	/**
   302  	 * The next index in the iterator.
   303  	 */
   304  	private var _nextIndex:Number = null ;
   305  	
   306  	/**
   307  	 * The internal expected mod count value.
   308  	 */
   309  	private var _expectedModCount:Number = null ;
   310  
   311  }
