    24  import vegas.data.iterator.ArrayIterator;
    25  
    26  /**
    27   * Converts an array to an iterator but this iterator return the value of a specific field if the array is an array of objects.
    28   * <p><b>Example :</b></p>
    29   * {@code
    30   * import vegas.data.iterator.ArrayFieldIterator ;
    31   * 
    32   * var ar:Array = [
    33   *     { label : "item1", date : new Date(2005, 10, 12) } ,
    34   *     { label : "item2", date : new Date(2004, 2, 22) } ,
    35   *     { label : "item3", date : new Date(2005, 4, 3) }
    36   * ] ;
    37   * 
    38   * trace (" --- browse 'label' field") ;
    39   * 
    40   * var it:ArrayFieldIterator = new ArrayFieldIterator(ar, "label") ;
    41   * while (it.hasNext()) 
    42   * {
    43   *     trace (it.next() + " : " + it.key()) ;
    44   * }
    45   * 
    46   * trace(" --- browse 'date' field") ;
    47   * 
    48   * var it:ArrayFieldIterator = new ArrayFieldIterator(ar, "date") ;
    49   * while (it.hasNext())
    50   * {
    51   *     trace (it.next() + " : " + it.key()) ;
    52   * }
    53   * }
    54   * @author eKameleon
    55   */
    56  class vegas.data.iterator.ArrayFieldIterator extends ArrayIterator 
    57  {
    58  
    59  	/**
    60  	 * Creates a new ArrayFieldIterator instance.
    61  	 */
    62  	public function ArrayFieldIterator(p_a:Array, p_fieldName:String) 
    63  	{
    64  		super(p_a) ;
    65  		fieldName = p_fieldName ;
    66  	}
    67  	
    68  	/**
    69  	 * The field used in the next method to return the next value in the array.
    70  	 */	
    71  	public var fieldName:String ;
    72  	
    73  	/**
    74  	 * Returns the next field element in the iteration.
    75  	 * @return the next field element in the iteration.
    76  	 */
    77  	public function next() 
    78  	{
    79  		var o = _a[++_k] ;
    80  		return (fieldName) ? o[fieldName] : o ;
    81  	}
    82  	
    83  }
