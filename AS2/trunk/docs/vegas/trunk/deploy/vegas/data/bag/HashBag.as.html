    24  import vegas.data.bag.AbstractBag;
    25  import vegas.data.Collection;
    26  import vegas.data.map.HashMap;
    27  import vegas.util.serialize.Serializer;
    28  
    29  /**
    30   * Implements Bag, using a HashMap to provide the data storage. This is the standard implementation of a bag.
    31   * <p><b>Example : </b>
    32   * {@code
    33   *	import vegas.data.Bag ;
    34   *	import vegas.data.bag.HashBag ;
    35   *	import vegas.data.Collection ;
    36   *	import vegas.data.collections.SimpleCollection ;
    37   *	import vegas.data.Set ;
    38   *	
    39   *	var c1:Collection = new SimpleCollection( ["item1", "item1", "item3"] ) ;
    40   *	var c2:Collection = new SimpleCollection( ["item1", "item2", "item3", "item4", "item5"] ) ;
    41   *	
    42   *	trace ("c1 collection : " + c1) ;
    43   *	trace ("c2 collection : " + c2) ;
    44   *	
    45   *	trace ("---- create a new HashBag") ;
    46   *	var bag:Bag = new HashBag ;
    47   *		
    48   *	trace ("---- insert") ;
    49   *	trace (" + bag insertAll c1 : " + bag.insertAll(c1)) ;
    50   *	trace (" + bag insertAll c2 : " + bag.insertAll(c2)) ;
    51   *	trace (" > bag : " + bag) ;
    52   *	trace (" > bag.touSource : " + bag.toSource()) ;
    53   *	
    54   *	trace ("---- contains") ;
    55   *	trace (" > bag containsAll c2 : " + bag.containsAll(c2)) ;
    56   *	
    57   *	trace ("---- insert") ;
    58   *	trace (" + bag insert item2 : " + bag.insert("item2")) ;
    59   *	trace (" > bag : " + bag) ;
    60   *	trace (" + bag insertCopies 2xitem2 : " + bag.insertCopies("item2", 2)) ;
    61   *	trace (" > bag : " + bag) ;
    62   *	
    63   *	trace ("---- remove") ;
    64   *	trace (" > bag removeCopies 1 x item2 : " + bag.removeCopies("item2", 1)) ;
    65   *	
    66   *	trace ("---- size") ;
    67   *	trace (" - bag getCount item2 : " + bag.getCount("item2")) ;
    68   *	trace (" > bag size : " + bag.size()) ;
    69   *	
    70   *	trace ("---- retain") ;
    71   *	trace (" > bag : " + bag) ;
    72   *	trace (" > bag retainAll c1 : " + bag.retainAll(c1)) ;
    73   *	trace (" > bag : " + bag) ;
    74   *		
    75   *	trace ("----") ;
    76   *	var s:Set = bag.uniqueSet() ;
    77   *	trace("bag uniqueSet : " + s) ;
    78   *}
    79   *</p>
    80   *@author eKameleon
    81   */
    82  class vegas.data.bag.HashBag extends AbstractBag 
    83  {
    84  	
    85  	/**
    86  	 * Creates a new HashBag instance.
    87  	 * @param c a {@code Collection} to constructs a bag containing all the members of the given collection.
    88  	 */
    89  	public function HashBag( c:Collection ) 
    90  	{
    91  		super(new HashMap()) ;
    92  		if (c) 
    93  		{
    94  			insertAll(c) ;
    95  		}
    96  	}
    97  	
    98  	/**
    99  	 * Returns the shallow copy of this bag.
   100  	 * @return the shallow copy of this bag.
   101  	 */
   102  	public function clone() 
   103  	{
   104  		return new HashBag(_map.clone()) ;
   105  	}
   106  
   107  	/**
   108  	 * Returns a Eden representation of the object.
   109  	 * @return a string representing the source code of the object.
   110  	 */
   111  	/*override*/ public function toSource(indent:Number, indentor:String):String 
   112  	{
   113  		return Serializer.getSourceOf(this, [Serializer.toSource(_extractList())]) ;
   114  	}
   115  	
   116  }
