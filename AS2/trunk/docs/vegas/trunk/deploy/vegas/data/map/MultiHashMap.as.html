    36  /**
    37   * The default implementation of the {@code MultiMap} interface.
    38   * <p><b>Example :</b></p>
    39   * {@code
    40   * import vegas.data.Collection ;
    41   * import vegas.data.iterator.Iterator ;
    42   * import vegas.data.map.HashMap ;
    43   * import vegas.data.Map ;
    44   * import vegas.data.MultiMap ;
    45   * import vegas.data.map.MultiHashMap ;
    46   * 
    47   * var map1:HashMap = new HashMap() ;
    48   * map1.put("key1", "valueD1") ;
    49   * map1.put("key2", "valueD2") ;
    50   * trace ("> map1 : " + map1) ;
    51   * trace ("> map1 containsKey 'key1' : " + map1.containsKey("key1")) ;
    52   * 
    53   * trace ("--- use a map argument in constructor") ;
    54   * 
    55   * var map:MultiHashMap = new MultiHashMap(map1) ;
    56   * map.put("key1", "valueA1") ;
    57   * map.put("key1", "valueA2") ;
    58   * map.put("key1", "valueA3") ;
    59   * map.put("key2", "valueA2") ;
    60   * map.put("key2", "valueB1") ;
    61   * map.put("key2", "valueB2") ;
    62   * map.put("key3", "valueC1") ;
    63   * map.put("key3", "valueC2") ;
    64   * 
    65   * trace ("init map : " + map) ;
    66   * trace ("\r--- toSource MultiMap") ;
    67   * 
    68   * trace("map toSource : " + map.toSource()) ;
    69   * 
    70   * trace ("\r--- put values in MultiMap") ;
    71   * 
    72   * trace ("key1 >> " + map.get("key1")) ;
    73   * 
    74   * trace ("key2 >> " + map.get("key2")) ;
    75   * 
    76   * trace ("key3 >> " + map.get("key3")) ;
    77   * 
    78   * trace ("\r--- toString MultiMap") ;
    79   * 
    80   * trace (map) ;
    81   * 
    82   * map.remove("key1", "valueA2") ;
    83   * 
    84   * trace ("\r--- remove a value in key1 >> " + map.get("key1")) ;
    85   * 
    86   * trace ("\r--- use iterator") ;
    87   * 
    88   * var it:Iterator = map.iterator() ;
    89   * while(it.hasNext()) 
    90   * {
    91   *     trace ("\t :: " + it.next()) ;
    92   * }
    93   * 
    94   * trace ("\r--- use a key iterator : key1") ;
    95   * var it:Iterator = map.iterator("key1") ;
    96   * while(it.hasNext()) 
    97   * {
    98   *     trace ("\t :: " + it.next()) ;
    99   * }
   100   * 
   101   * trace ("\r--- putCollection key2 in key1") ;
   102   * map.putCollection("key1", map.get("key2")) ;
   103   * trace ("key1 >> " + map.get("key1")) ;
   104   * 
   105   * trace ("\r--- different size") ;
   106   * trace ("map size : " + map.size()) ;
   107   * trace ("map totalSize : " + map.totalSize()) ;
   108   * 
   109   * trace ("\r--- clone MultiMap") ;
   110   * var clone:MultiMap = map.clone() ;
   111   * clone.remove("key1") ;
   112   * 
   113   * trace("clone : " + clone) ;
   114   * trace ("clone size : " + clone.totalSize()) ;
   115   * trace ("map size : " + map.totalSize()) ;
   116   * 
   117   * trace ("\r--- valueIterator") ;
   118   * var it:Iterator = map.valueIterator() ;
   119   * while(it.hasNext()) 
   120   * {
   121   *      trace("\t> " + it.next()) ;
   122   * }
   123   * 
   124   * } 
   125   * @author eKameleon
   126   * @see MultiMap
   127   */
   128  class vegas.data.map.MultiHashMap extends HashMap implements Iterable, MultiMap, IFormattable

   131  	/**
   132  	 * Creates a new MultiHashMap instance.
   133  	 */
   134  	public function MultiHashMap() 
   135  	{
   136  		_map = new HashMap() ;
   137  		if (arguments[0] instanceof Map) 
   138  		{
   139  			var m:Map = arguments[0].clone() ;
   140  			if (m) putAll(m) ;
   141  		}
   142  	}

   144  	/**
   145  	 * Removes all elements in this map.
   146  	 */
   147  	public function clear():Void 
   148  	{
   149  		_map.clear() ;
   150  	}

   152  	/**
   153  	 * Creates a new instance of the map value Collection container.
   154  	 * This method can be overridden to use your own collection type.
   155  	 * @return a new instance of the map value Collection container.
   156  	 */
   157  	public function createCollection():Collection 
   158  	{
   159  		return new SimpleCollection() ;	
   160  	}

   162  	/**
   163  	 * Returns a shallow copy of this object.
   164  	 * @return a shallow copy of this object.
   165  	 */
   166  	public function clone() 
   167  	{
   168  		var m:MultiHashMap = new MultiHashMap() ;
   169  		var vItr:Iterator = valueIterator() ;
   170  		var kItr:Iterator = keyIterator() ;
   171  		while (kItr.hasNext()) 
   172  		{
   173  			var key = kItr.next() ;
   174  			var value = vItr.next() ;
   175  			m.putCollection(key, value) ;
   176  		}
   177  		return m ;
   178  	}

   180  	/**
   181  	 * Checks whether the map contains the key specified.
   182  	 */
   183  	public function containsKey( key ):Boolean 
   184  	{
   185  		return _map.containsKey( key ) ;
   186  	}

   188  	/**
   189  	 * Checks whether the map contains the value specified or at the specified key contains the value.
   190  	 * <p><b>Example :</b></p>
   191  	 * {@code
   192  	 * var b:Boolean = map.containsValue(key, value) ;
   193  	 * var b:Boolean = map.containsValue(value) ;
   194  	 * }
   195  	 * @return {@code true} if the map contains the specified value. 
   196  	 */
   197  	public function containsValue( value ):Boolean 
   198  	{
   199  		var len:Number = arguments.length ;
   200  		if (len == 1) 
   201  		{
   202  			var it:Iterator = _map.iterator() ;
   203  			while (it.hasNext()) 
   204  			{
   205  				var cur = it.next() ;
   206  				if (cur.contains(value)) return true;
   207  			}
   208  		} 
   209  		else if (len == 2) 
   210  		{
   211  			return ( get(arguments[0] ).contains(arguments[1]) == true);
   212  		}
   213  		return false ;
   214  	}

   216  	/**
   217  	 * Gets the collection mapped to the specified key. This method is a convenience method to typecast the result of get(key).
   218  	 * @return the collection mapped to the specified key
   219  	 */
   220  	public function get( key ) /*Collection*/ 
   221  	{
   222  		return _map.get(key) || null ;
   223  	}

   225  	/**
   226  	 * Checks whether the map contains the key specified .

   233  	/**
   234  	 * This returns an array containing the combination of values from all keys.
   235  	 * @return returns an array containing the combination of values from all keys.
   236  	 */
   237  	public function getValues():Array 
   238  	{
   239  		var result:Array = [] ;
   240  		var values = this._map.getValues() ;
   241  		var l:Number = values.length ;
   242  		for (var i:Number = 0 ; i<l ; i++) 
   243  		{
   244  			result = result.concat( values[i].toArray() ) ;
   245  		}
   246  		return result ;
   247  	}

   254  	public function isEmpty():Boolean

   259  	/**
   260  	 * Gets an iterator for the collection mapped to the specified key.
   261  	 */
   262  	public function iterator( /*key*/ ):Iterator 
   263  	{
   264  		var key = arguments[0] ;
   265  		if (key) 
   266  		{
   267  			return _map.get(key).iterator() ;
   268  		}
   269  		else 
   270  		{
   271  			return _map.iterator() ;
   272  		}
   273  	}

   281  	}

   283  	/**
   284  	 * Adds the value to the collection associated with the specified key.
   285  	 */
   286  	public function put(key, value) 
   287  	{
   288  		if (!containsKey(key)) 
   289  		{
   290  			_map.put(key , createCollection()) ;
   291  		}
   292  		var b:Boolean = _map.get(key).insert(value) ;
   293  		return b ? value : null ;
   294  	}
   295  
   296  	/**
   297  	 * Override superclass to ensure that MultiMap instances are correctly handled.
   298  	 */
   299  	public function putAll(map:Map):Void 
   300  	{
   301  		var it:Iterator = map.iterator() ;
   302  		while (it.hasNext()) 
   303  		{
   304  			var value = it.next() ;
   305  			var key = it.key() ;
   306  			if (value instanceof Collection) 
   307  			{
   308  				putCollection(key, value) ;
   309  			}
   310  			else 
   311  			{
   312  				put(key, value) ;
   313  			}
   314  		}
   315  	}
   316  	
   317  	/**
   318  	 * Adds a collection of values to the collection associated with the specified key.
   319  	 */
   320  	public function putCollection(key, c:Collection):Void 
   321  	{
   322  		if (!containsKey(key)) 
   323  		{
   324  			_map.put(key , createCollection()) ;
   325  		}
   326  		_map.get(key).insertAll(c) ;
   327  	}

   329  	/**
   330  	 * Removes a specific value from map.
   331  	 * @param key the key to remove in the map
   332  	 * @param value (optional) if this value is defined removes a specific value from map else removes all values associated with the specified key.
   333  	 * @return the removed value.
   334  	 */
   335  	public function remove(key /*, value*/ ) 
   336  	{
   337  		var value = arguments[1] ;
   338  		if (key && value) 
   339  		{
   340  			var c:Collection = Collection( this.get(key)) ;
   341  			if (c != null)
   342  			{
   343  				var b:Boolean = c.remove(value) ;
   344  				if (c.isEmpty())
   345  				{
   346  					_map.remove(key) ;	
   347  				}
   348  				return (b) ? value : null ;
   349  			}
   350  			else
   351  			{
   352  				return null ;	
   353  			}
   354  		}
   355  		else 
   356  		{
   357  			return _map.remove(key) ;
   358  		}
   359  	}

   389  	/**
   390  	 * Gets the total size of the map by counting all the values.
   391  	 */
   392  	public function totalSize():Number 
   393  	{
   394  		var result:Number = 0 ;
   395  		var it:Iterator = _map.iterator() ;
   396  		while (it.hasNext()) 
   397  		{
   398  			result += (it.next().size() || 0) ;
   399  		}
   400  		return result ;
   401  	}

   409  		return new SimpleCollection(ar) ;
   410  	}

   413  	 * Returns the iterator of all values in the MultiHashMap.

   415  	public function valueIterator():Iterator 
   416  	{
   417  		return new ArrayIterator(_map.getValues()) ;
