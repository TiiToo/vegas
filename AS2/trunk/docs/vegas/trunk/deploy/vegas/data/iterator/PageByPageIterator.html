Class vegas.data.iterator.PageByPageIterator

OrderedIterator

Description

An iterator page by page over an array who return an new array of elements. If the step size value is 1 the next and previous methods returns the single value element in the data array.

Example :

 import vegas.data.iterator.PageByPageIterator
 
 var ar:Array = [1, 2, 3, 4, 5, 6, 7, 8] ;
 
 var it:PageByPageIterator = new PageByPageIterator(1, ar) ;

 var next:Function = function()
 {
     if (!it.hasNext())
     {
         it.reset() ;
     }
     var next = it.next()
     trace( "> " + next + " : " + it.currentPage() ) ;
 }
 
 var prev:Function = function()
 {
     if ( !it.hasPrevious())
     {
         it.lastPage() ;
     }
     var prev = it.previous() ;
     trace( "> " + prev + " : " + it.currentPage() ) ;
 }
 
 Key.addListener(this) ;
 
 onKeyDown = function()
 {
     var code:Number = Key.getCode() ;
     switch (code)
     {
         case Key.LEFT :
         {
             prev() ;
             break ;
         }
         case Key.RIGHT :
         {
             next() ;
             break ;
         }
     }
 }
