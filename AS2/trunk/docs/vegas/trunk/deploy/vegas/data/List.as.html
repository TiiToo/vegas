    24  import vegas.data.Collection;
    25  import vegas.data.iterator.ListIterator;
    26  
    27  /**
    28   * An ordered collection (also known as a sequence). The user of this interface has precise control over where in the list each element is inserted. The user can access elements by their integer index (position in the list), and search for elements in the list.
    29   * @author eKameleon
    30   */
    31  interface vegas.data.List extends Collection 
    32  {
    33  	
    34  	/**
    35  	 * Returns {@code true} if this list contains all of the elements of the specified collection.
    36  	 */
    37  	function containsAll(c:Collection):Boolean ;
    38  
    39  	/**
    40  	 * Indicates whether some other object is "equal to" this one.
    41  	 */
    42  	function equals(o):Boolean ;
    43  
    44  	/**
    45  	 * Returns the index in this list of the first occurrence of the specified element, or -1 if this list does not contain this element.
    46  	 */
    47  	function indexOf(o):Number ;
    48  	
    49  	/**
    50  	 * Appends all of the elements in the specified collection to the end of this list, in the order that they are returned by the specified collection's iterator (optional operation).
    51  	 */
    52  	function insertAll(c:Collection):Boolean ;
    53  	
    54  	/**
    55  	 * Inserts all of the elements in the specified collection into this list at the specified position (optional operation).
    56  	 */
    57  	function insertAllAt(id:Number, c:Collection):Boolean ;
    58  
    59  	/**
    60  	 * Inserts the specified element at the specified position in this list (optional operation).
    61  	 */
    62  	function insertAt(id:Number, o):Void ;
    63  
    64  	/**
    65  	 * Returns the index in this list of the last occurrence of the specified element, or -1 if this list does not contain this element.
    66  	 */
    67  	function lastIndexOf(o):Number ;
    68  	
    69  	/**
    70  	 * Returns a list iterator of the elements in this list (in proper sequence).
    71  	 */
    72  	function listIterator():ListIterator ;
    73  
    74  	/**
    75  	 * Removes from this list all the elements that are contained in the specified collection (optional operation).
    76  	 */
    77  	function removeAll(c:Collection):Boolean ;
    78  
    79  	/**
    80  	 * Removes the element at the specified position in this list (optional operation).
    81  	 */
    82  	function removeAt(id:Number) ;
    83  
    84  	/**
    85  	 * Removes from this list all the elements that are contained between the specific {@code from} and the specific {@code to} position in this list (optional operation).
    86  	 */
    87  	function removeRange(from:Number, to:Number):Void ;
    88  	
    89  	/**
    90  	 * Removes from this list all the elements that are contained between the specific {@code id} position and the end of this list (optional operation).
    91  	 */
    92  	function removesAt(id:Number, len:Number) ;
    93  
    94  	/**
    95  	 * Retains only the elements in this list that are contained in the specified collection (optional operation).
    96  	 */
    97  	function retainAll(c:Collection):Boolean ;
    98  
    99  	/**
   100  	 * Replaces the element at the specified position in this list with the specified element (optional operation).
   101  	 */
   102  	function setAt(id:Number, o) ;
   103  	
   104  	/**
   105  	 * Returns a view of the portion of this list between the specified fromIndex, inclusive, and toIndex, exclusive.
   106  	 */
   107  	function subList(fromIndex:Number, toIndex:Number):List ;
   108  
   109  }
