    36  /**
    37   * The CircularQueue class allows for storing objects in a circular queue of a predefined size.
    38   * <p><b>Example :</b></p>
    39   * {@code
    40   * import vegas.data.iterator.Iterator ;
    41   * import vegas.data.queue.CircularQueue ;
    42   * 
    43   * var q:CircularQueue = new CircularQueue(5) ;
    44   * 
    45   * trace ("maxSize : " + q.maxSize()) ;
    46   * trace ("enqueue item1 : " + q.enqueue ("item1")) ;
    47   * trace ("enqueue item2 : " + q.enqueue ("item2")) ;
    48   * trace ("enqueue item3 : " + q.enqueue ("item3")) ;
    49   * trace ("enqueue item4 : " + q.enqueue ("item4")) ;
    50   * trace ("enqueue item5 : " + q.enqueue ("item5")) ;
    51   * trace ("enqueue item6 : " + q.enqueue ("item6")) ;
    52   * 
    53   * trace ("element : " + q.element()) ;
    54   * trace ("dequeue : " + q.dequeue()) ;
    55   * trace ("element : " + q.element()) ;
    56   * trace ("size : " + q.size()) ;
    57   * trace ("isFull : " + q.isFull()) ;
    58   * trace ("array : " + q.toArray()) ;
    59   * 
    60   * trace("") ;
    61   * 
    62   * trace ("queue : " + q) ;
    63   * 
    64   * trace("") ;
    65   * 
    66   * trace ("dequeue : " + q.dequeue()) ;
    67   * trace ("enqueue item6 : " + q.enqueue("item6")) ;
    68   * trace ("enqueue item7 : " + q.enqueue("item7")) ;
    69   * trace ("peek : " + q.peek()) ;
    70   * trace ("size : " + q.size()) ;
    71   * trace ("isFull : " + q.isFull()) ;
    72   * 
    73   * trace("") ;
    74   * 
    75   * trace ("q : " + q) ;
    76   * 
    77   * trace ("------- clone") ;
    78   * 
    79   * var clone:CircularQueue = q.clone() ;
    80   * trace ("dequeue clone : " + clone.dequeue()) ;
    81   * trace ("enqueue clone item8 : " + clone.enqueue("item8")) ;
    82   * trace ("original queue : " + q) ;
    83   * trace ("clone queue : " + clone) ;
    84   * trace ("clone iterator :") ;
    85   * var i:Iterator = clone.iterator() ;
    86   * while (i.hasNext()) 
    87   * {
    88   *     trace ("\t+ " + i.next()) ;
    89   * }
    90   * trace("clone.toSource : " + clone.toSource()) ;
    91   * 
    92   * }
    93   * @author eKameleon
    94   */
    95  class vegas.data.queue.CircularQueue extends CoreObject implements BoundedQueue, ICloneable, Collection, ISerializable {
    96  
    97  	/**
    98  	 * Creates a new CircularQueue instance.
    99  	 * @param qSize the max number of element in the queue
   100  	 * @param elaments an array with elements to enqueue in the current stack.
   101  	 */
   102  	public function CircularQueue( qSize:Number , elements:Array) {
   103  		_qSize = (qSize || CircularQueue.DEFAULT_SIZE) + 1 ;
   104  		clear() ;
   105  		var l:Number = elements.length ;
   106  		if (l > 0) for (var i:Number = 0 ; i<l ; i++) 
   107  		{
   108  			enqueue(elements[i]) ;
   109  		}
   110  	}
   111  	
   112  	/**
   113  	 * The default numbers of elements in the queue.
   114  	 */
   115  	static public var DEFAULT_SIZE:Number = Number.MAX_VALUE ;
   116  
   117  	/**
   118  	 * Clear all elements in the queue.
   119  	 */
   120  	public function clear():Void {
   121  		_queue = new Array(_qSize) ;
   122  		_count = 0 ;
   123  		_rear = 0 ;
   124  		_front = 0 ;
   125  	}
   126  
   127  	/**
   128  	 * Returns a shallow copy of the queue.
   129  	 */
   130  	public function clone() 
   131  	{
   132  		var s:Number = _qSize - 1 ;
   133  		var a:Array = toArray() ;
   134  		return new CircularQueue(s , a) ;
   135  	}
   136  
   137  	/**
   138  	 * Returns {@code true} if the queue contains the object passed in argument.
   139  	 */
   140  	public function contains(o):Boolean {
   141  		var l:Number = _queue.length ;
   142  		for (var i:Number = 0 ; i<l ; i++) if (_queue[i] == o) return true ;
   143  		return false ; 
   144  	}
   145  	
   146  	/**
   147  	 * Retreives the first element in the queue object, return a boolean.
   148  	 * @return {@code true} if the first element in the queue is dequeue.
   149  	 */
   150  	public function dequeue():Boolean 
   151  	{
   152  		return poll() != null  ;
   153  	}
   154  
   155  	/**
   156  	 * Returns the value of the first element in the queue.
   157  	 * @return the value of the first element in the queue.
   158  	 */
   159  	public function element() 
   160  	{
   161  		return _queue[_front] ;
   162  	}	
   163  
   164  	/**
   165  	 * Enqueue a new element in the queue if the que is not full, return a boolean.
   166  	 */
   167  	public function enqueue(o):Boolean 
   168  	{
   169  		var next:Number = _rear + 1 ;
   170  		if ( (next == _front) || ( ( next == _qSize) && (_front == 0) )) 
   171  		{
   172  			return false ;
   173  		}
   174  		else 
   175  		{
   176  			_queue[_rear++] = o ;
   177  			_count ++ ;
   178  			if (_rear == _qSize) 
   179  			{
   180  				_rear = 0 ;
   181  			}
   182  		}
   183  		return true ;
   184  	}
   185  
   186  	/**
   187  	 * Returns an element in the queue with the id passed in argument. This method is unsupported in a CircularQueue.
   188  	 * @throws UnsupportedOperation
   189  	 */
   190  	public function get(id:Number) 
   191  	{ 
   192  		throw new UnsupportedOperation(this + " 'get' method us unsupported in a CircularQueue") ;
   193  	}
   194  
   195  	/**
   196  	 * Insert an element in the queue. This method is unsupported in a CircularQueue.
   197  	 * @throws UnsupportedOperation
   198  	 */
   199  	public function insert(o):Boolean 
   200  	{
   201  		throw new UnsupportedOperation(this + " 'insert' method us unsupported in a CircularQueue") ;
   202  		return false ;
   203  	}
   204  
   205  	/**
   206  	 * Returns {@code true} if the queue is empty.
   207  	 * @return {@code true} if the queue is empty.
   208  	 */
   209  	public function isEmpty():Boolean 
   210  	{
   211  		return _count == 0 ;
   212  	}
   213  
   214  	/**
   215  	 * Returns {@code true} if the queue is full.
   216  	 * @return {@code true} if the queue is full.
   217  	 */
   218  	public function isFull():Boolean 
   219  	{
   220  		return _count == maxSize() ;
   221  	}
   222  	
   223  	/**
   224  	 * Returns the iterator of the queue.
   225  	 * @return the iterator of the queue.
   226  	 * @see {@code vegas.data.iterator.ProtectedIterator}
   227  	 */
   228  	public function iterator():Iterator 
   229  	{
   230  		return (new ProtectedIterator(new ArrayIterator(toArray()))) ;
   231  	}
   232  
   233  	/**
   234  	 * Returns the max number of occurrences in the given queue.
   235  	 * @return the max number of occurrences in the given queue.
   236  	 */
   237  	public function maxSize():Number 
   238  	{
   239  		return _qSize -1 ;
   240  	}
   241  	
   242  	/**
   243  	 * Returns the value of the first element in the queue or {@code null} if the queue is empty.
   244  	 * @return the value of the first element in the queue or {@code null} if the queue is empty.
   245  	 */
   246  	public function peek() 
   247  	{
   248  		return isEmpty() ? null : _queue[_front] ;
   249  	}
   250  	
   251  	/**
   252  	 * Returns the value of the first element in the queue and remove this value in the queue.
   253  	 * @return the value of the first element in the queue and remove this value in the queue.
   254  	 */
   255  	public function poll() 
   256  	{
   257  		if (_front == _qSize) 
   258  		{
   259  			_front = 0 ; // loop back
   260  		}
   261          if (_front == _rear) 
   262          {
   263          	return null;  // queue is empty
   264          }
   265          else  
   266          {
   267              _count-- ;
   268              var mem = _queue[_front] ;
   269              _queue[_front] = undefined ;
   270              _front++ ;
   271              return mem ; // return mem object
   272          }
   273  	}	
   274  	
   275  	/**
   276  	 * Remove the passed argument value. This method is unsupported in a CircularQueue.
   277  	 * @throws UnsupportedOperation
   278  	 */
   279  	public function remove(o):Boolean 
   280  	{
   281  		throw new UnsupportedOperation(this + " 'remove' method us unsupported in a CircularQueue") ;
   282  		return false ;
   283  	}
   284  	
   285  	/**
   286  	 * Returns the number of elements in the CircularQueue.
   287  	 * @return the number of elements in the CircularQueue.
   288  	 */
   289  	public function size():Number 
   290  	{
   291  		return _count ;
   292  	}
   293  	
   294  	/**
   295  	 * Returns the array representation of the CircularQueue.
   296  	 * @return the array representation of the CircularQueue.
   297  	 */
   298  	public function toArray():Array 
   299  	{
   300  		if (_count == 0) {
   301  			return new Array() ;
   302  		} else {
   303  			var r:Array = new Array(_count) ;
   304  			var i:Number = (_front == _qSize) ? 0 : _front ;
   305  			var cpt:Number = 0 ;
   306  			while (cpt < _count) {
   307  				r[cpt++] = _queue[i++] ;
   308  				if (i == _qSize) i = 0 ;
   309  			}
   310  			return r ;
   311  		}
   312  	}
   313  
   314  	/**
   315  	 * Returns a Eden reprensation of the object.
   316  	 * @return a string representing the source code of the object.
   317  	 */
   318  	public function toSource(indent:Number, indentor:String):String 
   319  	{
   320  		var sourceA:String = Serializer.toSource(_qSize - 1) ;
   321  		var sourceB:String = Serializer.toSource(toArray()) ;
   322  		return Serializer.getSourceOf(this, [sourceA, sourceB]) ;
   323  	}
   324  	
   325  	/**
   326  	 * Returns the string representation of this instance.
   327  	 * @return the string representation of this instance
   328  	 */
   329  	public function toString():String 
   330  	{
   331  		return (new QueueFormat()).formatToString(this) ;
   332  	}
   333  
   334  	/**
   335  	 * The internal array.
   336  	 */
   337  	private var _queue:Array ;
   338  	
   339  	/**
   340  	 * The array index for the next object to be stored in the queue.
   341  	 */
   342      private var _rear:Number ;
   343      
   344      /**
   345       * The array index for the next object to be removed from the queue.
   346       */ 
   347  	private var _front:Number ;
   348  	
   349  	/**
   350  	 * The number of objects currently stored in the queue.
   351  	 */
   352  	private var _count:Number ;
   353  
   354  	/**
   355  	 * The number of objects in the array : queue size + 1
   356  	 */
   357      private var _qSize:Number ;
   358  
   359  }
