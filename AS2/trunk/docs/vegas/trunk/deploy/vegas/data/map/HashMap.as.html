    24  import vegas.core.CoreObject;
    25  import vegas.core.ICloneable;
    26  import vegas.data.iterator.ArrayIterator;
    27  import vegas.data.iterator.Iterable;
    28  import vegas.data.iterator.Iterator;
    29  import vegas.data.iterator.MapIterator;
    30  import vegas.data.Map;
    31  import vegas.data.map.MapFormat;

    34  /**
    35   * Hash table based implementation of the Map interface. 
    36   * <p><b>Attention :</b> this class is the ArrayMap class in the AS3 version of VEGAS.</p>
    37   * @author eKameleon
    38   */
    39  class vegas.data.map.HashMap extends CoreObject implements ICloneable, Iterable, Map 
    40  {
    41  
    42  	/**
    43  	 * Creates a new HashMap instance.
    44  	 */
    45  	public function HashMap() 
    46  	{
    47  		var k:Array = arguments[0] ;
    48  		var v:Array = arguments[1] ;
    49  		var b:Boolean = (k.length > 0 && k.length == v.length) ;
    50  		_keys = b ? [].concat(k) : ((_keys != null) ? [].concat(_keys) : []) ;
    51  		_values = b ? [].concat(v) : ((_values != null) ? [].concat(_values) : [])  ;
    52  	}
    53  	
    54  	/**
    55  	 * Removes all mappings from this map.
    56  	 */
    57  	public function clear():Void 
    58  	{
    59  		_keys = [] ;
    60  		_values = [] ;
    61  	}
    62  	
    63  	/**
    64  	 * Returns a shallow copy of this HashMap instance: the keys and values themselves are not cloned.
    65  	 * @return a shallow copy of this HashMap instance: the keys and values themselves are not cloned.
    66  	 */
    67  	public function clone() 
    68  	{
    69  		var m:HashMap = new HashMap() ;
    70  		m.putAll(this) ;
    71  		return m ;
    72  	}
    73  	
    74  	/**
    75  	 * Returns {@code true} if this map contains a mapping for the specified key.
    76  	 * @return {@code true} if this map contains a mapping for the specified key.
    77  	 */
    78  	public function containsKey( key ):Boolean 
    79  	{
    80  		return indexOfKey(key) > -1 ;
    81  	}
    82  	
    83  	/**
    84  	 * Returns {@code true} if this map maps one or more keys to the specified value.
    85  	 * @return {@code true} if this map maps one or more keys to the specified value.
    86  	 */
    87  	public function containsValue( value ):Boolean 
    88  	{
    89  		return indexOfValue(value) > -1 ;
    90  	}
    91  
    92  	/**
    93  	 * Returns the value to which this map maps the specified key.
    94  	 * @return the value to which this map maps the specified key.
    95  	 */
    96  	public function get(key) 
    97  	{
    98  		return _values[indexOfKey(key)] ;
    99  	}
   100  
   101  	/**
   102  	 * Returns an array representation of all keys in the map.
   103  	 * @return an array representation of all keys in the map.
   104  	 */
   105  	public function getKeys():Array 
   106  	{
   107  		return _keys.slice() ;
   108  	}
   109  
   110  	/**
   111  	 * Returns an array representation of all values in the map.
   112  	 */
   113  	public function getValues():Array 
   114  	{
   115  		return _values.slice() ;
   116  	}
   117  	
   118  	/**
   119  	 * Returns the index of the specified key in argument.
   120  	 * @param key the key in the map to search.
   121  	 * @return the index of the specified key in argument.
   122  	 */
   123  	public function indexOfKey(key):Number 
   124  	{
   125  		var l:Number = _keys.length ;
   126  		while (_keys[--l] != key && l>-1);
   127  		return l ;
   128  	}
   129  
   130  	/**
   131  	 * Returns the index of the specified value in argument.
   132  	 * @param value the value in the map to search.
   133  	 * @return the index of the specified value in argument.
   134  	 */
   135  	public function indexOfValue(value):Number 
   136  	{
   137  		var l:Number = _values.length ;
   138  		while (_values[--l] != value && l>-1) ;
   139  		return l ;
   140  	}
   141  	
   142  	/**
   143  	 * Returns true if this map contains no key-value mappings.
   144  	 * @return true if this map contains no key-value mappings.
   145  	 */
   146  	public function isEmpty():Boolean 
   147  	{
   148  		return (size() < 1 || size == undefined) ;
   149  	}
   150  
   151  	/**
   152  	 * Returns the values iterator of this map.
   153  	 * @return the values iterator of this map.
   154  	 */
   155  	public function iterator():Iterator 
   156  	{
   157  		return new MapIterator(this) ;
   158  	}
   159  
   160  	/**
   161  	 * Returns the keys iterator of this map.
   162  	 * @return the keys iterator of this map.
   163  	 */
   164  	public function keyIterator():Iterator 
   165  	{
   166  		return new ArrayIterator(_keys) ;
   167  	}
   168  
   169  	/**
   170  	 * Associates the specified value with the specified key in this map.
   171  	 * @param key the key to register the value.
   172  	 * @param value the value to be mapped in the map.
   173  	 */
   174  	public function put(key, value) 
   175  	{
   176  		var r ;
   177  		var i:Number = indexOfKey(key) ;
   178  		if (i<0) 
   179  		{
   180  			_keys.push(key) ;
   181  			_values.push(value) ;
   182  			return null ;
   183  		}
   184  		else 
   185  		{
   186  			r = _values[i] ;
   187  			_values[i] = value ;
   188  			return r ;
   189  		}
   190  	}
   191  	
   192  	/**
   193  	 * Copies all of the mappings from the specified map to this one.
   194  	 */
   195  	public function putAll(m:Map):Void 
   196  	{
   197  		var aV:Array = m.getValues() ;
   198  		var aK:Array = m.getKeys() ;
   199  		var l:Number = aK.length ;
   200  		for (var i:Number = 0 ; i<l ; i = i - (-1) ) 
   201  		{
   202  			put(aK[i], aV[i]) ;
   203  		}
   204  	}
   205  
   206  	/**
   207  	 * Removes the mapping for this key from this map if present.
   208  	 */
   209  	public function remove(key) 
   210  	{
   211  		var r = null ;
   212  		var i:Number = indexOfKey(key) ;
   213  		if (i > -1) 
   214  		{
   215  			r = _values[i] ;
   216  			_values.splice(i, 1) ;
   217  			_keys.splice(i, 1) ;
   218  		}
   219  		return r ;
   220  	}
   221  
   222  	/**
   223  	 * Returns the number of key-value mappings in this map.
   224  	 * @return the number of key-value mappings in this map.
   225  	 */
   226  	public function size():Number 
   227  	{
   228  		return _keys.length ;
   229  	}
   230  
   231  	/**
   232  	 * Returns the Eden representation of this map.
   233  	 * @return the Eden representation of this map.
   234  	 */
   235  	public function toSource(indent:Number, indentor:String):String 
   236  	{
   237  		var sourceA:String = Serializer.toSource(_keys) ;
   238  		var sourceB:String = Serializer.toSource(_values) ;
   239  		return Serializer.getSourceOf(this, [sourceA, sourceB]) ;
   240  	}
   241  
   242  	/**
   243  	 * Returns the string representation of this map.
   244  	 * @return the string representation of this map.
   245  	 */
   246  	public function toString():String 
   247  	{
   248  		return (new MapFormat()).formatToString(this) ;
   249  	}
   250  
   251  	private var _keys:Array ;
   252  
   253  	private var _values:Array ;
   254  
   255  	
   256  }
