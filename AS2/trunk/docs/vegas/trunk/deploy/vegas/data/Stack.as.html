    24  import vegas.core.ISerializable;
    25  import vegas.data.iterator.Iterator;
    26  
    27  /**
    28   * A collection designed for holding elements prior to processing. 
    29   * <p>Stacks typically, but do not necessarily, order elements in a LIFO (last-in-first-out) manner.</p>
    30   * @author eKameleon
    31   */
    32  interface vegas.data.Stack extends ISerializable 
    33  {
    34  
    35  	/**
    36  	 * Removes all of the elements from this stack (optional operation).
    37  	 */
    38  	function clear():Void ;
    39  
    40  	/**
    41  	 * Returns the shallow copy of this stack.
    42  	 */
    43  	function clone() ;
    44  
    45  	/**
    46  	 * Returns {@code true} if this stack contains no elements.
    47  	 * @return {@code true} if this stack is empty else {@code false}.
    48  	 */
    49  	function isEmpty():Boolean ;
    50  
    51  	/**
    52  	 * Returns the iterator of this stack.
    53  	 */
    54  	function iterator():Iterator ;
    55  
    56  	/**
    57  	 * Returns the lastly pushed value without removing it.
    58  	 * @throws the lastly pushed value.
    59  	 */
    60  	function peek() ;
    61  
    62  	/**
    63  	 * Removes and returns the lastly pushed value.
    64  	 * @return the lastly pushed value
    65  	 */
    66  	function pop() ;
    67  
    68  	/**
    69  	 * Pushes the passed-in value to this stack.
    70  	 */
    71  	function push(o):Void ;
    72  	
    73  	/**
    74  	 * Search a value in the stack.
    75  	 * @return the index position of a value in the stack.
    76  	 */
    77  	function search(o):Number ;
    78  
    79  	/**
    80  	 * Returns the number of pushed values.
    81  	 * @return the number of pushed values.
    82  	 */
    83  	function size():Number ;
    84  
    85  	/**
    86  	 * Returns the array representation of this stack.
    87  	 * @return the array representation of this stack
    88  	 */
    89  	function toArray():Array ;
    90  	
    91  	/**
    92  	 * Returns the string representation of this stack.
    93  	 * @return the string representation of this stack.
    94  	 */
    95  	function toString():String ;
    96  
    97  }
