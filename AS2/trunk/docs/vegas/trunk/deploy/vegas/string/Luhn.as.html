    24  /**
    25   * It is a simple checksum formula used to validate a variety of account numbers, such as credit card numbers, etc.
    26   * <p>The Luhn algorithm or Luhn formula, also known as the "modulus 10" or "mod 10" algorithm, was developed in the 1960s as a method of validating identification numbers.</p>
    27   * <p><b>example</b><br>
    28   * {@code
    29   * import vegas.string.Luhn ;
    30   * 
    31   * var code:String = "456565654" ;
    32   * trace (code + " isValid : " + Luhn.isValid(code)) ;
    33   * }
    34   * </p>
    35   * @see <a href='http://fr.wikipedia.org/wiki/Formule_de_Luhn'>Luhn Formula</a> 
    36   * @author eKameleon
    37   */
    38  class vegas.string.Luhn 
    39  {
    40  
    41  	/**
    42  	 * Returns {@code true} if the expression in argument is a valid Luhn value.
    43  	 * @return {@code true} if the expression in argument is a valid Luhn value.
    44  	 */
    45  	public static function isValid(str:String):Boolean 
    46  	{	
    47  		str = new String(str) ;
    48  		var	n:Number ;
    49  		var sum:Number = 0 ;
    50  		var l:Number = str.length ;
    51  		for (var i:Number = 0 ; i<l ; i++)
    52  		{
    53  			n = Number(str.charAt(i)) * ( i%2 == 1 ? 2 : 1) ;
    54  			sum += n - ((n > 9) ? 9 : 0) ;
    55  		}
    56  		return sum%10 == 0 ;
    57  	}
    58  
    59  }
