    24  import vegas.core.HashCode;
    25  import vegas.core.IFormattable;
    26  import vegas.core.IHashable;
    27  import vegas.errors.ErrorElement;
    28  import vegas.errors.ErrorFormat;
    29  import vegas.logging.ILogable;
    30  import vegas.logging.ILogger;
    31  import vegas.logging.Log;
    32  import vegas.logging.LogEventLevel;
    33  import vegas.util.ConstructorUtil;
    34  
    35  /**
    36   * This class provides a Abstract implementation to creates Error classes with an internal logging model. 
    37   * @author eKameleon
    38   */
    39  class vegas.errors.AbstractError extends Error implements IFormattable, IHashable, ILogable
    40  {
    41      
    42  	/**
    43  	 * Creates a new AbstractError only if you inherit this class.
    44  	 */
    45  	private function AbstractError(message:String, e:ErrorElement) 
    46  	{
    47  		this.message = message ;
    48  		this.errorElement = e ;
    49  		this.name = ConstructorUtil.getName(this) ;
    50  		this._logger = Log.getLogger( ConstructorUtil.getPath(this) ) ;
    51  	}
    52  
    53  	/**
    54  	 * Launch HashCode 'initialize' method.
    55  	 */	
    56  	private static var _initHashCode:Boolean = HashCode.initialize(AbstractError.prototype) ;
    57  
    58  	/**
    59  	 * The internal ErrorElement reference.
    60  	 */
    61  	public var errorElement:ErrorElement ;
    62  	
    63  	/**
    64  	 * Launch the external log of this error.
    65  	 */
    66  	public function log():Void
    67  	{
    68  		getLogger().log( getLevel() , "# " + name + " : " + message + " #" ) ;
    69  	}
    70  	
    71  	/**
    72  	 * Returns the internal ILogger of the current Error.
    73  	 * @return the internal ILogger of the current Error.
    74  	 */
    75  	public function getLogger():ILogger
    76  	{
    77  		return _logger ; 	
    78  	}
    79  	
    80  	/**
    81  	 * Returns the internal LogEventLevel used in the constructor of this instance.
    82  	 * You can overrides this method if you want change the internal LogEventLevel of the error.
    83  	 */
    84  	public function getLevel():Number
    85  	{
    86  		return LogEventLevel.ERROR ;	
    87  	}
    88  	
    89  	/**
    90  	 * Returns a hashcode value for the object.
    91  	 * @return a hashcode value for the object.
    92  	 */
    93  	public function hashCode():Number 
    94  	{
    95  		return null ;
    96  	}
    97  	
    98  	/**
    99  	 * Sets the internal {@code ILogger} reference of this {@code ILogable} object.
   100  	 */
   101  	public function setLogger( log:ILogger ):Void 
   102  	{
   103  		_logger = log ;
   104  	}
   105  
   106  	/**
   107  	 * Returns the string representation of this instance.
   108  	 * @return the string representation of this instance
   109  	 */
   110  	public function toString():String 
   111  	{
   112  		if ( arguments.caller == null)
   113  		{
   114  			log() ;
   115  		}
   116  		return (new ErrorFormat()).formatToString(this) ;
   117  	}
   118  
   119  	private var _logger:ILogger ;
   120  
   121  }
