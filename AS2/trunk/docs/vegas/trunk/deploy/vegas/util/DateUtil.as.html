    24  /**
    25   * The {@code DateUtil} utility class is an all-static class with methods for working with dates.
    26   * @author eKameleon
    27   */
    28  class vegas.util.DateUtil 
    29  {
    30  
    31  	/**
    32  	 * Returns a shallow copy of the date object passed in argument.
    33  	 * @return a shallow copy of the date object passed in argument.
    34  	 */
    35  	static public function clone(d:Date):Date 
    36  	{
    37  		return d ;
    38  	}
    39  
    40  	/**
    41  	 * Returns a deep copy of the date object passed in argument.
    42  	 * @return a deep copy of the date object passed in argument.
    43  	 */
    44  	static public function copy(d:Date):Date 
    45  	{
    46  		return new Date(d.valueOf()) ;
    47  	}
    48  
    49  	/**
    50  	 * Compares the twno specified Date objects for equality.
    51  	 * @return {@code true} if the the two specified Date object are equals.
    52  	 */
    53  	static public function equals( d1:Date, d2:Date ):Boolean 
    54  	{
    55  		if (!d2) return false ;
    56  		return d1.getTime() == d2.getTime() ;
    57      }
    58  	
    59  }
