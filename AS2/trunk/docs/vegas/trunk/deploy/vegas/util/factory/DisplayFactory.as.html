    24  import vegas.util.ConstructorUtil;
    25  import vegas.util.TypeUtil;
    26  
    27  /**
    28   * This factory static class is an all-static class with methods to creates dynamic {@code MovieClip} or {@code Textfield} who used a custom class. 
    29   * @author eKameleon
    30   * @see ConstructorUtil
    31   */
    32  class vegas.util.factory.DisplayFactory 
    33  {
    34  	
    35  	/**
    36  	 * Attachs a MovieClip in a target and register a custom constructor who inherit MovieClip in this instance.
    37  	 * @see ConstructorUtil.createVisualInstance
    38  	 */
    39  	public static function attachChild
    40  	( 
    41  		fConstructor:Function, sID:String, sName:String,
    42  		 nDepth:Number, mcTarget:MovieClip, init 
    43  	) 
    44  	{
    45  		var oChild = mcTarget.attachMovie(sID, sName, nDepth, null) ;
    46  		return ConstructorUtil.createVisualInstance( fConstructor, oChild, init) ;
    47  	}
    48  	
    49  	/**
    50  	 * Creates a new child object (MovieClip, TextField) with a custom constructor or an id link.
    51  	 * @see ConstructorUtil
    52  	 */
    53  	public static function createChild ( oChild , p_name:String , p_depth:Number, p_target, p_init) 
    54  	{
    55  		
    56  		var child ;
    57  		
    58  		if (oChild == null) 
    59  		{
    60  			child = p_target.createEmptyMovieClip(p_name, p_depth) ;
    61  			for (var each in p_init) 
    62  			{
    63  				child[each] = p_init[each] ;
    64  			}
    65  			return child ;
    66  			
    67  		}
    68  		else if (oChild instanceof Function) 
    69  		{
    70  			var p_class:Function = oChild ;
    71  			if (ConstructorUtil.isSubConstructorOf(p_class, MovieClip))  
    72  			{
    73  				child = p_target.createEmptyMovieClip (p_name, p_depth) ;
    74  			}
    75  			else if (ConstructorUtil.isSubConstructorOf(p_class, TextField)) 
    76  			{
    77  				p_target.createTextField (p_name, p_depth, 0, 0, 0, 0) ;
    78  				child = p_target[p_name] ;
    79  			}
    80  			return ConstructorUtil.createVisualInstance(p_class, child, p_init) ;
    81  		}
    82  		else if (TypeUtil.typesMatch(oChild, String)) 
    83  		{
    84  			return p_target.attachMovie(oChild, p_name, p_depth, p_init) ;
    85  		}
    86  	}
    87  	
    88  	/**
    89  	 * Returns the depth of the display instance (MovieClip, TextField, Video).
    90  	 * @param display The display reference (MovieClip, TextField, Video).
    91  	 * @return the depth of the display instance (MovieClip, TextField, Video).
    92  	 */
    93  	public static function getDepth( display ):Number
    94  	{
    95  		return MovieClip.prototype.getDepth.call( display ) ;
    96  	}
    97  	
    98  	/**
    99  	 * Swaps the stacking, or depth level (z-order), of the specified display with display that is specified by the target parameter, or with the display that currently occupies the depth level that is specified in the target parameter.
   100  	 * @param display An MovieClip, TextField or Video display object.
   101  	 * @param target This parameter can take one of two forms:
   102  	 * <bl>
   103  	 * <li>A Number that specifies the depth level where the display is to be placed.</li>
   104  	 * <li>A String that specifies the display instance whose depth is swapped with the movie clip for which the method is being applied. Both displays must have the same parent movie clip.</li>
   105  	 * </bl>
   106  	 */
   107  	public static function swapDepths( display, target ):Void
   108  	{
   109  		MovieClip.prototype.swapDepths.call( display, target ) ;
   110  	}
   111  	
   112  }
