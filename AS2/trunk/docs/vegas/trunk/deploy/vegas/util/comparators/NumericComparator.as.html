    32  /**
    33   * A Comparator which deals with alphabet characters 'naturally', but deals with numerics numerically. 
    34   * Leading 0's are ignored numerically, but do come into play if the number is equal. 
    35   * Thus aaa119yyyy comes before aaa0119xxxx regardless of x or y.
    36   * 
    37   * The comparison should be very performant as it only ever deals with issues at a character level 
    38   * and never tries to consider the numerics as numbers.
    39   * <p><b>Example :</b></p>
    40   * {@code
    41   * import vegas.events.Delegate ;
    42   * import vegas.util.comparators.NumericComparator ;
    43   * 
    44   * var ar:Array =
    45   * [
    46   *     "a0" , "a1",  "a2",  "a10" , "a11" , "b1" , "b2" , "b0" , "b01", "b001", "c0"
    47   * ] ;
    48   * 
    49   * var c:NumericComparator = new NumericComparator.getIgnoreCaseInstance() ;
    50   * 
    51   * ar.sort( Delegate.create( c , c.compare ) , Array.CASEINSENSITIVE ) ;
    52   * trace( ar ) ; // a0,a1,a2,a10,a11,b1,b2,b0,b01,b001,c0
    53   * 
    54   * ar.sort( Delegate.create( c , c.compare ) , Array.CASEINSENSITIVE | Array.DESCENDING ) ;
    55   * trace(ar ) ; // c0,b001,b01,b0,b2,b1,a11,a10,a2,a1,a0
    56   * }
    57   * @author eKameleon
    58   */
    59  class vegas.util.comparators.NumericComparator extends CoreObject implements IComparator

    76  	/**
    77  	 * Returns an integer value to compare two String objects.
    78  	 * @param o1 the first String object to compare.
    79  	 * @param o2 the second String object to compare.
    80  	 * @return <p>
    81  	 * <li>-1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
    82  	 * <li> 1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
    83  	 * <li> 0 if o1 and o2 are equal.</li>
    84  	 * </p>
    85  	 * @throws ClassCastError if compare(a, b) and 'a' or 'b' aren't String objects.
    86  	 */
    87  	public function compare(o1, o2):Number 
    88  	{
    89  		if ( o1 == null || o2 == null) 
    90  		{
    91  			if (o1 == o2) 
    92  			{
    93  				return 0 ;
    94  			}
    95  			else if (o1 == null) 
    96  			{
    97  				return -1 ;
    98  			}
    99  			else 
   100  			{
   101  				return 1 ;
   102  			}
   103  		}
   104  		else 
   105  		{
   106  			if ( !TypeUtil.typesMatch(o1, String) || !TypeUtil.typesMatch(o2, String)) 
   107  			{
   108  				throw new ClassCastError(this + " compare method failed, Arguments string expected.") ;
   109  			}
   110  			else 
   111  			{
   112  				
   113  				var comp:IComparator = ( ignoreCase ) ? StringComparator.getIgnoreCaseStringComparator() : StringComparator.getStringComparator() ;
   114  				
   115  				o1 = o1.toString() ;
   116  				o2 = o2.toString() ;
   117  				
   118  				// find the first digit.
   119          		var idx1:Number = _getFirstDigitIndex( o1 );
   120          		var idx2:Number = _getFirstDigitIndex( o2 );
   121  				
   122  				if( ( idx1 == -1 ) || ( idx2 == -1 ) || ( ! ( o1.substring(0,idx1) == o2.substring(0,idx2) ) ) )
   123  		        {
   124  		        	comp.compare(o1, o2) ;
   125          		}
   126  			
   127  		        // find the last digit
   128  		        var edx1:Number = _getLastDigitIndex( o1, idx1 ) ;
   129  		        var edx2:Number = _getLastDigitIndex( o2, idx2 ) ;
   130  		        
   131  		        var sub1:String = null ;
   132  		        var sub2:String = null ;
   133  		        
   134  		 		if( edx1 == -1 ) 
   135  		 		{
   136              		sub1 = o1.substring(idx1) ;
   137          		} 
   138          		else 
   139          		{
   140              		sub1 = o1.substring( idx1, edx1 ) ;
   141          		}
   142  
   143  		        if( edx2 == -1 ) 
   144  		        {
   145              		sub2 = o2.substring(idx2) ;
   146          		}
   147          		else 
   148          		{
   149              		sub2 = o2.substring(idx2, edx2) ;
   150  		        }
   151  		        
   152  		        // deal with zeros at start of each number
   153          		
   154          		var zero1:Number = _countZeroes( sub1 ) ;
   155          		var zero2:Number = _countZeroes( sub2 ) ;
   156  
   157          		sub1 = sub1.substring( zero1 ) ;
   158          		sub2 = sub2.substring( zero2 ) ;
   159  		        
   160  		        // if equal, then recurse with the rest of the string need to deal with zeroes so that 00119 appears after 119.
   161          		if( sub1 == sub2 ) 
   162          		{
   163              		
   164              		var ret:Number = 0 ;
   165              		
   166              		if( zero1 > zero2 ) 
   167              		{
   168                  		ret = 1;
   169              		}
   170              		else if( zero1 < zero2 ) 
   171              		{
   172                  		ret = -1 ; 
   173              		}
   174              		
   175              		if(edx1 != -1) 
   176              		{
   177                  		
   178                  		var r:Number = comp.compare( o1.substring(edx1) , o2.substring(edx2) ) ;
   179                  		if ( r != 0 )
   180                  		{
   181                  			ret = r ;
   182                  		}
   183  	                }
   184  		           	
   185  		           	return ret ;
   186  				} 
   187          		else // if a numerical string is smaller in length than another then it must be less. 
   188          		{
   189  		            if( sub1.length != sub2.length ) 
   190  		            {
   191                  		return ( sub1.length < sub2.length ) ? -1 : 1 ;
   192              		}
   193          		
   194          		}
   195  		
   196  		        var chr1:Array = sub1.split("");
   197          		var chr2:Array = sub2.split("") ;
   198  
   199          		var sz:Number = chr1.length ;	
   200          		for( var i:Number =0 ; i<sz ; i++ ) 
   201          		{
   202   		           // this should give better speed
   203  					if( chr1[i] != chr2[i] ) 
   204              		{
   205  	                	return ( chr1[i] < chr2[i] ) ? -1 : 1 ;
   206      	        	}
   207          		}
   208  	        	
   209  	        	return 0 ;
   210  	        	
   211  			}
   212  		}
   213  	}

   215  	/**
   216  	 * Returns the {@code NumericComparator} singleton with the a {@code false} ignoreCase property.
   217  	 * Clients are encouraged to use the value returned from this method instead of constructing a new instance to reduce allocation and garbage collection overhead when multiple StringComparators may be used in the same application.
   218  	 * @return the {@code NumericComparator} singleton with the a {@code false} ignoreCase property.
   219  	 */
   220  	public static function getInstance():NumericComparator
   221  	{
   222  		if ( _comparator == null )
   223  		{
   224  			_comparator = new NumericComparator(false) ;	
   225  		}
   226  		return _comparator ;
   227  	}

   233  	 * @return the {@code NumericComparator} singleton with the a {@code true} ignoreCase property.

   238  		{
   239  			_ignoreCaseComparator = new NumericComparator(true) ;

   242  	}

   263  	/**
   264  	 * Counts and returns the number of '0' in the passed-in string.
   265  	 * @return the number of '0' in the passed-in string.
   266  	 */
   267  	private function _countZeroes( str:String ) 
   268  	{
   269          var count:Number = 0 ;
   270          var size:Number  = str.length ;
   271          // The str must be small please
   272          for( var i:Number=0; i<size ; i++ ) 
   273          {
   274              if(str.charAt(i) == "0")
   275              {
   276                  count++;
   277              }
   278              else 
   279              {
   280                  break ;
   281              }
   282          }
   283          return count ;
   284      }

   286  	/**
   287  	 * Returns the first digit index position in a char list (String or Array of chars)
   288  	 * @return the first digit index position in a char list (String or Array of chars)
   289  	 */
   290      private function _getFirstDigitIndex( chars , start:Number ):Number
   291      {
   292      	if ( isNaN(start) )
   293      	{
   294      		start = 0 ;
   295      	}
   296          
   297          var chrs:Array ;
   298          
   299          if ( TypeUtil.typesMatch( chars, String ) )
   300          {
   301          	chrs = String(chars).split("") ;	
   302          }
   303          else if ( chars instanceof Array )
   304          {
   305          	chrs = [].concat(chars) ;
   306          }
   307  		else
   308  		{
   309  			return -1 ;	
   310  		}
   311          
   312          var size:Number = chrs.length ;
   313  
   314          for( var i:Number = start ; i<size ; i++ ) 
   315          {
   316              if( _isDigit(chrs[i])) 
   317              {
   318                  return i ;
   319              }
   320          }
   321          
   322          return -1;
   323      }

   325  	/**
   326  	 * Returns the last digit index position in a char list (String or Array of chars)
   327  	 * @return the last digit index position in a char list (String or Array of chars)
   328  	 */
   329      private function _getLastDigitIndex( chars, start:Number ):Number 
   330      {
   331  
   332  		var chrs:Array ;
   333          
   334          if ( TypeUtil.typesMatch( chars, String ) )
   335          {
   336          	chrs = String(chars).split("") ;	
   337          }
   338          else if ( chars instanceof Array )
   339          {
   340          	chrs = [].concat(chars) ;
   341          }
   342  		else
   343  		{
   344  			return -1 ;	
   345  		}
   346          var size:Number = chrs.length ;
   347  
   348          for( var i:Number = start ; i<size ; i++ ) 
   349          {
   350              if( !_isDigit(chrs[i])) 
   351              {
   352                  return i ;
   353              }
   354          }
   355          
   356          return -1;
   357      }

   359  	/**
   360  	 * Returns {@code true} if the passed-in character is a digit.
   361  	 * @return {@code true} if the passed-in character is a digit.
   362  	 */
   363  	private function _isDigit( /*char*/ c:String ):Boolean
   364      {
   365           return( ("0" <= c) && (c <= "9") );
   366      }
