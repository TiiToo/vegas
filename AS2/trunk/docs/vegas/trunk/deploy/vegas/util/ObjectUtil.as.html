    24  import vegas.util.Copier;
    25  import vegas.util.TypeUtil;
    26  
    27  /**
    28   * The {@code ObjectUtil} utility class is an all-static class with methods for working with object.
    29   * @author eKameleon
    30   */
    31  class vegas.util.ObjectUtil 
    32  {
    33  
    34  	/**
    35  	 * Returns the shallow copy of the object.
    36  	 * @return the shallow copy of the object.
    37  	 */
    38  	public static function clone(o) 
    39  	{
    40  		return o ;	
    41  	}
    42  
    43  	/**
    44  	 * Returns the deep copy of the object.
    45  	 * @return the deep copy of the object.
    46  	 */
    47  	public static function copy(o) 
    48  	{
    49  		var obj:Object = {} ;
    50  		var prop:String ;
    51  		for (prop in o) 
    52  		{
    53  			if( !o.hasOwnProperty( prop ) ) 
    54  			{
    55  				continue ;
    56  			}
    57  		  	else if ( o[prop] == undefined ) 
    58  			{
    59  				obj[prop] = undefined ;
    60  			}
    61  		  	else if ( o[prop] == null ) 
    62  			{
    63  				obj[prop] = null ;
    64  			}
    65  			else 
    66  			{
    67  				obj[prop] = Copier.copy( o[prop] ) ; 
    68  			}
    69  		}
    70  		return obj ;
    71  	}
    72  
    73  	/**
    74  	 * Returns a Boolean value indicating whether an object has a property with the specified name (*ECMA-262*). 
    75  	 * @return {@code true} whether the property is in the prototype chain or not.
    76  	 */
    77  	public static function hasProperty(o, prop:String):Boolean 
    78  	{
    79  		return ( o[prop] != undefined );
    80  	}
    81  
    82  	/**
    83  	 * Returns {@code true} if the passed object is empty of enumerable property.
    84  	 * @return {@code true} if the passed object is empty of enumerable property.
    85  	 */
    86  	public static function isEmpty(o):Boolean 
    87  	{
    88  		for (var each:String in o) 
    89  		{
    90  			return false ;
    91  			each = undefined ; // FDT 3 hack ^-^	
    92  		}
    93  		return true ;
    94  	}
    95  
    96  	/**
    97  	 * Creates a shallow copy of the current Object.
    98  	 */
    99  	public static function memberwiseClone( o ) 
   100  	{
   101  		var obj = {} ;
   102  		for( var prop:String in o ) 
   103  		{
   104  			obj[prop] = o[prop];
   105  		}
   106  		return obj;
   107  	}
   108  
   109  	/**
   110  	 * Determines whether the specified Object instance is the same instance as the current object.
   111  	 */
   112  	public static function referenceEquals( o1, o2 ):Boolean
   113  	{
   114  		if( ( o1.valueOf( ) === null) && (o2 === null) )
   115  		{
   116  			return true;
   117  		}
   118  
   119  		if( o1 === o2 )
   120  		{
   121  			return true ;
   122  		}
   123      
   124  		if( ( o1 == null) || TypeUtil.typesMatch( o2, Object ) )
   125  		{
   126  			return false ;
   127  		}
   128      
   129  		if( o1 !== o2 )
   130  		{
   131  			return false;
   132  		}
   133  
   134  		return true;	
   135  	}
   136  
   137  	/**
   138  	 * Converts an object to an equivalent Boolean value.
   139  	 */
   140  	public static function toBoolean(o):Boolean 
   141  	{
   142  		return (new Boolean( o.valueOf( ) )).valueOf( ) ;
   143  	}
   144  
   145  	/**
   146  	 * Converts an object to an equivalent Number value.
   147  	 */
   148  	public static function toNumber(o):Number 
   149  	{
   150  		return (new Number( o.valueOf( ) )).valueOf( ) ;
   151  	}
   152  
   153  	/**
   154  	 * Converts an object to an equivalent Object value.
   155  	 */
   156  	public static function toObject(o):Object 
   157  	{
   158  		return new Object( o.valueOf( ) ) ;
   159  	}
   160  }
