    24  import vegas.events.Delegate;
    25  
    26  /**
    27   * This factory static class is an all-static class with methods to creates different {@code ContextMenuItem} objects.
    28   * @author eKameleon
    29   * @see ContextMenu
    30   * @see ContextMenuItem
    31   */
    32  class vegas.util.factory.ContextMenuFactory 
    33  {
    34  
    35  	/**
    36  	 * Creates a {@code ContextMenuItem} who launch an exteral url when the user select this item in a ContextMenu.
    37  	 * @return a {@code ContextMenuItem} reference.
    38  	 */
    39  	public static function createItemURL(label:String, url:String, target:String, separator:Boolean):ContextMenuItem 
    40  	{
    41  		var f:Function = Delegate.create(ContextMenuFactory, _getURL, url, target) ;
    42  		var c:ContextMenuItem = new ContextMenuItem(label, f) ;
    43  		c.separatorBefore = separator ;
    44  		return c ;
    45  	}
    46  	
    47  	/**
    48  	 * Creates a {@code ContextMenuItem} who use a proxy method when the user select this item in a ContextMenu.
    49  	 * @return a {@code ContextMenuItem} reference.
    50  	 */
    51  	public static function createItemProxy( label:String, scope , method:Function, separator:Boolean , args:Array):ContextMenuItem 
    52  	{
    53  		var c:ContextMenuItem = new ContextMenuItem(label, Delegate.create.apply(null, [scope, method].concat(args)) ) ;
    54  		c.separatorBefore = separator ;
    55  		return c ;
    56  	}
    57  	
    58  	/**
    59  	 * Internal proxy method to launch {@code getURL} method.
    60  	 */
    61  	private static function _getURL(target, item, url:String, where:String) 
    62  	{
    63  		getURL(url, where || "_blank") ;	
    64  	}
    65  	
    66  }
