    24  import vegas.core.CoreObject;
    25  import vegas.util.AttributeType;
    26  
    27  /**
    28   * This tool static class define a basic attribute structure.
    29   * SSAS FMS Framework inspiration and Zwetan : Core2 AS2 Framework inspiration for refactoring !
    30   * @author eKameleon
    31   * @see ASSetPropFlags global and no documented function.
    32   */
    33  class vegas.util.Attribute extends CoreObject
    34  {
    35  
    36  	/**
    37  	 * Creates a new Attribute instance.
    38  	 * <p><b>Example :</b></p>
    39  	 * {@code
    40  	 * import vegas.util.Attribute ;
    41  	 * var attr:Attribute = new Attribute(true, true, true) ;
    42  	 * }
    43  	 * @param dontEnum a boolean to defined if the object is enumerable or not.
    44  	 * @param dontDelete a boolean to defined if the object can be remove or not.
    45  	 * @param readOnly a boolean to defined if the object is readable only or not.
    46  	 */
    47  	public function Attribute(dontEnum:Boolean, dontDelete:Boolean, readOnly:Boolean) 
    48  	{
    49  		this.dontEnum = dontEnum == true ;
    50          this.dontDelete = dontDelete == true ;
    51  		this.readOnly = readOnly == true ;
    52  	}
    53  
    54  	/**
    55  	 * The dontEnum value to defined if the object is enumerable or not.
    56  	 */
    57  	public var dontEnum:Boolean ;
    58  	
    59  	/**
    60  	 * The dontDelete value to defined if the object can be remove or not. 
    61  	 */
    62  	public var dontDelete:Boolean ;
    63  	
    64  	/**
    65  	 * The readOnly value to defined if the object is readable only or not.
    66  	 */
    67  	public var readOnly:Boolean ;
    68  
    69  	/**
    70  	 * Returns the attribut instance and configuration of an object.
    71  	 * @param obj the object to defined the {@code Attribute} instance.
    72  	 * @param property the property in the object to check.
    73  	 * @return the attribut instance and configuration of an object.
    74  	 */
    75      static public function getAttribute(obj:Object, property:String):Attribute 
    76      {
    77          var dontEnum:Boolean = Attribute.isDontEnum(obj, property);
    78          var readOnly:Boolean = Attribute.isReadOnly(obj, property);
    79          var dontdelete:Boolean = Attribute.isDontDelete(obj, property);
    80          var attrib:Attribute = new Attribute( dontEnum, dontdelete, readOnly ) ;
    81          Attribute.setAttribute( obj, property, attrib ) ;
    82          return attrib ;
    83  	}
    84  
    85  	/**
    86  	 * Returns {@code true} if the property in the specified object can't be deleted.
    87  	 * @return {@code true} if the property in the specified object can't be deleted.
    88  	 */
    89      static public function isDontDelete(obj, property:String):Boolean 
    90      {
    91          var save = obj[property] ;
    92  		delete obj[property] ;
    93          if( obj[property] === undefined ) 
    94          {
    95              obj[property] = save ;
    96              return false;
    97  		}
    98          return true;
    99  	}
   100  
   101  	/**
   102  	 * Returns {@code true} if the property in the specified object isn't enumerable.
   103  	 * @return {@code true} if the property in the specified object isn't enumerable.
   104  	 */
   105      static public function isDontEnum(obj, property:String):Boolean 
   106      {
   107          return !obj.isPropertyEnumerable(property) ;
   108  	}
   109      
   110  	/**
   111  	 * Returns {@code true} if the property in the specified object can be read only.
   112  	 * @return {@code true} if the property in the specified object can be read only.
   113  	 */
   114      static public function isReadOnly(obj, property:String):Boolean 
   115      {
   116          var dummy:String = "__uFFFCuFFFDuFFFCuFFFD__"; // we use ORC char to prevent string colision
   117          var save = obj[property] ;
   118          obj[property] = dummy ;
   119          if( obj[property] == dummy ) 
   120          {
   121              obj[property] = save ;
   122              return false ;
   123  		}
   124          return true ;
   125  	}
   126  	
   127  	/**
   128  	 * Sets the attribute of the property in the specified object.
   129  	 * @param obj the specified object to change the property attribute.
   130  	 * @param property the string name of the property of the specified object.
   131  	 * @param attrib the Attribute instance to change the current attribute of the property.
   132  	 * @param override to defined the override value. This value is the last parameter of the {@code ASSetPropFlags} global function.
   133  	 * @see ASSetPropFlags
   134  	 */
   135  	static public function setAttribute(obj:Object, property, attrib, override):Void 
   136  	{
   137          if( attrib == null ) attrib = AttributeType.NONE ;
   138  		if( override == null ) 
   139  		{
   140  			override = AttributeType.NONE ;
   141  		}
   142  		if( attrib instanceof Attribute ) 
   143  		{
   144  			attrib = attrib.valueOf() ;
   145  		}
   146          if( override instanceof Attribute ) 
   147          {
   148          	override = override.valueOf();
   149          }
   150          _global.ASSetPropFlags( obj, property, attrib, override );
   151      }
   152      
   153      /**
   154       * Returns the string representation of this object.
   155       * @return the string representation of this object.
   156       */
   157  	public function toString():String 
   158  	{
   159          var data:Array = [];
   160          var sep:String = ",";
   161          if(readOnly) 
   162          {
   163          	data.push( "readOnly" ) ;
   164          }
   165  		if(dontDelete) 
   166  		{
   167  			data.push( "dontDelete" ) ;
   168  		}
   169  		if(dontEnum) 
   170  		{
   171  			data.push( "dontEnum" ) ;
   172  		}
   173          if(data.length == 0) 
   174          {
   175          	data.push("none") ;
   176          }
   177          return "[Attribute:" + data.join( sep ) + "]";
   178  	}
   179  	
   180  	/**
   181  	 * Returns the value of this object.
   182  	 * @return the value of this object.
   183  	 */
   184  	public function valueOf():Number 
   185  	{
   186          return ((Number(readOnly) << 2) | (Number(dontDelete) << 1) | Number(dontEnum) );
   187  	}
   188  	
   189  	
   190  }
