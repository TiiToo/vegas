    90  	/**
    91  	 * Compares the two caracteres passed in argument for order.
    92  	 * @return <p>
    93  	 * <li>-1 if charA is "lower" than (less than, before, etc.) charB ;</li>
    94  	 * <li> 1 if charA is "higher" than (greater than, after, etc.) charB ;</li>
    95  	 * <li> 0 if charA and charB are equal.</li>
    96  	 * </p>
    97  	 */
    98  	public static function compareChars( charA:String, charB:String ):Number
    99  	{
   100  		var a:String = charA.charAt(0) ;
   101  		var b:String = charB.charAt(0) ;
   102  		if ( caseValue(a) < caseValue(b) ) 
   103  		{
   104  			return -1;
   105  		}
   106  		if ( caseValue(a) > caseValue(b) ) 
   107  		{
   108  			return 1 ;
   109  		}
   110  		if ( a < b ) 
   111  		{
   112  			return -1;
   113  		}
   114  		if ( a > b ) 
   115  		{
   116  			return 1;
   117  		}
   118  		return 0 ;
   119  	}

   208  	 * Inserts a specified instance of String at a specified index position in this instance.
   209  	 * <p><b>Example :</b></p>
   210  	 * {@code
   211  	 * import vegas.util.StringUtil ;
   212  	 * var result:String ;
   213  	 * result = StringUtil.insert("hello", 0, "a" ) ;  // ahello
   214  	 * result = StringUtil.insert("hello", -1, "a" ) ; // ahello
   215  	 * result = StringUtil.insert("hello", 10, "a" ) ; // helloa
   216  	 * result = StringUtil.insert("hello", 1, "a" ) ;  // haello
   217  	 * }
   218  	 * @return the string modified by the method.
   219  	 */
   220  	public static function insert( str:String, startIndex:Number, value:String):String 
   221  	{
   222  		str = str.valueOf() ;
   223  		if( value == null ) 
   224  		{
   225  			return str ;
   226  		}
   227  		if( str == "" ) 
   228  		{
   229  			return value ;
   230  		}
   231          if( startIndex <= 0 )
   232          {
   233          	return value + str ;
   234          }
   235         	else if( isNaN(startIndex) || (startIndex >= str.length) ) 
   236         	{
   237         		return str + value ;
   238         	}
   239         	var strA:String = str.substr( 0, startIndex );
   240      	var strB:String = str.substr( startIndex ) ;
   241      	return strA + value + strB ;
   242  	}
