    24  import vegas.core.CoreObject;
    25  import vegas.core.IComparator;
    26  import vegas.errors.ClassCastError;
    27  import vegas.util.comparators.NumberComparator;
    28  import vegas.util.TypeUtil;
    29  	
    30  /**
    31   * This comparator compare Date objects.
    32   * <p><b>Example :</b></p>
    33   * {@code
    34   * import vegas.util.comparators.DateComparator;
    35   * 
    36   * var comp:DateComparator = new DateComparator() ;
    37   * 
    38   * var d1:Date   = new Date(2007, 1, 1) ;
    39   * var d2:Number =  1170284400000 ;
    40   * var d3:Date   = new Date(2007, 2, 2) ;
    41   * var d4:Number = 1172790000000 ;
    42   * 
    43   * trace( comp.compare(d1, d1) ) ; // 0
    44   * trace( comp.compare(d1, d2) ) ; // 0
    45   * trace( comp.compare(d2, d1) ) ; // 0
    46   * trace( comp.compare(d1, d3) ) ; // -1
    47   * trace( comp.compare(d1, d4) ) ; // -1
    48   * trace( comp.compare(d3, d1) ) ; // 1
    49   * trace( comp.compare(d4, d1) ) ; // 1
    50   * }
    51   * @author eKameleon
    52   */
    53  class vegas.util.comparators.DateComparator extends CoreObject implements IComparator
    54  {
    55  
    56  	/**
    57  	 * Creates a new DateCompator instance.
    58  	 */
    59  	public function DateComparator() 
    60  	{
    61  		//
    62  	}
    63  	
    64  	/**
    65  	 * Returns an integer value to compare two Date objects.
    66  	 * @param o1 the first Date object to compare.
    67  	 * @param o2 the second Date object to compare.
    68  	 * @return <p>
    69  	 * <li>-1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
    70  	 * <li> 1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
    71  	 * <li> 0 if o1 and o2 are equal.</li>
    72  	 * </p>
    73  	 * @throws ClassCastError if compare(a, b) and 'a' and 'b' must be Date or uint objects.
    74  	 */
    75  	public function compare(o1, o2):Number 
    76  	{
    77  		var b1:Boolean = TypeUtil.typesMatch(o1, Number) || TypeUtil.typesMatch(o1, Date) ;
    78  		var b2:Boolean = TypeUtil.typesMatch(o2, Number) || TypeUtil.typesMatch(o2, Date) ;
    79  		if ( b1 && b2 ) 
    80  		{
    81  			var a:Number = (o1 instanceof Date) ? o1.valueOf() : o1 ;
    82  			var b:Number = (o2 instanceof Date) ? o2.valueOf() : o2 ;
    83  			return NumberComparator.getInstance().compare(a, b) ;
    84  		}
    85  		else 
    86  		{
    87  			throw new ClassCastError(this + ".compare(a, b), 'a' and 'b' must be Date or Number objects.") ;
    88  		}
    89  	}
    90  
    91  	/**
    92  	 * Returns the singleton instance of a DateComparator.
    93  	 * Developers are encouraged to use the comparator returned from this method instead of constructing a new instance to reduce allocation and GC overhead when multiple comparable comparators may be used in the same application.
    94   	 * @return the singleton instance of a DateComparator.
    95  	 */
    96  	static public function getInstance():DateComparator
    97  	{
    98  		if (_instance == null)
    99  		{
   100  			_instance = new DateComparator() ;
   101  		}
   102  		return _instance ;	
   103  	}
   104  
   105  	/**
   106    	 * The internal static singleton of this class.
   107   	 */
   108  	static private var _instance:DateComparator ;
   109  
   110  }
