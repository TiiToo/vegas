    24  import vegas.core.CoreObject;
    25  import vegas.core.IComparator;
    26  import vegas.errors.ClassCastError;
    27  import vegas.errors.NullPointerError;
    28  import vegas.util.TypeUtil;
    29  
    30  /**
    31   * This comparator compare two Objects by field.
    32   * <p><b>Example :</b></p>
    33   * {@code
    34   * 
    35   * import vegas.events.Delegate ;
    36   * import vegas.util.comparators.GenericComparator ;
    37   * import vegas.util.comparators.StringComparator ; 
    38   * 
    39   * var write:Function = function( ar:Array ):Void
    40   * {
    41   *     var result:Array = [] ;
    42   *     var size:Number = ar.length ;
    43   *     for (var i:Number = 0 ; i<size ; i++)
    44   *     {
    45   *         result[i] = ar[i].label ;
    46   *     }
    47   *     trace( result.join('|') ) ;
    48   * }
    49   * 
    50   * var ar:Array = 
    51   * [
    52   *     { id : 0 , label : "Paris" } ,
    53   *     { id : 0 , label : "Marseille" } ,
    54   *     { id : 0 , label : "Lyon" } ,
    55   *     { id : 0 , label : "Bordeaux" }
    56   * ] ;
    57   * 
    58   * var c:GenericComparator = new GenericComparator( "label", StringComparator.getIgnoreCaseStringComparator() ) ;
    59   * 
    60   * ar.sort( Delegate.create( c , c.compare ) , Array.CASEINSENSITIVE ) ;
    61   * write(ar , "label") ; // Bordeaux|Lyon|Marseille|Paris
    62   * 
    63   * ar.sort( Delegate.create( c , c.compare ) , Array.CASEINSENSITIVE | Array.DESCENDING ) ;
    64   * write(ar , "label") ; // Paris|Marseille|Lyon|Bordeaux
    65   * }
    66   * @author eKameleon
    67   */
    68  class vegas.util.comparators.GenericComparator extends CoreObject implements IComparator
    69  {
    70  
    71  	/**
    72  	 * Creates a new GenericComparator instance.
    73  	 * @param sortBy A String who represents a property name to compare the two objects.
    74  	 * @param comparator An IComparator use to compare the two objects with the specified property names. 
    75  	 * @throws ClassCastError If the 'sortBy' argument is not a String value.
    76  	 * @throws NullPointerError If the comparator argument is 'null' or 'undefined'.
    77  	 */
    78  	public function GenericComparator( sortBy , comparator:IComparator ) 
    79  	{
    80  		
    81  		if ( TypeUtil.typesMatch( sortBy, String ) )
    82  		{
    83  			this.sortBy = sortBy ;
    84  		}
    85  		else
    86  		{
    87  			throw new ClassCastError(this + " constructor failed, the 'sortBy' argument must be a String value.") ;
    88  		}
    89  		
    90  		if ( comparator != null )
    91  		{
    92  			this.comparator = comparator ;
    93  		}
    94  		else
    95  		{
    96  			throw new NullPointerError( this + " constructor failed, the 'comparator' argument not must be 'null' or 'undefined'.") ;	
    97  		}
    98  
    99  	}
   100  	
   101  	/**
   102  	 * The IComparator used by this IComparator to compare two objects with this specified field.
   103  	 */
   104  	public var comparator:IComparator ;
   105  	
   106  	/**
   107  	 * A String who represents a property name to compare the two objects.
   108  	 */
   109  	public var sortBy:String;
   110  
   111  	/**
   112  	 * Returns an integer value to compare two objects.
   113  	 * @param o1 the first Number object to compare.
   114  	 * @param o2 the second Number object to compare.
   115  	 * @return <p>
   116  	 * <li>-1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
   117  	 * <li> 1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
   118  	 * <li> 0 if o1 and o2 are equal.</li>
   119  	 * </p>
   120  	 * @throws NullPointerError
   121  	 */
   122  	public function compare(o1, o2):Number 
   123  	{
   124  		if ( o1 != null && o2 != null ) 
   125  		{
   126  			if ( comparator == null )
   127  			{
   128  				throw new NullPointerError( "GenericComparator failed, the 'comparator' argument not must be 'null' or 'undefined'.") ;
   129  			}
   130  			else
   131  			{
   132  				return comparator.compare( o1[sortBy] , o2[sortBy] ) ;
   133  			}
   134  		}
   135  		else 
   136  		{
   137  			throw new NullPointerError( "GenericComparator compare method failed, The two arguments not must be null.") ;
   138  		}
   139  	}
   140  	
   141  	/**
   142  	 * Returns the singleton instance of a GenericComparator.
   143  	 * Developers are encouraged to use the comparator returned from this method instead of constructing a new instance to reduce allocation and GC overhead when multiple comparable comparators may be used in the same application.
   144   	 * @return the singleton instance of a GenericComparator.
   145  	 */
   146  	public static function getInstance():GenericComparator
   147  	{
   148  		if (_instance == null)
   149  		{
   150  			_instance = new GenericComparator() ;
   151  		}
   152  		return _instance ;	
   153  	}
   154  
   155  	/**
   156    	 * The internal static singleton of this class.
   157   	 */
   158  	private static var _instance:GenericComparator ;
   159  
   160  }
