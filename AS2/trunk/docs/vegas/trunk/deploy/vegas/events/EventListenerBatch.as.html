    24  import vegas.data.collections.SimpleCollection;
    25  import vegas.data.collections.TypedCollection;
    26  import vegas.data.iterator.Iterator;
    27  import vegas.events.Event;
    28  import vegas.events.EventListener;
    29  
    30  /**
    31   * It handles several {@code EventListener} as one {@code EventListener}.
    32   * <p><b>Example : </b></p>
    33   * {@code
    34   * 	
    35   * import vegas.events.* ;
    36   * 
    37   * var EVENT_TYPE:String = "onTest" ;
    38   * 
    39   * var action1:Function = function (e:Event):Void 
    40   * {
    41   *    trace ("> action1 : " + e.getType()) ;
    42   * }
    43   * 
    44   * var action2:Function = function (e:Event):Void 
    45   * {
    46   *     trace ("> action2 : " + e.getType()) ;
    47   * }
    48   * 
    49   * var oListener1:EventListener = new Delegate(this, action1) ;
    50   * var oListener2:EventListener = new Delegate(this, action2) ;
    51   * 
    52   * var batch:EventListenerBatch = new EventListenerBatch() ;
    53   * batch.insert(oListener1) ;
    54   * batch.insert(oListener2) ;
    55   * 	
    56   * var e:Event = new BasicEvent( EVENT_TYPE , this ) ;
    57   * 	
    58   * EventDispatcher.getInstance().addEventListener(EVENT_TYPE, batch) ;
    59   * EventDispatcher.getInstance().dispatchEvent( e ) ;
    60   * }
    61   */
    62  class vegas.events.EventListenerBatch extends TypedCollection implements EventListener 
    63  {
    64  
    65  	/**
    66  	 * Creates a new EventListenerBatch instance.
    67  	 */
    68   	public function EventListenerBatch() 
    69   	{
    70  		super(EventListener, new SimpleCollection()) ;
    71    	}
    72  
    73  	/**
    74  	 * Returns a shallow copy of this instance.
    75  	 * @return a shallow copy of this instance.
    76  	 */	
    77  	public function clone() 
    78  	{
    79  		var b:EventListenerBatch = new EventListenerBatch() ;
    80  		var it:Iterator = iterator() ;
    81  		while (it.hasNext()) 
    82  		{
    83  			b.insert(it.next()) ;
    84  		}
    85  		return b ;
    86  	}
    87  	
    88  	/**
    89  	 * Handles the event.
    90  	 */
    91  	public function handleEvent(e:Event) 
    92  	{
    93  		var ar:Array = toArray() ;
    94  		var i:Number = -1 ;
    95  		var l:Number = ar.length ;
    96  		if (l>0) while (++i < l) 
    97  		{ 
    98  			ar[i].handleEvent.apply(ar[i], arguments) ; 
    99  		}
   100    	}
   101  	
   102  }
