    42  /**
    43   * Stores the listeners object an notifies them with the DOM Events level 2/3 of the W3C.
    44   * The EventDispatcher class implements the IEventDispatcher interface. 
    45   * This object allows any object to be an {@code EventTarget}.
    46   * <p><b>Thanks</b>:</p>
    47   * <p>{@code EventDispatcher} is an AS2 port of the <b>Java.schst.net EventDispatcher</b>. Inspired by the NotificationCenter of Apple's Cocoa-Framework.
    48   * <li>EventDispatcher JAVA : Stephan Schmid - http://schst.net/</li><li>Cocoa-Framework : http://developer.apple.com/cocoa/</li><li>Notification center : http://developer.apple.com/documentation/Cocoa/Conceptual/Notifications/index.html</li>
    49   * </p>
    50   * @author eKameleon

   111  	/**
   112  	 * Allows the registration of event listeners on the event target.
   113  	 * @param eventName A string representing the event type to listen for. If eventName value is "ALL" addEventListener use addGlobalListener
   114  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the {@code EventListener} interface.
   115  	 * @param useCapture Determinates if the event flow use capture or not.
   116  	 * @param priority Determines the priority level of the event listener.
   117  	 * @param autoRemove Apply a removeEventListener after the first trigger
   118  	 */
   119  	public function addEventListener( eventName:String, listener:EventListener, useCapture:Boolean, priority:Number, autoRemove:Boolean):Void 
   120  	{
   121  		
   122  		priority = isNaN(priority) ? 0 : priority ;
   123  		
   124  		if (eventName == "ALL") 
   125  		{
   126  			addGlobalEventListener(listener, priority, autoRemove) ;
   127  		}
   128  		else 
   129  		{
   130  			var map:HashMap = (!useCapture) ? _listeners : _captures ;
   131  			if (!map.containsKey(eventName)) 
   132  			{
   133  				map.put(eventName, new EventListenerCollection()) ;
   134  			}
   135  			var col:EventListenerCollection = map.get(eventName) ;
   136  			col.addListener(listener, autoRemove, priority) ;	
   137  			_dispatchQueuedEvents() ;
   138  		}
   139  	}

   141  	/**
   142  	 * Allows the registration of global event listeners on the event target.
   143  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the {@code EventListener} interface.
   144  	 * @param priority Determines the priority level of the event listener.
   145  	 * @param autoRemove Apply a removeEventListener after the first trigger
   146  	 */
   147  	public function addGlobalEventListener(listener:EventListener, priority:Number, autoRemove:Boolean):Void 
   148  	{
   149  		_globalListeners.addListener(listener, autoRemove, priority ) ;
   150          _dispatchQueuedEvents() ;

   153  	/**
   154  	 * Dispatches an event into the event flow.
   155  	 * @param event The Event object that is dispatched into the event flow.
   156  	 * @param isQueue if the EventDispatcher isn't register to the event type the event is bufferized.
   157  	 * @param target the target of the event.
   158  	 * @param contect the context of the event.
   159  	 * @return the reference of the event dispatched in the event flow.
   160  	 */
   161  	public function dispatchEvent( event , isQueue:Boolean, target, context):Event 
   162  	{
   163  		if (!event) 
   164  		{
   165  			return null ;
   166  		}
   167  		var e:Event = EventFactory.create(event, target || this, context) ;
   168  		if (e == null) 
   169  		{
   170  			return null ;
   171  		}
   172  		var phase:Number = e.getEventPhase() ;
   173  		if (phase == EventPhase.AT_TARGET) 
   174  		{
   175  			_capture(e) ; // CAPTURING_PHASE
   176  			e.setEventPhase(EventPhase.AT_TARGET) ;
   177  			e.setCurrentTarget(this) ;
   178  			_propagate(e, isQueue || false ) ; // AT_TARGET
   179  			_bubble(e) ; // BUBBLING_PHASE
   180  			e.setEventPhase(EventPhase.AT_TARGET) ; // TODO inclure cette initialisation dans initEvent ??
   181  		}
   182  		else if (phase == EventPhase.BUBBLING_PHASE) 
   183  		{
   184  			_propagateBubble(e) ;
   185  		}
   186  		else if (phase == EventPhase.CAPTURING_PHASE) 
   187  		{
   188  			_propagateCapture(e) ;
   189  		}
   190  		return e ;
   191  	}

   297  	/** 
   298  	 * Removes a listener from the EventDispatcher object.
   299  	 * If there is no matching listener registered with the EventDispatcher object, then calling this method has no effect.
   300  	 * @param Specifies the type of event.
   301  	 * @param the class name(string) or a EventListener object.
   302  	 */
   303      public function removeEventListener(eventName:String, listener , useCapture:Boolean):EventListener 
   304      {
   305  		if (eventName == "ALL") 
   306  		{
   307  			return removeGlobalEventListener(listener) ;
   308  		}
   309  		if (listener instanceof EventListener || TypeUtil.typesMatch(listener, String))  
   310  		{
   311  			var map:HashMap = (!useCapture) ? _listeners : _captures ;
   312  			if (!map.containsKey(eventName) ) 
   313  			{
   314  				return null ;
   315  			}
   316  			var col:EventListenerCollection = map.get(eventName) ;
   317  			var container:EventListenerContainer = col.removeListener(listener) ;
   318  			if (container != null) 
   319  			{
   320  				return container.getListener() ;
   321  			}
   322  		}
   323          return null ;
   324      }

   326  	/** 
   327  	 * Removes a global listener from the EventDispatcher object.
   328  	 * If there is no matching listener registered with the EventDispatcher object, then calling this method has no effect.
   329  	 * @param the class name(string) or a EventListener object.
   330  	 */
   331  	public function removeGlobalEventListener( listener ):EventListener 
   332  	{
   333  		if (listener instanceof EventListener || typeof(listener) == "string") 
   334  		{
   335  			var container:EventListenerContainer = EventListenerContainer(_globalListeners.removeListener(listener));
   336  			if (container != null) 	
   337  			{
   338  				return container.getListener();
   339  			}
   340  			return null;
   341  		}
   342  	}

   394  	private function _bubble(e:Event):Boolean 
   395  	{
   396  		if (e["stop"] >= EventPhase.STOP ) return false ;
   397  		var parents:Array = _getParents() ;
   398  		if (parents != null) 
   399  		{
   400  			var l:Number = parents.length ;
   401  			var i:Number = 0 ;
   402  			var current:EventDispatcher ;
   403  			while (i<l) 
   404  			{
   405  				if (e.getBubbles()) 
   406  				{
   407  					current = parents[i] ;
   408  					e.setCurrentTarget(current.getTarget()) ;
   409  					e.setEventPhase(EventPhase.BUBBLING_PHASE) ;
   410  					current.dispatchEvent(e) ;
   411  					if (e["stop"] >= EventPhase.STOP ) 
   412  					{
   413  						return false ;
   414  					}
   415  				}
   416  				i++ ;
   417  			}
   418  		}
   419  		return true ;
   420  	}
   421  	
   422  	private function _capture(e:Event):Boolean 
   423  	{
   424  		var parents:Array = _getParents() ;
   425  		if (parents != null) 
   426  		{
   427  			var l:Number = parents.length ;
   428  			var current:EventDispatcher ;
   429  			while (--l > -1) 
   430  			{
   431  				current = parents[l] ;
   432  				e.setCurrentTarget(current.getTarget()) ;
   433  				e.setEventPhase(EventPhase.CAPTURING_PHASE) ;
   434  				current.dispatchEvent(e) ;
   435  				if (e["stop"] >= EventPhase.STOP ) 
   436  				{
   437  					return false ;
   438  				}
   439  			}
   440  		}
   441  		return true ;
   442  	}
   443  	
   444  	private function _dispatchQueuedEvents():Void 
   445  	{
   446  		var q:Queue = _queue.getQueuedEvents();
   447  		if (q.size() > 0) 
   448  		{
   449  			var ar:Array = q.toArray() ;
   450  			var len:Number = ar.length ;
   451  			for (var i:Number = 0 ; i<len ; i++) 
   452  			{
   453  				var e:Event = ar[i] ;
   454  				dispatchEvent(e, e.isQueued()) ;
   455  			}
   456  		}
   457  	}
   458  
   459  	private function _getParents():Array 
   460  	{
   461  		if (parent == null) 
   462  		{
   463  			return null ;
   464  		}
   465  		var ar:Array = [] ;
   466  		var tmp:EventDispatcher = parent ;
   467  		while(tmp != null) 
   468  		{
   469  			ar.push(tmp) ;
   470  			tmp = tmp.parent ;
   471  		}
   472  		return ar ;
   473  	}
   474  
   475  	private function _propagate(e:Event, isQueue:Boolean):Event 
   476  	{
   477  		
   478  		if (e["stop"] >= EventPhase.STOP ) 
   479  		{
   480  			return e ; // hack the interface limitation
   481  		}
   482  		
   483  		if (_listeners.containsKey(e.getType())) 
   484  		{
   485              var col:EventListenerCollection = EventListenerCollection(_listeners.get(e.getType())) ;
   486              col.propagate(e) ;
   487          }
   488          
   489          if (e.isCancelled()) 
   490          {
   491          	return e ;
   492          }
   493          
   494          _globalListeners.propagate(e);
   495          
   496          if (isQueue == false || e.isCancelled()) 
   497          {
   498          	return e ;
   499          }
   500          
   501          _queue.enqueue(e) ;
   502          
   503          return e ;
   504          
   505  	}
   506  
   507  	private function _propagateBubble(e:Event):Void 
   508  	{
   509  		if (e.getEventPhase(EventPhase.BUBBLING_PHASE)) 
   510  		{
   511  			e.setCurrentTarget(getTarget()) ;
   512  			_propagate(e) ;
   513  		}
   514  	}
   515  	
   516  	private function _propagateCapture(e:Event):Void 
   517  	{
   518  		if (_captures.containsKey(e.getType())) 
   519  		{
   520  			e.getEventPhase(EventPhase.CAPTURING_PHASE) ;
   521              var col:EventListenerCollection = EventListenerCollection(_captures.get(e.getType())) ;
   522              col.propagate(e) ;
   523          }
   524  	}
