    24  import vegas.core.CoreObject;
    25  import vegas.data.Set;
    26  import vegas.events.Event;
    27  import vegas.events.EventDispatcher;
    28  import vegas.events.EventListener;
    29  import vegas.events.EventListenerCollection;
    30  import vegas.events.IEventDispatcher;
    31  
    32  // TODO test bubbling/capturing
    33  
    34  /**
    35   * This abstract class is used to create concrete {@code IEventDispatcher} implementations. This class used an internal {@code EventDispatcher} object by composition.
    36   * <p>You can overrides the internal {@code EventDispatcher} instance with the {@code initEventDispatcher} method. Used a global singleton reference in this method to register all events in a {@code FrontController} for example.</p>
    37   * @author eKameleon
    38   */
    39  class vegas.events.AbstractCoreEventDispatcher extends CoreObject implements IEventDispatcher
    40  {
    41  
    42  	/**
    43  	 * Creates a new AbstractCoreEventDispatcher.
    44  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    45  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    46  	 */
    47  	private function AbstractCoreEventDispatcher( bGlobal:Boolean , sChannel:String  ) 
    48  	{
    49  		_oED = initEventDispatcher() ;
    50  		setGlobal( bGlobal , sChannel ) ;	
    51  	}		
    52  
    53  	/**
    54  	 * (read-only) Returns the value of the isGlobal flag of this model. Use the {@code setGlobal} method to modify this value.
    55  	 * @return {@code true} if the instance use a global EventDispatcher to dispatch this events.
    56  	 */
    57  	public function get isGlobal():Boolean 
    58  	{
    59  		return getIsGlobal() ;
    60  	}
    61  
    62  	/**
    63  	 * Allows the registration of event listeners on the event target.
    64  	 * @param eventName A string representing the event type to listen for. If eventName value is "ALL" addEventListener use addGlobalListener
    65  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the {@code EventListener} interface.
    66  	 * @param useCapture Determinates if the event flow use capture or not.
    67  	 * @param priority Determines the priority level of the event listener.
    68  	 * @param autoRemove Apply a removeEventListener after the first trigger
    69  	 */
    70  	public function addEventListener( eventName:String, listener:EventListener, useCapture:Boolean, priority:Number, autoRemove:Boolean):Void 
    71  	{
    72  		_oED.addEventListener.apply(_oED, arguments) ;
    73  	}
    74  
    75  	/**
    76  	 * Allows the registration of global event listeners on the event target.
    77  	 * 
    78  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the <b>EventListener</b> interface.
    79  	 * @param priority Determines the priority level of the event listener.
    80  	 * @param autoRemove Apply a removeEventListener after the first trigger
    81  	 */
    82  	public function addGlobalEventListener(listener:EventListener, priority:Number, autoRemove:Boolean):Void 
    83  	{
    84  		_oED.addGlobalEventListener.apply(_oED, arguments) ;
    85  	}		
    86  
    87  	/**
    88  	 * Dispatches an event into the event flow.
    89  	 * @param event The Event object that is dispatched into the event flow.
    90  	 * @param isQueue if the EventDispatcher isn't register to the event type the event is bufferized.
    91  	 * @param target the target of the event.
    92  	 * @param contect the context of the event.
    93  	 * @return the reference of the event dispatched in the event flow.
    94  	 */
    95  	public function dispatchEvent(event, isQueue:Boolean, target, context):Event 
    96  	{
    97  		return _oED.dispatchEvent.apply(_oED, arguments) ;	
    98  	}
    99  
   100  	/**
   101  	 * Returns the internal {@code EventDispatcher} reference.
   102  	 * @return the internal {@code EventDispatcher} reference.
   103  	 */
   104  	public function getEventDispatcher():EventDispatcher 
   105  	{
   106  		return _oED ;
   107  	}
   108  
   109  	/**
   110  	 * Returns the {@code EventListenerCollection} of the specified event name.
   111  	 * @return the {@code EventListenerCollection} of the specified event name.
   112  	 */
   113  	public function getEventListeners(eventName:String):EventListenerCollection 
   114  	{
   115  		return _oED.getEventListeners.apply(_oED, arguments) ;
   116  	}
   117  
   118  	/**
   119  	 * Returns the {@code EventListenerCollection} of this EventDispatcher.
   120  	 * @return the {@code EventListenerCollection} of this EventDispatcher.
   121  	 */
   122  	public function getGlobalEventListeners():EventListenerCollection 
   123  	{
   124  		return _oED.getGlobalEventListeners.apply(_oED, arguments) ;
   125  	}
   126  	
   127  	/**
   128  	 * Returns the value of the isGlobal flag of this model.
   129  	 * @return {@code true} if the instance use a global EventDispatcher to dispatch this events.
   130  	 */
   131  	public function getIsGlobal():Boolean 
   132  	{
   133  		return _isGlobal ;
   134  	}
   135  
   136  	/**
   137  	 * Returns the EventDispatcher reference of the parent of this instance.
   138  	 * @return the EventDispatcher reference of the parent of this instance.
   139  	 */
   140  	public function getParent():EventDispatcher 
   141  	{
   142  		return _oED.parent ;
   143  	}
   144  
   145  	/**
   146  	 * Returns a {@code Set} of all register event's name in this EventListener.
   147  	 * @return a {@code Set} of all register event's name in this EventListener.
   148  	 */
   149  	public function getRegisteredEventNames():Set 
   150  	{
   151  		return _oED.getRegisteredEventNames.apply(_oED, arguments) ;
   152  	}
   153  
   154  	/**
   155  	 * Returns the target of this instance.
   156  	 * @return the target of this instance.
   157  	 */
   158  	public function getTarget() 
   159  	{
   160  		return _oED.getTarget() ;	
   161  	}
   162  
   163  	/**
   164  	 * Checks whether the EventDispatcher object has any listeners registered for a specific type of event.
   165  	 * This allows you to determine where altered handling of an event type has been introduced in the event flow heirarchy by an EventDispatcher object.
   166  	 */ 
   167  	public function hasEventListener(eventName:String):Boolean 
   168  	{
   169  		return _oED.hasEventListener.apply(_oED, arguments) ;
   170  	}
   171  
   172  	/**
   173  	 * Creates and returns the internal {@code EventDispatcher} reference (this method is invoqued in the constructor).
   174  	 * You can overrides this method if you wan use a global {@code EventDispatcher} singleton.
   175  	 * @return the internal {@code EventDispatcher} reference.
   176  	 */
   177  	public function initEventDispatcher():EventDispatcher 
   178  	{
   179  		return new EventDispatcher( this ) ;
   180  	}
   181  
   182  	/** 
   183  	 * Removes a listener from the EventDispatcher object.
   184  	 * If there is no matching listener registered with the {@code EventDispatcher} object, then calling this method has no effect.
   185  	 * @param Specifies the type of event.
   186  	 * @param the class name(string) or a {@code EventListener} object.
   187  	 */
   188  	public function removeEventListener(eventName:String, listener, useCapture:Boolean):EventListener 
   189  	{
   190  		return _oED.removeEventListener.apply(_oED, arguments) ;
   191  	}
   192  
   193  	/** 
   194  	 * Removes a global listener from the EventDispatcher object.
   195  	 * If there is no matching listener registered with the EventDispatcher object, then calling this method has no effect.
   196  	 * @param the string representation of the class name of the EventListener or a EventListener object.
   197  	 */
   198  	public function removeGlobalEventListener( listener ):EventListener 
   199  	{
   200  		return _oED.removeGlobalEventListener.apply(_oED, arguments) ;
   201  	}
   202  
   203  	/**
   204  	 * Sets the internal {@code EventDispatcher} reference.
   205  	 */
   206  	public function setEventDispatcher( e:EventDispatcher ):Void 
   207  	{
   208  		_oED = e || initEventDispatcher() ;
   209  	}
   210  
   211  	/**
   212  	 * Sets if the instance use a global {@code EventDispatcher} to dispatch this events, if the {@code flag} value is {@code false} the instance use a local EventDispatcher.
   213  	 * @param flag the flag to use a global event flow or a local event flow.
   214  	 * @param channel the name of the global event flow if the {@code flag} argument is {@code true}.  
   215  	 */
   216  	public function setGlobal( flag:Boolean , channel:String ):Void 
   217  	{
   218  		_isGlobal = flag ;
   219  		setEventDispatcher( flag ? EventDispatcher.getInstance( channel ) : null ) ;
   220  	}
   221  
   222  	/**
   223  	 * Sets the parent EventDispatcher reference of this instance.
   224  	 */
   225  	public function setParent( parent:EventDispatcher ):Void 
   226  	{
   227  		_oED.parent = parent ;
   228  	}
   229  
   230  	/**
   231  	 * The protected internal EventDispatcher reference.
   232  	 */
   233  	private var _oED:EventDispatcher ;
   234  
   235  	/**
   236  	 * The internal flag to indicate if the event flow is global.
   237  	 */
   238  	private var _isGlobal:Boolean ;
   239  	
   240  }
