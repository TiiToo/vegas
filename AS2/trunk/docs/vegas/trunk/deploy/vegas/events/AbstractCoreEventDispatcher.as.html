    24  import vegas.core.CoreObject;
    25  import vegas.data.Set;
    26  import vegas.events.Event;
    27  import vegas.events.EventDispatcher;
    28  import vegas.events.EventListener;
    29  import vegas.events.EventListenerCollection;
    30  import vegas.events.IEventDispatcher;
    31  
    32  // TODO test bubbling/capturing
    33  
    34  /**
    35   * This abstract class is used to create concrete {@code IEventDispatcher} implementations. This class used an internal {@code EventDispatcher} object by composition.
    36   * <p>You can overrides the internal {@code EventDispatcher} instance with the {@code initEventDispatcher} method. Used a global singleton reference in this method to register all events in a {@code FrontController} for example.</p>
    37   * @author eKameleon
    38   */
    39  class vegas.events.AbstractCoreEventDispatcher extends CoreObject implements IEventDispatcher
    40  {
    41  
    42  	/**
    43  	 * Creates a new AbstractCoreEventDispatcher.
    44  	 */
    45  	private function AbstractCoreEventDispatcher() 
    46  	{
    47  		super() ;
    48  		_oED = initEventDispatcher() ;
    49  	}		
    50  
    51  	/**
    52  	 * Allows the registration of event listeners on the event target.
    53  	 * @param eventName A string representing the event type to listen for. If eventName value is "ALL" addEventListener use addGlobalListener
    54  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the {@code EventListener} interface.
    55  	 * @param useCapture Determinates if the event flow use capture or not.
    56  	 * @param priority Determines the priority level of the event listener.
    57  	 * @param autoRemove Apply a removeEventListener after the first trigger
    58  	 */
    59  	public function addEventListener( eventName:String, listener:EventListener, useCapture:Boolean, priority:Number, autoRemove:Boolean):Void 
    60  	{
    61  		_oED.addEventListener.apply(_oED, arguments) ;
    62  	}
    63  
    64  	/**
    65  	 * Allows the registration of global event listeners on the event target.
    66  	 * 
    67  	 * @param listener The object that receives a notification when an event of the specified type occurs. This must be an object implementing the <b>EventListener</b> interface.
    68  	 * @param priority Determines the priority level of the event listener.
    69  	 * @param autoRemove Apply a removeEventListener after the first trigger
    70  	 */
    71  	public function addGlobalEventListener(listener:EventListener, priority:Number, autoRemove:Boolean):Void 
    72  	{
    73  		_oED.addGlobalEventListener.apply(_oED, arguments) ;
    74  	}		
    75  
    76  	/**
    77  	 * Dispatches an event into the event flow.
    78  	 * @param event The Event object that is dispatched into the event flow.
    79  	 * @param isQueue if the EventDispatcher isn't register to the event type the event is bufferized.
    80  	 * @param target the target of the event.
    81  	 * @param contect the context of the event.
    82  	 * @return the reference of the event dispatched in the event flow.
    83  	 */
    84  	public function dispatchEvent(event, isQueue:Boolean, target, context):Event 
    85  	{
    86  		return _oED.dispatchEvent.apply(_oED, arguments) ;	
    87  	}
    88  
    89  	/**
    90  	 * Returns the internal {@code EventDispatcher} reference.
    91  	 * @return the internal {@code EventDispatcher} reference.
    92  	 */
    93  	public function getEventDispatcher():EventDispatcher 
    94  	{
    95  		return _oED ;
    96  	}
    97  
    98  	/**
    99  	 * Returns the {@code EventListenerCollection} of the specified event name.
   100  	 * @return the {@code EventListenerCollection} of the specified event name.
   101  	 */
   102  	public function getEventListeners(eventName:String):EventListenerCollection 
   103  	{
   104  		return _oED.getEventListeners.apply(_oED, arguments) ;
   105  	}
   106  
   107  	/**
   108  	 * Returns the {@code EventListenerCollection} of this EventDispatcher.
   109  	 * @return the {@code EventListenerCollection} of this EventDispatcher.
   110  	 */
   111  	public function getGlobalEventListeners():EventListenerCollection 
   112  	{
   113  		return _oED.getGlobalEventListeners.apply(_oED, arguments) ;
   114  	}
   115  
   116  	/**
   117  	 * Returns the EventDispatcher reference of the parent of this instance.
   118  	 * @return the EventDispatcher reference of the parent of this instance.
   119  	 */
   120  	public function getParent():EventDispatcher 
   121  	{
   122  		return _oED.parent ;
   123  	}
   124  
   125  	/**
   126  	 * Returns a {@code Set} of all register event's name in this EventListener.
   127  	 * @return a {@code Set} of all register event's name in this EventListener.
   128  	 */
   129  	public function getRegisteredEventNames():Set 
   130  	{
   131  		return _oED.getRegisteredEventNames.apply(_oED, arguments) ;
   132  	}
   133  
   134  	/**
   135  	 * Returns the target of this instance.
   136  	 * @return the target of this instance.
   137  	 */
   138  	public function getTarget() 
   139  	{
   140  		return _oED.getTarget() ;	
   141  	}
   142  
   143  	/**
   144  	 * Checks whether the EventDispatcher object has any listeners registered for a specific type of event.
   145  	 * This allows you to determine where altered handling of an event type has been introduced in the event flow heirarchy by an EventDispatcher object.
   146  	 */ 
   147  	public function hasEventListener(eventName:String):Boolean 
   148  	{
   149  		return _oED.hasEventListener.apply(_oED, arguments) ;
   150  	}
   151  
   152  	/**
   153  	 * Creates and returns the internal {@code EventDispatcher} reference (this method is invoqued in the constructor).
   154  	 * You can overrides this method if you wan use a global {@code EventDispatcher} singleton.
   155  	 * @return the internal {@code EventDispatcher} reference.
   156  	 */
   157  	public function initEventDispatcher():EventDispatcher 
   158  	{
   159  		return new EventDispatcher( this ) ;
   160  	}
   161  
   162  	/** 
   163  	 * Removes a listener from the EventDispatcher object.
   164  	 * If there is no matching listener registered with the {@code EventDispatcher} object, then calling this method has no effect.
   165  	 * @param Specifies the type of event.
   166  	 * @param the class name(string) or a {@code EventListener} object.
   167  	 */
   168  	public function removeEventListener(eventName:String, listener, useCapture:Boolean):EventListener 
   169  	{
   170  		return _oED.removeEventListener.apply(_oED, arguments) ;
   171  	}
   172  
   173  	/** 
   174  	 * Removes a global listener from the EventDispatcher object.
   175  	 * If there is no matching listener registered with the EventDispatcher object, then calling this method has no effect.
   176  	 * @param the string representation of the class name of the EventListener or a EventListener object.
   177  	 */
   178  	public function removeGlobalEventListener( listener ):EventListener 
   179  	{
   180  		return _oED.removeGlobalEventListener.apply(_oED, arguments) ;
   181  	}
   182  
   183  	/**
   184  	 * Sets the internal {@code EventDispatcher} reference.
   185  	 */
   186  	public function setEventDispatcher( e:EventDispatcher ):Void 
   187  	{
   188  		_oED = e || initEventDispatcher() ;
   189  	}
   190  
   191  	/**
   192  	 * Sets the parent EventDispatcher reference of this instance.
   193  	 */
   194  	public function setParent( parent:EventDispatcher ):Void 
   195  	{
   196  		_oED.parent = parent ;
   197  	}
   198  
   199  	/**
   200  	 * The protected internal EventDispatcher reference.
   201  	 */
   202  	private var _oED:EventDispatcher ;
   203  
   204  }
