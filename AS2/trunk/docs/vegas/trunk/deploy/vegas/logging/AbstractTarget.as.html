    39  /**
    40   * This class provides the basic functionality required by the logging framework for a target implementation. It handles the validation of filter expressions and provides a default level property. No implementation of the logEvent() method is provided.
    41   * @author eKameleon
    42   */
    43  class vegas.logging.AbstractTarget extends CoreObject implements EventListener, ITarget 
    44  {
    45  
    46  	/**
    47  	 * Creates a new AbstractTarget instance.
    48  	 */
    49  	private function AbstractTarget()
    50  	{
    51  		_level = LogEventLevel.ALL ;
    52  		_filters = [ "*" ] ;
    53  		_set = new HashSet() ;
    54  	}
    55  
    56  	/**
    57  	 * The static field used when throws an Error when a character is invalid.
    58  	 */        
    59  	static public var CHARS_INVALID:String = "The following characters are not valid : []~$^&/\\(){}<>+=`!#%?,:;'\"@" ;
    60  
    61  	/**
    62  	 * The static field used when throws an Error when filter failed.
    63  	 */        
    64  	static public var ERROR_FILTER:String = "Error for filter ''{0}'" ;
    65          
    66  	/**
    67  	 * The static field used when throws an Error when the character placement failed.
    68  	 */        
    69  	static public var CHAR_PLACEMENT:String = "'*' must be the right most character." ;
    70  
    71  	/**
    72  	 * In addition to the level setting, filters are used to provide a pseudo hierarchical mapping for processing only those events for a given category.
    73  	 */
    74  	public function get filters():Array 
    75  	{
    76  		return _filters ;	
    77  	}
    78  	
    79  	/**
    80  	 * (read-write) Sets the filters array of this target.
    81  	 * If you use this virtual property the target is register in the {@code Log} singleton automatically.
    82  	 */
    83  	public function set filters( value:Array ):Void
    84  	{
    85  		
    86  		if ( value != null && value.length > 0 )
    87  		{
    88  			var filter:String ;
    89  			var index:Number ;
    90  			var len:Number = value.length ;
    91  			for (var i:Number = 0; i<len ; i++)
    92  			{
    93  				
    94  				filter = value[i] ;
    95  				// check for invalid characters
    96  				if ( Log.hasIllegalCharacters(filter) )
    97  				{
    98  					throw new InvalidFilterError( (new StringFormatter(ERROR_FILTER)).format(filter) + CHARS_INVALID ) ;
    99  				}
   100  				index = filter.indexOf("*") ;
   101   				if ((index >= 0) && (index != (filter.length -1)))
   102                  {                        
   103  					throw new InvalidFilterError( (new StringFormatter(ERROR_FILTER)).format( filter) + CHAR_PLACEMENT ) ;
   104  				}
   105  			}
   106  		}
   107  		else
   108  		{
   109  			value = ["*"] ; // if null was specified then default to all
   110  		}
   111  
   112  		if ( _set.size() > 0 )
   113  		{
   114  			Log.removeTarget(this);
   115  			_filters = value;
   116  			Log.addTarget(this);
   117  		}
   118  		else
   119  		{
   120  			_filters = value;
   121  		}
   122  	}
   123          
   124  	/**
   125  	 * Provides access to the level this target is currently set at.
   126  	 */
   127  	public function get level():Number
   128  	{
   129  		return _level ;	
   130  	}
   131  
   132  	/**
   133  	 * (read-write) Sets the level of this target.
   134  	 * If you use this virtual property the target is register in the {@code Log} singleton automatically.
   135  	 */
   136  	public function set level( value:Number ):Void
   137  	{
   138  		Log.removeTarget(this) ;
   139  		_level = value ;
   140  		Log.addTarget(this) ;     
   141  	} 
   142  
   143  	/**
   144  	 * Inserts a category in the fllters if this category don't exist.
   145  	 * Returns a boolean if the category is add in the list.
   146  	 */
   147  	public function addCategory( category:String ):Boolean 
   148  	{
   149  		if ( _filters.indexOf( category ) == -1 ) 
   150  		{
   151  			return false ;
   152  		}
   153  		_filters.push( category ) ;
   154  		return true ;
   155  	}
   156  
   157  	/**
   158  	 * Sets up this target with the specified logger.
   159  	 * Note : this method is called by the framework and should not be called by the developer.
   160  	 */
   161  	public function addLogger( logger:ILogger ):Void 
   162  	{
   163  		if ( !_set.contains(logger) )
   164  		{
   165  			_set.insert(logger) ;
   166  			logger.addEventListener( LogEvent.LOG, this ) ;
   167  		}
   168  	}
   169  
   170  	/**
   171  	 * Add a new namespace in the filters array.
   172  	 */
   173  	public function addNamespace(namespace:String):Boolean 
   174  	{
   175  		if (filters == undefined) 
   176  		{
   177  			filters = [] ;
   178  		}
   179  		if ( ArrayUtil.contains(filters, namespace) ) 
   180  		{
   181  			return false ;
   182  		}
   183  		else
   184  		{
   185  			filters.push(namespace) ;
   186  			return true ;
   187  		}
   188  	}
   189  
   190  	/**
   191  	 * This method is called whenever an event occurs of the type for which the EventListener interface was registered.
   192  	 */
   193  	public function handleEvent(e:Event) 
   194  	{
   195  		var event:LogEvent = LogEvent(e) ; 
   196  		if (  event.level >= level )
   197  		{
   198  			var category:String = LogLogger(event.getTarget()).category ;
   199  			var isValid:Boolean = _isValidCategory( category ) ;
   200  			if (isValid)
   201  			{
   202  				logEvent( event ) ;
   203  			}
   204  		}
   205  	}
   206  
   207      /**
   208       *  This method handles a <code>LogEvent</code> from an associated logger.
   209       *  <p>A target uses this method to translate the event into the appropriate format for transmission, storage, or display.</p>
   210       *  <p>This method will be called only if the event's level is in range of the target's level.</p>
   211       *  <p><b><i>Descendants need to override this method to make it useful.</i></b></p>
   212       */		
   213  	public function logEvent(e:LogEvent):Void 
   214  	{
   215  		//
   216  	}
   217  
   218  	/**
   219  	 * Removes a category in the fllters if this category exist.
   220  	 * Returns a boolean if the category is remove.
   221  	 */
   222  	public function removeCategory( category:String ):Boolean
   223  	{
   224  		var pos:Number = _filters.indexOf( category ) ;
   225  		if ( pos > -1) 
   226  		{
   227  			_filters.splice(pos, 1) ;
   228  			return true ;
   229  		}
   230  		else 
   231  		{
   232  			return false ;
   233  		}
   234  	}
   235  
   236  	/**
   237  	 * Stops this target from receiving events from the specified logger.
   238  	 * Note : this method is called by the framework and should not be called by the developer.
   239  	 */
   240  	public function removeLogger(logger:ILogger):Void 
   241  	{
   242  		if (_set.contains(logger))
   243  		{
   244  			_set.remove( logger ) ;
   245  			logger.removeEventListener(LogEvent.LOG, this) ;
   246  		}
   247  	}
   248  
   249  	/**
   250  	 * Removes an existing namespace in the filters array.
   251  	 */
   252  	public function removeNamespace(namespace:String):Boolean 
   253  	{
   254  		var pos:Number = ArrayUtil.indexOf(filters, namespace) ;
   255  		if ( pos > -1) 
   256  		{
   257  			filters.splice(pos, 1) ;
   258  			return true ;
   259  		} 
   260  		else 
   261  		{
   262  			return false ;
   263  		}
   264  	}
   265  	
   266  	/**
   267  	 * Returns 'true' is the passed argument is a valid category.
   268  	 */
   269  	private function _isValidCategory(category:String):Boolean 
   270  	{
   271  		
   272  		if ( category == Log.DEFAULT_CATEGORY ) 
   273  		{
   274  			return true ;
   275  		}
   276  		
   277  		var l:Number = filters.length ;
   278  		if (l > 0) 
   279  		{
   280  			for (var i:Number = 0 ; i<l ; i++) 
   281  			{
   282  				var pattern:String = filters[i] ;
   283  				var we:WildExp = new WildExp( pattern, WildExp.IGNORECASE | WildExp.MULTIWORD );
   284  				var result = we.test( category ) ;
   285  				if (result == true || pattern == category) 
   286  				{
   287  					return true ;
   288  				}
   289  			}
   290  			return false ;
   291  		} 
   292  		else 
   293  		{
   294  			return true ;
   295  		}
   296  	}
   297  		
   298  	/**
   299  	 * Storage for the filters property.
   300  	 */
   301  	private var _filters:Array ;
   302  	
   303  	/**
   304  	 * Storage for the filters property.
   305  	 */
   306  	private var _level:Number ;
   307  	
   308  	private var _set:HashSet ;
   309  
   310  }
