    24  import vegas.logging.LogEvent;
    25  import vegas.logging.targets.LineFormattedTarget;
    26  import vegas.util.MathsUtil;
    27  
    28  /**
    29   * Provides a logger target that uses the FlashInspector console to output log messages. 
    30   * Thanks Pablo Costantini and LuminicBox <a href='http://www.luminicbox.com/blog/default.aspx?page=post&id=2'>FlashInspector</a>.
    31   * <p><b>Example :</b></p>
    32   * <p>
    33   * {@code
    34   * import vegas.logging.* ;
    35   * import vegas.logging.targets.LuminicTarget ;
    36   * 
    37   * var target:LuminicTarget = new LuminicTarget(2, false) ;
    38   * target.includeLines = true ;
    39   * target.includeLevel = true ;
    40   * target.filters = ["__TEST__"] ;
    41   * target.level = LogEventLevel.ALL ; // level filter !
    42   * 
    43   * var logger:ILogger = Log.getLogger("__TEST__") ;
    44   * 
    45   * logger.debug("hello") ;
    46   * logger.warn(2) ;
    47   * 
    48   * logger.debug("------------") ;
    49   * target.isCollapse = true ;
    50   * logger.error([2, 3, ["coucou", "hello", "item"]]) ;
    51   * logger.debug({ prop1:"coucou" , prop2:[3, 4, 5] , prop3:true } ) ;
    52   * logger.debug(new XML("<item><subitem>coucou</subitem><subitem>hello</subitem></item>")) ;
    53   * }
    54   * </p>
    55   * @author eKameleon
    56   */
    57  class vegas.logging.targets.LuminicTarget extends LineFormattedTarget 
    58  {
    59  	
    60  	/**
    61  	 * Creates a new LuminicTarget instance.
    62  	 */
    63  	public function LuminicTarget(depth:Number, collapse:Boolean) 
    64  	{
    65  		_lc = new LocalConnection() ;
    66  		isCollapse = collapse ;
    67  		setMaxDepth( depth || 4 ) ;
    68  	}
    69  
    70  	/**
    71  	 * The id of the local connection.
    72  	 */
    73      public static var CONNECTION_ID:String = "_luminicbox_log_console" ;
    74         
    75      /**
    76       * The name of the dispatch message.
    77       */
    78      public static var DISPATCH_MESSAGE:String = "log" ;
    79  
    80  	/**
    81  	 * Indicated if the console use collapse property or not.
    82  	 */
    83  	public var isCollapse:Boolean ;
    84  	
    85  	/**
    86  	 * (read-only) Returns the max depth to collaspe the structure of an object in the console.
    87  	 */
    88  	public function get maxDepth():Number
    89  	{
    90  		return getMaxDepth() ;	
    91  	}
    92  
    93  	/**
    94  	 * (read-only) Sets the max depth to collaspe the structure of an object in the console.
    95  	 */
    96  	public function set maxDepth( value:Number ):Void
    97  	{
    98  		setMaxDepth(value) ;	
    99  	}
   100  
   101  	/**
   102  	 * Returns the max depth to collaspe the structure of an object in the console.
   103  	 */
   104  	public function getMaxDepth():Number 
   105  	{
   106  		return _maxDepth ;
   107  	}
   108  	
   109  	/**
   110       * Descendants of this class should override this method to direct the specified message to the desired output.
   111       *
   112       * @param message String containing preprocessed log message which may include time, date, category, etc. 
   113       *        based on property settings, such as <code>includeDate</code>, <code>includeCategory</code>, etc.
   114  	 */
   115  	/*override*/ public function internalLog( message , level:Number ):Void
   116  	{
   117  		_lc.send( CONNECTION_ID , DISPATCH_MESSAGE, message ) ;
   118  	}
   119  	
   120  	/**
   121  	 * This method handles a LogEvent from an associated logger.
   122  	 */
   123  	/*override*/ public function logEvent(e:LogEvent) 
   124  	{
   125  		var message = e.message ;
   126          var level:String = LogEvent.getLevelString(e.level) ;
   127  		var category:String = e.currentTarget.category ;
   128  		var time:Date = new Date() ;
   129  
   130  		var context:Object = 
   131  		{
   132  			loggerId     : null ,
   133  			levelName    : level ,
   134  			time : time
   135  		} ;
   136  	    	
   137  		if (isCollapse) 
   138  		{
   139  			context.argument = _serializeObj( message , 1) ;
   140  		} 
   141  		else 
   142  		{
   143  			var data:Object = 
   144  			{
   145  				type  : "string" ,
   146  				value : formatMessage( message, level, category, time)
   147  			} ;
   148  			context.argument = data ;
   149  		}
   150  		
   151  		internalLog( context ) ;
   152  		
   153  	}
   154  
   155  	/**
   156  	 * Sets the max depth to collaspe the structure of an object in the console.
   157  	 */
   158  	public function setMaxDepth(value:Number) 
   159  	{
   160  		_maxDepth = MathsUtil.clamp(value, 1, 255) ;
   161  	}
   162  
   163  	/**
   164  	 * Internal LocalConnection reference.
   165  	 */
   166  	private var _lc:LocalConnection ;
   167  	
   168  	/**
   169  	 * Internal max depth value.
   170  	 */
   171  	private var _maxDepth:Number ;
   172  
   173  	/**
   174  	 * Returns an object with all the config of the current log content.
   175  	 */
   176  	private function _getType(o):Object 
   177  	{
   178  		var tof:String = typeof(o);
   179  		var type = new Object();
   180  		type.inspectable = true ;
   181  		type.name = tof ;
   182  		switch( tof ) 
   183  		{
   184  			case "string" :
   185  			case "boolean" :
   186  			case "number" :
   187  			case "undefined" :
   188  			case "null" :
   189  			{
   190  				type.inspectable = false ;
   191  				break ;
   192  			}
   193  			default :
   194  			{	
   195  				if(o instanceof Date) 
   196  				{
   197  					type.inspectable = false ;
   198  					type.name = "date" ;
   199  				}
   200  				else if(o instanceof Array) 
   201  				{
   202  					type.name = "array" ;
   203  				}
   204  				else if(o instanceof Button) 
   205  				{
   206  					type.name = "button";
   207  				}
   208  				else if(o instanceof MovieClip) 
   209  				{
   210  					type.name = "movieclip";
   211  				}
   212  				else if(o instanceof XML) 
   213  				{
   214  					type.name = "xml" ;
   215  					type.stringify = true ;
   216  				}
   217  				else if(o instanceof XMLNode) 
   218  				{
   219  					type.name = "xmlnode" ;
   220  					type.stringify = true ;
   221  				}
   222  				else if(o instanceof Color) 
   223  				{
   224  					type.name = "color" ;
   225  				}
   226  			}
   227  		}
   228  		return type ;
   229  	}
   230  	
   231  
   232  	/**
   233  	 * Serialize un object before send the message in the console.
   234  	 */
   235  	private function _serializeObj(o, depth:Number) : Object {
   236  		
   237  		var type = _getType(o) ;
   238  		var serial:Object = {} ;
   239  		
   240  		if (!type.inspectable) 
   241  		{
   242  			serial.value = o ;
   243  		}
   244  		else if(type.stringify) 
   245  		{
   246  			serial.value = o + "" ;
   247  		}
   248  		else 
   249  		{
   250  			if (depth <= _maxDepth) 	
   251  			{
   252  				if(type.name == "movieclip" || type.name == "button") 
   253  				{
   254  					serial.id = o + "" ;
   255  				}
   256  				var items:Array = [] ;
   257  				if(o instanceof Array) 
   258  				{
   259  					var len:Number = o.length ;
   260  					for(var i:Number=0; i<len ; i++) 
   261  					{
   262  						items[i] = 
   263  						{
   264  							property : i ,
   265  							value : _serializeObj( o[i], (depth+1) )
   266  						} ;
   267  					}
   268  					
   269  				}
   270  				else 
   271  				{
   272  					for (var prop:String in o) 
   273  					{
   274  						items.push
   275  						( 
   276  							{
   277  								property : prop ,
   278  								value : _serializeObj( o[prop], (depth+1) )
   279  							} 
   280  						) ;
   281  					} 
   282  				}
   283  				
   284  				serial.value = items ;
   285  				
   286  			}
   287  			else 
   288  			{
   289  				serial.reachLimit = true ;
   290  			}
   291  		}
   292  		serial.type = type.name ;
   293  		return serial ;
   294  	}
