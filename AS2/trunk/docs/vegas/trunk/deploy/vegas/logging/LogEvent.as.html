    24  import vegas.events.DynamicEvent;
    25  import vegas.logging.LogEventLevel;
    26  import vegas.util.serialize.Serializer;
    27  
    28  /**
    29   * Represents the log information for a single logging event. The loging system dispatches a single event each time a process requests information be logged. This event can be captured by any object for storage or formatting.
    30   * @author eKameleon 
    31   */
    32  dynamic class vegas.logging.LogEvent extends DynamicEvent 
    33  {
    34  	
    35  	/**
    36  	 * Creates a new LogEvent.
    37  	 */
    38  	public function LogEvent( message , level:Number ) 
    39  	{
    40  		super( LogEvent.LOG ) ;
    41  		this.message = message || "" ;
    42  		this.level = isNaN(level) ? LogEventLevel.ALL : level ;
    43  	}
    44  
    45  	/**
    46  	 * Event type constant, identifies a logging event.
    47  	 */
    48  	static public var LOG:String = "log" ;
    49  	
    50  	/**
    51  	 * Provides access to the level for this log event.
    52  	 */
    53  	public var level:Number ;
    54  	
    55  	/**
    56  	 * Provides access to the message that was logged.
    57  	 */
    58  	public var message ;
    59  	
    60  	/**
    61  	 * Returns the shallow copy of the event.
    62  	 * @return the shallow copy of the LogEvent event.
    63  	 */
    64  	/*override*/ public function clone() 
    65  	{
    66  		return new LogEvent( message, level ) ;
    67  	}
    68  	
    69  	/**
    70  	 * Returns a string value representing the level specified.
    71  	 */
    72  	static public function getLevelString( value:Number ):String 
    73  	{
    74  		
    75  		switch (value) 
    76  		{
    77  			
    78  			case LogEventLevel.ALL :
    79  			{
    80  				return "ALL" ;
    81  			}
    82  			
    83  			case LogEventLevel.DEBUG :
    84  			{
    85  				return "DEBUG" ;
    86  			}
    87  			
    88  			case LogEventLevel.ERROR :
    89  			{
    90  				return "ERROR" ;
    91  			}
    92  			
    93  			case LogEventLevel.FATAL :
    94  			{
    95  				return "FATAL" ;
    96  			}
    97  			
    98  			case LogEventLevel.INFO :
    99  			{
   100  				return "INFO" ;
   101  			}
   102  			
   103  			case LogEventLevel.WARN :
   104  			{
   105  				return "WARN" ;
   106  			}
   107  			
   108  			default :
   109  			{
   110  				return "UNKNOWN" ;
   111  			}
   112  			
   113  		}
   114  	}
   115  
   116  	/**
   117  	 * Internal methods used by toSource method.
   118  	 */
   119  	/*protected*/ private function _getParams():Array 
   120  	{
   121  		var ar1:Array = 
   122  		[
   123  			Serializer.toSource(message) ,
   124  			Serializer.toSource(level)
   125  		] ;
   126  		var ar2:Array = super._getParams() ;
   127  		return ar1.concat( ar2 ) ;
   128  	}
   129  
   130  }
