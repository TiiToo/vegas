    26  /**
    27   *	Constructor: GenericParser
    28   */
    29  class buRRRn.eden.GenericParser
    30      {
    31      
    32      var source:String;
    33      var pos:Number;
    34      var ch:String;
    35      
    36      var log:Function;
    37      var onParsed:Function;
    38      var isAuthorized:Function;
    39      var addAuthorized:Function;
    40      var removeAuthorized:Function;
    41      
    42      function GenericParser( source:String, callback )
    43          {
    44          if( callback == null )
    45              {
    46              callback = buRRRn.eden.Application;
    47              }
    48          
    49          //delegates
    50          this.log = function( message )
    51              {
    52              callback.log( message );
    53              } ;
    54          
    55          this.onParsed = function( value )
    56              {
    57              return callback.onParsed( value );
    58              } ;
    59          
    60          this.isAuthorized = function( value )
    61              {
    62              return callback.isAuthorized( value );
    63              } ;
    64          
    65          this.addAuthorized = function()
    66              {
    67              callback.addAuthorized.apply( callback, arguments );
    68              } ;
    69          
    70          this.removeAuthorized = function()
    71              {
    72              callback.removeAuthorized.apply( callback, arguments );
    73              } ;
    74          
    75          this.source = source;
    76          this.pos    = 0;
    77          this.ch     = "";
    78          }
    79      
    80      /* Method: getCharAt
    81      */
    82      function getCharAt( pos:Number ):String
    83          {
    84          if( pos == null )
    85              {
    86              pos = this.pos;
    87              }
    88          
    89          return source.charAt( pos );
    90          }
    91      
    92      /* Method: getChar
    93      */
    94      function getChar():String
    95          {
    96          return source.charAt( pos );
    97          }
    98      
    99      /* Method: next
   100      */
   101      function next():String
   102          {
   103          ch = getChar();
   104          pos++;
   105          return ch;
   106          }
   107      
   108      /* Method: hasMoreChar
   109      */
   110      function hasMoreChar():Boolean
   111          {
   112          return( pos <= (source.length-1) );
   113          }
   114      
   115      /* Method: eval
   116         To override.
   117      */
   118      function eval()
   119          {
   120          
   121          }
   122      
   123      /* StaticMethod: evaluate
   124         To override.
   125      */
   126      static function evaluate( source:String, callback )
   127          {
   128          var parser = new buRRRn.eden.GenericParser( source, callback );
   129          return parser.eval();
   130          }
   131      
   132      /* Method: isAlpha
   133      */
   134      function isAlpha( /*char*/ c:String ):Boolean
   135          {
   136          return( (("A" <= c) && (c <= "Z")) || (("a" <= c) && (c <= "z")) );
   137          }
   138  
   139      /* Method: isASCII
   140      */
   141      function isASCII( /*char*/ c:String ):Boolean
   142          {
   143          return( c.charCodeAt( 0 ) <= 255 );
   144          }
   145      
   146      /* Method: isDigit
   147      */
   148      function isDigit( /*char*/ c:String ):Boolean
   149          {
   150          return( ("0" <= c) && (c <= "9") );
   151          }
   152      
   153      /* Method: isHexDigit
   154      */
   155      function isHexDigit( /*char*/ c:String ):Boolean
   156          {
   157          return( isDigit( c ) || (("A" <= c) && (c <= "F")) || (("a" <= c) && (c <= "f")) );
   158          }
   159      
   160      /* Method: isOctalDigit
   161      */
   162      function isOctalDigit( /*char*/ c:String ):Boolean
   163          {
   164          return( ("0" <= c) && (c <= "7") );
   165          }
   166      
   167      /* Method: isUnicode
   168      */
   169      function isUnicode( /*char*/ c:String ):Boolean
   170          {
   171          return( c.charCodeAt( 0 ) > 255 );
   172          }
   173      
   174      }
