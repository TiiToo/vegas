    24  import buRRRn.ASTUce.Config;
    25  import buRRRn.ASTUce.MiniRunner;
    26  
    27  /**
    28   * You must defined the unittests classpath to localize the Application class of your application.
    29   * @author eKameleon
    30   */
    31  class buRRRn.ASTUce.Application 
    32  {
    33  
    34  	static public var separator:String = "----------------------------------------------------------------";
    35  
    36  	/**
    37  	 * Returns information about the ASTUce framework.
    38  	 */
    39  	static public function getInfo():String
    40  	{
    41  		var str:String = "" ;
    42  		var CRLF:String = "\n" ;
    43      
    44      	if( Config.verbose )
    45          {
    46          	str += _name + ": " + _fullName + " v" + _version;
    47          }
    48      	else
    49          {
    50          	str += _name + " v" + _version;
    51          }
    52      
    53  		if( Config.verbose )
    54          {
    55          	str += CRLF;
    56          	str += "Copyright (c) 2004-2007 Zwetan Kjukov, All right reserved." + CRLF ;
    57          	str += "Made in the EU.";
    58          }
    59      
    60      	return str;
    61  	}
    62  	
    63  	/**
    64  	 * Execute the main process of the ASTUce framework.
    65  	 */
    66  	static public function main():Void
    67  	{
    68  		
    69  		trace( separator );
    70      
    71      	var runner:MiniRunner = new MiniRunner( "Main Tests" );
    72      
    73      	if( Config.testMyself )
    74          {
    75          
    76          	runner.setUp = function()
    77              {
    78              	// this.suite.addTest( Tests.AllTests.suite() );
    79              } ;
    80          
    81          }
    82      
    83      	if( arguments.length > 0 )
    84          {
    85          	for( var i=0; i<arguments.length; i++ )
    86              {
    87              	runner.suite.addTest( arguments[i] );
    88              }
    89          }
    90      
    91      	trace( getInfo() );
    92      	
    93      	trace( separator );
    94      	
    95      	runner.run();
    96      	
    97      	trace( separator );
    98      
    99      	if( Config.showConstructorList )
   100          {
   101          	trace( runner.suite );
   102          	trace( separator );
   103          }
   104  	}
   105  
   106      /**
   107       * The language extension.
   108       */
   109      static private var _ext:String = "AS2";
   110  
   111  	/**
   112  	 * The code name of the framework.
   113   	 */
   114  	static public var _name:String = "ASTUce";
   115  		
   116  	/***
   117  	 * The full name of the framework.
   118  	 */
   119  	static public var _fullName:String = "ActionScript Test Unit compact edition";
   120  
   121      /**
   122       * The target platform of the framework.
   123       */
   124      static private var _platform:String = "Flash ActionScript v2.0";
   125      	
   126  	/**
   127  	 * The version as string of the framework.
   128  	 */
   129  	static public var _version:String = "1.0.0.0 (VEGAS version)";
   130  	
   131  }
