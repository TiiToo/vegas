    25  /* Interface: IMoney
    26     The common interface for simple Monies and MoneyBags.
    27  */
    28  interface buRRRn.ASTUce.samples.IMoney
    29  {
    30      
    31      
    32      /* Method: plus
    33         Adds a money to this money.
    34      */
    35      function plus( m:IMoney ):IMoney;
    36      
    37      /* Method: addMoney
    38         Adds a simple Money to this money.
    39         
    40         This is a helper method for implementing double dispatch.
    41      */
    42      function addMoney( m:Money ):IMoney;
    43  
    44      /* Method: equals
    45      */
    46      function equals( /*untyped*/ obj ):Boolean ;
    47  
    48      
    49      /* Method: addMoneyBag
    50         Adds a MoneyBag to this money.
    51         
    52         This is a helper method for implementing double dispatch
    53      */
    54      function addMoneyBag( mb:MoneyBag ):IMoney;
    55      
    56      /* Method: isZero
    57         Tests whether this money is zero.
    58      */
    59      function isZero():Boolean;
    60      
    61      /* Method: multiply
    62         Multiplies a money by the given factor.
    63      */
    64      function multiply( factor:Number ):IMoney;
    65      
    66      /* Method: negate
    67         Negates this money.
    68      */
    69      function negate():IMoney;
    70      
    71      /* Method: minus
    72         Subtracts a money from this money.
    73      */
    74      function minus( m:IMoney ):IMoney;
    75      
    76      /* Method: appendTo
    77         Append this to a MoneyBag.
    78      */
    79      function appendTo( mb:MoneyBag ):Void;
    80      
    81  }
