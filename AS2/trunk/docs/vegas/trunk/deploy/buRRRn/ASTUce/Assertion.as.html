    25  import buRRRn.ASTUce.AssertionFailedError;
    26  import buRRRn.ASTUce.ComparisonFailure;
    27  import buRRRn.ASTUce.Config;
    28  import buRRRn.ASTUce.Strings;
    29  
    30  import vegas.core.CoreObject;
    31  import vegas.core.types.NullObject;
    32  import vegas.string.StringFormatter;
    33  import vegas.util.Comparater;
    34  import vegas.util.ObjectUtil;
    35  import vegas.util.serialize.Serializer;
    36  import vegas.util.TypeUtil;

    38  /**
    39   * A set of assert methods. Messages are only displayed when an assert fails.
    40   * Assertion is used because Assert is a reserved ECMAScript futur keyword.
    41   * @author eKameleon
    42   */
    43  class buRRRn.ASTUce.Assertion extends CoreObject
    44  {
    45  
    46  	/**
    47  	 * Asserts that two objects are equal.
    48  	 * If they are not an <AssertionFailedError> is thrown/trace with the given message.
    49  	 */
    50      static function assertEquals( expected, actual, message:String ):Void
    51  	{
    52  		
    53  		if( (expected == null) && (actual == null) )
    54  		{
    55  			return ;
    56  		}
    57  		
    58  		if ( (expected != null) && Comparater.compare(expected, actual ) )
    59  		{
    60  			return ;
    61  		}
    62  		
    63  		if ((expected != null) && (expected == actual))
    64  		{
    65  			return ;	
    66  		}
    67  		
    68  		if( TypeUtil.typesMatch( expected , String ) || TypeUtil.typesMatch( actual , String ) )
    69  		{
    70              throw new ComparisonFailure( expected, actual, message );
    71  		}
    72          else
    73  		{
    74  			_failNotEquals( expected, actual, message );
    75  		}
    76  		
    77  	}
    78   	
    79   	/**
    80  	 * Asserts that a condition is false. If it isn't it throws/trace an <AssertionFailedError> with the given message.
    81  	 */
    82  	public static function assertFalse( condition:Boolean , message:String ):Void
    83  	{
    84  		assertTrue( !condition, message );
    85  	}
    86  
    87  	/**
    88  	 * Asserts that an object is not null.
    89  	 * If it is an <AssertionFailedError> is thrown with the given message.
    90  	 */
    91      static function assertNotNull( obj, message:String ):Void
    92  	{
    93  		assertTrue( obj != null, message ) ;
    94  	}
    95  
    96  	/**
    97       * Asserts that two objects refer to the same object.
    98       * If they are not an <AssertionFailedError> is thrown with the given message.
    99       */
   100  	static function assertNotSame( expected, actual, message:String ):Void
   101  	{
   102  		if( expected == null )
   103  		{
   104  			expected = new NullObject() ;
   105  		}
   106  		if(  ObjectUtil.toBoolean( ObjectUtil.referenceEquals( expected, actual ) ) )
   107  		{ 
   108  			_failSame( expected, actual, message );
   109  		}
   110  	}  
   111  
   112      /**
   113       * Asserts that an object is not undefined.
   114       * If it is not an <AssertionFailedError> is thrown with the given message.
   115       */
   116  	static function assertNotUndefined( obj, message:String ):Void
   117  	{
   118  		assertTrue( obj != undefined, message );
   119  	}
   120  	
   121  	/**
   122  	 * Asserts that an object is null.
   123  	 * If it is an <AssertionFailedError> is thrown with the given message.
   124  	 */
   125  	static function assertNull( obj, message:String ):Void
   126  	{
   127  		assertTrue( obj == null, message );
   128  	}
   129  
   130      /**
   131       * Asserts that two objects refer to the same object.
   132       * If they are not an <AssertionFailedError> is thrown with the given message.
   133       * Note : same object mean same reference so the comparison is by reference not by value.
   134       */
   135  	static function assertSame( expected, actual, message:String ):Void
   136  	{
   137  		
   138  		if( expected == null )
   139  		{
   140  			expected = new NullObject() ;
   141  		}
   142          
   143          if( ObjectUtil.referenceEquals( expected, actual ) )
   144  		{
   145  			return;
   146  		}
   147          
   148  		_failNotSame( expected, actual, message );
   149  		
   150  	}
   151  
   152  	/**
   153  	 * Asserts that a condition is true. If it isn't it throws/trace an <AssertionFailedError> with the given message.
   154  	 */
   155  	public static function assertTrue( condition:Boolean , message:String ):Void
   156  	{
   157  		if ( !condition )
   158  		{
   159  			fail(message) ;	
   160  		}
   161  	}
   162      
   163      /**
   164       * Asserts that an object is undefined.
   165       * If it is not an <AssertionFailedError> is thrown with the given message.
   166       */
   167      static function assertUndefined( obj, message:String ):Void
   168  	{
   169  		assertTrue( obj == undefined, message );
   170  	}
   171      
   172  	/**
   173  	 * Fails a test with the given message.
   174  	 */
   175  	public static function fail( message:String ):Void
   176  	{
   177  		throw new AssertionFailedError( message );
   178  	}
   179  
   180  	/**
   181  	 * Format the message.
   182  	 */
   183  	public static function format( expected, actual, message )
   184  	{
   185  		var formatted:String = "" ;
   186  		if ( (message != null) && (message != "") )
   187  		{
   188  			formatted = message + " " ;	
   189  		}
   190  		return (new StringFormatter( Strings.expectedButWas )).format(formatted, expected, actual) ;
   191  	}
   192  	
   193  	/**
   194  	 * Fails a test when 2 objects are not equals.
   195  	 */
   196  	private static function  _failNotEquals( expected, actual, message:String )
   197      {
   198      	
   199      	if( Config.showObjectSource && (expected != null) && (actual != null) )
   200          {
   201          	expected = Serializer.toSource(expected);
   202          	actual   = Serializer.toSource(actual) ;
   203          }
   204      
   205      	if( Config.invertExpectedActual )
   206  	    {
   207          	var tmp  = expected;
   208          	expected = actual;
   209          	actual   = tmp;
   210          }
   211      
   212      	fail( format( expected, actual, message ) );
   213      }
   214  
   215  	/**
   216  	 * Fails a test when 2 objects are not the same.
   217  	 */
   218      static function _failNotSame( expected, actual, message:String ):Void
   219  	{
   220  		var formatted:String = "";
   221  		if( message != null )
   222  		{
   223  			formatted= message + " ";
   224  		}
   225          fail( (new StringFormatter( Strings.expectedSame)).format( formatted, expected, actual ) ) ;
   226  	}
   227  
   228  	/**
   229  	 * Fails a test when 2 objects are the same.
   230  	 */
   231      static function _failSame( expected, actual, message:String ):Void
   232  	{
   233  		var formatted:String = "";
   234  		if( message != null )
   235  		{
   236  			formatted = message + " ";
   237  		}
   238          
   239  		fail( (new StringFormatter( Strings.expectedNotSame )).format( formatted ) ) ;
   240  	}
   241  
   242  }
