    24  import buRRRn.ASTUce.AssertionFailedError;
    25  import buRRRn.ASTUce.ITest;
    26  import buRRRn.ASTUce.ITestListener;
    27  import buRRRn.ASTUce.Strings;
    28  import buRRRn.ASTUce.TestFailure;
    29  import buRRRn.ASTUce.TestResult;
    30  
    31  import vegas.core.CoreObject;
    32  import vegas.string.StringFormatter;
    33  import vegas.util.TypeUtil;
    34  
    35  /**
    36   * This implementation of the ITestListener interface creates an ouput printer for the resut of all tests.
    37   * @author eKameleon
    38   */
    39  class buRRRn.ASTUce.ResultPrinter extends CoreObject implements ITestListener 
    40  {
    41  	
    42  	/**
    43  	 * Creates a new ResultPrinter instance.
    44  	 */
    45  	public function ResultPrinter( writer:Function ) 
    46  	{
    47  		
    48  		_writer = Function( _global["trace"] ) ; //default writer
    49          
    50          _column = 0 ;
    51          
    52          if( (writer != null) && TypeUtil.typesMatch( writer , Function ) )
    53          {
    54              _writer = writer ;
    55  		}
    56  		
    57  	}
    58  
    59  	/**
    60  	 * Returns the internal writer method of this instance.
    61  	 */
    62      public function get writer():Function
    63  	{
    64  		return _writer ;
    65  	}
    66  
    67      /**
    68       * An error occurred.
    69       */
    70  	public function addError( test:ITest, e : Error) : Void 
    71  	{
    72  		writeLine( "E" );
    73  	}
    74  
    75      /**
    76       * A failure occurred.
    77       */
    78  	public function addFailure(test : ITest, afe : AssertionFailedError) : Void 
    79  	{
    80  		writeLine( "F" );
    81  	}
    82  
    83  	/**
    84  	 * Elapsed a time as string representation.
    85  	 * @return the time as string.
    86  	 */
    87      public function elapsedTimeAsString( runTime:Number ):String
    88  	{
    89  		var dat:Date  = new Date( runTime.valueOf() );
    90          var ms:Number = dat.getUTCMilliseconds();
    91          var s:Number  = dat.getUTCSeconds();
    92          var m:Number  = dat.getUTCMinutes();
    93          var h:Number  = dat.getUTCHours();
    94          return (new StringFormatter( Strings.PrtElapsedTime )).format( h, m, s, ms ) ;
    95  	}
    96  
    97      /**
    98       * A test ended.
    99       */
   100  	public function endTest(test : ITest) : Void 
   101  	{
   102  		//
   103  	}
   104  	
   105  	/**
   106  	 * Print a result in the printer.
   107  	 */
   108      public function print( result:TestResult, runTime:Number ):Void
   109  	{
   110          printHeader( runTime );
   111          printErrors( result );
   112          printFailures( result );
   113          printFooter( result );
   114  	}
   115  
   116  	/**
   117  	 * Print an error.
   118   	 */
   119      public function printErrors( result:TestResult ):Void
   120  	{
   121  		printDefects( result.errors, result.errorCount, Strings.nameError ) ;
   122  	}
   123  
   124  	/**
   125  	 * Print a failure.
   126  	 */
   127      public function printFailures( result:TestResult ):Void
   128  	{
   129          printDefects( result.failures, result.failureCount, Strings.nameFailure ) ;
   130  	}
   131  
   132  	/**
   133  	 * The runTime parameter can either be a Number or a Date. 
   134   	 */
   135      public function printHeader( runTime:Number ):Void
   136  	{
   137  		writeLine( "" );
   138  		writeLine( (new StringFormatter( Strings.PrtTime )).format( elapsedTimeAsString( runTime ) ) ) ;
   139  	}
   140  
   141  	/**
   142  	 * Print a defect message.
   143  	 */	
   144      public function printDefects( booBoos:Array, /*Int*/ count:Number, type:String ):Void
   145  	{
   146          var i:Number ;
   147          if( count == 0 )
   148  		{
   149  			return ;
   150  		}
   151  		if( count == 1 )
   152  		{
   153  			writeLine( "" );
   154  			writeLine( (new StringFormatter( Strings.PrtOneDefect )).format( count, type ) ) ;
   155  		}
   156          else
   157  		{
   158              writeLine( "" );
   159              writeLine( (new StringFormatter( Strings.PrtMoreDefects )).format( count, type ) ) ;
   160  		}
   161  		var len:Number = booBoos.length ;
   162  	    for( i = 0 ; i < len ; i++ )
   163  		{
   164  			printDefectHeader( booBoos[i], i );
   165  			printDefectTrace( booBoos[i] );
   166  		}
   167  	}
   168      
   169  	/**
   170  	 * Print a defect header.
   171  	 */
   172      public function printDefectHeader( booBoo:TestFailure, /*Int*/ count:Number ):Void
   173  	{
   174          writeLine( count + ") " + booBoo.failedTest );
   175  	}
   176      
   177  	/**
   178  	 * Print a defect trace.
   179  	 */
   180      public function printDefectTrace( booBoo:TestFailure ):Void
   181  	{
   182          /*!## TODO: find a way to have more detais about the code stack */
   183          //writeLine( booBoo.exceptionMessage ); //short error message
   184          writeLine( booBoo.thrownException ); //long error message
   185          writeLine( "" );
   186  	}
   187      
   188      /**
   189       * Print the footer.
   190       */
   191  	public function printFooter( result:TestResult ):Void
   192  	{
   193          if( result.wasSuccessful() == true )
   194  		{
   195  			writeLine( "" );
   196  			writeLine( (new StringFormatter( Strings.PrtOK )).format( result.runCount, (result.runCount == 1 ? "": "s") ) ) ;
   197  		}
   198          else
   199  		{
   200  			writeLine( "" );
   201  			writeLine( Strings.PrtFailure ) ;
   202  			writeLine( (new StringFormatter( Strings.PrtFailureDetails)).format( result.runCount, result.failureCount, result.errorCount ) ); //core2
   203  		}
   204          writeLine( "" );
   205  	}
   206      
   207      /**
   208       * A test started.
   209       */
   210  	public function startTest(test : ITest) : Void 
   211  	{
   212  		//
   213  	}
   214  
   215  	/**
   216  	 * Write a new message in the printer.
   217  	 */
   218      public function writeLine( message:String ):Void
   219  	{
   220  		writer.call(this, message) ;
   221  	}
   222      
   223      private var _column:Number;
   224      
   225      private var _writer:Function;
   226  
   227  }
