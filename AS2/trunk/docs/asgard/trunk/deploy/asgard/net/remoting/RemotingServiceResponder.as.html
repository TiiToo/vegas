    24  import asgard.events.RemotingEvent;
    25  
    26  import vegas.core.CoreObject;
    27  import vegas.events.Delegate;
    28  
    29  /**
    30   * The RemotingServiceResponder class provides an object that is used in the RemotingService instances to handle return values from the server.
    31   * @author eKameleon
    32   */
    33  class asgard.net.remoting.RemotingServiceResponder extends CoreObject 
    34  {
    35  	
    36  	/**
    37  	 * Creates a new RemotingServiceResponder.
    38  	 * @param scope the object who handle the values from the server.
    39  	 * @param resultMethod The function invoked if the call to the server succeeds and returns a result.
    40  	 * @param faultMethod The function invoked if the server returns an error.
    41  	 */	
    42  	public function RemotingServiceResponder( scope, resultMethod:Function, faultMethod:Function) 
    43  	{
    44  		
    45  		super();
    46  		
    47  		if (!faultMethod) 
    48  		{
    49  			faultMethod = scope["onFault"] ;
    50  		}
    51  		
    52  		if (!resultMethod) 
    53  		{
    54  			resultMethod = scope["onResult"] ;
    55  		}
    56  		
    57  		_eFault  = new RemotingEvent( RemotingEvent.FAULT, null) ;
    58  		_eResult = new RemotingEvent( RemotingEvent.RESULT, null) ;
    59  		_fault   = new Delegate(scope, faultMethod) ;
    60  		_result  = new Delegate(scope, resultMethod) ;
    61  		
    62  	}
    63  	
    64  	/**
    65  	 * Returns the name of the method.
    66  	 * @return the name of the method.
    67  	 */
    68  	public function getMethodName():String 
    69  	{
    70  		return _methodName ;
    71  	}
    72  	
    73  	/**
    74  	 * Returns the service reference of this responder.
    75  	 * @return the service reference of this responder.
    76  	 */
    77  	public function getService() 
    78  	{
    79  		return _service ;	
    80  	}
    81  	
    82  	/**
    83  	 * Invoked if the call to the server succeeds and returns a result.
    84  	 */
    85  	public function onResult( oResult ):Void 
    86  	{
    87  		_eResult.setResult( oResult, _methodName ) ;
    88  		_result.setArguments( _eResult ) ;
    89  		_result.run() ;
    90  	}
    91  
    92  	/**
    93  	 * Invoked if the server returns an error.
    94  	 */
    95  	public function onStatus( oFault:Object ):Void 
    96  	{
    97  		_eFault.setFault(oFault, _methodName) ;
    98  		_fault.setArguments( _eFault );
    99  		_fault.run();
   100  	}
   101  	
   102  	/**
   103  	 * Sets the name of the method.
   104  	 */
   105  	public function setMethodName(sName:String):Void 
   106  	{
   107  		_methodName = sName ;
   108  	}
   109  
   110  	/**
   111  	 * Sets the service reference.
   112  	 */
   113  	public function setService( service ) 
   114  	{
   115  		_service = service ;
   116  	}
   117  
   118  	private var _eFault:RemotingEvent ;
   119  	private var _eResult:RemotingEvent ;
   120  	
   121  	private var _fault:Delegate ;
   122  	private var _methodName:String ;
   123  	private var _result:Delegate ;
   124  	
   125  	private var _service ;
   126  	
   127  }
