    24  import vegas.core.CoreObject;
    25  
    26  /**
    27   * The NetStream status object who contains the level and the status value of all NetStreams in the application.
    28   * @author eKameleon
    29   * @version 1.0.0.0
    30   */	
    31  class asgard.net.NetStreamStatus extends CoreObject 
    32  {
    33  	
    34  	/**
    35  	 * Creates a new NetStreamStatus instance.
    36  	 * @param sCode the {@code String} code value.
    37  	 * @param sLevel the {@code String} level value.
    38  	 */
    39  	public function NetStreamStatus( sCode:String , sLevel:String) 
    40  	{
    41  		code = sCode ;
    42  		level = sLevel ;
    43  	}
    44  
    45  	/**
    46  	 * The 'error' level code.
    47  	 */
    48  	static public var ERROR:String = "error" ;
    49  	
    50  	/**
    51  	 * The 'status' level code.
    52  	 */
    53  	static public var STATUS:String = "status" ;
    54  	
    55  	/**
    56  	 * Data is not being received quickly enough to fill the buffer. Data flow will be interrupted until the buffer refills.
    57  	 */
    58  	static public var BUFFER_EMPTY:NetStreamStatus = new NetStreamStatus("NetStream.Buffer.Empty", STATUS) ;
    59  	
    60  	/**
    61  	 * The buffer is full and the stream will begin playing.
    62  	 */
    63  	static public var BUFFER_FULL:NetStreamStatus = new NetStreamStatus("NetStream.Buffer.Full", STATUS) ;
    64  	
    65  	/**
    66  	 * Data has finished streaming, and the remaining buffer will be emptied.
    67  	 */
    68  	static public var BUFFER_FLUSH:NetStreamStatus = new NetStreamStatus("NetStream.Buffer.Flush", STATUS) ;
    69  
    70  	/**
    71  	 * The stream is paused.
    72  	 */
    73  	static public var PAUSE:NetStreamStatus = new NetStreamStatus("NetStream.Pause.Notify", STATUS) ;	
    74  
    75  	/**
    76  	 * An error has occurred in playback for a reason other than those listed elsewhere in this table,
    77  	 * such as the subscriber not having read access.
    78  	 */
    79  	static public var PLAY_FAILED:NetStreamStatus = new NetStreamStatus("NetStream.Play.Failed", ERROR) ;	
    80  	
    81  	/**
    82  	 * Caused by a play list reset.
    83  	 */
    84  	static public var PLAY_RESET:NetStreamStatus = new NetStreamStatus("NetStream.Play.Start", STATUS) ;	
    85  
    86  	/**
    87  	 * The initial publish to a stream is sent to all subscribers.
    88  	 */
    89  	static public var PLAY_PUBLISH:NetStreamStatus = new NetStreamStatus("NetStream.Play.PublishNotify", STATUS) ;	
    90  	
    91  	/**
    92  	 * Playback has started.
    93  	 */
    94  	static public var PLAY_START:NetStreamStatus = new NetStreamStatus("NetStream.Play.Start", STATUS) ;	
    95  		
    96  	/**
    97  	 * Playback has stopped.
    98  	 */
    99  	static public var PLAY_STOP:NetStreamStatus = new NetStreamStatus("NetStream.Play.Stop", STATUS) ;	
   100  	
   101  	/**
   102  	 * The FLV passed to the play() method can't be found.
   103  	 */
   104  	static public var PLAY_STREAM_NOT_FOUND:NetStreamStatus = new NetStreamStatus("NetStream.Play.StreamNotFound", ERROR) ;	
   105  
   106  	/**
   107  	 * An unpublish from a stream is sent to all subscribers.
   108  	 */
   109  	static public var PLAY_UNPUBLISH:NetStreamStatus = new NetStreamStatus("NetStream.Play.UnpublishNotify", STATUS) ;	
   110  	
   111  	/**
   112  	 * Publish was successful.
   113  	 */
   114  	static public var PUBLISH_START:NetStreamStatus = new NetStreamStatus("NetStream.Publish.Start", STATUS) ;	
   115  	
   116  	/**
   117  	 *  Attempt to publish a stream which is already being published by someone else.
   118  	 */
   119  	static public var PUBLISH_BAD_NAME:NetStreamStatus = new NetStreamStatus("NetStream.Publish.BadName", ERROR) ;		
   120  
   121  	/**
   122  	 * The publisher of the stream has been idling for too long.
   123  	 */
   124  	static public var PUBLISH_IDLE:NetStreamStatus = new NetStreamStatus("NetStream.Publish.Idle", STATUS) ;	
   125  
   126  	/**
   127  	 * An attempt to record a stream failed.
   128  	 */
   129  	static public var RECORD_FAILED:NetStreamStatus = new NetStreamStatus("NetStream.Record.Failed", ERROR) ;
   130  	
   131  	/**
   132  	 * Attempt to record a stream that is still playing or the client has no access right.
   133  	 */
   134  	static public var RECORD_NOACCESS:NetStreamStatus = new NetStreamStatus("NetStream.Record.NoAccess", ERROR) ;
   135  
   136  	/**
   137  	 * Recording has started.
   138  	 */
   139  	static public var RECORD_START:NetStreamStatus = new NetStreamStatus("NetStream.Record.Start", STATUS) ;
   140  	
   141  	/**
   142  	 * Recording stopped.
   143  	 */
   144  	static public var RECORD_STOP:NetStreamStatus = new NetStreamStatus("NetStream.Record.Stop", STATUS) ;
   145  
   146  	/**
   147  	 * The seek fails, which happens if the stream is not seekable.
   148  	 */
   149  	static public var SEEK_FAILED:NetStreamStatus = new NetStreamStatus("NetStream.Seek.Failed", ERROR) ;
   150  
   151  	/**
   152  	 * For video downloaded with progressive download, 
   153  	 * the user has tried to seek or play past the end of the video data that has downloaded thus far,
   154  	 * or past the end of the video once the entire file has downloaded. 
   155  	 * The message.details  property contains a time code that indicates the last valid position to which the user can seek.
   156  	 */
   157  	static public var SEEK_INVALID_TIME:NetStreamStatus = new NetStreamStatus("NetStream.Seek.InvalidTime", ERROR) ;
   158  
   159  	/**
   160  	 * The seek operation is complete.
   161  	 */
   162  	static public var SEEK_NOTIFY:NetStreamStatus = new NetStreamStatus("NetStream.Seek.Notify", STATUS) ;
   163  
   164  	/**
   165  	 * The stream is unpaused.
   166  	 */
   167  	static public var UNPAUSE:NetStreamStatus = new NetStreamStatus("NetStream.UnPause.Notify", STATUS) ;
   168  
   169  	/**
   170  	 * The unpublish operation was successfull.
   171  	 */
   172  	static public var UNPUBLISH_SUCCESS:NetStreamStatus = new NetStreamStatus("NetStream.Unpublish.Success", STATUS) ;	
   173  	
   174  	/**
   175  	 * The code of the NetStream information.
   176  	 */
   177  	public var code:String ;
   178  	
   179  	/**
   180  	 * The level of the NetStream status information.
   181  	 */
   182  	public var level:String ;
   183  
   184  	/**
   185  	 * Compares the specified object with this object for equality.
   186  	 * @return {@code true} if the the specified object is equal with this object.
   187  	 */
   188  	public function equals(o):Boolean
   189  	{
   190  		return o.toString() == toString() ;	
   191  	}
   192  
   193  	/**
   194  	 * Returns the string representation of this object.
   195  	 * @return the string representation of this object.
   196  	 */
   197  	public function toString():String 
   198  	{
   199  		return code ;	
   200  	}
   201  
   202  }
