    24  import asgard.net.NetServerConnection;
    25  import asgard.net.remoting.RemotingAuthentification;
    26  import asgard.net.remoting.RemotingConnectionCollector;
    27  
    28  /**
    29   * The NetServerConnection object used in the RemotingService class to connect the client with the server.
    30   * @author eKameleon
    31   * @version 1.0.0.0
    32   */	
    33  class asgard.net.remoting.RemotingConnection extends NetServerConnection 
    34  {
    35  
    36  	/**
    37  	 * Creates a new RemotingConnection instance.
    38  	 * @param sURL the url of the connection.
    39  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    40  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    41  	 */	
    42  	function RemotingConnection( sURL:String , bGlobal:Boolean , sChannel:String ) 
    43  	{
    44  		super( bGlobal , sChannel ) ;
    45  		if ( sURL != null && sURL.length > 0 ) 
    46  		{
    47  			this.connect( sURL );
    48  		}
    49  	}
    50  
    51  	/**
    52  	 * The string value of the amf server debug attribut.
    53  	 */
    54  	static public var AMF_SERVER_DEBUG:String = "amf_server_debug" ;
    55  	
    56  	/**
    57  	 * The string representation of the Credentials namespace.
    58  	 */
    59  	static public var CREDENTIALS:String = "Credentials" ;
    60  	
    61  	/**
    62  	 * Returns the shallow copy of this object.
    63  	 * @return the shallow copy of this object.
    64  	 */
    65  	public function clone() 
    66  	{
    67  		return new RemotingConnection( uri ) ;	
    68  	}
    69  	
    70  	/**
    71  	 * Returns the RemotingConnection reference defines with the specified url representation.
    72  	 * @return the RemotingConnection reference defines with the specified url representation.
    73  	 */
    74  	static public function getConnection( sUrl:String ):RemotingConnection 
    75  	{
    76  		if ( ! RemotingConnectionCollector.contains(sUrl) ) 
    77  		{
    78  			RemotingConnectionCollector.insert(sUrl, new RemotingConnection( sUrl ));
    79  		}
    80  		return RemotingConnectionCollector.get(sUrl) ;
    81  	}
    82  
    83  	/**
    84  	 * Sets the credentials authentification value of this connection.
    85  	 */
    86  	public function setCredentials( authentification:RemotingAuthentification ):Void  
    87  	{
    88  		var o:Object = (authentification == null) ? null : authentification.toObject() ;
    89  		addHeader( RemotingConnection.CREDENTIALS , false, o);
    90  	}
    91  
    92  	/**
    93  	 * Start the debug mode of this connection.
    94  	 */
    95  	public function startDebug():Void 
    96  	{
    97  		var oDebug:Object = {
    98  			amf:false, 
    99  			error:true,
   100  			trace:true,
   101  			coldfusion:false, 
   102  			m_debug:true,
   103  			httpheaders:false, 
   104  			amfheaders:false, 
   105  			recordset:true
   106  		};
   107  		addHeader( RemotingConnection.AMF_SERVER_DEBUG, true, oDebug) ;
   108  	}
   109  
   110  	/**
   111  	 * Stop the debug mode of this connection.
   112  	 */
   113  	public function stopDebug():Void 
   114  	{
   115  		addHeader( RemotingConnection.AMF_SERVER_DEBUG, true, undefined) ;
   116  	}
   117  	
   118  }
