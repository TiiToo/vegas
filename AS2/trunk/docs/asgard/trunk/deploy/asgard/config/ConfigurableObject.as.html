    24  import asgard.config.ConfigCollector;
    25  import asgard.config.IConfigurable;
    26  
    27  import vegas.errors.Warning;
    28  import vegas.events.AbstractCoreEventDispatcher;
    29  
    30  /**
    31   * This core class extend the CoreObject class and implement the IConfigurable interface.
    32   * A IConfigurable object handle a notification of the ConfigCollector class with the method setup(), you must override this method in your concrete class.
    33   * The IConfigurable objects are registered in the ConfigCollector to launch the setup of all IConfigurable object one time with the {@code ConfigCollector.run()} method when the Config is loaded for example. 
    34   * @author eKameleon
    35   */
    36  class asgard.config.ConfigurableObject extends AbstractCoreEventDispatcher implements IConfigurable
    37  {
    38  
    39  	/**
    40  	 * Creates a new ConfigurableObject instance.
    41  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    42  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    43  	 */
    44  	public function ConfigurableObject( bGlobal:Boolean , sChannel:String ) 
    45  	{
    46  		super( bGlobal , sChannel ) ;	
    47  		isConfigurable = true ;
    48  	}
    49  
    50  	/**
    51  	 * (read-write) Returns {@code true} if the object is configurable and receive the notification of the ConfigCollector in the setup() method.
    52  	 * @return {@code true} if the object is configurable and receive the notification of the ConfigCollector.
    53  	 */
    54  	public function get isConfigurable():Boolean
    55  	{
    56  		
    57  		return _isConfigurable ;
    58  			
    59  	}
    60  
    61  	/**
    62  	 * (read-write) Sets if the object is configurable and receive the notification of the ConfigCollector in the setup() method.
    63  	 * @param b the flag boolean value to define if the object is configurable.
    64  	 */
    65  	public function set isConfigurable(b:Boolean):Void
    66  	{
    67  		_isConfigurable = (b == true) ;
    68  		if (_isConfigurable)
    69  		{
    70  			ConfigCollector.insert(this) ;	
    71  		}
    72  		else
    73  		{
    74  			ConfigCollector.remove(this) ;
    75  		}
    76  	}
    77  
    78  	/**
    79  	 * Invoqued when this object when the ConfigCollector is run.
    80  	 */
    81  	 public function setup():Void
    82  	 {
    83  	 	throw new Warning( this + ".setup(), you must override this method !" ) ;
    84  	 }
    85  	
    86  	/**
    87  	 * Determinates if the object is configurable.
    88  	 */
    89  	 private var _isConfigurable:Boolean ;
    90  	 
    91  }
