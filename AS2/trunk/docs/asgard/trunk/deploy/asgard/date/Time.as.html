    24  import vegas.core.CoreObject;
    25  import vegas.util.MathsUtil;
    26  
    27  /**
    28   * The {@code Time} object is a holder for a time difference.
    29   * <p>{@code Time} splits a time difference (distance between two dates) into 
    30   * days, hours, minutes, seconds and milliseconds to offers methods to access the time difference value.
    31   * @author eKameleon
    32   */
    33  class asgard.date.Time extends CoreObject
    34  {
    35  	
    36  	/**
    37  	 * Creates a new {@code Time} instance.
    38       * @param timeDifference The time size of the time difference for the passed-in {@code format}.
    39       * @param format (optional) "d"/"h"/"m"/"s"/"ms" for the unit of the amout, default case is "ms".
    40       */
    41  	public function Time( timeDifference:Number, format:String ) 
    42  	{
    43  		setValue( timeDifference, format) ;
    44  	}
    45  
    46  	/** 
    47  	 * Factor from ms to second. 
    48  	 */
    49  	static public var SECOND:Number = 1000;
    50  	
    51  	/** 
    52  	 * Factor from ms to minute.
    53  	 */
    54  	static public var MINUTE:Number = SECOND*60;
    55  	
    56  	/** 
    57  	 * Factor from ms to hour.
    58  	 */
    59  	static public var HOUR:Number = MINUTE * 60 ;
    60  	
    61  	/** 
    62  	 * Factor from ms to day.
    63  	 */
    64  	static public var DAY:Number = HOUR * 24 ;
    65  
    66  	/** 
    67  	 * Amount of days.
    68  	 */
    69  	public var days:Number;
    70  
    71  	/**
    72  	 * Time difference in ms.
    73  	 */
    74  	public var ms:Number;
    75  	
    76  	/**  
    77  	 * Amount of hours.
    78  	 */
    79  	public var hours:Number;
    80  	
    81  	/**
    82  	 *  Amount of minutes.
    83  	 */
    84  	public var minutes:Number;
    85  	
    86  	/**
    87  	 *  Amount of seconds.
    88  	 */
    89  	public var seconds:Number;
    90  	
    91  	/** 
    92  	 * Amount of milliseconds.
    93  	 */
    94  	public var milliSeconds:Number;
    95  	
    96  	/** 
    97  	 * Flag if the instance need to be evaluated by {@link #evaluate}.
    98  	 */
    99  	public var doEval:Boolean = true;	
   100  
   101  	/**
   102  	 * Splits the time distance from ms (source value) into the different units.
   103  	 */
   104  	public function evaluate():Void 
   105  	{
   106  		var negative = MathsUtil.sign(ms) ;
   107  		var rest:Number = ms ;
   108  		
   109  		days = rest / DAY ;
   110  		rest -= negative * Math.floor(days) * DAY ;
   111  		
   112  		hours = rest / HOUR;
   113  		rest -= negative * Math.floor(hours) * HOUR ;
   114  		
   115  		minutes = rest / MINUTE;
   116  		rest -= negative * Math.floor(minutes) * MINUTE ;
   117  		
   118  		seconds = rest / SECOND ;
   119  		rest -= negative * Math.floor(seconds)*SECOND ;
   120  		
   121  		milliSeconds = rest;
   122  		
   123  		doEval = false ;
   124  	}
   125  	
   126  	/**
   127  	 * Returns the amount of days are contained within the time.
   128  	 * <p>It will not round the result if you pass-in nothing.</p>
   129  	 * @param round (optional) the number of decimal spaces
   130  	 * @return The time difference in days.
   131  	 */
   132  	public function getDays(round:Number):Number 
   133  	{
   134  		if (doEval) 
   135  		{
   136  			evaluate() ;
   137  		}
   138  		return isNaN(round) ? days : MathsUtil.floor(days, round) ;
   139  	}
   140  
   141  	/**
   142  	 * Returns the amount of hours are contained within the time.
   143  	 * <p>It will not round the result if you pass-in nothing.</p>
   144  	 * @param round (optional) the number of decimal spaces
   145  	 * @return The time difference in hours.
   146  	 */
   147  	public function getHours(round:Number):Number 
   148  	{
   149  		if (doEval) 
   150  		{
   151  			evaluate() ;
   152  		}
   153  		return isNaN(round) ? hours : MathsUtil.floor(hours, round);
   154  	}
   155  
   156  	/**
   157  	 * Returns the amount of milliseconds are contained within the time.
   158  	 * <p>It will not round the result if you pass-in nothing.</p>
   159  	 * @param round (optional) the number of decimal spaces.
   160  	 * @return The time difference in milliseconds.
   161  	 */
   162  	public function getMilliSeconds(round:Number):Number 
   163  	{
   164  		if (doEval) 
   165  		{
   166  			evaluate();
   167  		}
   168  		return isNaN(round) ? milliSeconds : MathsUtil.floor(milliSeconds, round) ;
   169  	}
   170  
   171  	/**
   172  	 * Returns the amount of minutes are contained within the time.
   173  	 * <p>It will not round the result if you pass-in nothing.</p>
   174  	 * @param round (optional) the number of decimal spaces
   175  	 * @return The time difference in minutes.
   176  	 */
   177  	public function getMinutes(round:Number):Number 
   178  	{
   179  		if (doEval) 
   180  		{
   181  			evaluate() ;
   182  		}
   183  		return isNaN(round) ? minutes :MathsUtil.floor(minutes, round) ;
   184  	}
   185  	/**
   186  	 * Returns the amount of seconds are contained within the time.
   187  	 * <p>It will not round the result if you pass-in nothing.</p>
   188  	 * @param round (optional) the number of decimal spaces
   189  	 * @return The time difference in seconds
   190  	 */
   191  	public function getSeconds(round:Number):Number 
   192  	{
   193  		if (doEval) 
   194  		{
   195  			evaluate();
   196  		}
   197  		return isNaN(round) ? seconds : MathsUtil.floor(seconds, round) ;
   198  	}
   199  
   200  	/**
   201  	 * Returns the time distance in days.
   202  	 * @return The time difference in days.
   203  	 */
   204  	public function inDays():Number 
   205  	{
   206  		return ms / DAY ;
   207  	}
   208  
   209  	/**
   210  	 * Returns the time distance in hours.
   211  	 * @return The time difference in hours.
   212  	 */
   213  	public function inHours():Number 
   214  	{
   215  		return ms / HOUR ;
   216  	}
   217  
   218  	/**
   219  	 * Returns the time difference in milliseconds.
   220  	 * @return the time difference in milliseconds.
   221  	 */
   222  	public function inMilliSeconds():Number 
   223  	{
   224  		return ms;
   225  	}
   226  
   227  	/**
   228  	 * Returns the time distance in minutes.
   229  	 * @return The time difference in minutes.
   230  	 */
   231  	public function inMinutes():Number 
   232  	{
   233  		return ms / MINUTE;
   234  	}
   235  	
   236  	/**
   237  	 * Returns The time difference in seconds.
   238  	 * @return The time difference in seconds.
   239  	 */
   240  	public function inSeconds():Number 
   241  	{
   242  		return ms / SECOND ;
   243  	}
   244  
   245  	/**
   246  	 * Removes the passed-in {@code timeDifference} from the current time.
   247  	 * @param timeDifference time difference to be removed from the current time
   248  	 * @return A new instance with the resulting amount of time
   249  	 */
   250  	public function minus(timeDifference:Time):Time 
   251  	{
   252  		return new Time(ms-timeDifference.valueOf());
   253  	}
   254  
   255  	/**
   256  	 * Adds the passed-in {@code timedistance} to the current time.
   257  	 * @param timeDifference time difference to be added to the current time.
   258  	 * @return A new instance with the resulting amount of time.
   259  	 */
   260  	public function plus( timeDifference:Time ):Time 
   261  	{
   262  		return new Time( ms + timeDifference.valueOf() );
   263  	}
   264  
   265  	/** 
   266  	 * Sets the time of the instance.
   267  	 * 
   268  	 * <p>Uses "ms" if no format or a wrong format was passed-in.</p>
   269  	 * <p>Uses {@code Number.MAX_VALUE} if {@code Infinity} was passed-in.</p>
   270       * @param time size of the time difference for the passed-in {@code format}
   271       * @param format (optional) "d"/"h"/"m"/"s"/"ms" for the unit of the amout. Default value is ms.
   272  	 */
   273  	public function setValue(timeDifference:Number, format:String):Time 
   274  	{
   275  		
   276  		if (timeDifference == Infinity) 
   277  		{
   278  			timeDifference = Number.MAX_VALUE;
   279  		}
   280  		
   281  		switch (format) 
   282  		{
   283  			case "d" :
   284  			{
   285  				ms = timeDifference * DAY ;
   286  				break;
   287  			}
   288  			case "h" :
   289  			{
   290  				ms = timeDifference * HOUR ;
   291  				break;
   292  			}
   293  			case "m" :
   294  			{
   295  				ms = timeDifference * MINUTE ;
   296  				break;
   297  			}
   298  			case "s":
   299  			{
   300  				ms = timeDifference * SECOND ;
   301  				break;
   302  			}
   303  			default:
   304  			{
   305  				ms = timeDifference ;
   306  			}
   307  		}
   308  		
   309  		doEval = true ;
   310  		
   311  		return this ;
   312  		
   313  	}
   314  
   315  	/**
   316  	 * Returns the value of the time distance (in ms).
   317  	 * @return The value of this object in ms.
   318  	 */
   319  	public function valueOf():Number 
   320  	{
   321  		return ms;
   322  	}
