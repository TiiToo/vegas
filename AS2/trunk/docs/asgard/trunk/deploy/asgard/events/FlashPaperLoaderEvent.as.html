    34  class asgard.events.FlashPaperLoaderEvent extends DisplayLoaderEvent 
    35  {
    36  
    37  	/**
    38  	 * Creates a new FlashPaperLoaderEvent instance.
    39  	 */
    40  	public function FlashPaperLoaderEvent
    41  	(
    42  		type : String, fpLoader:FlashPaperLoader
    43  		, p_currentZoom:Number, p_isEnabledScrolling:Boolean, p_newPageNumber:Number, p_newVisibleArea 
    44  		, p_code:Number, p_error:String, context
    45  		, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number
    46  	) 
    47  	{
    48  		super(type, fpLoader, p_code, p_error, context, bubbles, eventPhase, time, stop);
    49  		currentZoom = isNaN(p_currentZoom) ? null : p_currentZoom ;
    50  		isEnabledScrolling = p_isEnabledScrolling || null ;
    51  		newPageNumber = isNaN(p_newPageNumber) ? null : p_newPageNumber ;
    52  		newVisibleArea = p_newVisibleArea || null ;
    53  	}

   133  	public var isEnabledScrolling:Boolean ;

   138  	public var newPageNumber:Number ;

   148  	public function clone()

   159  	}
   160  
   161  	/**
   162  	 * This protected method is used by the toSource method.
   163  	 */
   164  	/*protected*/ private function _getParams():Array 
   165  	{
   166  		var ar:Array = super._getParams() ;
   167  		ar.splice(2, null, Serializer.toSource(currentZoom)) ;
   168  		ar.splice(3, null, Serializer.toSource(isEnabledScrolling)) ;
   169  		ar.splice(4, null, Serializer.toSource(newPageNumber)) ;
   170  		ar.splice(5, null, Serializer.toSource(newVisibleArea)) ;
   171  		return ar ;
   172  	}
