    47   * import asgard.data.remoting.RecordSet ;
    48   * import asgard.events.RemotingEvent ;

    59   *     var type:String = e.getType() ;

    65   *     trace("> description : " +  e.getDescription()) ;
    66   * } ;
    67   * 
    68   * var onFinish:Function = function (e:RemotingEvent):Void 
    69   * {
    70   *     trace(">> " +  e.getType()) ;
    71   * } ;

    73   * var onResult:Function = function(e:RemotingEvent):Void 
    74   * {

   111  	public function RecordSet( o ) {
   112  		
   113  		_eAdd    = new RecordSetEvent(RecordSetEvent.ADD_ITEMS, this) ;
   114  		_eClear  = new RecordSetEvent(RecordSetEvent.CLEAR_ITEMS, this);
   115  		_eRemove = new RecordSetEvent(RecordSetEvent.REMOVE_ITEMS, this) ;
   116  		_eSort   = new RecordSetEvent(RecordSetEvent.SORT_ITEMS, this) ;
   117  		_eUpdate = new RecordSetEvent(RecordSetEvent.UPDATE_ITEMS, this) ;
   118  		
   119  		_items = [] ;
   120  		mTiles = [] ;
   121  		
   122  		if (serverInfo == null) 
   123  		{
   124  			if (serverinfo != null) 
   125  			{
   126  				serverInfo = serverinfo ;
   127  			}
   128  		}
   129  		
   130  		if (o) 
   131  		{
   132  		
   133  			if (o instanceof Array)
   134  			{
   135  				mTiles = o ;
   136  			} 
   137  			else 
   138  			{
   139  				parse(o) ;
   140  			}		
   141  		}
   142  		
   143  		// TODO ??? voir ici ce qui cloche en comparaison avec script MM.
   144  		
   145  		if (serverInfo) 
   146  		{
   147  			parse(this) ;	
   148  		}
   149  
   150  	}

   163  	public function addItemAt(index:Number, oItem) 
   164  	{
   165  		var l:Number = size() ;
   166  		if (index >= 0 && index < l ) {
   167  			
   168  			_items.splice(index, 0, oItem) ;
   169  			
   170  		} else {
   171  			
   172  			 if (index == l) {
   173  			 	_items[index] = oItem ;	
   174  			 } else {
   175  			 	return null ;	
   176  			 }
   177  		}
   178  		oItem["__ID__"] = _id++ ;
   179  		_eAdd.index = index ;
   180  		notifyChanged( _eAdd ) ;
   181  		return oItem ;
   182  	}

   226  	public function filter(filter:Function, context):RecordSet 
   227  	{
   228  		
   229  		if ( checkLocal() ) {
   230  		  return null ;
   231  		}
   232  
   233  		var rs:RecordSet = new RecordSet() ;
   234  		rs.setColumnNames(mTiles) ;
   235  	
   236  		var len:Number = size() ;
   237  		for(var i:Number = 0; i < len ; i++) {
   238  			var item:Object = getItemAt(i) ;
   239  			if ((item != null) && (item != 1) && filter(item, context))	{
   240  				rs.addItem(item);
   241  			}
   242  		}
   243  		return rs ;
   244  		
   245  	}

   264  	/**
   265  	 * Returns a unique ID corresponding to the record, at the specified index.
   266  	 * The RecordSet object assigns each record a unique ID. The ID is not part 
   267  	 * of the record; it is a separate item that is associated with the record
   268  	 * internally within the RecordSet object. Unlike a record index, its ID will
   269  	 * not change when the RecordSet object is sorted or when records are added or
   270  	 * deleted. When a record is deleted, its ID is retired and will never be used
   271  	 * again in this RecordSet object.
   272  	 */
   273  	public function getItemID( index:Number )

   349  	public function parse( oRaw ):Void 
   350  	{
   351  		
   352  		clear();
   353  		
   354  		mTiles = oRaw.serverInfo.columnNames ;
   355  		
   356  		mRecordsAvailable = 0 ;
   357  		
   358  		// implements setData !
   359  		// setData((serverInfo.cursor == null) ? 0 : (serverInfo.cursor - 1), serverInfo.initialData);
   360  		
   361  		var aItems:Array = oRaw.serverInfo.initialData ;
   362  		
   363  		
   364  		var l:Number = aItems.length ;
   365  		for (var i:Number = 0 ; i<l ; i++) {
   366  			var item:Object = {} ;
   367  			var aProperties:Array = aItems[i] ;
   368  			var count:Number = aProperties.length ;
   369  			for (var j:Number = 0 ; j<count ; j++) {
   370  				item[ mTiles[ j ] ] = aProperties[j] ;
   371  			}
   372  			_items.push( item );
   373  		}
   374  		
   375  		//serverInfo = null ;
   376  		
   377  	}

   404  	public function removeItemsAt(index:Number, len:Number):Array 
   405  	{
   406  		var oldItems = _items.splice(index, len) ;
   407  		_eRemove.firstItem = index ;
   408  		_eRemove.lastItem = index + len - 1 ;
   409  		_eRemove.removedItems = [].concat(oldItems) ;
   410  		notifyChanged ( _eRemove ) ;
   411  		return oldItems ;
   412  	}

   423  	public function replaceItemAt( index:Number, item:Object ):Void 
   424  	{
   425  		if (index>=0 && index<=size()) {
   426  			var tmpID = getItemID(index) ;
   427  			_items[index] = item ;
   428  			_items[index].__ID__ = tmpID ;
   429  			_eUpdate.index = index ;
   430  			notifyChanged( _eUpdate ) ;
   431  		}
   432  	}

   434  	public function setItemIndex( oItem, index:Number):Void 
   435  	{
   436  		var id:Number = indexOf(oItem) ;
   437  		if (id == -1 || id == index) return ;
   438  		else {
   439  			var tmp = oItem ;
   440  			_items.splice(id, 1) ;
   441  			_items.splice(index, 0, tmp) ;
   442  			_eUpdate.index = index ;
   443  			notifyChanged( _eUpdate ) ;
   444  		}
   445  	}
