    35  /**
    36   * The DisplayObject class is the base class for all objects that can be displayed in the Flash Player.
    37   * <p><b>Example :</b></p>
    38   * {@code
    39   * import asgard.display.DisplayObject ;
    40   * import asgard.display.DisplayObjectCollector ;
    41   * 
    42   * var mc:MovieClip = createEmptyMovieClip("display", 1) ;
    43   * 
    44   * var UIList = {} ;
    45   * UIList.CANVAS = "canvas" ;
    46   * 
    47   * var dCanvas = new DisplayObject( "canvas", mc ) ;
    48   * 
    49   * trace("> DisplayObjectCollector contains canvas : " + DisplayObjectCollector.containsDisplay(canvas))  ;
    50   * 
    51   * var contains:Boolean = DisplayObjectCollector.contains(UIList.CANVAS) ;
    52   * trace("contains 'canvas' DisplayObject : " + contains) ;
    53   * 
    54   * var canvas = DisplayObjectCollector.get(UIList.CANVAS) ;
    55   * canvas.move(25, 25) ;
    56   * 
    57   * var square:MovieClip = canvas.view ;
    58   * square.lineStyle(3, 0xFF0000, 100, true, "none", "square" ) ;
    59   * square.beginFill(0xF2F10D, 100) ;
    60   * square.lineTo(150,0) ;
    61   * square.lineTo(150,150) ;
    62   * square.lineTo(0,150) ;
    63   * square.lineTo(0,0) ;
    64   * 
    65   * trace("canvas : " + canvas) ;
    66   * trace("canvas name : '" + canvas.getName() + "'" ) ;
    67   * trace("The root DisplayObject : " + canvas.root + " : " + canvas.root.view) ;
    68   * 
    69   * this.onKeyDown = function ()
    70   * {
    71   *     var code:Number = Key.getCode() ;
    72   *     switch (code)
    73   *     {
    74   *         case Key.UP :
    75   *         {
    76   *             canvas.hide() ;
    77   *             trace("canvas idHide : " + canvas.isVisible()) ;
    78   *             break ;
    79   *         }
    80   *         case Key.DOWN :
    81   *         {
    82   *             canvas.show() ;
    83   *             break ;
    84   *         }
    85   *         case Key.RIGHT :
    86   *         {
    87   *             canvas.rotation = 25 ;
    88   *             break ;
    89   *         }
    90   *         case Key.LEFT :
    91   *         {
    92   *             canvas.alpha = 30 ;
    93   *             break ;
    94   *         }
    95   *     }
    96   * }
    97   * 
    98   * Key.addListener(this) ;
    99   * }
   100   * @author eKameleon
   101   */

   106  	 * Creates a new DisplayObject instance.
   107  	 * @param sName the name of the display.
   108  	 * @param target the DisplayObject instance control this target.
   109  	 * @param bGlobal the flag to use a global event flow or a local event flow.
   110  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
   111  	 */
   112      public function DisplayObject( sName:String , target , bGlobal:Boolean , sChannel:String ) 
   113      {
   114          
   115          super( bGlobal, sChannel  ) ;
   116  		
   117  		if (target) 
   118  		{
   119  			view = target ;
   120  		}
   121  		else 
   122  		{
   123  			_loader = DisplayLoaderCollector.get( sName ) ;
   124  			
   125  			if ( _loader ) 
   126  			{
   127  				this.view = _loader.getView() ;
   128  			}
   129  			else 
   130  			{
   131  				throw new IllegalArgumentError("invalid arguments for " + this + " constructor.");
   132  			}
   133  		}
   134  		
   135  		_setName(sName) ;
   136  
   137      }

   499  	/**
   500  	 * Returns an instance in a visual DisplayObject if exist else throws a Warning error.
   501  	 * @param name the name of the instance in the view scope of the display. 
   502  	 * @param depth (optional) the depth value of the visual reference.
   503  	 * @return an instance in a visual DisplayObject if exist else throws a Warning error.
   504  	 * @throws Warning if the resolve failed with the specified instance name.
   505  	 */
   506  	public function resolve( name:String , depth )
   507  	{
   508  		if ( view[ name ] != null)
   509  		{
   510  			if ( !isNaN(depth) )
   511  			{
   512  				DisplayFactory.swapDepths( view[name] , depth ) ;	
   513  			}
   514  			return view[name] ;
   515  		}
   516  		else
   517  		{
   518  			throw new Warning( this + " resolve failed with the view : '" + name + "'" );
   519  		}
   520  	}

   526  	 * @param depth (optional) the depth value of the visual reference.
   527  	 * @throws Warning if the resolve failed with the specified instance name.
   528  	 */
   529  	public function resolveVisual( clazz:Function , name:String , depth )
   530  	{
   531  		return ConstructorUtil.createVisualInstance( clazz , resolve( name , depth ) ) ;
