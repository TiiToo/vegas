    24  import asgard.display.DisplayObject;
    25  
    26  import flash.display.BitmapData;
    27  
    28  /**
    29   * Display a BitmapData object in a MovieClip.
    30   * <p><b>Example :</b></p>
    31   * {@code
    32   * import asgard.display.Bitmap ; 
    33   * import flash.display.BitmapData ;
    34   * 
    35   * var url:String = "pic/picture.jpg" ;
    36   * var container:MovieClip = createEmptyMovieClip("container", 1) ;
    37   * container._x = 10 ;
    38   * container._y = 10 ;
    39   * 
    40   * var loader:MovieClipLoader = new MovieClipLoader() ;
    41   * loader.addListener(this) ;
    42   * 
    43   * var bmp:BitmapData ;
    44   * var bmpMC:MovieClip = createEmptyMovieClip("bmpMC", 2) ;
    45   * 
    46   * var bitmap:Bitmap = new Bitmap("bmp1", bmpMC) ;
    47   * 
    48   * var onLoadInit:Function = function ( target ):Void
    49   * {
    50   *     bmp = new BitmapData(target._width, target._height) ;
    51   *     bmp.draw(target) ;
    52   *     
    53   *     bitmap.x = container._x + container._width + 10 ;
    54   *     bitmap.y = container._y ;
    55   *     
    56   *     trace(bitmap.width + " : " + bitmap.height) ;
    57   *     
    58   *     bitmap.bitmapData = bmp ;
    59   *     
    60   *     trace(bitmap.width + " : " + bitmap.height) ;
    61   * }
    62   * 
    63   * loader.loadClip(url, container) ;
    64   * 
    65   * var onKeyDown:Function = function():Void
    66   * {
    67   *     // bitmap.clear() ; // test clear method
    68   *     bitmap.bitmapData = null ;
    69   *     trace(bitmap.width + " : " + bitmap.height) ;
    70   * }
    71   * Key.addListener(this) ;
    72   * 
    73   * }
    74   * @author eKameleon
    75   */
    76  class asgard.display.Bitmap extends DisplayObject 
    77  {
    78  	
    79  	/**
    80  	 * Creates a new Bitmap instance.
    81  	 */
    82  	public function Bitmap( sName:String, target:MovieClip, bmp:BitmapData, pixelSnapping:String, smoothing:Boolean ) 
    83  	{
    84  		
    85  		super( sName, target ) ;
    86  		
    87  		clear() ;
    88  		
    89  		setPixelSnapping(pixelSnapping, true) ;
    90  		
    91  		setSmoothing(smoothing, true) ;
    92  		
    93  		setBitmapData(bmp) ;
    94  		
    95  	}
    96  
    97  	/**
    98  	 * The depth of the internal container of this display.
    99  	 * Overrides this protected constant if you want change this level in your extended display of this Bitmap display.
   100  	 */
   101  	/*protected*/ private static var CONTAINER_DEPTH:Number = 100 ;
   102  
   103  	/**
   104  	 * Returns the BitmapData object being referenced.
   105  	 * @return the BitmapData object being referenced.
   106  	 */
   107  	public function get bitmapData():BitmapData
   108  	{
   109  		return getBitmapData() ;	
   110  	}
   111  
   112  	/**
   113  	 * Sets the BitmapData object being referenced.
   114  	 */
   115  	public function set bitmapData(bmp:BitmapData):Void
   116  	{
   117  		setBitmapData(bmp) ;	
   118  	}
   119  	
   120  	/**
   121  	 * (read-only) Returns the bitmap container reference.
   122  	 * @return the bitmap container reference. 
   123  	 */
   124  	public function get container():MovieClip
   125  	{
   126  		return _container ;	
   127  	}
   128  	
   129  	/**
   130  	 * (read-write) Controls whether or not the Bitmap object is snapped to the nearest pixel.
   131  	 * <p>The PixelSnapping class includes possible values :
   132  	 * <ul>
   133  	 * <li>PixelSnapping.NEVER No pixel snapping occurs.</li>
   134  	 * <li>PixelSnapping.ALWAYS The image is always snapped to the nearest pixel, independent of transformation.</li>
   135  	 * <li>PixelSnapping.AUTO The image is snapped to the nearest pixel if it is drawn with no rotation or skew and it is drawn at a scale factor of 99.9% to 100.1%. If these conditions are satisfied, the bitmap image is drawn at 100% scale, snapped to the nearest pixel. Internally, this value allows the image to be drawn as fast as possible using the vector renderer.</li>
   136  	 * </ul>
   137  	 * </p>
   138  	 * @since Flash Player 8.5
   139  	 */
   140  	public function get pixelSnapping():String
   141  	{
   142  		return getPixelSnapping() ;	
   143  	}
   144  
   145  	/**
   146  	 * (read-write) Controls whether or not the Bitmap object is snapped to the nearest pixel.
   147  	 * <p>The PixelSnapping class includes possible values :
   148  	 * <ul>
   149  	 * <li>PixelSnapping.NEVER No pixel snapping occurs.</li>
   150  	 * <li>PixelSnapping.ALWAYS The image is always snapped to the nearest pixel, independent of transformation.</li>
   151  	 * <li>PixelSnapping.AUTO The image is snapped to the nearest pixel if it is drawn with no rotation or skew and it is drawn at a scale factor of 99.9% to 100.1%. If these conditions are satisfied, the bitmap image is drawn at 100% scale, snapped to the nearest pixel. Internally, this value allows the image to be drawn as fast as possible using the vector renderer.</li>
   152  	 * </ul>
   153  	 * </p>
   154  	 * @since Flash Player 8.5
   155  	 */
   156  	public function set pixelSnapping( snapping:String ):Void
   157  	{
   158  		setPixelSnapping( snapping ) ;	
   159  	}
   160  	
   161  	/**
   162  	 * (read-write) Controls whether or not the bitmap is smoothed when scaled. 
   163  	 * If true, the bitmap is smoothed when scaled. If false, the bitmap is not smoothed when scaled. 
   164  	 */ 
   165  	public function get smoothing():Boolean
   166  	{
   167  		return getSmoothing() ;	
   168  	}
   169  
   170  	/**
   171  	 * (read-write) Controls whether or not the bitmap is smoothed when scaled. 
   172  	 * If true, the bitmap is smoothed when scaled. If false, the bitmap is not smoothed when scaled. 
   173  	 */
   174  	public function set smoothing(b:Boolean):Void
   175  	{
   176  		setSmoothing(b) ;	
   177  	}
   178  	
   179  	/**
   180  	 * Clear the display.
   181  	 */
   182  	public function clear():Void
   183  	{
   184  		_container = view.createEmptyMovieClip("__container__", CONTAINER_DEPTH ) ;
   185  		_container.view = _container.createEmptyMovieClip("view", 1) ; 
   186  	}
   187  	
   188  	/**
   189  	 * Disposes the BitmapData in the display.
   190  	 */
   191  	public function dispose():Void
   192  	{
   193  		_bmp.dispose() ;	
   194  	}
   195  	
   196  	/**
   197  	 * Returns the BitmapData reference.
   198  	 * @return the BitmapData reference.
   199  	 */
   200  	public function getBitmapData():BitmapData
   201  	{
   202  		return _bmp ;	
   203  	}
   204  
   205  	/**
   206  	 * Returns the bmp content target who contains the view of the BitmapData.
   207  	 * @return the bmp content target.
   208  	 */
   209  	public function getContent():MovieClip
   210  	{
   211  		return _container.view ;	
   212  	}
   213  
   214  	/**
   215  	 * Returns the pixel snapping value.
   216  	 */
   217  	public function getPixelSnapping():String
   218  	{
   219  		return _pixelSnapping ; 	
   220  	} 
   221  
   222  	/**
   223  	 * Returns the smoothing value.
   224  	 */
   225  	public function getSmoothing():Boolean
   226  	{
   227  		return _smoothing ; 	
   228  	} 
   229  
   230  	/**
   231  	 * Release the display.
   232  	 */
   233  	/*override*/ public function release():Void
   234  	{
   235  		super.release() ;
   236  		dispose() ;
   237  		_container.removeMovieClip() ;	
   238  	}
   239  
   240  	/**
   241  	 * Sets the bitmapData reference in this display.
   242  	 */
   243  	public function setBitmapData(bmp:BitmapData):Void
   244  	{
   245  		if (_bmp != null)
   246  		{
   247  			_bmp.dispose() ;
   248  			_bmp = null ;	
   249  		}
   250  		if ( bmp != null )
   251  		{
   252  			_bmp = bmp.clone() ;
   253  		}
   254  		update() ;	
   255  	}
   256  
   257  	/**
   258  	 * Sets the pixel snapping value in this display.
   259  	 */
   260  	public function setPixelSnapping( snapping:String, noUpdate:Boolean ):Void
   261  	{
   262  		_pixelSnapping = snapping ;
   263  		if ( noUpdate ) return ;
   264  		update() ;
   265  	} 
   266  
   267  	/**
   268  	 * Sets the smoothing value in this display.
   269  	 */
   270  	public function setSmoothing(b:Boolean, noUpdate:Boolean):Void
   271  	{
   272  		_smoothing = b ;
   273  		if ( noUpdate ) return ;
   274  		update() ; 	
   275  	} 
   276  
   277  	/**
   278  	 * Update the display.
   279  	 */
   280  	public function update():Void
   281  	{
   282  		_container.view.attachBitmap(_bmp, 1, getPixelSnapping(), getSmoothing()) ;
   283  	}
   284  
   285  	/**
   286  	 * The internal BitmapData reference.
   287  	 */	
   288  	private var _bmp:BitmapData ;
   289  
   290  	/**
   291  	 * The internal movieclip reference to attach the bitmap.
   292  	 */
   293  	private var _container:MovieClip ;
   294  
   295  	/**
   296  	 * The default value of the BitmapData pixelSnapping property.
   297  	 */
   298  	private var _pixelSnapping:String = "auto" ; 
   299  
   300  	/**
   301  	 * The default value of the BitmapData smoothing propety.
   302  	 */
   303  	private var _smoothing:Boolean = false ;
   304  
   305  }
