    24  /**
    25   * The InterpolationMethod class provides values for the interpolationMethod parameter in the MovieClip.beginGradientFill() and MovieClip.lineGradientStyle() methods.
    26   * @author eKameleon
    27   */
    28  class asgard.display.InterpolationMethod 
    29  {
    30  
    31  	/**
    32  	 * Specifies that the linear RGB interpolation method should be used. 
    33  	 * This means that Flash Player uses an RGB color space based on a linear RGB color model.
    34   	 */
    35  	public static var LINEAR_RGB:String = "linearRGB" ;
    36  	
    37  	/**
    38  	 * Specifies that the RGB interpolation method should be used. 
    39  	 * This means that Flash Player uses the exponential sRGB (standard RGB) space when rendering the gradient. 
    40  	 * The sRGB space is a W3C-endorsed standard that defines a non-linear conversion between red, green, and blue component values and the actual intensity of the visible component color.
    41  	 */
    42  	public static var RGB:String = "rgb" ;
    43  	
    44  	private static var __ASPF__ = _global.ASSetPropFlags(InterpolationMethod, null , 7, 7) ;
    45  
    46  }
