   310  	/**
   311  	 * Scale in the container.
   312  	 */
   313  	public function scaleIn( p:Point ):Void
   314  	{
   315  
   316  		if ( _scale * _scaleStep > _rScale.max )  
   317  		{
   318  			return ;
   319  		}
   320  
   321  		if (p == null) 
   322  		{
   323  			p = getCenter() ;	
   324  		}
   325  
   326  		var p1:Point = new Point( p.x - _offset.x , p.y - _offset.y ) ;
   327  		var p2:Point = new Point( p1.x * _scaleStep , p1.y * _scaleStep ) ;
   328  		
   329  		_offset.x -= p2.x - p1.x ;
   330  		_offset.y -= p2.y - p1.y ;
   331  
   332  		_container._x = _offset.x ;
   333  		_container._y = _offset.y ;
   334  		
   335  		_container._xscale = _scaleReference * ( _scale *= _scaleStep ) ;
   336  		_container._yscale = _container._xscale ;
   337  		
   338  		_boundPosition() ;
   339  		
   340  		dispatchEvent ( new UIEvent(UIEventType.CHANGE, this) ) ; // TODO : ScaleEvent
   341  		
   342  	}
   343  	
   344  	/**
   345  	 * Scale out the container.
   346  	 */
   347  	public function scaleOut( p:Point ):Void
   348  	{
   349  
   350  		if ( ( _scale / _scaleStep ) < _rScale.min ) 
   351  		{
   352  			return ;
   353  		}
   354  
   355  		if (p == null) 
   356  		{
   357  			p = getCenter() ;	
   358  		}
   359  
   360  		var p1:Point = new Point( p.x - _offset.x , p.y - _offset.y  ) ;
   361  		var p2:Point = new Point( p1.x / _scaleStep , p1.y / _scaleStep ) ;
   362  		
   363  		_offset.x -= p2.x - p1.x  ;
   364  		_offset.y -= p2.y - p1.y ;
   365  		
   366  		_container._x = _offset.x ;
   367  		_container._y = _offset.y ;
   368  		
   369  		_container._xscale = _scaleReference * ( _scale /= _scaleStep ) ;
   370  		_container._yscale = _container._xscale ;
   371  		
   372  		_boundPosition() ;
   373  		
   374  		dispatchEvent ( new UIEvent(UIEventType.CHANGE, this) ) ; // TODO : ScaleEvent
   375  	
   376  	}

   417  	/**
   418  	 * Sets the scale value.
   419  	 */
   420  	public function setScale( value:Number, p:Point ):Void 
   421  	{
   422  
   423  		value = _rScale.clamp(value) ;
   424  	
   425  		if (p == null) 
   426  		{
   427  			p = getCenter() ;	
   428  		}
   429  	
   430  		var coef:Number = value / _scale ;
   431  		
   432  		var p1:Point = new Point( p.x - _offset.x , p.y - _offset.y ) ;
   433  		var p2:Point = new Point( p1.x * coef , p1.y * coef ) ;
   434  		
   435  		_offset.x -= p2.x - p1.x;
   436  		_offset.y -= p2.y - p1.y;
   437  		_container._x = _offset.x ;
   438  		_container._y = _offset.y ;
   439  		_container._xscale = _scaleReference * ( _scale *= coef ); 
   440  		_container._yscale = _container._xscale ;
   441  		
   442  		_boundPosition() ;
   443  		
   444  		dispatchEvent ( new UIEvent(UIEventType.CHANGE, this) ) ; // TODO : ScaleEvent
   445  		
   446  	}

   496  	/**
   497  	 * Update the display.
   498  	 */
   499  	public function update():Void
   500  	{
   501  		if ( isNaN(_container._width) || _container._width == 0)
   502  		{
   503  			_recMask.width = _w ;
   504  		}
   505  		else if (_w < _container._width)
   506  		{
   507  			_recMask.width = _w ;	
   508  		}
   509  		else
   510  		{
   511  			_recMask.width = _container._width ;	
   512  		}
   513  
   514  		if ( isNaN(_container._height) || _container._height == 0)
   515  		{
   516  			_recMask.height = _h ;
   517  		}
   518  		else if (_h < _container._height)
   519  		{
   520  			_recMask.height = _h ;	
   521  		}
   522  		else
   523  		{
   524  			_recMask.height = _container._height ;	
   525  		}
   526  		view.scrollRect = _recMask.toFlash() ;
   527  		
   528  		_boundPosition() ;
   529  		
   530  	}

   587  	/**
   588  	 * Bounds the container if the view is out of the bounds of the visible area.
   589  	 */
   590  	private function _boundPosition():Void
   591  	{
   592  
   593  		if (_container._x + _container._width < _recMask.width) 
   594  		{
   595  			_container._x = _recMask.width - _container._width ;
   596  		}
   597  		if ( _container._y + _container._height < _recMask.height ) 
   598  		{
   599  			_container._y = _recMask.height - _container._height ;
   600  		}
   601  		if ( _container._x > _recMask.x ) 
   602  		{
   603  			_container._x = _recMask.x ;
   604  		}
   605  		if ( _container._y > _recMask.y ) 
   606  		{
   607  			_container._y = _recMask.y ;
   608  		}
   609  
   610  	}

   618  		_offset.y = _container._y ;
   619  	 }

   621  	/**
   622  	 * Activate the drag handler if the use press the container.
   623  	 * This method is avite only if the display is "isRegisterDrag".
   624  	 * @see registerDrag
   625  	 * @see unregisterDrag
   626  	 */ 
   627  	private function _startDrag():Void
   628  	{
   629  		var $w:Number = - (_container._width - _recMask.width) ;
   630  		var $h:Number = - (_container._height - _recMask.height) ;
   631  		this.startDrag(false, 0, 0, $w, $h) ; 
   632  	}

   640  	}
