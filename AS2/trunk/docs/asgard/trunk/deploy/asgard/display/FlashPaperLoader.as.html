   398  	/**
   399  	 * Refresh the view.
   400  	 */
   401  	public function update():Void 
   402  	{
   403  		if (getIsFlashPaperDocument()) 
   404  		{
   405  			var w:Number = isNaN(_w) ? _default_w : _w ; 
   406  			var h:Number = isNaN(_h)? _default_h : _h ;
   407  			enableScrolling(_isEnableScrolling) ;
   408  			_fp.setSize(w, h) ;
   409  			if (!isLogoVisible()) 
   410  			{
   411  				hideLogo() ;
   412  			}
   413  			if (!isToolBarBackgroundVisible()) 
   414  			{
   415  				hideToolBarBackground() ;
   416  			}
   417  		}
   418  	}

   491  	/**
   492  	 * Fix the selection management.
   493  	 */
   494  	static private function _fixSelectionManagement(mc:MovieClip):Void 
   495  	{
   496  		var main:MovieClip = mc.gMainView.m_mainMC ;
   497  		
   498  		if (typeof (main.onMouseDown) == "function" && main.hasSafeSelectionManagement === undefined) 
   499  		{
   500        		main.onMouseDown =  function() 
   501        		{
   502           		var fpfocus:MovieClip = _global.FPUI.Component.focusedComponent;
   503           		
   504           		if (fpfocus != null) 
   505           		{
   506  		            var hitFocused:Boolean = false;
   507              		if (fpfocus._visible) 
   508              		{
   509  	               		var p:Point = new Point(_root._xmouse, _root._ymouse);
   510  	               		_root.localToGlobal(p) ;
   511  	               		hitFocused = fpfocus.hitTest(p.x, p.y, true);
   512              		}
   513              		if (hitFocused == false) 
   514              		{
   515  						//Selection.setFocus(null) ;
   516  						if (typeof (fpfocus.onComponentKillFocus) == "function") 
   517  						{
   518                    			fpfocus.onComponentKillFocus();
   519                 			}
   520              		}
   521           		}
   522        		} ;
   523        		
   524     		}
   525  	}
