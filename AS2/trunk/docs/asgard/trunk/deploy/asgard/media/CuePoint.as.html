    26  /**

    30  class asgard.media.CuePoint extends CoreObject

    32  
    33  	/**
    34  	 * Creates a new CuePoint instance.
    35  	 * @param o An Object with the properties 'name', 'parameters', 'time' and 'type' used to define this CuePoint object.
    36  	 */	
    37  	public function CuePoint( o ) 
    38  	{
    39  		if (o != null)
    40  		{
    41  			this.name = o.name ;
    42  			this.time = o.time ;
    43  			this.type = o.type ;
    44  			this.parameters = o.parameters ;
    45  		}
    46  	}
    47  	
    48  	/**
    49  	 * The name of the CuePoint.
    50  	 */
    51  	public var name:String ;
    52  	
    53  	/**
    54  	 * A associative array of name/value pair strings specified for this cue point. 
    55  	 * Any valid string can be used for the parameter name or value.
    56  	 */
    57  	public var parameters:Array ;
    58  
    59  	/**
    60  	 * The time of the CuePoint.
    61  	 */
    62  	public var time:String ;
    63  	
    64  	/**
    65  	 * The type of the CuePoint.
    66  	 */
    67  	public var type:String ;
    68  	
    69  	/**
    70  	 * Returns the String representation of this object.
    71  	 * @return the String representation of this object.
    72  	 */
    73  	public function toString():String
    74  	{
    75  		var txt:String = "[CuePoint" ;
    76  		if (name != null)
    77  		{
    78  			txt += ",name:" + name ;
    79  		}
    80  		if (time != null)
    81  		{
    82  			txt += ",time:" + time ;
    83  		}
    84  		if (type != null)
    85  		{
    86  			txt += ",type:" + type ;
    87  		} 
    88  		if (parameters != null)
    89  		{
    90  			txt += ",parameters:" + parameters ;
    91  		} 
    92  		txt += "]" ;
    93  		return txt ; 	
    94  	}
    95  }
