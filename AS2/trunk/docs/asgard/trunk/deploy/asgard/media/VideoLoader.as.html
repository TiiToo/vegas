   484  	/**
   485  	 * Internal load method.
   486  	 * @throws if the url property is invalid.
   487  	 */
   488  	private function _load():Void 
   489  	{
   490  
   491  		if ( this.getUrl() == null ) 
   492  		{
   493  			throw new UnsupportedOperation( this + " can't play without any valid url property, loading fails.");
   494  		}
   495  		
   496  		setLoaded(true) ;
   497  		
   498  		_oMetaData = null ;
   499  
   500  		_oNS = new NetStream( _oNC ) ;
   501  		_oNS.setBufferTime( _nBufferTime ) ;
   502  		_oNS.onStatus = Delegate.create(this, _onNetStreamStatus) ;
   503  		_oNS["onCuePoint"] = Delegate.create(this, _onCuePoint) ; 
   504  		_oNS["onMetaData"] = Delegate.create(this, _onMetaData) ;
   505  		
   506  		_oNS.toString = function()
   507  		{
   508  			return "[NetStream]" ;
   509  		} ;
   510  		
   511  		setContent( _oNS ) ;
   512  		
   513  		_oVideo.attachVideo(_oNS) ;
   514  		_mcTarget.attachAudio(_oNS) ;
   515  				
   516  		_oNS.play( this.getUrl() );
   517  
   518  		startProgress() ;
   519  
   520  		if ( isAutoPlay() == false ) 
   521  		{
   522  			this.pause(true) ;
   523  		} 
   524  
   525  		super.load();
   526  		
   527  	}

   560

   564  		}
   565  		
   566  		_oMetaData = new FLVMetaData(info) ;

   568  		setDuration( isNaN(info.duration) ? 0 : parseInt(info.duration) ) ;

   577  	private function _onNetServerStatus( e:NetServerEvent ):Void
   578  	{
   579  		
   580  		var status:NetServerStatus = e.getStatus() ;
   581  
   582  		switch ( true ) 
   583  		{
   584  
   585  			case status.equals( NetServerStatus.CLOSED )  :
   586  			{ 
   587  				getLogger().info(this + " connect closed.") ;
   588  				stopProgress() ;
   589  				break ;
   590  			}	
   591  			
   592  			case status.equals( NetServerStatus.SUCCESS )  :
   593  			{
   594  				getLogger().info(this + " connect success.") ;
   595  				break ;
   596  			}	
   597  			
   598  			case status.equals( NetServerStatus.FAILED )  :
   599  			{ 
   600  				getLogger().warn(this + " connect failed.") ;
   601  				break ;
   602  			}
   603  		}	
   604  	}

   606  	private var _isFull:Boolean = false ;

   608  	/**
   609  	 * Invoqued when the status of the stream change.
   610  	 */
   611  	private function _onNetStreamStatus( info:Object ):Void 
   612  	{
   613  		
   614  		var code:String = info.code ;
   615  		
   616  		getLogger().debug( this + " stream status : " + code + ", level : " + (info.level || "") ) ;
   617  		
   618  		switch ( true ) 
   619  		{
   620  			
   621  			case NetStreamStatus.BUFFER_EMPTY.equals(code) :
   622  			{
   623  				if (_isFull)
   624  				{
   625  					_checkStop() ;
   626  				}
   627  				break ;
   628  			}
   629  			
   630  			case NetStreamStatus.PLAY_START.equals(code) :
   631  			{
   632  				_isFull = false ;
   633  				// getLogger().info( this + " stream starts playing.");
   634  				break;
   635  			}		
   636  			case NetStreamStatus.PLAY_STOP.equals(code) :
   637  			{
   638  				_isFull = true ;
   639  				break;
   640  			}
   641  			
   642  			case NetStreamStatus.PLAY_STREAM_NOT_FOUND.equals(code) :
   643  			{
   644  				getLogger().warn( this + " stream not found.");
   645  				stopProgress() ;
   646  				break ;
   647  			}
   648  			
   649  			case NetStreamStatus.SEEK_INVALID_TIME.equals(code) :
   650  			{
   651  				getLogger().warn( this + " seeks invalid time in '" + this.getUrl() + "'.");
   652  				break;
   653  			}	
   654  			
   655  			case NetStreamStatus.BUFFER_FULL.equals(code) :
   656  			{
   657  				getLogger().info( this + " stream buffer is full." );
   658  				break;
   659  			}
   660  
   661  		}
   662  	}
   663  	
   664  	private function _checkStop():Void
   665  	{
   666  		if (_isFull)
   667  		{
   668  			_isFull = false ;
   669  			notifyEvent(MediaEvent.MEDIA_FINISH) ;
   670  			if (isLoop()) 
   671  			{
   672  				this.play(0) ;
   673  			}
   674  			else 
   675  			{
   676  				this.stop() ;	
   677  			}
   678  		}
   679  	}
