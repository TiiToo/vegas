    23  
    24  import pegas.maths.Range;
    25  
    26  import vegas.errors.IllegalArgumentError;
    27  import vegas.events.NumberEvent;
    28  import vegas.util.factory.DisplayFactory;
    29  
    30  /**
    31   * This display control a Video object and this attached Sound object.
    32   * @author eKameleon
    33   */
    34  class asgard.media.AbstractVideoDisplay extends BackgroundDisplay
    35  {
    36  	
    37  	/**
    38  	 * Creates a new AbstractVideoDisplay instance.
    39  	 * If the target reference contains a {@ode Video} instance with the name "video", this object is used in the display.
    40  	 * @param sName the name of the display.
    41  	 * @param target the target of this display.
    42  	 * @param (optional) the {@code Video} object of this display. 
    43  	 */	
    44  	public function AbstractVideoDisplay( sName:String, target:MovieClip, video:Video) 
    45  	{
    46  		
    47  		super(sName, target);
    48  
    49  		_oVideo = video ? video : view.video ;
    50  		
    51  		DisplayFactory.swapDepths( _oVideo, DEFAULT_VIDEO_DEPTH ) ;
    52  		
    53  		setSize( view._width , view._height ) ;
    54  		
    55  		_oVideo.toString = function():String
    56  		{
    57  			return "[Video]" ;	
    58  		} ;
    59  		
    60  		_oSound = new Sound( view ) ;
    61  		_oSound.setVolume( DEFAULT_VOLUME ) ;
    62  		_oSound.toString = function()
    63  		{
    64  			return "[Sound]" ;	
    65  		} ;
    66  
    67  		if ( !_oVideo ) 
    68  		{
    69  			throw new IllegalArgumentError( this + " constructor failed. Invalid video object was passed in the constructor.");
    70  		}
    71  		
    72  	}
    73  
    74  	/**
    75  	 * The depth of the video reference.
    76  	 */
    77  	public static var DEFAULT_VIDEO_DEPTH:Number = 1 ;
    78  
    79  	/**
    80  	 * The default value of the volume in the display.
    81  	 */
    82  	public static var DEFAULT_VOLUME:Number = 80 ;
    83  
    84  	/**
    85  	 * const Defined the event name of the event 
    86  	 */
    87  	public static var SOUND_VOLUME_CHANGE:String = "onVolumeChange" ;
    88  
    89  	/**
    90  	 * The alpha value of the screen.
    91  	 */
    92  	public var themeAlpha:Number = null ;
    93  	
    94  	/**
    95  	 * The color value of the screen.
    96  	 */
    97  	public var themeColor:Number = null ;
    98  
    99  	/**
   100  	 * The view reference of this display.
   101  	 */
   102  	public var view:MovieClip ;
   103  	
   104  	/**
   105  	 * Returns the sound's volume value of the UI.
   106  	 * @return the sound's volume value of the UI.
   107  	 */
   108  	public function get volume():Number
   109  	{
   110  		return _oSound.getVolume() ;	
   111  	}
   112  	
   113  	/**
   114  	 * Sets the sound's volume value of the display.
   115  	 * This property dispatch a {@code NumberEvent} object with the volume value returns by the getNumber() method of the event.
   116  	 * To handle this event you can use the {@code SOUND_VOLUME_CHANGE} static property.
   117  	 * @param value the Number value of the volume of this display.
   118  	 */
   119  	public function set volume( value:Number ):Void
   120  	{
   121  		var n:Number = Range.PERCENT_RANGE.clamp(value) ;
   122  		_oSound.setVolume( n ) ;
   123  		dispatchEvent( new NumberEvent( SOUND_VOLUME_CHANGE , _oSound.getVolume() , this ) ) ;
   124  	}
   125  
   126  	/**
   127  	 * Attaching an audio NetStream or a Microphone source to play.
   128  	 * @param oSource the audio source.  
   129  	 */
   130  	public function attachAudio( oSource ):Void
   131  	{
   132  		view.attachAudio( oSource ) ;	
   133  	}
   134  	
   135  	/**
   136  	 * Attaching the client camera
   137  	 * @param cam The specified {@code Camera} object.
   138  	 */
   139  	public function attachCamera( cam:Camera ):Void
   140  	{
   141  		_oVideo.attachVideo( cam ) ;	
   142  	}
   143  	
   144  	/**
   145  	 * Attaching a NetStream object.
   146  	 * @param stream The specified {@code NetStream} object.
   147  	 */
   148  	public function attachStream( stream:NetStream ):Void
   149  	{
   150  		_oVideo.attachVideo( stream ) ;	
   151  	}
   152  	
   153  	/**
   154  	 * Clear the video in the display.
   155  	 */
   156  	public function clear():Void
   157  	{
   158  		view.attachAudio(null) ;
   159  		_oVideo.attachVideo(null) ;
   160  		_oVideo.clear() ;
   161  	}
   162  
   163  	/**
   164  	 * Returns the sound reference of this display.
   165  	 * @return the sound reference of this display.
   166  	 */
   167  	public function getSound():Sound
   168  	{
   169  		return _oSound ;	
   170  	}
   171  	
   172  	/**
   173  	 * Returns the video reference of this display.
   174  	 * @return the video reference of this display.
   175  	 */
   176  	public function getVideo():Video
   177  	{
   178  		return _oVideo ;
   179  	}
   180  
   181  	/**
   182  	 * Sets the virtual height value of the component.
   183  	 */
   184  	public function setH( n:Number ) : Void 
   185  	{
   186  		view._height = n ;
   187  		super.setH(n) ;
   188  	}
   189  
   190  	/**
   191  	 * Sets the virtual width and height values of the component.
   192  	 */
   193  	public function setSize( w:Number, h:Number ) : Void 
   194  	{
   195  		
   196  		_oVideo._width  = w;
   197  		_oVideo._height = h ;
   198  		_oVideo.width  = w;
   199  		_oVideo.height = h ;
   200  		
   201  		super.setSize( w, h ) ;
   202  	}
   203  
   204  	/**
   205  	 * Sets the virtual width value of the component.
   206  	 */
   207  	public function setW( n:Number ):Void 
   208  	{
   209  		view._width = n ;
   210  		super.setW(n) ;
   211  	}
   212  
   213  	/**
   214  	 * Internal sound object.
   215  	 */
   216  	private var _oSound : Sound;
   217  	
   218  	/**
   219  	 * Internal video object.
   220  	 */
   221  	private var _oVideo : Video;
   222  
   223  
   224  }
