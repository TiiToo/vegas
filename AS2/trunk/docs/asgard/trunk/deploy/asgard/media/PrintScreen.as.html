    24  import asgard.display.DisplayObject;

    38  /**
    39   * This class capture a BitmapData over a movieclip or a DisplayObject.
    40   * @author eKameleon
    41   */
    42  class asgard.media.PrintScreen extends AbstractAction implements Action
    43  {
    44  
    45  	/**
    46  	 * Creates a new PrintScreen instance.
    47  	 */	
    48  	function PrintScreen( display , x:Number, y:Number, width:Number, height:Number ) 
    49  	{
    50  		
    51  		_buffer = [] ;
    52  		
    53  		_eClear = new PrintScreenEvent(PrintScreenEvent.CLEAR, this) ;
    54  		_eFinish = new PrintScreenEvent(PrintScreenEvent.FINISH, this) ;
    55  		_eProgress = new PrintScreenEvent(PrintScreenEvent.PROGRESS, this) ;
    56  		_eStart = new PrintScreenEvent(PrintScreenEvent.START, this) ;
    57  
    58  		_rec = new Rectangle() ;
    59  		
    60  		_timer = new Timer( DEFAULT_PROGRESS_DELAY ) ;
    61  		_timer.addEventListener( TimerEvent.TIMER , new Delegate(this, _onTimer ) ) ;
    62  		
    63  		setTarget( display ) ;
    64  		
    65  		if (isNaN(x))
    66  		{
    67  			x = 0 ;	
    68  		}
    69  		
    70  		if (isNaN(y))
    71  		{
    72  			y = 0 ;	
    73  		}
    74  		
    75  		var w:Number = isNaN(width)  ? _target._width  : width ;
    76  		var h:Number = isNaN(height) ? _target._height : height ;
    77  		
    78  		setRectangle( x, y, w, h ) ;
    79  		
    80  	}
    81  
    82  	/**
    83  	 * The default delay to notify the PrintScreen.PROGRESS event.
    84  	 */	
    85  	static public var DEFAULT_PROGRESS_DELAY:Number = 5 ;
    86  
    87  	/**
    88  	 * Returns the height of the capture.
    89  	 * @return the height of the capture.
    90  	 */
    91  	public function get height():Number
    92  	{
    93  		return _rec.height ;	
    94  	}
    95  
    96  	/**
    97  	 * (read-only) Returns {@code true} if the process is in progress.
    98  	 * @return {@code true} if the process is in progress.
    99  	 */
   100  	public function get running():Boolean 
   101  	{
   102  		return getRunning() ;	
   103  	}
   104  
   105  	/**
   106  	 * Returns the width of the capture.
   107  	 * @return the width of the capture.
   108  	 */
   109  	public function get width():Number
   110  	{
   111  		return _rec.width ;	
   112  	}
   113  
   114  	/**
   115  	 * Clear the buffer of this PrintScreen object.
   116  	 */
   117  	public function clear():Void
   118  	{
   119  		if (_bmp)
   120  		{
   121  			_bmp.dispose() ;
   122  		}
   123  		_bmp = new BitmapData( _rec.width , _rec.height ) ;
   124  		_cpt = 0 ;
   125  		_time = 0 ;
   126  		_buffer = [] ;
   127  		notifyCleared() ;
   128  	}
   129  	
   130  	/**
   131  	 * Returns a shallow copy of this object.
   132  	 * @return a shallow copy of this object.
   133  	 */
   134  	public function clone() 
   135  	{
   136  		var rec:Rectangle = getRectangle() ;
   137  		return  new PrintScreen( getTarget() , rec.x, rec.y, rec.width, rec.height) ;
   138  	}
   139  
   140  	/**
   141  	 * Returns the internal BitmapData reference.
   142  	 * @return the internal BitmapData reference.
   143  	 */
   144  	public function getBitmapData():BitmapData
   145  	{
   146  		return _bmp ;	
   147  	}
   148  
   149  	/**
   150  	 * Returns the Array representation of the PrintScreen's buffer.
   151  	 * @return the Array representation of the PrintScreen's buffer.
   152  	 */
   153  	public function getBuffer():Array
   154  	{
   155  		return _buffer ;	
   156  	}
   157  
   158  	/**
   159  	 * Returns the delay in ms between the capture and the end of the process.
   160  	 * @return the delay in ms between the capture and the end of the process.
   161  	 */
   162  	public function getDelay():Number
   163  	{
   164  		return _timer.getDelay() ;
   165  	}
   166  	
   167  	/**
   168  	 * Returns the Rectangle reference of this PrintScreen object.
   169  	 * @return the Rectangle reference of this PrintScreen object.
   170  	 */
   171  	public function getRectangle():Rectangle
   172  	{
   173  		return _rec ;	
   174  	}
   175  
   176  	/**
   177  	 * Returns the progress percent value of the capture process.
   178  	 * @return the progress percent value of the capture process.
   179  	 */
   180  	public function getPercent():Number
   181  	{
   182  		var n:Number = Math.min(100, Math.ceil( _cpt / ( _size / 100 ) ) );
   183  		return (isNaN(n)) ? 0 : n ;
   184  	}
   185  
   186  	/**
   187  	 * Returns the MovieClip display of this PrintScreen object.
   188  	 * @return the MovieClip display of this PrintScreen object.
   189  	 */
   190  	public function getTarget():MovieClip
   191  	{
   192  		return _target ;
   193  	}
   194  
   195  	/**
   196  	 * Returns {@code true} if the process is in progress.
   197  	 * @return {@code true} if the process is in progress.
   198  	 */
   199  	public function getRunning():Boolean 
   200  	{
   201  		return _isRunning ;	
   202  	}
   203  	
   204  	/**
   205  	 * Returns the elapsed time during the capture (in ms).
   206  	 * @return the elapsed time during the capture (in ms).
   207  	 */
   208  	public function getTime():Number
   209  	{
   210  		return isNaN(_time) ? 0 : _time ;	
   211  	}
   212  	
   213  	/**
   214  	 * Notify an PrintScreenEvent when the process is cleared.
   215  	 */
   216  	public function notifyCleared():Void 
   217  	{
   218  		dispatchEvent(_eClear) ;
   219  	}	
   220  
   221  	/**
   222  	 * Notify an PrintScreenEvent when the process is finished.
   223  	 */
   224  	public function notifyFinished():Void 
   225  	{
   226  		_timer.stop() ;
   227  		_setRunning(false) ;
   228  		dispatchEvent(_eFinish) ;
   229  	}
   230  
   231  	/**
   232  	 * Notify an PrintScreenEvent when the process is in progress.
   233  	 */
   234  	public function notifyProgress():Void 
   235  	{
   236  		dispatchEvent(_eProgress) ;
   237  	}
   238  	
   239  	/**
   240  	 * Notify an PrintScreenEvent when the process is started.
   241  	 */
   242  	public function notifyStarted():Void 
   243  	{
   244  		dispatchEvent(_eStart) ;
   245  		_setRunning(true) ;
   246  		_timer.start() ;
   247  	}
   248  
   249  	/**
   250  	 * Run the capture process.
   251  	 */
   252  	public function run():Void
   253  	{
   254  		
   255  		clear() ;
   256  		
   257  		if ( _target == null )
   258  		{
   259  			return ;	
   260  		}
   261  		
   262  		notifyStarted() ;
   263  		
   264  		_time = getTimer() ;
   265  		_bufferize() ;
   266  		_time = getTimer() - _time ;
   267  		
   268  		notifyFinished() ;
   269  		
   270  	}
   271  	
   272  	/**
   273  	 * Sets the delay in ms between the capture and the end of the process.
   274  	 */
   275  	public function setDelay( time:Number ):Void
   276  	{
   277  		_timer.setDelay( time ) ;	
   278  	}
   279  	
   280  	/**
   281  	 * Sets the internal rectangle area properties.
   282  	 * @param x the x origin position of the internal capture area.
   283  	 * @param y the y origin position of the internal capture area.
   284  	 * @param width the width of the internal capture area.
   285  	 * @param height the height of the internal capture area.
   286  	 */
   287  	public function setRectangle( x:Number, y:Number, width:Number, height:Number ):Void
   288  	{
   289  		_rec.x = isNaN(x) ? 0 : x ;
   290  		_rec.y = isNaN(y) ? 0 : y ;
   291  		_rec.width = isNaN(width) ? 0 : width ;
   292  		_rec.height = isNaN(height) ? 0 : height ;
   293  		if (_bmp != null)
   294  		{
   295  			_bmp.dispose() ;	
   296  		}
   297  		
   298  		_size = _rec.width * _rec.height ;
   299  	}
   300  	
   301  	/**
   302  	 * Sets the display of this PrintScreen object.
   303  	 * @param display can be a DisplayObject (use the view property) or a MovieClip.
   304  	 */
   305  	public function setTarget( display ):Void
   306  	{
   307  		if ( display instanceof MovieClip )
   308  		{
   309  			_target = display ;
   310  		}
   311  		else if ( display instanceof DisplayObject )
   312  		{
   313  			_target = display.view ;
   314  		} 
   315  		else
   316  		{
   317  			_target = null ;	
   318  		}
   319  	}
   320  	
   321  	/**
   322  	 * Returns the size of the buffer.
   323  	 * @return the size of the buffer.
   324  	 */
   325  	public function size():Number
   326  	{
   327  		return _size ;	
   328  	}

   330  	/**
   331  	 * The internal BitmapData reference to copy the picture.
   332  	 */
   333  	private var _bmp:BitmapData ;
   334  	
   335  	/**
   336  	 * The internal array buffer to capture all pixel colors of the display.
   337  	 */
   338  	private var _buffer:Array ;
   339  	
   340  	/**
   341  	 * Defines when the process is in progress the number of pixel captured.
   342  	 */
   343  	private var _cpt:Number ;
   344  	
   345  	/**
   346  	 * The internal delay between the end of the capture and the PrintScreenEvent.FINISH event.
   347  	 */
   348  	private var _delay:Number ;
   349  
   350  	/**
   351  	 * The event notify when the capture is cleared.
   352  	 */
   353  	private var _eClear:PrintScreenEvent ;
   354  
   355  	/**
   356  	 * The event notify when the capture is finished.
   357  	 */
   358  	private var _eFinish:PrintScreenEvent ;
   359  
   360  	/**
   361  	 * The event notify when the capture is in progress.
   362  	 */
   363  	private var _eProgress:PrintScreenEvent ;
   364  
   365  	/**
   366  	 * The event notify when the capture is started.
   367  	 */
   368  	private var _eStart:PrintScreenEvent ;
   369  
   370  	/**
   371  	 * The event notify when the capture is cleared.
   372  	 */
   373  	private var _isRunning:Boolean ;
   374  
   375  	/**
   376  	 * The internal Rectangle area to capture the bitmap.
   377  	 */
   378  	private var _rec:Rectangle ;
   379  
   380  	/**
   381  	 * The numbers of pixels to capture.
   382  	 */
   383  	private var _size:Number ;
   384  	
   385  	/**
   386  	 * The internal target of this PrintScreen object.
   387  	 */
   388  	private var _target:MovieClip ;
   389  
   390  	/**
   391  	 * The elapsed time during the capture (in ms).
   392  	 */
   393  	private var _time:Number ;
   394  
   395  	/**
   396  	 * The internal timer to notify the progress of the capture process.
   397  	 */
   398  	private var _timer:Timer ;

   400  	/**
   401  	 * Creates the buffer of all pixels.
   402  	 */
   403  	private function _bufferize():Void
   404  	{
   405  		var x:Number = _rec.x ;
   406  		var y:Number = _rec.y ;
   407  		var h:Number = _rec.height ;
   408  		var w:Number = _rec.width  ;
   409  		var m:Matrix = new Matrix() ;
   410  		m.translate( -x, -y ) ;
   411  		_bmp.draw( _target , m, null, 1 , _rec ) ;
   412  		for ( var i:Number = 0 ; i < h ; i++ ) 
   413  		{
   414  			for ( var j:Number = 0 ; j < w ; j++ ) 
   415  			{
   416  				_buffer[_cpt] = _bmp.getPixel(j, i) ;
   417  				_cpt ++ ;
   418  			}
   419  		}
   420  	}	
   421  
   422  	private function _onTimer():Void
   423  	{
   424  		notifyProgress() ;
   425  	}
   426  
   427  	/**
   428  	 * This protected method is an internal method to change the _isRunning value.
   429  	 */
   430  	/*protected*/ private function _setRunning(b:Boolean):Void 
   431  	{
   432  		_isRunning = b ;	
   433  	}
