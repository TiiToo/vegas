    24  import pegas.events.ValidatorEvent;
    25  
    26  import vegas.core.IValidator;
    27  import vegas.data.iterator.Iterator;
    28  import vegas.data.set.HashSet;
    29  import vegas.events.AbstractCoreEventDispatcher;
    30  
    31  /**
    32   * Set a collection of keys to validate an action.
    33   * <p><b>Example :</b></p>
    34   * {@code
    35   * import vegas.events.Delegate ;
    36   * import vegas.events.EventListener ;
    37   * import vegas.events.ValidatorEvent ;
    38   * 
    39   * import pegas.ui.Keyboard ;
    40   * import pegas.ui.KeysValidator ;
    41   * 
    42   * var onDebug:Function = function (ev:ValidatorEvent):Void
    43   * {
    44   *     trace ( this + " : " + ev.type ) ;
    45   * }
    46   * 
    47   * var debug:EventListener = new Delegate(this, onDebug) ;
    48   * 
    49   * var code_F:Number = Keyboard.getCharCode("f") ;
    50   * 
    51   * trace ("code F key : " + code_F) ;
    52   * trace ("code CTRL key : " + Keyboard.CONTROL) ;
    53   * 
    54   * var validator:KeysValidator = new KeysValidator() ;
    55   * validator.addEventListener(ValidatorEvent.VALID, debug) ;
    56   * validator.addEventListener(ValidatorEvent.INVALID, debug) ;
    57   * 
    58   * validator.codePolicy = KeysValidator.ASCII ; // ASCII or CODE
    59   *
    60   * validator.insert( Keyboard.CONTROL ) ;
    61   * validator.insert(code_F) ;
    62   * 
    63   * trace ("count keys : " + validator.countKeys()) ;
    64   * 
    65   * // Disabled Shortcuts in the flashplayer to test this example !
    66   * 
    67   * }
    68   * @author eKameleon
    69   * @see pegas.ui.Keyboard (key manager with F1..F12, etc. key codes)
    70   */
    71  class pegas.ui.KeysValidator extends AbstractCoreEventDispatcher implements IValidator 
    72  {
    73  	
    74  	/**
    75  	 * Creates a new KeysValidator instance.
    76  	 * @param keys An array of key codes to initialize the validator.
    77  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    78  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    79  	 */
    80  	public function KeysValidator( keys:Array ,bGlobal:Boolean , sChannel:String ) 
    81  	{
    82  		super( bGlobal , sChannel ) ;
    83  		_cp  = KeysValidator.CODE ;
    84  		_set = new HashSet( keys ) ;
    85  		_reset() ;
    86  		enabled = true ;
    87  	}
    88  
    89  	/**
    90  	 * Indicates the code policy when the validator use the Key.getCode() method.
    91  	 */
    92  	public static var CODE:Number = 0 ;
    93  
    94  	/**
    95  	 * Indicates the code policy when the validator use the Key.ASCII() method.
    96  	 */
    97  	public static var ASCII:Number = 1 ;
    98  	
    99  	private static var __ASPF__ = _global.ASSetPropFlags(KeysValidator, null, 7, 7) ;
   100  	
   101  	/**
   102  	 * Switch the validate process of this validator to use the {@code Key.getCode()} or the {@code Key.getASCII()} method.
   103  	 * This virtual property value is KeysValidator.CODE (0) ou KeysValidator.ASCII (1).
   104  	 * The default value of this property is KeysValidator.CODE (0).
   105  	 */
   106  	public function get codePolicy():Number 
   107  	{
   108  		return _cp ;
   109  	}
   110  
   111  	/**
   112  	 * Switch the validate process of this validator to use the {@code Key.getCode()} or the {@code Key.getASCII()} method.
   113  	 * This virtual property value is KeysValidator.CODE (0) ou KeysValidator.ASCII (1).
   114  	 * The default value of this property is KeysValidator.CODE (0).
   115  	 */
   116  	public function set codePolicy(n:Number):Void 
   117  	{
   118  		setCodePolicy(n) ;
   119  	}
   120  	
   121  	/**
   122  	 * Indicates if the validator is active (true) or not (false).
   123  	 */
   124  	public function get enabled():Boolean
   125  	{
   126  		return _enabled ;	
   127  	}
   128  	
   129  	/**
   130  	 * Indicates if the validator is active (true) or not (false).
   131  	 */
   132  	public function set enabled(b:Boolean):Void
   133  	{
   134  		_enabled = b ;
   135  		if( _enabled ) 
   136  		{
   137  			Key.addListener(this) ;
   138  		}
   139  		else
   140  		{
   141  			Key.removeListener(this) ;	
   142  		}	
   143  	}
   144  	
   145  	/**
   146  	 * Clear all elements in this key validator.
   147  	 */
   148  	public function clear():Void 
   149  	{
   150  		_set.clear() ;
   151  	}
   152  	
   153  	/**
   154  	 * Returns the number of key codes in the internal collection of this validator.
   155  	 * @return the number of key codes in the internal collection of this validator.
   156  	 */
   157  	public function countKeys():Number 
   158  	{
   159  		return _set.size() ;
   160  	}
   161  	
   162  	/**
   163  	 * Inserts a new keyCode in the internal {@code Set} of this validator if it is not already present..
   164  	 */
   165  	public function insert(keyCode:Number):Boolean 
   166  	{
   167  		var b:Boolean = _set.insert(keyCode) ;
   168  		if (b) _reset() ;
   169  		return b ;
   170  	}
   171  	
   172  	/**
   173  	 * Removes the specified key code in the validator if it's exist.
   174  	 */
   175  	public function remove(keyCode:Number):Boolean 
   176  	{
   177  		var b:Boolean = _set.remove(keyCode) ;
   178  		if (b) _reset() ;
   179  		return b ;
   180  	}
   181  	
   182  	/**
   183  	 * Sets and Switch the validate process of this validator to use the {@code Key.getCode()} or the {@code Key.getASCII()} method.
   184  	 * This virtual property value is KeysValidator.CODE (0) ou KeysValidator.ASCII (1).
   185  	 * The default value of this property is KeysValidator.CODE (0).
   186  	 */
   187  	public function setCodePolicy(code:Number):Void 
   188  	{
   189  		_cp = code ;
   190  	}
   191  	
   192  	/**
   193  	 * Returns {@code true} if the passed-in value is support by the validator.
   194  	 * @return {@code true} if the passed-in value is support by the validator.
   195  	 */
   196  	public function supports(value):Boolean 
   197  	{
   198  		return _it.next() == value ;
   199  	}
   200  
   201  	/**
   202  	 * Returns an Array representation of all key codes in this validator.
   203  	 * @return an Array representation of all key codes in this validator.
   204  	 */
   205  	public function toArray():Array
   206  	{
   207  		return _set.toArray() ;
   208  	}
   209  	
   210  	/**
   211  	 * Validates the specified value in argument.
   212  	 */	
   213  	public function validate(value):Void 
   214  	{
   215  		if ( supports(value) ) 
   216  		{
   217  			if (!_it.hasNext()) 
   218  			{
   219  				_reset() ;
   220  				dispatchEvent(new ValidatorEvent( ValidatorEvent.VALID, this)) ;
   221  			}
   222  		} 
   223  		else 
   224  		{
   225  			_reset() ;
   226  			dispatchEvent( new ValidatorEvent( ValidatorEvent.INVALID, this) ) ;
   227  		}
   228  	}
   229  	
   230  	private var _cp:Number ;
   231  
   232  	/**
   233  	 * The enabled flag of this validator.
   234  	 */
   235  	private var _enabled:Boolean ;
   236  	
   237  	/**
   238  	 * The internal {@code Set} of this validator.
   239  	 */
   240  	private var _set:HashSet ;
   241  	
   242  	/**
   243  	 * The internal iterator of this validator.
   244  	 */
   245  	private var _it:Iterator ;
   246  	
   247  	/**
   248  	 * Reset the validator iterator.
   249  	 */
   250  	private function _reset():Void 
   251  	{
   252  		_it = _set.iterator() ;
   253  	}
   254  	
   255  	/**
   256  	 * Invoqued when the key is down.
   257  	 */
   258  	private function onKeyDown():Void 
   259  	{
   260  		var code:Number ;
   261  		if (_cp == KeysValidator.ASCII) 
   262  		{
   263  			code = Key.getAscii() ;
   264  			if (code == 0 || code == undefined ) 
   265  			{
   266  				code = Key.getCode() ;
   267  			}
   268  		} 
   269  		else 
   270  		{
   271  			code = Key.getCode() ;
   272  		}
   273  		validate(code) ;
   274  	}
   275  	
   276  	/**
   277  	 * Invoqued when the keys are up.
   278  	 */
   279  	private function onKeyUp():Void 
   280  	{
   281  		_reset() ;
   282  	}
   283  
   284  }
