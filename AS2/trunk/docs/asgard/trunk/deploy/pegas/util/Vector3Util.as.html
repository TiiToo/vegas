    24  import pegas.geom.Vector3;
    25  
    26  /**
    27   * Static tool class to manipulate and transform {@code Vector3} references.
    28   * @author eKameleon
    29   */
    30  class pegas.util.Vector3Util 
    31  {
    32  
    33  	/**
    34  	 * Computes the addition of two Vector3 and returns the first vector.
    35  	 * @param v1 the first Vector3.
    36  	 * @param v2 the second Vector3.
    37  	 * @return the addition result of two Vector3.
    38  	 */
    39  	static public function addition( v1:Vector3, v2:Vector3 ):Vector3
    40  	{
    41  		 v1.x += v2.x ;
    42  		 v1.y += v2.y ;
    43  		 v1.z += v2.z ;
    44  		 return v1 ;	
    45  	}
    46  
    47  	/**
    48  	 * Computes the addition of two Vector3.
    49  	 * @param v1 a Vector3 to concat.
    50  	 * @param v2 a Vector3 to concat.
    51  	 * @return the addition result of two Vector3.
    52  	 */
    53  	static public function getAddition( v1:Vector3, v2:Vector3 ):Vector3
    54  	{
    55  		return new Vector3( (v1.x + v2.x) , (v1.y + v2.y) , (v1.z + v2.z) ) ;	
    56  	}
    57  
    58  	/**
    59  	* Returns the angle in radian between the two 3D vectors. The formula used here is very simple.
    60  	* It comes from the definition of the dot product between two vectors.
    61  	* @param v1	Vector3 The first Vector3.
    62  	* @param v2	Vector3 The second Vector3.
    63  	* @return the angle in radian between the two vectors.
    64  	*/
    65  	static public function getAngle ( v1:Vector3, v2:Vector3 ):Number
    66  	{
    67  		var ncos:Number = ( Vector3Util.getDot( v1, v2 ) ) / ( getNorm(v1) * getNorm(v2) );
    68  		var sin2:Number = 1 - (ncos * ncos) ;
    69  		if (sin2<0)
    70  		{
    71  			// trace(" wrong "+ ncos ) ;
    72  			sin2 = 0 ;
    73  		}
    74  		// I took long time to find this bug. Who can guess that (1-cos*cos) is negative ? !
    75  		// sqrt returns a NaN for a negative value !
    76  		return  Math.atan2( Math.sqrt(sin2), ncos );
    77  	}
    78  
    79  	/**
    80  	 * Computes the cross product of the two Vector3s.
    81  	 * @param v1 a {@code Vector3}.
    82  	 * @param v2 a {@code Vector3}.
    83  	 * @return the {@code Vector3} resulting of the cross product.
    84  	 */
    85  	static public function getCross( v1:Vector3, v2:Vector3 ):Vector3
    86  	{
    87  		return new Vector3
    88  		( 	
    89  			(v2.y * v1.z) - (v2.z * v1.y) ,
    90  			(v2.z * v1.x) - (v2.x * v1.z) ,
    91  			(v2.x * v1.y) - (v2.y * v1.x)
    92  		);
    93  	}
    94  	
    95  	/**
    96  	 * Computes the dot product of the two Vector3s.
    97  	 * @param v1 a {@code Vector3}.
    98  	 * @param v2 a {@code Vector3}.
    99  	 * @return the dot product of the 2 Vector3.
   100  	 */
   101  	static public function getDot( v1:Vector3, v2:Vector3 ):Number
   102  	{
   103  		return (v1.x * v2.x) + (v1.y * v2.y) + (v1.z * v2.z) ;	
   104  	}
   105  
   106  	/**
   107  	 * Returns the length of the vector.
   108  	 * @param v the vector.
   109  	 * @return the length of the vector.
   110  	 */
   111  	static public function getLength( v:Vector3 ):Number
   112  	{
   113  		return Math.sqrt( (v.x * v.x) + (v.y * v.y) + (v.z * v.z) ) ;
   114  	}
   115  
   116  	/**
   117  	 * Computes the oposite Vector3 of the {@code Vector3}.
   118  	 * @param v the Vector3 reference to negate.
   119  	 * @return a new negate {@code Vector3} reference.
   120  	 */
   121  	public static function getNegate( v:Vector3 ):Vector3
   122  	{
   123  		return new Vector3( - v.x, - v.y, - v.z );
   124  	}	
   125  
   126  	/**
   127  	 * Computes the norm of the {@code Vector3}.
   128  	 * @param v a Vector3 reference.
   129  	 * @return the norm of the specified {@code Vector3}.
   130  	 */
   131  	static public function getNorm( v:Vector3 ):Number
   132  	{
   133  		return Math.sqrt( (v.x * v.x) + (v.y * v.y) + (v.z * v.z) );
   134  	}
   135  
   136  	/**
   137  	 * Computes the power of the specified Vector3.
   138  	 * @param v the Vector3 reference.
   139  	 * @param value the value of the pow..
   140  	 * @return A new Vector3 powered by the method.
   141  	 */
   142  	static public function getPow( v:Vector3, value:Number ):Vector3
   143  	{
   144  		return new Vector3( Math.pow( v.x, value ) , Math.pow( v.y, value ) ,  Math.pow( v.z, value ) ) ;
   145  	}
   146  
   147  	/**
   148  	 * Scales the specified Vector3 with the input value.
   149  	 * @param vector the Vector3 reference to transform.
   150  	 * @param value a real number to scale the current Vector3.
   151  	 * @return A new Vector3 scaled by the value passed in second argument in this method.
   152  	 */
   153  	static public function getScale( v:Vector3, value:Number ):Vector3
   154  	{
   155  		return new Vector3 ( v.x * value , v.y * value , v.z * value ) ;
   156  	}
   157  
   158  	/**
   159  	 * Returns the squared length of this vector.
   160  	 * @param v the vector.
   161  	 * @return the squared length of this vector.
   162  	 */
   163  	static public function getSquaredLength( v:Vector3 ):Number
   164  	{
   165  		return (v.x * v.x) + (v.y * v.y) + (v.z * v.z) ;
   166  	}
   167  
   168  	/**
   169  	 * Computes the substraction of two Vector3.
   170  	 * @param v1 a Vector3 to substract.
   171  	 * @param v2 a Vector3 to substract.
   172  	 * @return the substraction result of two Vector3.
   173  	 */
   174  	static public function getSubstraction( v1:Vector3 , v2:Vector3 ):Vector3
   175  	{
   176  		return new Vector3( (v1.x - v2.x) , (v1.y - v2.y) , (v1.z - v2.z) ) ;
   177  	}
   178  
   179  	/**
   180  	 * Normalize the specified {@code Vector3} in parameter.
   181  	 * @param v a Vector3 reference.
   182  	 * @return {@code true} of the normalize method is success else false for mistake.
   183  	 */	
   184  	static public function normalize( v:Vector3 ):Boolean
   185  	{
   186  		var norm:Number = getNorm( v );
   187  		if( norm == 0 || norm == 1) 
   188  		{
   189  			return false ;
   190  		}
   191  		v.x /= norm ;
   192  		v.y /= norm ;
   193  		v.z /= norm ;
   194  		return true ;
   195  	}
   196  
   197  	/**
   198  	 * Sets the specified {@code Vector3} object with the second {@code Vector3} object passed in argument.
   199  	 * @param v1 the first {@code Vector3}.
   200  	 * @param v2 the second {@code Vector3}.
   201  	 * @return the first {@code Vector3} transformed.
   202  	 */
   203  	static public function setByVector3( v1:Vector3, v2:Vector3):Vector3
   204  	{
   205  		v1.x = v2.x ;
   206  		v1.y = v2.y ;
   207  		v1.z = v2.z ;
   208  		return v1 ;
   209  	}
   210  
   211  	/**
   212  	 * Scales the specified Vector3 with the input value.
   213  	 * @param vector the Vector3 reference to transform.
   214  	 * @param value a real number to scale the current Vector3.
   215  	 */
   216  	static public function scale( v:Vector3, value:Number ):Void
   217  	{
   218  		v.x *= value ;
   219  		v.y *= value ;
   220  		v.z *= value ;
   221  	}
   222  
   223  	/**
   224  	 * Computes the substraction of two Vector3.
   225  	 * @param v1 the first Vector3.
   226  	 * @param v2 the second Vector3.
   227  	 * @return the substraction result of two Vector3.
   228  	 */
   229  	static public function substraction( v1:Vector3 , v2:Vector3 ):Vector3
   230  	{
   231  		 v1.x -= v2.x ;
   232  		 v1.y -= v2.y ;
   233  		 v1.z -= v2.z ;
   234  		 return v1 ;	
   235  	}
   236  
   237  }
