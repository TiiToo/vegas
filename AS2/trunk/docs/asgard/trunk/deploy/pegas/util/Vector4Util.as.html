    24  import pegas.geom.Vector4;
    25  
    26  /**
    27   * Static tool class to manipulate and transform {@code Vector4} references.
    28   * @author eKameleon
    29   */
    30  class pegas.util.Vector4Util 
    31  {
    32  
    33  	/**
    34  	 * Computes the addition of two Vector4 and returns the first vector.
    35  	 * @param v1 the first Vector4.
    36  	 * @param v2 the second Vector4.
    37  	 * @return the addition result of two Vector4.
    38  	 */
    39  	public static function addition( v1:Vector4, v2:Vector4 ):Vector4
    40  	{
    41  		 v1.x += v2.x ;
    42  		 v1.y += v2.y ;
    43  		 v1.z += v2.z ;
    44  		 v1.w += v2.w ;
    45  		 return v1 ;	
    46  	}
    47  
    48  	/**
    49  	 * Returns the cross product of two Vector4.
    50  	 * The returned Vector4 is orthogonal to the other two Vector4 and when ignoring the fourth 
    51  	 * component the resulting Vector3 is also orthogonal.
    52  	 */
    53  	public static function cross(v1:Vector4, v2:Vector4):Vector4
    54  	{
    55  		v1.x = (v2.y * v1.z) - (v2.z * v1.y) ;
    56  		v1.y = (v2.z * v1.x) - (v2.x * v1.z) ;
    57  		v1.z = (v2.x * v1.y) - (v2.y * v1.x) ;
    58  		v1.w = 0 ;
    59  		return v1 ;
    60  	}
    61  
    62  	/**
    63  	 * Computes the addition of two Vector4.
    64  	 * @param v1 a Vector4 to concat.
    65  	 * @param v2 a Vector4 to concat.
    66  	 * @return the addition result of two Vector4.
    67  	 */
    68  	public static function getAddition( v1:Vector4, v2:Vector4 ):Vector4
    69  	{
    70  		return new Vector4( (v1.x + v2.x) , (v1.y + v2.y) , (v1.z + v2.z) , (v1.w + v2.w)) ;	
    71  	}
    72  
    73  	/**
    74  	* Returns the angle in radian between the two 3D vectors. The formula used here is very simple.
    75  	* It comes from the definition of the dot product between two vectors.
    76  	* @param v1	Vector4 The first Vector4.
    77  	* @param v2	Vector4 The second Vector4.
    78  	* @return the angle in radian between the two vectors.
    79  	*/
    80  	public static function getAngle ( v1:Vector4, v2:Vector4 ):Number
    81  	{
    82  		var ncos:Number = ( Vector4Util.getDot( v1, v2 ) ) / ( getNorm(v1) * getNorm(v2) );
    83  		var sin2:Number = 1 - (ncos * ncos) ;
    84  		if (sin2<0)
    85  		{
    86  			trace(" wrong "+ ncos ) ;
    87  			sin2 = 0 ;
    88  		}
    89  		// I took long time to find this bug. Who can guess that (1-cos*cos) is negative ? !
    90  		// sqrt returns a NaN for a negative value !
    91  		return  Math.atan2( Math.sqrt(sin2), ncos );
    92  	}
    93  
    94  	/**
    95  	 * Computes the cross product of the two Vector4s.
    96  	 * @param v1 a {@code Vector4}.
    97  	 * @param v2 a {@code Vector4}.
    98  	 * @return the {@code Vector4} resulting of the cross product.
    99  	 */
   100  	public static function getCross( v1:Vector4, v2:Vector4 ):Vector4
   101  	{
   102  		return new Vector4
   103  		( 	
   104  			(v2.y * v1.z) - (v2.z * v1.y) ,
   105  			(v2.z * v1.x) - (v2.x * v1.z) ,
   106  			(v2.x * v1.y) - (v2.y * v1.x) ,
   107  			0
   108  		);
   109  	}
   110  
   111  	/**
   112  	 * Computes the dot product of the two Vector4s.
   113  	 * @param v1 a {@code Vector4}.
   114  	 * @param v2 a {@code Vector4}.
   115  	 * @return the dot product of the 2 Vector4.
   116  	 */
   117  	public static function getDot( v1:Vector4, v2:Vector4 ):Number
   118  	{
   119  		return (v1.x * v2.x) + (v1.y * v2.y) + (v1.z * v2.z) + (v1.w * v2.w) ;	
   120  	}
   121  
   122  	/**
   123  	 * Returns the length of the vector.
   124  	 * @param v the vector.
   125  	 * @return the length of the vector.
   126  	 */
   127  	public static function getLength( v:Vector4 ):Number
   128  	{
   129  		return Math.sqrt( (v.x * v.x) + (v.y * v.y) + (v.z * v.z) + (v.w * v.w) ) ;
   130  	}
   131  
   132  	/**
   133  	 * Computes the oposite Vector4 of the {@code Vector4}.
   134  	 * @param v the Vector4 reference to negate.
   135  	 * @return a new negate {@code Vector4} reference.
   136  	 */
   137  	public static function getNegate( v:Vector4 ):Vector4
   138  	{
   139  		return new Vector4( - v.x, - v.y, - v.z , - v.w );
   140  	}	
   141  
   142  	/**
   143  	 * Computes the norm of the {@code Vector4}.
   144  	 * @param v a Vector4 reference.
   145  	 * @return the norm of the specified {@code Vector4}.
   146  	 */
   147  	public static function getNorm( v:Vector4 ):Number
   148  	{
   149  		return Math.sqrt( (v.x * v.x) + (v.y * v.y) + (v.z * v.z) + (v.w * v.w) );
   150  	}
   151  
   152  	/**
   153  	 * Computes the power of the specified Vector4.
   154  	 * @param v the Vector4 reference.
   155  	 * @param value the value of the pow..
   156  	 * @return A new Vector4 powered by the method.
   157  	 */
   158  	public static function getPow( v:Vector4, value:Number ):Vector4
   159  	{
   160  		return new Vector4( Math.pow( v.x, value ) , Math.pow( v.y, value ) ,  Math.pow( v.z, value ) , Math.pow( v.w, value ) ) ;
   161  	}
   162  
   163  	/**
   164  	 * Scales the specified Vector4 with the input value.
   165  	 * @param vector the Vector4 reference to transform.
   166  	 * @param value a real number to scale the current Vector4.
   167  	 * @return A new Vector4 scaled by the value passed in second argument in this method.
   168  	 */
   169  	public static function getScale( v:Vector4, value:Number ):Vector4
   170  	{
   171  		return new Vector4 ( v.x * value , v.y * value , v.z * value , v.w * value ) ;
   172  	}
   173  
   174  	/**
   175  	 * Returns the squared length of this vector.
   176  	 * @param v the vector.
   177  	 * @return the squared length of this vector.
   178  	 */
   179  	public static function getSquaredLength( v:Vector4 ):Number
   180  	{
   181  		return (v.x * v.x) + (v.y * v.y) + (v.z * v.z) + + (v.w * v.w) ;
   182  	}
   183  
   184  	/**
   185  	 * Computes the substraction of two Vector4.
   186  	 * @param v1 a Vector4 to substract.
   187  	 * @param v2 a Vector4 to substract.
   188  	 * @return the substraction result of two Vector4.
   189  	 */
   190  	public static function getSubstraction( v1:Vector4 , v2:Vector4 ):Vector4
   191  	{
   192  		return new Vector4( (v1.x - v2.x) , (v1.y - v2.y) , (v1.z - v2.z) , (v1.w - v2.w) ) ;
   193  	}
   194  
   195  	/**
   196  	 * Normalize the specified {@code Vector4} in parameter.
   197  	 * @param v a Vector4 reference.
   198  	 * @return {@code true} of the normalize method is success else false for mistake.
   199  	 */	
   200  	public static function normalize( v:Vector4 ):Boolean
   201  	{
   202  		var norm:Number = getNorm( v );
   203  		if( norm == 0 || norm == 1) 
   204  		{
   205  			return false ;
   206  		}
   207  		v.x /= norm ;
   208  		v.y /= norm ;
   209  		v.z /= norm ;
   210  		v.w /= norm ;
   211  		return true ;
   212  	}
   213  	
   214  	/**
   215  	 * Sets the specified {@code Vector4} object with the second {@code Vector4} object passed in argument.
   216  	 * @param v1 the first {@code Vector4}.
   217  	 * @param v2 the second {@code Vector4}.
   218  	 * @return the first {@code Vector4} transformed.
   219  	 */
   220  	public static function setByVector4( v1:Vector4, v2:Vector4):Vector4
   221  	{
   222  		v1.x = v2.x ;
   223  		v1.y = v2.y ;
   224  		v1.z = v2.z ;
   225  		v1.w = v2.w ;
   226  		return v1 ;
   227  	}
   228  
   229  	/**
   230  	 * Scales the specified Vector4 with the input value.
   231  	 * @param vector the Vector4 reference to transform.
   232  	 * @param value a real number to scale the current Vector4.
   233  	 */
   234  	public static function scale( v:Vector4, value:Number ):Void
   235  	{
   236  		v.x *= value ;
   237  		v.y *= value ;
   238  		v.z *= value ;
   239  		v.w *= value ;
   240  	}
   241  	
   242  	/**
   243  	 * Computes the substraction of two Vector4.
   244  	 * @param v1 the first Vector3.
   245  	 * @param v2 the second Vector3.
   246  	 * @return the substraction result of two Vector3.
   247  	 */
   248  	public static function substraction( v1:Vector4 , v2:Vector4 ):Vector4
   249  	{
   250  		 v1.x -= v2.x ;
   251  		 v1.y -= v2.y ;
   252  		 v1.z -= v2.z ;
   253  		 v1.w -= v2.w ;
   254  		 return v1 ;	
   255  	}

   257  }
