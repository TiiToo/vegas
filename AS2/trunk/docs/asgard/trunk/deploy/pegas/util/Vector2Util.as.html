    24  import pegas.geom.Vector2;
    25  
    26  /**
    27   * Static tool class to manipulate and transform {@code Vector2} references.
    28   * @author eKameleon
    29   */
    30  class pegas.util.Vector2Util 
    31  {
    32  
    33  	/**
    34  	 * Computes the addition of two Vector2 and returns the first vector.
    35  	 * @param v1 the first Vector2.
    36  	 * @param v2 the second Vector2.
    37  	 * @return the addition result of two Vector2.
    38  	 */
    39  	public static function addition( v1:Vector2, v2:Vector2 ):Vector2
    40  	{
    41  		 v1.x += v2.x ;
    42  		 v1.y += v2.y ;
    43  		 return v1 ;	
    44  	}
    45  
    46  	/**
    47  	 * Computes the addition of two Vector2.
    48  	 * @param v1 a Vector2 to concat.
    49  	 * @param v2 a Vector2 to concat.
    50  	 * @return the addition result of two Vector2.
    51  	 */
    52  	public static function getAddition( v1:Vector2, v2:Vector2 ):Vector2
    53  	{
    54  		return new Vector2( (v1.x + v2.x) , (v1.y + v2.y) ) ;	
    55  	}
    56  
    57  	/**
    58  	 * Returns the distance between the 2 vectors in argument.
    59  	 * @param v1 the first vector.
    60  	 * @param v2 the second vector.
    61  	 * @return the distance between the 2 vectors in argument.
    62   	 */
    63  	public static function getDistance( v1:Vector2, v2:Vector2 ):Number
    64  	{
    65  		return Math.sqrt( Vector2Util.getSquaredDistance( v1, v2 ) ) ;
    66  	}
    67  
    68  	/**
    69  	 * Returns the length of the vector.
    70  	 * @param v the vector.
    71  	 * @return the length of the vector.
    72  	 */
    73  	public static function getLength( v:Vector2 ):Number
    74  	{
    75  		return Math.sqrt( (v.x * v.x) + (v.y * v.y) ) ;
    76  	}
    77  
    78  	/**
    79  	 * Returns the middle Vector2 between 2 Points.
    80  	 * <p><b>Example :</b></p>
    81  	 * {@code
    82  	 * var p1:Vector2 = new Vector2(10,10) ;
    83  	 * var p2:Vector2 = new Vector2(20,20) ;
    84  	 * var middle:Vector2 = Vector2Util.getMiddle(p1,p2) ;
    85  	 * trace(middle) ;
    86  	 * }
    87  	 * @return the middle Point between 2 Points.
    88  	 */
    89  	public static function getMiddle(p1:Vector2, p2:Vector2):Vector2 
    90  	{
    91  		return new Vector2( (p1.x + p2.x) / 2 , (p1.y + p2.y) / 2) ;
    92  	}
    93  	
    94  	/**
    95  	 * Normalizes the Vector2 instance passed in argument..
    96  	 * @param v the Vector2 to normalize.
    97  	 * @return a normalized Vector2, with length 1.
    98  	 */
    99  	public static function getNormalize( v:Vector2 ):Vector2
   100  	{
   101  		var len:Number = 1 / Vector2Util.getLength(v) ;
   102  		return Vector2Util.getScale( v , len ) ;
   103  	}
   104  
   105  	/**
   106  	 * Computes the power of the specified Vector2.
   107  	 * @param v the Vector2 reference.
   108  	 * @param value the value of the pow..
   109  	 * @return A new Vector2 powered by the method.
   110  	 */
   111  	public static function getPow( v:Vector2, value:Number ):Vector2
   112  	{
   113  		return new Vector2( Math.pow( v.x, value ) , Math.pow( v.y, value ) ) ;
   114  	}
   115  
   116  	/**
   117  	 * Scales the specified Vector2 with the input value.
   118  	 * @param vector the Vector2 reference to transform.
   119  	 * @param value a real number to scale the current Vector2.
   120  	 * @return A new Vector2 scaled by the value passed in second argument in this method.
   121  	 */
   122  	public static function getScale( v:Vector2, value:Number ):Vector2
   123  	{
   124  		return new Vector2 ( v.x * value , v.y * value ) ;
   125  	}
   126  
   127  	/**
   128  	 * Returns the squared distance between 2 vectors.
   129  	 * @param v1 the first vector.
   130  	 * @param v2 the second vector.
   131  	 * @return the squared distance between 2 vectors.
   132  	 */
   133  	public static function getSquaredDistance( v1:Vector2, v2:Vector2 ):Number
   134  	{
   135  		var dx:Number= v2.x - v1.x ;
   136  		var dy:Number= v2.y - v1.y ; 
   137  		return (dx * dx) + (dy * dy) ;
   138  	}
   139  
   140  	/**
   141  	 * Returns the squared length of this vector.
   142  	 * @param v the vector.
   143  	 * @return the squared length of this vector.
   144  	 */
   145  	public static function getSquaredLength( v:Vector2 ):Number
   146  	{
   147  		return (v.x * v.x) + (v.y * v.y) ;
   148  	}
   149  
   150  	/**
   151  	 * Computes the substraction of two Vector2.
   152  	 * @param v1 a Vector2 to substract.
   153  	 * @param v2 a Vector2 to substract.
   154  	 * @return the substraction result of two Vector2.
   155  	 */
   156  	public static function getSubstraction( v1:Vector2 , v2:Vector2 ):Vector2
   157  	{
   158  		return new Vector2( (v1.x - v2.x) , (v1.y - v2.y) ) ;
   159  	}
   160  
   161  
   162  
   163  	/**
   164  	 * Sets the specified {@code Vector2} object with the second {@code Vector2} object passed in argument.
   165  	 * @param v1 the first {@code Vector2}.
   166  	 * @param v2 the second {@code Vector2}.
   167  	 * @return the first {@code Vector2} transformed.
   168  	 */
   169  	public static function setByVector2( v1:Vector2, v2:Vector2):Vector2
   170  	{
   171  		v1.x = v2.x ;
   172  		v1.y = v2.y ;
   173  		return v1 ;
   174  	}
   175  	
   176  	/**
   177  	 * Computes the substraction of two Vector2.
   178  	 * @param v1 the first Vector2.
   179  	 * @param v2 the second Vector2.
   180  	 * @return the substraction result of two Vector2.
   181  	 */
   182  	public static function substraction( v1:Vector2 , v2:Vector2 ):Vector2
   183  	{
   184  		 v1.x -= v2.x ;
   185  		 v1.y -= v2.y ;
   186  		 return v1 ;	
   187  	}
   188  
   189  }
