    41      /**
    42       * Returns the Plane reference defines by the specified object with the properties a, b, c and d.
    43       * <p><b>Example :</b></p>
    44       * {@code
    45       * import pegas.geom.Plane ;
    46       * import pegas.util.PlaneUtil ;
    47       * 
    48       * var p:Plane = PlaneUtil.getPlaneByObject( { a:10 , b:10 , c:100 , d:100 } ) ;
    49       * trace(p) ; // [Plane:{10,10,100,100}]
    50       * }
    51       * @return the Plane reference defines by the specified object with the properties a, b, c and d.
    52       */
    53      public static function getPlaneByObject( o:Object ):Plane
    54      {
    55          return new Plane( o.a, o.b, o.c, o.d ) ;
    56      } 
    57      
    58      /**
    59       * Returns the Plane reference defines by the specified object with the two vectors in argument.
    60       * <p><b>Example :</b></p>
    61       * {@code
    62       * import pegas.geom.Plane ;
    63       * import pegas.geom.Vector2 ;
    64       * import pegas.util.PlaneUtil ;
    65       * 
    66       * var v1:Vector2 = new Vector2(10,10) ;
    67       * var v2:Vector2 = new Vector2(100,100) ;
    68       * var p:Plane = PlaneUtil.getPlaneByVector( v1, v2 ) ;
    69       * trace(p) ; // [Plane:{10,10,100,100}]
    70       * }
    71       * @return the Plane reference defines by the specified object with the two vectors in argument.
    72       */
    73      public static function getPlaneByVector( v1:Vector2 , v2:Vector2 ):Plane
    74      {
    75          return new Plane( v1.x, v1.y, v2.x, v2.y ) ;
    76      } 
    77  
    78  }
