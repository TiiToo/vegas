    26  /**
    27   * <code>ColorHSV</code> provide static color conversion methods, in Hue Saturation Value color space.
    28   * {@link http://en.wikipedia.org/wiki/HSV_color_space}
    29   * <p><b>Example :</b></p>
    30   * {@code
    31   * import asgard.colors.ColorHSV;
    32   * var o:Object = ColorHSV.rgb2hsv(255, 0, 0);
    33   * for (var prop:String in o) 
    34   * {
    35   *     trace( prop + " : " + o[prop] ) ;
    36   * }
    37   * var oRGB : Object = {r:250,g:0,b:0};
    38   * var o2 : Object = ColorHSV.hsv2rgb (oRGB, 50, 100);
    39   * trace ("--")
    40   * for (var prop:String:String in o2) 
    41   * {
    42   * 	   trace( prop + " : " + o2[prop] ) ;
    43   * }
    44   * }
    45   * </pre>
    46   * @author eKameleon
    47   */
    48  class pegas.colors.ColorHSV 
    49  {
    50  	
    51  	/**
    52  	 * Converts a HSV color in a RGB color.
    53  	 * <p><b>Example :</b></p>
    54  	 * {@code
    55  	 * import pegas.colors.ColorHSV;
    56  	 * var oRGB : Object = {r:250,g:0,b:0};
    57  	 * var o2 : Object = ColorHSV.hsv2rgb (oRGB, 50, 100); // desaturate 50%
    58  	 * for (var prop:String in o2) 
    59  	 * {
    60  	 *     trace( prop + " : " + o2[prop] );
    61  	 * }
    62  	 * }
    63  	 * @param oColor an Object with r,g,b properties ( range from 0 - 255 ), this is the Hue.
    64  	 * @param s a Number represent the Saturation value ( range from 0 - 100 ).
    65  	 * @param v a Number represent the Brightness ( Value ) ( range from 0 - 100 ).
    66  	 * @return an Object with r,g,b properties.
    67  	 */
    68  	public static function hsv2rgb ( oColor, s:Number, v:Number) :Object 
    69  	{
    70  		var tmp:Array = ["r:"+oColor.r,"g:"+oColor.g,"b:"+oColor.b].sort(_fsort) ;
    71  		var order:Array = [ tmp[0].split(":")[0] , tmp[1].split(":")[0] , tmp[2].split(":")[0] ];
    72  		var o:Object = { r:oColor.r , g:oColor.g , b:oColor.b } ;
    73  		o[order[2]] += ( 100 - s ) * 2.55 ;
    74  		o[order[1]] += ( 255 - o[order[1]] ) * ( 100 - s ) / 100 ;
    75  		o[order[0]] *= v / 100 ; 
    76  		o[order[1]] *= v / 100 ; 
    77  		o[order[2]] *= v / 100 ;
    78  		return o ;
    79  	}
    80  
    81  	/**
    82  	 * Converts a RGB color in a HSV color.
    83  	 * <p><b>Example :</b></p>
    84  	 * {@code
    85  	 * import pegas.colors.ColorHSV;
    86  	 * var o:Object = ColorHSV.rgb2hsv(255, 0, 0);
    87  	 * for (var i in o) {
    88  	 * 	if (typeof (o[i]) == "object") 
    89  	 * 	{
    90  	 * 		trace(i+" : ");
    91  	 * 		for (var j in o[i]) 
    92  	 * 		{
    93  	 * 			trace("	"+j+" : "+o[i][j]);
    94  	 * 		}
    95  	 * 	}
    96  	 * 	else 
    97  	 * 	{
    98  	 * 		trace(i+" : "+o[i]);
    99  	 * 	}
   100  	 * }
   101  	 * }
   102  	 * @param r a Number represent red value ( range from 0 - 255 ).
   103  	 * @param g a Number represent red value ( range from 0 - 255 ).
   104  	 * @param b a Number represent red value ( range from 0 - 255 ).
   105  	 * @return an Object with h ( Hue ), s ( Saturation ), v ( Value / Brightness ) properties. h is an Object with r,g,b properties.
   106  	 */
   107  	public static function rgb2hsv ( r:Number, g:Number ,b:Number):Object 
   108  	{
   109  		var temp:Array = ["r:" + r,"g:" + g,"b:"+b].sort(_fsort);
   110  		var order:Array =[ temp[0].split(":")[0] , temp[1].split(":")[0] , temp[2].split(":")[0] ];
   111  		var o:Object= {r:r,g:g,b:b} ;
   112  		var v:Number = o[order[0]] * 100 /255 ;
   113  		o.r *= 100 / v ;
   114  		o.g *= 100 / v ;
   115  		o.b *= 100 / v ;
   116  		var s:Number =  (255 - o[order[2]]) * 100 / 255 ; // saturation
   117  		o[order[1]] = Math.round ( 255* (1 - (255-o[order[1]])/(255-o[order[2]]) ));
   118  		o[order[0]]= 255 ; o[order[2]] = 0 ;
   119  		return( { h:o , s:s<<0 ,v:v<<0} ) ;
   120  	}
   121  	
   122  	private static function _fsort ( a:String , b:String ) :Number 
   123  	{
   124  		var val1:Number = Number(a.split(":")[1]);
   125  		var val2:Number = Number(b.split(":")[1]);
   126  		if (val1 > val2) return -1;
   127  		else if (val1 < val2) return 1;
   128  		else return 0;
   129  	}
   130  	
   131  }
