    38  	/**
    39  	 * Calculates an array representation of Point elements to create a quadratic Bezier curve looped or not in n-points.
    40  	 * @return an array representation of Point elements to create a quadratic Bezier curve looped or not in n-points.
    41  	 */
    42  	public static function createBezier(step:Number,points:Array,precision:Number,loop:Boolean):Array 
    43  	{
    44  		
    45  		precision= isNaN(precision) ? 0 : precision ;
    46  		
    47  		if (loop == undefined) 
    48  		{
    49  			loop = true ;
    50  		}
    51  		
    52  		var pts:Array = [];
    53  		var nbp:Number = points.length ;
    54  		var i:Number=0 ;
    55  		var lastpoint:Point = null ;
    56  		
    57  		var _p0:Point ;
    58  		var _p1:Point ;
    59  		var _p2:Point ;
    60  		var t:Number ;
    61  		
    62  		var c1,c2, c3, nx, ny, u1:Number ;
    63  		var npoint:Point ;
    64  		
    65  		while (i<nbp)
    66  		{
    67  
    68  			_p0 = points[i] ;
    69  			_p1 = (loop)? points[(i+1)%nbp]:(points[(i+1)])? points[(i+1)]:_p0 ;
    70  			_p2 = (loop)? points[(i+2)%nbp]:(points[(i+2)])? points[(i+2)]:_p1 ;
    71  
    72  			t = 0 ;
    73  
    74  			while (t<=1)
    75  			{
    76  
    77  				var p0:Point = new Point((_p0.x+_p1.x)/2,(_p0.y+_p1.y)/2);
    78  				var p1:Point = new Point(_p1.x,_p1.y);
    79  				var p2:Point = new Point((_p2.x+_p1.x)/2,(_p2.y+_p1.y)/2);
    80  				
    81  				u1 = 1 - t ;
    82  				c1 = u1 * u1 ;
    83  				c2 = 2 * t * u1 ;
    84  				c3 = t * t ;
    85  			
    86  				nx = c1 * p0.x + c2 * p1.x + c3 * p2.x ;
    87  				ny = c1 * p0.y + c2 * p1.y + c3 * p2.y ;
    88  				
    89  				npoint =new Point(nx,ny);
    90  			
    91  				if (lastpoint!=null) 
    92  				{
    93  					if ( Point.distance(lastpoint,npoint) > precision )
    94  					{
    95  						lastpoint=npoint;
    96  						pts.push(npoint);
    97  					}
    98  				} 
    99  				else 
   100  				{
   101  					lastpoint = npoint ;
   102  					pts.push(npoint) ;
   103  				}
   104  
   105  				t+=step ;
   106  			}
   107  
   108  			i++ ;
   109  		}
   110  		return pts ;
   111  	}

   113  	/**
   114  	 * Returns the baryCenter of a collection of points.
   115  	 * @return the baryCenter of a collection of points.
   116  	 */
   117  	public static function getBaryCenter(pts:Array):Point 
   118  	{
   119  		var nbp:Number = pts.length ;
   120  		var x:Number=0;
   121  		var y:Number=0;
   122  		while (--nbp>=0)
   123  		{
   124  			x += pts[nbp].x ;
   125  			y += pts[nbp].y ;
   126  		}
   127  		return new Point(x/pts.length,y/pts.length) ;
   128  	}

   130  	public static function split(p0:Vector2, p1:Vector2, p2:Vector2, p3:Vector2):Object
   131  	{
   132  		var p0_1:Vector2 = Line.getMiddle(p0, p1) ;
   133  		var p1_2:Vector2 = Line.getMiddle(p1, p2);
   134  		var p2_3:Vector2 = Line.getMiddle(p2, p3);
   135  		var p0_2:Vector2 = Line.getMiddle(p0_1, p1_2);
   136  		var p1_3:Vector2 = Line.getMiddle(p1_2, p2_3);
   137  		var p0_3:Vector2 = Line.getMiddle(p0_2, p1_3);
   138  		var o:Object = 
   139  		{
   140  			b0 : { a:p0  , b:p0_1 , c:p0_2 , d:p0_3 } ,
   141  			b1 : { a:p0_3 , b:p1_3 , c:p2_3 , d:p3 }  
   142  		} ;
   143  		return o ; 
   144  	}

   146  	public static function getCubicDerivative(c0:Number, c1:Number, c2:Number, c3:Number, t:Number):Number 
   147  	{
   148  		var g:Number = 3 * (c1 - c0) ;
   149  		var b:Number = (3 * (c2 - c1)) - g ;
   150  		var a:Number = c3 - c0 - b - g ;
   151  		return ( 3*a*t*t + 2*b*t + g ) ;
   152  	}
   153  
   154  	public static function getCubicPt(c0:Number, c1:Number, c2:Number, c3:Number, t:Number):Number 
   155  	{
   156  		var ts:Number = t*t ;
   157  		var g:Number = 3 * (c1 - c0);
   158  		var b:Number = (3 * (c2 - c1)) - g;
   159  		var a:Number = c3 - c0 - b - g;
   160  		return ( a*ts*t + b*ts + g*t + c0 ) ;
   161  	}
   162  
   163  	public static function getCubicTgt(p0:Vector2, p1:Vector2, p2:Vector2, p3:Vector2, t:Number):Object 
   164  	{
   165  		var p:Vector2 = new Vector2() ;
   166  		p.x = getCubicPt(p0.x, p1.x, p2.x, p3.x, t);
   167  		p.y = getCubicPt(p0.y, p1.y, p2.y, p3.y, t);
   168  		var v:Vector2 = new Vector2() ;
   169  		v.x = getCubicDerivative(p0.x, p1.x, p2.x, p3.x, t);
   170  		v.y = getCubicDerivative(p0.y, p1.y, p2.y, p3.y, t);
   171  		var l:Line = Line.getVectorLine(p, v);
   172  		return { p : p , l : l } ;
   173  	}
