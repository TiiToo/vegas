    36   * <p>A Rectangle object is an area defined by its position, as indicated by its top-left corner point (x, y), and by its width and its height.</p>
    37   * <p>The x, y, width, and height properties of the Rectangle class are independent of each other; changing the value of one property has no effect on the others.</p>
    38   * <p>To used this class with a FP8 and > with flash.display.BitmapData class and other flash.* classes you can use the method toFlash() to return a compatible reference of this object.</p>

    48  	 * import pegas.geom.Rectangle;
    49  	 * var rec:Rectangle = new Rectangle(5, 10, 50, 100);

    69  		}
    70  	}

   273  	 * Determines whether the specified coordinates is contained within the rectangular region defined by this Rectangle object.
   274  	 * @return {@code true} if the specified coordinates is contained within the rectangular region defined by this Rectangle object.

   276  	public function contains( p_x:Number, p_y:Number):Boolean

   290  	/**
   291  	 * Determines whether the Rectangle object specified by the rect parameter is contained within this Rectangle object.
   292  	 * @return {@code true} if the specified Rectangle is contained within this Rectangle object. 
   293  	 */
   294  	public function containsRectangle(rec:Rectangle):Boolean 
   295  	{
   296  		var a:Number = rec.x + rec.width ;
   297  		var b:Number = rec.y + rec.height ;
   298  		var c:Number = x + width ;
   299  		var d:Number = y + height ;
   300  		return (rec.x >= x && rec.x < c && rec.y >= y && rec.y < d && a > this.x && a <= c && b > y && b <= d) ;
   301  	}

   303  	/**
   304  	 * Compares its two arguments for order.
   305  	 * @param o1 the first object to compare.
   306  	 * @param o2 the second object to compare.
   307  	 * @return <p>
   308  	 * <li>-1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
   309  	 * <li> 1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
   310  	 * <li> 0 if o1 and o2 are equal.</li>
   311  	 * </p>
   312  	 */
   313  	public function compare(o1, o2):Number 
   314  	{
   315  		var s1:Point = o1.getSize() ;
   316  		var s2:Point = o2.getSize() ;
   317  		var a1:Number = s1.x * s1.y ;
   318  		var a2:Number = s2.x * s2.y ;
   319  		if (a1 == a2) return 0 ;
   320  		else if(a1 > a2) return 1 ;
   321  		else return -1 ;
   322  	}

   445  	/**
   446  	 * If the Rectangle object specified in the toIntersect parameter intersects with this Rectangle object, 
   447  	 * the intersection() method returns the area of intersection as a Rectangle object.
   448  	 */
   449  	public function intersection(toIntersect:Rectangle):Rectangle 
   450  	{
   451  		var rec:Rectangle = new Rectangle() ;
   452  		if (isEmpty() || toIntersect.isEmpty()) {
   453  			rec.setEmpty();
   454  			return rec ;
   455  		}
   456  		rec.x = Math.max(x, toIntersect.x);
   457  		rec.y = Math.max(y, toIntersect.y);
   458  		rec.width = Math.min( x + width, toIntersect.x + toIntersect.width) - rec.x ;
   459  		rec.height = Math.min(y + height, toIntersect.y + toIntersect.height) - rec.y ;
   460  		if ( rec.width <= 0 || rec.height <= 0 ) rec.setEmpty();
   461  		return rec ;
   462  	}
