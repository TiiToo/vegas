    32  /**
    33   * Defines a linear equation of the form : {@code ax + by = c} with fixed real coefficients a, b and c such that a and b are not both zero.
    34   * <p><b>Example :</b></p>
    35   * {@code
    36   * var l:Line = new Line(10, 20, 5) ;
    37   * trace(l) ; // {a:10,b:20,c:5}
    38   * }
    39   */
    40  class pegas.geom.Line extends CoreObject implements ICloneable, IEquality 
    41  {
    42  	
    43  	/**
    44  	 * Creates a new Line object.
    45  	 * @param a the a component of the Line.
    46  	 * @param b the b component of the Line.
    47  	 * @param c the c component of the Line.
    48  	 */
    49  	public function Line( a:Number, b:Number, c:Number) 
    50  	{
    51  		this.a = a ;
    52  		this.b = b ;
    53  		this.c = c ;
    54  	}
    55  	
    56  	/**
    57  	 * Determinates the a component of the Line.
    58  	 */
    59  	public var a:Number ;
    60  
    61  	/**
    62  	 * Determinates the b component of the Line.
    63  	 */
    64  	public var b:Number ;
    65  
    66  	/**
    67  	 * Determinates the c component of the Line.
    68  	 */
    69  	public var c:Number ;

    71  	/**
    72  	 * Returns a shallow copy of this instance.
    73  	 * <p><b>Example :</b></p>
    74  	 * {@code
    75  	 * var l1:Line = new Line(10, 20, 30) ;
    76  	 * var l2:Line = l1.clone() ;
    77  	 * }
    78  	 * @return a shallow copy of this instance.
    79  	 */
    80  	public function clone() 
    81  	{
    82  		return new Line(a, b, c) ;
    83  	}

    85  	/**
    86  	 * Returns a deep copy of this instance.
    87  	 * <p><b>Example :</b></p>
    88  	 * {@code
    89  	 * var l1:Line = new Line(10, 20, 30) ;
    90  	 * var l2:Line = l1.copy() ;
    91  	 * }
    92  	 * @return a deep copy of this instance.
    93  	 */
    94  	public function copy()
    95  	{
    96  		return new Line(a, b, c) ;	
    97  	}

    99  	/**
   100  	 * Returns the distance between two vectors.
   101  	 * {@code
   102  	 * var v1:Vector2 = new Vector2(10,20) ;
   103  	 * var v2:Vector2 = new Vector2(40,60) ;
   104  	 * trace(Line.distance(p1,p2) ) ; // 50
   105  	 * }
   106  	 * @return the distance between two vectors.
   107  	 */
   108  	public static function distance( p1:Vector2, p2:Vector2 ):Number 
   109  	{
   110  		return Vector2Util.getDistance(p1, p2) ;
   111  	}
   112  
   113  	/**
   114  	 * Compares the specified object with this object for equality.
   115  	 * @return {@code true} if the the specified object is equal with this object.
   116  	 */
   117  	public function equals( o ):Boolean 
   118  	{
   119  		return (o instanceof Line && o.a == a && o.b == b && o.c == c) ;
   120  	}
   121  
   122  	/**
   123  	 * Returns a Line reference defines with the two vectors in argument. 
   124  	 * This line is a line equation as two properties (a,b) such that (y = a*x + b) for any x or a unique c property such that (x = c) for all y.
   125  	 * The function takes two points as parameter, p0 and p1 containing two properties x and y.
   126  	 * @return a line equation as two properties (a,b) such that (y = a*x + b) for any x or a unique c property such that (x = c) for all y.
   127  	 */
   128  	public static function getLine( p1:Vector2, p2:Vector2 ):Line 
   129  	{
   130  		var x0:Number = p1.x;
   131  		var y0:Number = p1.y;
   132  		var x1:Number = p2.x;
   133  		var y1:Number = p2.y;
   134  		var l:Line = new Line() ;
   135  		if (x0 == x1) 
   136  		{
   137  			if (y0 == y1)
   138  			{
   139  				l = null ;
   140  			}
   141  			else 
   142  			{
   143  				l.c = x0 ; // Otherwise, the line is a vertical line
   144  			}
   145  		}
   146  		else 
   147  		{
   148  			l.a = (y0 - y1) / (x0 - x1) ;
   149  			l.b = y0 - (l.a * x0) ;
   150  		}
   151  		return l ;
   152  	}
   153  
   154  	/**
   155  	 * Returns a point (x,y) that is the intersection of two lines.
   156  	 * A line is defined either by a and b parameters such that (y = a*x + b) for any x or 
   157  	 * a single parameter c such that (x = c) for all y.
   158  	 * @return a vector (x,y) that is the intersection of two lines.
   159  	 */
   160  	public static function getLineCross(l1:Line, l2:Line):Vector2
   161  	{
   162  		if ( l1 == null || l2 == null ) 
   163  		{
   164  			return null ;
   165  		}
   166  		var a0:Number = l1.a ;
   167  		var b0:Number = l1.b ;
   168  		var c0:Number = l1.c ;
   169  		var a1:Number = l2.a ;
   170  		var b1:Number = l2.b ;
   171  		var c1:Number = l2.c ;
   172  		var u:Number ;
   173  		if ( !c0 && !c1 ) 
   174  		{
   175  			if (a0 == a1) 
   176  			{
   177  				return null ;
   178  			} 
   179  			u = (b1 - b0) / (a0 - a1);		
   180  			return new Vector2 (u , a0 * u + b0 ) ;
   181  		} 
   182  		else 
   183  		{
   184  			if (c0) 
   185  			{
   186  				if (c1) 
   187  				{
   188  					return null ;
   189  				}
   190  				else 
   191  				{
   192  					return new Vector2 (c0, (a1*c0)+b1) ;
   193  				}
   194  			} 
   195  			else if (c1) 
   196  			{
   197  				return new Vector2(c1, (a0*c1) + b0) ;
   198  			}
   199  		}
   200  	}

   202  	/**
   203  	 * Returns the middle of a segment define by two points.
   204  	 * @return the middle of a segment define by two points.
   205  	 */
   206  	public static function getMiddle(v1:Vector2, v2:Vector2):Vector2 
   207  	{
   208  		return Vector2Util.getMiddle(v1,v2) ;
   209  	}
   210  	
   211  	/**
   212  	 * Returns a point on a segment [p1, p2] which distance from p1 is ratio of the length [p1, p2].
   213  	 * @return a point on a segment [p1, p2] which distance from p1 is ratio of the length [p1, p2].
   214  	 */
   215  	public static function getPointOnSegment(p1:Vector2, p2:Vector2, ratio:Number):Vector2 
   216  	{
   217  		return new Vector2( p1.x + ((p2.x - p1.x) * ratio) , p1.y + ((p2.y - p1.y) * ratio) ) ;
   218  	}
   219  	
   220  	/**
   221  	 * Returns a line equation as two properties (a,b) such that (y = a*x + b) for any x or a unique c property such that (x = c) for all y.
   222  	 * The function takes two parameters, a point p(x,y) through which the line passes and a direction vector v(x,y).
   223  	 * @return a line equation as two properties (a,b) such that (y = a*x + b) for any x.
   224  	 */
   225  	public static function getVectorLine(p:Vector2, v:Vector2):Line 
   226  	{
   227  		var l:Line = new Line() ;
   228  		var x:Number = p.x;
   229  		var vx = v.x ;
   230  		if (vx == 0) 
   231  		{
   232  			l.c = x ;
   233  		}
   234  		else 
   235  		{
   236  			l.a = v.y / vx ;
   237  			l.b = p.y - (l.a * x);
   238  		}
   239  		return l ;
   240  	}

   242  	/**
   243  	 * Returns the Object representation of this object.
   244  	 * @return the Object representation of this object.
   245  	 */
   246  	public function toObject():Object 
   247  	{
   248  		return { a:a , b:b , c:c } ;
   249  	}
   250  
   251  	/**
   252  	 * Returns a Eden reprensation of the object.
   253  	 * @return a string representing the source code of the object.
   254  	 */
   255  	public function toSource(indent:Number, indentor:String):String 
   256  	{
   257  		var params:Array = [ Serializer.toSource(a) , Serializer.toSource(b) ] ;
   258  		if ( !isNaN(c) ) 
   259  		{
   260  			params.push( Serializer.toSource(c) ) ;
   261  		}
   262  		return Serializer.getSourceOf(this, params ) ;
   263  	}
   264  
   265  	/**
   266  	 * Returns the string representation of the object.
   267  	 * @return the string representation of the object.
   268  	 */ 
   269  	public function toString():String 
   270  	{
   271  		var s:String = "[Line a:" + (isNaN(a) ? 0 : a) + ", b:" + (isNaN(b) ? 0 : b) ;
   272  		if (!isNaN(c))
   273  		{
   274  			s += ", c:" + (isNaN(c) ? 0 : c) ;
   275  		}
   276  		s +=  "]" ;
   277  		return s ;
   278  	}
