    32  /**
    33   * Represents a transformation matrix that determines how to map points from one coordinate space to another.
    34   * <p>By setting the properties of a Matrix object and applying it to a MovieClip or BitmapData object you can perform various graphical transformations on the object. These transformation functions include translation (x and y repositioning), rotation, scaling, and skewing.</p>
    35   * <p><b>Example :</b></p>
    36   * {@code
    37   * import pegas.geom.TransformMatrix;
    38   * var matrix_1:TransformMatrix = new TransformMatrix();
    39   * trace(matrix_1); // (a:1,b:0,c:0,d:1,tx:0,ty:0)
    40   * 
    41   * var matrix_2:TransformMatrix = new TransformMatrix(1, 2, 3, 4, 5, 6);
    42   * trace(matrix_2); // (a:1,b:2,c:3,d:4,tx:5,ty:6)
    43   * }
    44   * The following example creates matrix_1 by sending no parameters to the Matrix constructor and matrix_2 by sending parameters to it. The TransformMatrix object matrix_1, which is created with no parameters, is an identity Matrix with the values (a:1,b:0,c:0,d:1,tx:0,ty:0).
    45   * @author eKameleon
    46   */
    47  class pegas.geom.TransformMatrix extends CoreObject implements ICloneable, IEquality, ISerializable 
    48  {
    49  
    50  	/**
    51  	 * Creates a new TransformMatrix instance.
    52  	 */
    53  	public function TransformMatrix( a:Number, b:Number, c:Number, d:Number, tx:Number, ty:Number) 
    54  	{
    55  		if (arguments.length == 0) 
    56  		{
    57  			identity() ;
    58  		}
    59  		else 
    60  		{
    61  			this.a = a ;
    62  			this.b = b ;
    63  			this.c = c ;
    64  			this.d = d ;
    65  			this.tx = tx ;
    66  			this.ty = ty ;
    67  		}
    68  	}
    69  
    70  	/**
    71  	 * The value in the first row and first column of the Matrix object, which affects the positioning of pixels along the x axis when scaling or rotating an image.
    72  	 */
    73  	public var a:Number ;
    74  
    75  	/**
    76  	 * The value in the first row and second column of the Matrix object, which affects the positioning of pixels along the y axis when rotating or skewing an image.
    77  	 */
    78  	public var b:Number ;
    79  
    80  	/**
    81  	 * The value in the second row and first column of the Matrix object, which affects the positioning of pixels along the x axis when rotating or skewing an image.
    82  	 */
    83  	public var c:Number ;
    84  
    85  	/**
    86  	 * The value in the second row and second column of the Matrix object, which affects the positioning of pixels along the y axis when scaling or rotating an image.
    87  	 */
    88  	public var d:Number ;
    89  
    90  	/**
    91  	 * The distance by which to translate each point along the x axis.
    92  	 */
    93  	public var tx:Number ;
    94  
    95  	/**
    96  	 * The distance by which to translate each point along the y axis.
    97  	 */
    98  	public var ty:Number ;
    99  		
   100  	public function clone() 
   101  	{
   102  		return new TransformMatrix(a, b, c, d, tx, ty) ;
   103  	}
   104  
   105  	/**
   106  	 * Concatenates a matrix with the current matrix, effectively combining the geometric effects of the two.
   107  	 */
   108  	public function concat(m:TransformMatrix):Void 
   109  	{
   110  		var r_a:Number = a * m.a ;
   111  		var r_d:Number = d * m.d ;
   112  		var r_b:Number = 0 ;
   113  		var r_c:Number = 0 ;
   114  		var r_tx:Number = tx * m.a + m.tx ;
   115  		var r_ty:Number = ty * m.d + m.ty ;
   116  		if (b != 0 || c != 0 || m.b != 0 || m.c != 0) 
   117  		{
   118  			r_a = r_a + b * m.c ;
   119  			r_d = r_d + c * m.b ;
   120  			r_b = r_b + (a * m.b + b * m.d) ;
   121  			r_c = r_c + (c * m.a + d * m.c) ;
   122  			r_tx = r_tx + ty * m.c ;
   123  			r_ty = r_ty + tx * m.b ;
   124  		}
   125  		a = r_a ;
   126  		b = r_b ;
   127  		c = r_c ;
   128  		d = r_d ;
   129  		tx = r_tx ;
   130  		ty = r_ty ;
   131  	}
   132  
   133  	/**
   134  	 * Includes parameters for scaling, rotation, and translation.
   135  	 */
   136  	public function createBox(scaleX:Number, scaleY:Number, rotation:Number, x:Number, y:Number):Void 
   137  	{
   138  		var rr:Number = isNaN(rotation) ? 0 : rotation ;
   139  		var rx:Number = isNaN(x) ? 0 : x ;
   140  		var ry:Number = isNaN(y) ? 0 : y ;
   141  		identity() ;
   142  		rotate(rr) ;
   143  		scale(scaleX, scaleY) ;
   144  		tx = rx ;
   145  		ty = ry ;
   146  	}
   147  	
   148  	/**
   149  	 * Creates the specific style of matrix expected by the MovieClip.beginGradientFill() method.
   150  	 */
   151  	public function createGradientBox(width:Number, height:Number, rotation:Number, x:Number, y:Number):Void 
   152  	{
   153  		var rr = isNaN(rotation) ? 0 : rotation ;
   154  		var rx:Number = isNaN(x) ? 0 : x ;
   155  		var ry:Number = isNaN(y) ? 0 : y ;
   156  		createBox((width / 1638.400000) , (height / 1638.400000) , rr, rx + width / 2, ry + height / 2) ;
   157  	}
   158  	
   159  	/**
   160  	 * Given a point in the pretransform coordinate space, returns the coordinates of that point after the transformation occurs.
   161  	 */
   162  	public function deltaTransformPoint(p:Point):Point 
   163  	{
   164  		return new Point(a * p.x + c * p.y , d * p.y + b * p.x) ;
   165  	}
   166  	
   167  	public function equals(o):Boolean 
   168  	{
   169  		return ( o instanceof TransformMatrix && o.a == a && o.b == b && o.c == c && o.d == d && o.tx == tx && o.ty == ty) ;
   170  	}
   171  	
   172  	/**
   173  	 * Sets each matrix property to a value that cause a transformed movie clip or geometric construct to be identical to the original.
   174  	 */
   175  	public function identity():Void 
   176  	{
   177  		a = d = 1 ;
   178  		b = c = 0 ;
   179  		tx = ty = 0 ;
   180  	}
   181  	
   182  	/**
   183  	 * Performs the opposite transformation of the original matrix.
   184  	 */
   185  	public function invert():Void 
   186  	{
   187  		if (b == 0 && c == 0) 
   188  		{
   189  			a = 1 / a ;
   190  			d = 1 / d ;
   191  			b = 0 ;
   192  			c = 0 ;
   193  			tx = - a * tx ;
   194  			ty = - d * ty ;
   195  		}
   196  		else 
   197  		{
   198  			var a0:Number = a ;
   199  			var a1:Number = b ;
   200  			var a2:Number = c ;
   201  			var a3:Number = d ;
   202  			var det:Number = a0 * a3 - a1 * a2;
   203  			if (det == 0) identity() ;
   204  			else 
   205  			{
   206  				det = 1 / det ;
   207  				a = a3 * det ;
   208  				b = -a1 * det ;
   209  				c = -a2 * det ;
   210  				d = a0 * det ;
   211  				var p:Point = deltaTransformPoint( new Point(tx, ty) ) ;
   212  				tx = - p.x ;
   213  				ty = - p.y ;
   214  			}
   215  		}
   216  	}
   217  	
   218  	/**
   219  	 * Sets the values in the current matrix so that the matrix can be used to apply a rotation transformation.
   220  	 */
   221  	public function rotate( radians:Number ):Void 
   222  	{
   223  		var c:Number = Math.cos(radians);
   224  		var s:Number = Math.sin(radians);
   225  		concat(new TransformMatrix(c, s, -s, c, 0, 0)) ;
   226  	}
   227  	
   228  	/**
   229  	 * Sets the values in the current matrix so that the matrix can be used to apply a rotation transformation with a degrees value in argument.
   230  	 */
   231  	public function rotateD( degrees:Number ):Void 
   232  	{
   233  		rotate(Trigo.degreesToRadians(degrees)) ;
   234  	}
   235  	
   236  	/**
   237  	 * Modifies a matrix so that its effect, when applied, is to resize an image.
   238  	 */
   239  	public function scale(sx:Number, sy:Number):Void 
   240  	{
   241  		concat(new TransformMatrix(sx, 0, 0, sy, 0, 0));
   242  	}

   244  	/**
   245  	 * Returns a flash.geom.Matrix reference of this TransformMatrix object.
   246  	 * @return a flash.geom.Matrix reference of this TransformMatrix object.
   247  	 */
   248  	public function toFlash():flash.geom.Matrix
   249  	{
   250  		return new flash.geom.Matrix(a, b, c, d, tx, ty) ;
   251  	}

   253  	/**
   254  	 * Returns a Eden reprensation of the object.
   255  	 * @return a string representing the source code of the object.
   256  	 */
   257  	public function toSource( indent:Number, indentor:String):String 
   258  	{
   259  		return "new TransformMatrix(" + a + "," + b + "," + c + "," + d + "," + tx + "," + ty + ")" ;
   260  	}
   261  
   262  	/**
   263  	 * Returns the string representation of the object.
   264  	 * @return the string representation of the object.
   265  	 */ 
   266  	public function toString():String 
   267  	{
   268  		return "[a:" + a + ",b:" + b + ",c:" + c + ",d:" + d + ",tx:" + tx + ",ty:" + ty +  "]" ;
   269  	}
   270  
   271  	/**
   272  	 * Applies the geometric transformation represented by the Matrix object to the specified point.
   273  	 */
   274  	public function transformPoint(p:Point):Point 
   275  	{
   276  		return new Point( (a * p.x) + (c * p.y) + tx , (d * p.y) + (b * p.x) + ty ) ;
   277  	}
   278  	
   279  	/**
   280  	 * Modifies a Matrix object so that the effect of its transformation is to move an object along the x and y axes.
   281  	 */
   282  	public function translate(dx:Number, dy:Number):Void 
   283  	{
   284  		tx += dx ;
   285  		ty += dy ;
   286  	}
