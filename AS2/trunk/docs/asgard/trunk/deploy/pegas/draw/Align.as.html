    24  import vegas.util.ArrayUtil;
    25  
    26  /**
    27   * The Align enumeration class provides constant values to align displays or components.
    28   * @author eKameleon
    29   */
    30  class pegas.draw.Align 
    31  {
    32  
    33  	/**
    34  	 * Defines the CENTER value (0).
    35  	 */
    36  	static public var CENTER:Number = 0 ;
    37  
    38  	/**
    39  	 * Defines the LEFT value (2).
    40  	 */
    41  	static public var LEFT:Number = 2 ;
    42  
    43  	/**
    44  	 * Defines the RIGHT value (4).
    45  	 */
    46  	static public var RIGHT:Number = 4 ;
    47  
    48  	/**
    49  	 * Defines the TOP value (8).
    50  	 */
    51  	static public var TOP:Number = 8 ;
    52  
    53  	/**
    54  	 * Defines the BOTTOM value (16).
    55  	 */
    56  	static public var BOTTOM:Number = 16 ;
    57  
    58  	/**
    59  	 * Defines the REVERSE value (32).
    60  	 */
    61  	static public var REVERSE:Number = 32 ;
    62  
    63  	/**
    64  	 * Defines the BOTTOM_LEFT value (18).
    65  	 */
    66  	static public var BOTTOM_LEFT = BOTTOM | LEFT ;
    67  
    68  	/**
    69  	 * Defines the BOTTOM_RIGHT value (20).
    70  	 */
    71  	static public var BOTTOM_RIGHT = BOTTOM | RIGHT ;
    72  
    73  	/**
    74  	 * Defines the TOP_LEFT value (10).
    75  	 */
    76  	static public var TOP_LEFT = TOP | LEFT ;
    77  
    78  	/**
    79  	 * Defines the TOP_RIGHT value (12).
    80  	 */
    81  	static public var TOP_RIGHT = TOP | RIGHT ;
    82  
    83  	/**
    84  	 * Defines the LEFT_BOTTOM value (50).
    85  	 */
    86  	static public var LEFT_BOTTOM = BOTTOM_LEFT | REVERSE ;
    87  
    88  	/**
    89  	 * Defines the RIGHT_BOTTOM value (52).
    90  	 */
    91  	static public var RIGHT_BOTTOM = BOTTOM_RIGHT | REVERSE ;
    92  
    93  	/**
    94  	 * Defines the LEFT_TOP value (42).
    95  	 */
    96  	static public var LEFT_TOP = TOP_LEFT | REVERSE ;
    97  
    98  	/**
    99  	 * Defines the RIGHT_TOP value (44).
   100  	 */
   101  	static public var RIGHT_TOP = TOP_RIGHT | REVERSE ;
   102  
   103  	static private var __ASPF__ = _global.ASSetPropFlags(Align, null , 7, 7) ;
   104  
   105  	/**
   106  	 * Converts a string value in this Align value.
   107  	 * <p><b>Example :</b></p>
   108  	 * {@code
   109  	 * import pegas.draw.Align ;
   110  	 * 
   111  	 * var sAlign:String = "l" ;
   112  	 * 
   113  	 * trace( Align.toNumber("l") == Align.LEFT ) ; // true
   114  	 * 
   115  	 * }
   116  	 */
   117  	static public function toNumber(str:String):Number 
   118  	{
   119  		switch (str.toLowerCase()) 
   120  		{
   121  			case "l" : 
   122  			{
   123  				return Align.LEFT ;
   124  			}
   125  			case "r" :
   126  			{
   127  				return Align.RIGHT ;
   128  			}
   129  			case "t" :
   130  			{
   131  				return Align.TOP ;
   132  			}
   133  			case "b" :
   134  			{
   135  				return Align.BOTTOM ;
   136  			}
   137  			case "tl" :
   138  			{
   139  				return Align.TOP_LEFT ;
   140  			}
   141  			case "tr" :
   142  			{
   143  				return Align.TOP_RIGHT ;
   144  			}
   145  			case "bl" :
   146  			{
   147  				return Align.BOTTOM_LEFT ;
   148  			}
   149  			case "br" :
   150  			{
   151  				return Align.BOTTOM_RIGHT ;
   152  			}
   153  			case "lt" :
   154  			{
   155  				return Align.LEFT_TOP ;
   156  			}
   157  			case "rt" :
   158  			{
   159  				return Align.RIGHT_TOP ;
   160  			}
   161  			case "lb" :
   162  			{
   163  				return Align.LEFT_BOTTOM ;
   164  			}
   165  			case "rb" :
   166  			{
   167  				return Align.RIGHT_BOTTOM ;
   168  			}	
   169  			default :
   170  			{
   171  				return Align.CENTER ;
   172  			}
   173  		}
   174  	}
   175  
   176  	/**
   177  	 * Returns the string representation of the specified Align value passed in argument.
   178  	 * <p><b>Example :</b></p>
   179  	 * {@code
   180  	 * import pegas.draw.Align ;
   181  	 * trace( Align.toString(Align.LEFT)) ; // "l"
   182  	 * trace( Align.toString(Align.TOP_LEFT)) ; // "tl"
   183  	 * trace( Align.toString(Align.RIGHT_BOTTOM)) ; // "rb"
   184  	 * }
   185  	 * @return the string representation of the specified Align value passed in argument.
   186  	 */
   187  	static public function toString(n:Number):String 
   188  	{
   189  		switch (n) 
   190  		{
   191  			case Align.LEFT : return "l" ;
   192  			case Align.RIGHT : return "r" ;
   193  			case Align.TOP : return "t" ;
   194  			case Align.BOTTOM : return "b" ;
   195  			case Align.TOP_LEFT : return "tl" ; 
   196  			case Align.TOP_RIGHT : return "tr" ;
   197  			case Align.BOTTOM_LEFT  : return "bl" ;
   198  			case Align.BOTTOM_RIGHT : return "br" ;
   199  			case Align.LEFT_TOP : return "lt" ; 
   200  			case Align.RIGHT_TOP : return "rt" ;
   201  			case Align.LEFT_BOTTOM : return "lb" ;
   202  			case Align.RIGHT_BOTTOM : return "rb" ;
   203  			default : return "" ;
   204  		}
   205  	}
   206  
   207  	/**
   208  	 * Returns {@code true} if the specified Align value in argument is a valid Align value else returns {@code false}.
   209  	 * @return {@code true} if the specified Align value in argument is a valid Align value else returns {@code false}.
   210  	 */
   211  	static public function validate(n:Number):Boolean 
   212  	{
   213  		var a:Array = [ 
   214  			Align.CENTER, Align.LEFT , Align.RIGHT, 
   215  			Align.TOP, Align.BOTTOM, 
   216  			Align.TOP_LEFT, Align.TOP_RIGHT,
   217  			Align.BOTTOM_LEFT, Align.BOTTOM_RIGHT
   218  		] ;
   219  		return  ArrayUtil.contains(a, n) ;
   220  	}
   221  	
   222  }
