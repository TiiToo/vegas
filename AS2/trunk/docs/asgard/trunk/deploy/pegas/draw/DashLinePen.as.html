    24  import pegas.draw.LinePen;
    25  import pegas.geom.Vector2;
    26  
    27  /**
    28   * This pen is the basic tool to draw a dash line in a MovieClip reference.
    29   * @author eKameleon
    30   */
    31  class pegas.draw.DashLinePen extends LinePen 
    32  {
    33  
    34  	/**
    35  	 * Creates a new DashLinePen instance.
    36  	 */
    37  	public function DashLinePen(target:MovieClip, isNew:Boolean) 
    38  	{
    39  		super(target, isNew) ;
    40  	}
    41  
    42  	/**
    43  	 * The default length value of the dashs in the line.
    44  	 */
    45  	public static var DEFAULT_LENGTH:Number = 2 ;
    46  
    47  	/**
    48  	 * The default spacing value of the dashs in the line.
    49  	 */
    50  	public static var DEFAULT_SPACING:Number = 2 ;
    51  
    52  	/**
    53  	 * (read-write) Determinates the length of a dash in the line.
    54  	 */
    55  	public function get length():Number 
    56  	{
    57  		return getLength() ;	
    58  	}
    59  	
    60  	/**
    61  	 * @private
    62  	 */
    63  	public function set length(n:Number):Void 
    64  	{
    65  		setLength(n) ;	
    66  	}
    67  
    68  	/**
    69  	 * (read-write) Determinates the spacing value between two dashs in this line.
    70  	 */
    71  	public function get spacing():Number {
    72  		return getSpacing() ;	
    73  	}
    74  	
    75  	/**
    76  	 * @private
    77  	 */
    78  	public function set spacing(n:Number):Void 
    79  	{
    80  		setSpacing(n) ;	
    81  	}
    82  	
    83  	/**
    84  	 * Returns a shallow copy of this pen.
    85  	 * @return a shallow copy of this pen.
    86  	 */
    87  	public function clone() 
    88  	{
    89  		var p:DashLinePen = new DashLinePen(_target) ;
    90  		p.setLine( getStart(), getEnd(), getThickness(), getColor(), getAlpha(), getLength(), getSpacing()) ;
    91  		return p ;
    92  	}
    93  	
    94  	/**
    95  	 * Draw the dash line shape.
    96  	 * @param start The vector object who define the start position to draw the line.
    97  	 * @param end The vector object who define the end position to draw the line.
    98  	 * @param thickness The thickness value of the line.
    99  	 * @param color The color value of the line.
   100  	 * @param alpha The alpha value of the line.
   101  	 * @param length The length of a dash.
   102  	 * @param spacing The spacing between two dashs.
   103  	 */
   104  	public function draw( start:Vector2 , end:Vector2, thickness:Number, color:Number, alpha:Number, length:Number, spacing:Number ):Void 
   105  	{
   106  		
   107  		if (arguments.length > 0) 
   108  		{
   109  			setLine.apply(this, arguments) ;
   110  		}
   111  		
   112  		var s:Vector2 = _pStart ;
   113  		var e:Vector2 = _pEnd ;
   114  		var segl:Number = getLength() + getSpacing() ;
   115  		if (isNaN(segl)) segl = 0 ;
   116  		var dx:Number = e.x - s.x ;
   117  		var dy:Number = e.y - s.y ;
   118  		var delta = Math.sqrt((dx * dx) + (dy * dy));
   119  		var nbSegs:Number = Math.floor(Math.abs(delta / segl)) ;
   120  		_radians = Math.atan2 ( dy, dx ) ;
   121  		__x = s.x;
   122  		__y = s.y;
   123  		dx = Math.cos(_radians)* segl ;
   124  		dy = Math.sin(_radians)* segl ;
   125  		if (_autoClear) clear () ;
   126  		lineStyle (t, lc, la) ;
   127  		for (var i:Number = 0; i < nbSegs; i++) 
   128  		{
   129  			moveTo( __x, __y ) ;
   130  			_lineRadiansTo ( _length ) ;
   131  			__x += dx ;
   132  			__y += dy ;
   133  		}
   134  		moveTo(__x, __y) ; 
   135  		delta = Math.sqrt((e.x - __x)*(e.x - __x)+ (e.y - __y)* (e.y - __y)) ;
   136  		if(delta>_length) 
   137  		{
   138  			_lineRadiansTo ( _length ) ;
   139  		}
   140  		else if (delta > 0) 
   141  		{
   142  			_lineRadiansTo ( delta ) ;
   143  		}
   144  		moveTo(e.x, e.y);
   145  	}
   146  
   147  	/**
   148  	 * Returns the length of a dash in the line.
   149  	 * @return the length of a dash in the line.
   150  	 */
   151  	public function getLength():Number
   152  	{ 
   153  		return _length ;
   154  	}
   155  
   156  	/**
   157  	 * Returns the spacing value between two dashs.
   158  	 * @return the spacing value between two dashs.
   159  	 */
   160  	public function getSpacing():Number 
   161  	{ 
   162  		return _spacing ;
   163  	}
   164  	
   165  	/**
   166  	 * Returns the size of a dash.
   167  	 * @return the size of a dash.
   168  	 */
   169  	public function setLength(n:Number, noDraw:Boolean):Void 
   170  	{
   171  		_length = isNaN(n) ? DEFAULT_LENGTH : n ; 
   172  		if (!noDraw) draw() ;	
   173  	}
   174  
   175  	/**
   176  	 * Sets all the line properties before draw it.
   177  	 */
   178  	public function setLine( start:Vector2, end:Vector2, thickness:Number, color:Number, alpha:Number, length:Number, spacing:Number ):Void 
   179  	{
   180  		super.setLine(start, end, thickness, color, alpha) ;
   181  		if (!isNaN( length ) ) 
   182  		{
   183  			setLength( length, true) ;
   184  		}
   185  		if (!isNaN( spacing ) ) 
   186  		{
   187  			setSpacing( spacing, true) ;
   188  		}
   189  	}
   190  
   191  	/**
   192  	 * Sets the spacing value between two dashs in the line.
   193  	 */
   194  	public function setSpacing(n:Number, noDraw:Boolean):Void 
   195  	{ 
   196  		_spacing = isNaN( n ) ? DEFAULT_SPACING : n ; 
   197  		if (!noDraw) draw() ;	
   198  	}
   199  
   200  	private var _length:Number = 2 ;
   201  	
   202  	private var _radians:Number ; 
   203  	
   204  	private var _spacing:Number = 2 ;
   205  	
   206  	private var __x:Number ;
   207  	
   208  	private var __y:Number ;
   209  	
   210  	/**
   211  	 * @private
   212  	 */
   213  	private function _lineRadiansTo ( n:Number ) : Void 
   214  	{
   215  		lineTo ( __x + Math.cos(_radians) * n ,  __y + Math.sin(_radians) * n ) ;
   216  	}
   217  	
   218  }
