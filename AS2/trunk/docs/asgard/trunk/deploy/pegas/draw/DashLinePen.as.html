    24  import pegas.draw.LinePen;
    25  import pegas.geom.Point;
    26  
    27  /**
    28   * This pen is the basic tool to draw a dash line in a MovieClip reference.
    29   * @author eKameleon
    30   */
    31  class pegas.draw.DashLinePen extends LinePen 
    32  {
    33  
    34  	/**
    35  	 * Creates a new DashLinePen instance.
    36  	 */
    37  	public function DashLinePen(target:MovieClip, isNew:Boolean) 
    38  	{
    39  		super(target, isNew) ;
    40  	}
    41  
    42  	public var default_length:Number = 2 ;
    43  	
    44  	public var default_spacing:Number = 2 ;
    45  
    46  	public function get length():Number {
    47  		return getLength() ;	
    48  	}
    49  	
    50  	public function set length(n:Number):Void {
    51  		setLength(n) ;	
    52  	}
    53  
    54  	public function get spacing():Number {
    55  		return getSpacing() ;	
    56  	}
    57  	
    58  	public function set spacing(n:Number):Void {
    59  		setSpacing(n) ;	
    60  	}
    61  	
    62  
    63  	public function clone() 
    64  	{
    65  		var p:DashLinePen = new DashLinePen(_target) ;
    66  		p.setLine(getStart(), getEnd(), getThickness(), getColor(), getAlpha(), getLength(), getSpacing()) ;
    67  		return p ;
    68  	}
    69  	
    70  	public function draw
    71  	(
    72  	
    73  		p_start:Point, p_end:Point, 
    74  		p_thickness:Number, p_color:Number,
    75  		p_alpha:Number, p_length:Number, p_spacing:Number
    76  	
    77  	):Void 
    78  	{
    79  		
    80  		if (arguments.length > 0) setLine.apply(this, arguments) ;
    81  		
    82  		var s:Point = _pStart ;
    83  		var e:Point = _pEnd ;
    84  		var segl:Number = getLength() + getSpacing() ;
    85  		if (isNaN(segl)) segl = 0 ;
    86  		var dx:Number = e.x - s.x ;
    87  		var dy:Number = e.y - s.y ;
    88  		var delta = Math.sqrt((dx * dx) + (dy * dy));
    89  		var nbSegs:Number = Math.floor(Math.abs(delta / segl)) ;
    90  		__radians = Math.atan2 ( dy, dx ) ;
    91  		__x = s.x;
    92  		__y = s.y;
    93  		dx = Math.cos(__radians)* segl ;
    94  		dy = Math.sin(__radians)* segl ;
    95  		if (_autoClear) clear () ;
    96  		lineStyle (t, lc, la) ;
    97  		for (var i:Number = 0; i < nbSegs; i++) {
    98  			moveTo( __x, __y ) ;
    99  			_lineRadiansTo ( _length ) ;
   100  			__x += dx ;
   101  			__y += dy ;
   102  		}
   103  		moveTo(__x, __y) ; 
   104  		delta = Math.sqrt((e.x - __x)*(e.x - __x)+ (e.y - __y)* (e.y - __y)) ;
   105  		if(delta>_length) _lineRadiansTo ( _length ) ;
   106  		else if (delta > 0) _lineRadiansTo ( delta ) ;
   107  		moveTo(e.x, e.y);
   108  	}
   109  
   110  	public function getLength():Number
   111  	{ 
   112  		return _length ;
   113  	}
   114  
   115  	public function getSpacing():Number 
   116  	{ 
   117  		return _spacing ;
   118  	}
   119  
   120  	public function setLength(n:Number, noDraw:Boolean):Void 
   121  	{
   122  		_length = isNaN(n) ? default_length : n ; 
   123  		if (!noDraw) draw() ;	
   124  	}
   125  
   126  	public function setLine
   127  	(
   128  		p_start:Point, p_end:Point, 
   129  		p_thickness:Number, p_color:Number, p_alpha:Number, 
   130  		p_length:Number, p_spacing:Number 
   131  	):Void 
   132  	{
   133  		super.setLine(p_start, p_end, p_thickness, p_color, p_alpha) ;
   134  		if (!isNaN(p_length)) setLength(p_length, true) ;
   135  		if (!isNaN(p_spacing)) setSpacing(p_spacing, true) ;
   136  	}
   137  
   138  	public function setSpacing(n:Number, noDraw:Boolean):Void { 
   139  		_spacing = isNaN(n) ? default_spacing : n ; 
   140  		if (!noDraw) draw() ;	
   141  	}
   142  
   143  	private var _length:Number = 2 ;
   144  	private var __radians:Number ; 
   145  	private var _spacing:Number = 2 ;
   146  	private var __x:Number ;
   147  	private var __y:Number ;
   148  	
   149  	private function _lineRadiansTo ( n:Number ) : Void 
   150  	{
   151  		lineTo ( __x + Math.cos(__radians) * n ,  __y + Math.sin(__radians) * n ) ;
   152  	}
   153  }
