    26  /**
    27   * The Canvas pen is used to draw a complex shape with differents points in a data model.
    28   * <p><b>Example :</b></p>
    29   * {@code
    30   * import pegas.draw.Canvas ;
    31   * var mc:MovieClip = createEmptyMovieClip("container", 1) ;
    32   * mc._x = 200 ;
    33   * mc._y = 200 ;
    34   * 
    35   * // the model used to draw the shape in the canvas.
    36   * var data:Array = 
    37   * [
    38   *     ['S',[2, 0xFFFFFF, 100]] , // lineStyle(2, 0xFFFFFF, 100)
    39   *     ['F',[0xFF0000, 100]] , // beginFill(0xFF0000)
    40   *     ['M',[0,0]] , // moveTo (0,0)
    41   *     ['L',[80,90]] , // lineTo (80,90)
    42   *     ['L',[20,90]] , // lineTo (20,90)
    43   *     ['L',[20,50]],  // lineTo (20,50)
    44   *     ['EF'] // endFill ()
    45   * ] ;
    46   * 
    47   * var c:Canvas = new Canvas(mc, data) ;
    48   * trace ("canvas : " + c) ;
    49   * c.draw() ;
    50   * }
    51   * <p><b>Thanks :</b> Peter Hall - <a href='http://www.peterjoel.com/blog/'>Peter Joel Blog</a></p>
    52   * @author eKameleon
    53   */
    54  class pegas.draw.Canvas extends AbstractPen

    95  	/**
    96  	 * Converts the curves in the model to lines.
    97  	 */
    98  	public function curvesToLines():Void 
    99  	{
   100  		var d:Array = _data ;
   101  		var e:Array ;
   102  		var value:Array ;
   103  		var l:Number = d.length ;
   104  		while (--l > -1) 
   105  		{
   106  			e = d[l];
   107  			value = e[1];
   108  			if( e[0] == "C" )
   109  			{
   110  				d[l] = ["L",[value[2],value[3]]];
   111  			}
   112  		}
   113  	}

   121  	}

   123  	public function removeTransform():Void 
   124  	{
   125  		this.C = MovieClip.prototype.curveTo ;
   126  		this.L = MovieClip.prototype.lineTo ;
   127  		this.M = MovieClip.prototype.moveTo ;
   128  	}

   130  	/**
   131  	 * Draws the shape with the canvas pen.
   132  	 */
   133  	/*override*/ public function draw():Void 
   134  	{
   135  		Canvas.process(_target, this) ;
   136  	}

   138  	/**
   139  	 * Converts the lines in the model to curves.
   140  	 */
   141  	public function linesToCurves():Void 
   142  	{
   143  		var i:Number = 0 ;
   144  		var key:String ;
   145  		var value:Array ;
   146  		var e:Array ;
   147  		var d:Array = _data ;
   148  		var len:Number = d.length;
   149  		var ax:Number = 0 ;
   150  		var ay:Number = 0 ;
   151  		while ( i<len ) 
   152  		{
   153  			e = d[i] ;
   154  			key = e[0] ;
   155  			value = e[1] ;
   156  			if( key == "L" )
   157  			{
   158  				d[i] = ["C", [(ax+value[0])/2 , (ay+value[1])/2 , value[0], value[1]]] ;
   159  				ax = value[0] ;
   160  				ay = value[1] ;
   161  			} 
   162  			else if ( key == "C" )
   163  			{
   164  				ax = value[2] ;
   165  				ay = value[3] ;
   166  			}
   167  			else if ( key == "M" )
   168  			{
   169  				ax = value[0] ;
   170  				ay = value[1] ;
   171  			}
   172  			i++ ;
   173  		}
   174  	}
   175  
   176  	/**
   177  	 * This static method launch the process to draw in a specified movieclip target the canvas defined with the (@code draw} argument.
   178  	 * @param target the movieclip reference.
   179  	 * @param draw the {@code Canvas} reference used to draw in the {@code target} reference.
   180  	 */
   181  	public static function process( target:MovieClip , draw:Canvas ):Void 
   182  	{
   183  		var t:MovieClip = target ;
   184  		var d:Array = draw.getData() ;
   185  		var len:Number = draw.size() ;
   186  		if (t && len > 0) 
   187  		{
   188  			var c:Array ;
   189  			var i:Number = 0 ;
   190  			while (i < len) 
   191  			{
   192  				c = d[i++] ;
   193  				draw[c[0]].apply(t, c[1]) ;
   194  			}
   195  		}
   196  	}

   206  	/**
   207  	 * Apply a transform method in this Canvas. Uses the {@code CanvasTransform} tool class and this static methods.
   208  	 * @param transform the method effect used to transform the shape.
   209  	 * <p><b>Example :</b></p>
   210  	 * {@code
   211  	 * import pegas.draw.Canvas ;
   212  	 * import pegas.draw.CanvasTransform ;
   213  	 * 
   214  	 * var data:Array = 
   215  	 * [
   216  	 *     ['S',[2,0xFF0000,100]],
   217  	 *     ['F',[0x000000]],
   218  	 *     ['L',[80,0]],
   219  	 *     ['L',[160,0]],
   220  	 *     ['L',[160,50]],
   221  	 *     ['L',[160,100]],
   222  	 *     ['L',[80,100]],
   223  	 *     ['L',[0,100]],
   224  	 *     ['L',[0,50]],
   225  	 *     ['L',[0,0]],
   226  	 *     ['EF']
   227  	 * ] ;
   228  	 * 
   229  	 * this.createEmptyMovieClip("canvas_mc", 1) ;
   230  	 * canvas_mc._x = 400 ;
   231  	 * canvas_mc._y = 200 ;
   232  	 * 
   233  	 * var c:Canvas = new Canvas(canvas_mc, data) ;
   234  	 * trace (c + " : " + c.size()) ;
   235  	 * c.draw() ;
   236  	 * 
   237  	 * var bounds = canvas_mc.getBounds(canvas_mc);
   238  	 * 
   239  	 * var count:Number = 0;
   240  	 * onEnterFrame = function()
   241  	 * {
   242  	 *     count ++;
   243  	 *     var b = bounds ;
   244  	 *     var xAmount = Math.sin(count/8)*40 ;
   245  	 *     var yAmount = Math.cos(count/8)*50 ;
   246  	 *     var transform:Function = CanvasTransform.createPinch(b.xMin, b.yMin, b.xMax, b.yMax, xAmount, yAmount) ;
   247  	 *     c.setTransform( transform ) ;
   248  	 *     c.clear();
   249  	 *     c.draw() ;
   250  	 *     
   251  	 *     var speedX = (_xmouse - canvas_mc._x ) * 0.2 ;
   252  	 *     var speedY = (_ymouse - canvas_mc._y ) * 0.2 ;
   253  	 *     canvas_mc._x += speedX ;
   254  	 *     canvas_mc._y += speedY ;
   255  	 *     
   256  	 *     updateAfterEvent() ;
   257  	 * }
   258  	 * 
   259  	 * var cpt:Number = 0 ;
   260  	 * this.onKeyDown = function ()
   261  	 * {
   262  	 *     (cpt++ %2 == 0) ? c.linesToCurves() : c.curvesToLines() ;
   263  	 * }
   264  	 * Key.addListener(this) ;
   265  	 * }
   266  	 * @see CanvasTransform.
   267  	 */
   268  	public function setTransform( transform:Function ):Void 
   269  	{
   270  		if(typeof transform != "function")
   271  		{
   272  			removeTransform() ;
   273  		}
   274  		else 
   275  		{
   276  			this.C = function(cx:Number, cy:Number, ax:Number, ay:Number):Void 
   277  			{
   278  				var cp:Object = transform(cx,cy);
   279  				var ap:Object = transform(ax,ay);
   280  				this.curveTo(cp.x, cp.y, ap.x, ap.y);
   281  			} ;
   282  			this.L = function (x:Number, y:Number):Void 
   283  			{
   284  				var p:Object = transform(x,y);
   285  				this.lineTo(p.x,p.y);
   286  			} ;
   287  			this.M = function(x:Number, y:Number):Void 
   288  			{
   289  				var p:Object = transform(x,y) ;
   290  				this.moveTo(p.x,p.y);
   291  			} ;
   292  		}
   293  	}

   296  	 * Returns the number of elements in the model.

   301  		return _data.length ;
