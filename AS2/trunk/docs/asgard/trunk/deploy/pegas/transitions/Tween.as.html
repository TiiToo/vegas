    28  import pegas.transitions.Motion;
    29  import pegas.transitions.TweenEntry;
    30  import pegas.transitions.TweenProvider;
    31  
    32  /**
    33   * The Tween class lets you use ActionScript to move, resize, and fade movie clips easily on the Stage by specifying a property of the target movie clip to be tween animated over a number of frames or seconds.
    34   * The Tween class also lets you specify a variety of easing methods.
    35   * <p>Easing refers to gradual acceleration or deceleration during an animation, which helps your animations appear more realistic.</p>
    36   * <p><b>Example :</b></p>
    37   * {@code
    38   * import asgard.process.ActionEvent ;
    39   * import asgard.transitions.Tween ;
    40   * import asgard.transitions.easing.* ;
    41   * 
    42   * import vegas.events.Delegate ;
    43   * import vegas.events.EventListener ;
    44   * 
    45   * var onDebug:Function = function(ev:ActionEvent):Void
    46   * {
    47   *     trace (":: debug -> " + ev.type + " : " + ev.target ) ;
    48   * }
    49   * 
    50   * var debug:EventListener = new Delegate(this, onDebug) ;
    51   * 
    52   * var tw:Tween = new Tween (mc, "_x", Elastic.easeOut, mc._x, 400, 2, true, true) ;
    53   * tw.addEventListener("ALL", debug) ;
    54   * }
    55   * @author eKameleon
    56   */
    57  class pegas.transitions.Tween extends Motion 
    58  {
    59  
    60  	/**
    61  	 * Creates a new Tween instance.
    62  	 * <p><b>Usage :</b></p>
    63  	 * {@code
    64  	 * var tw:Tween = new Tween( obj, prop:String, e:Function, b:Number, f:Number, d:Number , u:Boolean, auto:Boolean) ;
    65  	 * var tw:Tween = new Tween( obj, ar:Array ) ;
    66  	 * }
    67  	 */
    68  	function Tween( args ) 
    69  	{
    70  		var obj = arguments[0] ;
    71  		if (obj) 
    72  		{
    73  			setTarget(obj) ;
    74  		}
    75  		else 
    76  		{
    77  			return ;
    78  		}
    79  		var l:Number = arguments.length ;
    80  		if (l == 2 && arguments[1] instanceof Array) 
    81  		{
    82  			setTweenProvider(arguments[1]) ;
    83  		} 
    84  		else if (l > 2) 
    85  		{
    86  			var p:String = arguments[1] ; // property
    87  			var e:Function = arguments[2] ; // easing
    88  			var b:Number = arguments[3] ; // begin
    89  			var f:Number = arguments[4] ; // finish
    90  			var d:Number = arguments[5] ; // duration
    91  			var u:Boolean = arguments[6] ; // useSeconds
    92  			var a:Boolean = arguments[7] ; // auto start
    93  			setDuration(d) ;
    94  			useSeconds = u ;
    95  			setTweenProvider( [new TweenEntry(p, e, b, f)] ) ;
    96  			if (a) run() ;
    97  		}
    98  	}
    99  
   100  	/**
   101  	 * (read-write) Returns the model of this Tween object.
   102  	 * @return the model of this Tween object.
   103  	 */
   104  	public function get tweenProvider():TweenProvider 
   105  	{
   106  		return getTweenProvider() ;
   107  	}
   108  
   109  	/**
   110  	 * (read-write) Sets the model of this Tween object.
   111  	 */
   112  	public function set tweenProvider( o ):Void 
   113  	{
   114  		setTweenProvider(o) ;
   115  	}
   116  
   117  	/**
   118  	 * Removes all entries in the model of this Tween object.
   119  	 */
   120  	public function clear():Void 
   121  	{
   122  		if (running) this.stop() ;
   123  		if (_model) _model.clear() ;
   124  		notifyCleared() ;
   125  	}
   126  
   127  	/**
   128  	 * Returns a shallow copy of this Tween object.
   129  	 * @return a shallow copy of this Tween object.
   130  	 */
   131  	/*override*/ public function clone() 
   132  	{
   133  		var t:Tween = new Tween() ;
   134  		t.setTarget(getTarget()) ;
   135  		t.duration = getDuration() ;
   136  		t.useSeconds = useSeconds ;
   137  		if (size() > 0) t.setTweenProvider(getTweenProvider().clone()) ;
   138  		return t ;
   139  	}
   140  
   141  	/**
   142  	 * Returns the TweenProvider model reference of this Tween object.
   143  	 * @return the TweenProvider model reference of this Tween object.
   144  	 */
   145  	public function getTweenProvider():TweenProvider 
   146  	{
   147  		return _model ;
   148  	}
   149  	
   150  	/**
   151  	 * Inserts a TweenEnry in the model of the Tween object.
   152  	 * @param entry a TweenEntry reference.
   153  	 */
   154  	public function insert( entry:TweenEntry ):Void 
   155  	{
   156  		if (!_model) setTweenProvider() ;
   157  		_model.insert( entry ) ;
   158  	}
   159  
   160  	/**
   161  	 * Inserts a new property in the Tween object.
   162  	 * @param prop the string representation of the number property.
   163  	 * @param easing the easing method used by the Tween on this property.
   164  	 * @param begin the begin value.
   165  	 * @param finish the finish value.
   166  	 * @return a TweenEntry defined by the specified arguments.
   167  	 */
   168  	public function insertProperty( prop , easing:Function, begin:Number, finish:Number):TweenEntry 
   169  	{
   170  		var e:TweenEntry = new TweenEntry(prop, easing, begin, finish) ;
   171  		insert(e) ;
   172  		return e ;
   173  	}
   174  
   175  	/**
   176  	 * Remove a TweenEntry in the Tween Object.
   177  	 */
   178  	public function remove(o:TweenEntry):Void 
   179  	{
   180  		if (running) stop() ;
   181  		_model.remove(o) ;
   182  	}	
   183  	
   184  	/**
   185  	 * Remove a property in the Tween Object.
   186  	 */
   187  	public function removeProperty(prop:String):Boolean 
   188  	{
   189  		if (running) stop() ;
   190  		return _model.remove(prop) ;
   191  	}
   192  
   193  	/**
   194  	 * Sets the model of this Tween object.
   195  	 */
   196  	public function setTweenProvider(o):Void 
   197  	{
   198  		if (o instanceof TweenProvider) 
   199  		{
   200  			_model = o ;
   201  		}
   202  		else if (o instanceof Array) 
   203  		{
   204  			_model = new TweenProvider(o) ;
   205  		}
   206  		else 
   207  		{
   208  			_model = new TweenProvider() ;
   209  		}
   210  	}
   211  	
   212  	/**
   213  	 * The numbers of elements(properties) in the model of this Tween.
   214  	 */
   215  	public function size():Number 
   216  	{
   217  		return _model.size() ;
   218  	}
   219  	
   220  	/**
   221  	 * Update the current Tween in the time.
   222  	 */
   223  	/*override*/ public function update():Void 
   224  	{
   225  		var o = _target ;
   226  		var t:Number = _time ;
   227  		var d:Number = _duration ;
   228  		var a:Array = _model.toArray() ;
   229  		var l:Number = a.length ;
   230  		while(--l > -1) 
   231  		{
   232  			var e:TweenEntry = a[l] ;
   233  			o[e.prop] = e.setPosition( e.getPosition(t, d) ) ;
   234  			updateAfterEvent() ;
   235  		}
   236  		notifyChanged() ;
   237  	}
   238  	
   239  	private var _model:TweenProvider ;
   240  
   241  }
