    24  import pegas.process.AbstractAction;
    25  
    26  import vegas.core.ITimer;
    27  import vegas.events.Delegate;
    28  import vegas.events.EventListener;
    29  import vegas.events.TimerEvent;
    30  import vegas.util.FrameTimer;
    31  import vegas.util.Timer;
    32  
    33  /**
    34   * The Motion class.
    35   * @author eKameleon
    36   */
    37  class pegas.transitions.Motion extends AbstractAction 
    38  {
    39  
    40  	/**
    41  	 * Creates a new Motion instance.
    42  	 */
    43  	function Motion() 
    44  	{
    45  		_oNext = new Delegate(this, nextFrame) ;
    46  	}
    47  
    48  	/**
    49  	 * Returns the duration of the tweened animation in frames or seconds.
    50  	 * @return the duration of the tweened animation in frames or seconds.
    51  	 */
    52  	public function get duration():Number 
    53  	{
    54  		return getDuration() ;
    55  	}
    56  	
    57  	/**
    58  	 * Sets the duration of the tweened animation in frames or seconds.
    59  	 */
    60  	public function set duration( n:Number ):Void 
    61  	{
    62  		setDuration(n) ;
    63  	}
    64  	
    65  	/**
    66  	 * Returns the number of frames per second of the tweened animation.
    67  	 * @return the number of frames per second of the tweened animation.
    68  	 */
    69  	public function get fps():Number 
    70  	{
    71  		return getFPS() ;
    72  	}
    73  	
    74  	/**
    75  	 * Sets the number of frames per second of the tweened animation.
    76  	 */
    77  	public function set fps( n:Number ):Void 
    78  	{
    79  		setFPS(n) ;
    80  	}
    81  
    82  	/**
    83  	 * Defined the internal prevTime value.
    84  	 */
    85  	public var prevTime:Number ;
    86  
    87  	/**
    88  	 * Returns the target reference of the object contrains by the Motion effect.
    89  	 * @return the target reference of the object contrains by the Motion effect.
    90  	 */
    91  	public function get target() 
    92  	{
    93  		return getTarget() ;	
    94  	}
    95  	
    96  	/**
    97  	 * Sets the target reference of the object contrains by the Motion effect.
    98  	 */
    99  	public function set target( oT ) 
   100  	{
   101  		setTarget( oT ) ;	
   102  	}
   103  
   104  	/**
   105  	 * Defined if the Motion used seconds or not.
   106  	 */
   107  	public var useSeconds:Boolean ;
   108  
   109  	/**
   110  	 * Returns a shallow copy of this object.
   111  	 * @return a shallow copy of this object.
   112  	 */
   113  	public function clone() 
   114  	{
   115  		//
   116  	}
   117  	
   118  	/**
   119  	 * Forwards the tweened animation directly to the end of the animation.
   120  	 */
   121  	public function fforward():Void 
   122  	{ 
   123  		setTime (_duration) ;
   124  		_fixTime() ;
   125  	}
   126  	
   127  	/**
   128  	 * Returns the duration of the tweened animation in frames or seconds.
   129  	 * @return the duration of the tweened animation in frames or seconds.
   130  	 */
   131  	public function getDuration():Number 
   132  	{ 
   133  		return _duration ;
   134  	}
   135  
   136  	/**
   137  	 * Returns the number of frames per second of the tweened animation.
   138  	 * @return the number of frames per second of the tweened animation.
   139  	 */
   140  	public function getFPS ():Number 
   141  	{ 
   142  		return _fps;
   143  	}
   144  
   145  	/**
   146  	 * Returns the target reference of the object contrains by the Motion effect.
   147  	 * @return the target reference of the object contrains by the Motion effect.
   148  	 */
   149  	public function getTarget() 
   150  	{
   151  		return _target ;
   152  	}
   153  
   154  	/**
   155  	 * Returns the current time within the duration of the animation.
   156  	 * @return the current time within the duration of the animation.
   157  	 */
   158  	public function getTime():Number 
   159  	{ 
   160  		return _time ;
   161  	}
   162  
   163  	/**
   164  	 * Forwards the tweened animation to the next frame.
   165  	 */
   166  	public function nextFrame():Void 
   167  	{ 
   168  		var t:Number = (useSeconds) ? ((getTimer() - _startTime) / 1000) : (_time + 1) ;
   169  		setTime(t) ;
   170  	}
   171  	
   172  	/**
   173  	 * Directs the tweened animation to the frame previous to the current frame.
   174  	 */
   175  	public function prevFrame():Void 
   176  	{
   177  		if (!useSeconds) setTime (_time - 1)  ;
   178  	}
   179  	
   180  	/**
   181  	 * Resumes a tweened animation from its stopped point in the animation.
   182  	 */
   183  	public function resume():Boolean 
   184  	{
   185  		if (_stopping && _time != duration) 
   186  		{
   187  			_fixTime() ;
   188  			startInterval() ;
   189  			return true ;
   190  			notifyResumed() ;
   191  		}
   192  		else 
   193  		{
   194  			return false ;
   195  		}
   196  	}
   197  	
   198  	/**
   199  	 * Rewinds a tweened animation to the beginning of the tweened animation.
   200  	 */
   201  	public function rewind(t:Number):Void 
   202  	{
   203  		_time = t || 0 ;
   204  		_fixTime() ;
   205  		update() ;
   206  	}
   207  	
   208  	/**
   209  	 * Runs the object.
   210  	 */
   211  	public function run():Void 
   212  	{
   213  		notifyStarted() ;
   214  		rewind() ;
   215  		startInterval() ;
   216  	}
   217  
   218  	/**
   219  	 * Sets the duration of the tweened animation in frames or seconds
   220  	 */
   221  	public function setDuration(n:Number):Void 
   222  	{ 
   223  		_duration = (isNaN(n) || n <= 0 ) ? _global.Infinity : n ;
   224  	}
   225  	
   226  	/**
   227  	 * Sets the number of frames per second of the tweened animation.
   228  	 */
   229  	public function setFPS (n:Number):Void 
   230  	{
   231  		var tmp:Boolean = running ;
   232  		_timer.stop() ;
   233  		_fps = n ;
   234  		if (tmp) _timer.start() ;
   235  	}
   236  	
   237  	/**
   238  	 * Sets the target reference of the object contrains by the Motion effect.
   239  	 */
   240  	public function setTarget(o):Void 
   241  	{
   242  		_target = o ;
   243  	}
   244  	
   245  	/**
   246  	 * Sets the current time within the duration of the animation.
   247  	 */
   248  	public function setTime(t:Number):Void 
   249  	{
   250  		prevTime = _time ;
   251  		if (t > _duration) {
   252  			if (looping) {
   253  				rewind (t - _duration);
   254  				this.update();
   255  				notifyLooped() ;
   256  			} else {
   257  				if (useSeconds) {
   258  					_time = _duration ;
   259  					update() ;
   260  				}
   261  				this.stop() ;
   262  				notifyFinished() ;
   263  			}
   264  		} else if (t<0) {
   265  			rewind() ;
   266  			update() ;
   267  		} else {
   268  			_time = t ;
   269  			update() ;
   270  		}
   271  	}
   272  
   273  	/**
   274  	 * Sets the internal timer of the tweened animation.
   275  	 */
   276  	public function setTimer(timer:ITimer):Void 
   277  	{
   278  		if (_timer) 
   279  		{
   280  			_timer.stop();
   281  			_timer.removeEventListener(TimerEvent.TIMER, _oNext) ;
   282  			_timer = null ;
   283  		}
   284  		_timer = timer ;
   285  		_timer.addEventListener(TimerEvent.TIMER, _oNext ) ;
   286  	}
   287  	
   288  	/**
   289  	 * Starts the tweened animation from the beginning.
   290  	 */
   291  	public function start():Void 
   292  	{
   293  		_stopping = false ; 
   294  		run() ;
   295  	}
   296  	
   297  	/**
   298  	 * Starts the intenral interval of the tweened animation.
   299  	 */
   300  	public function startInterval():Void 
   301  	{
   302  		if (_fps) 
   303  		{
   304  			setTimer(new Timer(1000 / _fps)) ;
   305  		}
   306  		else 
   307  		{
   308  			setTimer(new FrameTimer()) ;
   309  		}
   310  		_timer.start() ; 
   311  		_setRunning(true) ;
   312  	}
   313  	
   314  	/**
   315  	 * Stops the tweened animation at its current position.
   316  	 */
   317  	public function stop():Void 
   318  	{
   319  		stopInterval() ;
   320  		_stopping = true ;
   321  		notifyStopped() ;
   322  	}
   323  
   324  	/**
   325  	 * Stops the intenral interval of the tweened animation.
   326  	 */
   327  	public function stopInterval():Void 
   328  	{
   329  		_timer.stop() ;
   330  		_setRunning(false) ;
   331  	}
   332  	
   333  	/**
   334  	 * Update the current object.
   335  	 */
   336  	public function update():Void 
   337  	{
   338  		//
   339  	}
   340  	
   341  	private var _duration:Number ;
   342  	private var _fps:Number ;
   343  	private var _oNext:EventListener ;
   344  	private var _startTime:Number ;
   345  	private var _stopping:Boolean ;
   346  	private var _target ;
   347  	private var _time:Number ;
   348  	private var _timer:ITimer ;
   349  	
   350  	private function _fixTime():Void 
   351  	{
   352  		if ( useSeconds) _startTime = getTimer() - _time * 1000 ;
   353  	}
   354  	
   355  }
