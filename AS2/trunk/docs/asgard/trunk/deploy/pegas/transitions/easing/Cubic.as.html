    16  	/*override*/ static public function easeIn (t:Number, b:Number, c:Number, d:Number):Number 
    17  	{

    21  	/*override*/ static public function easeOut (t:Number, b:Number, c:Number, d:Number):Number 
    22  	{

    24  	}

    26  	/*override*/ static public function easeInOut (t:Number, b:Number, c:Number, d:Number):Number 
    27  	{
    28  		if ((t/=d/2) < 1) return c/2*t*t*t + b;
    29  		return c/2*((t-=2)*t*t + 2) + b;
    30  	}
