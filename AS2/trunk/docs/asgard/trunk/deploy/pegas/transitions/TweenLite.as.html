    24  import pegas.transitions.Motion;
    25  import pegas.transitions.TweenEntry;
    26  
    27  /**
    28   * The TweenLite class lets you use ActionScript to move, resize, and fade movie clips easily on the Stage by specifying a property of the target movie clip to be tween animated over a number of frames or seconds.
    29   * The TweenLite class also lets you specify a variety of easing methods.
    30   * <p>Easing refers to gradual acceleration or deceleration during an animation, which helps your animations appear more realistic.</p>
    31   * <p><b>Example :</b></p>
    32   * {@code
    33   * import pegas.process.ActionEvent ;
    34   * import pegas.transitions.TweenLite ;
    35   * import pegas.transitions.easing.* ;
    36   * 
    37   * import vegas.events.Delegate ;
    38   * import vegas.events.EventListener ;
    39   * 
    40   * var onDebug:Function = function( ev:ActionEvent ):Void
    41   * {
    42   *     trace (":: debug -> " + ev.type + " : " + ev.target ) ;
    43   * }
    44   * 
    45   * var debug:EventListener = new Delegate(this, onDebug) ;
    46   * 
    47   * var tw:TweenLite = new TweenLite (mc, "_x", Elastic.easeOut, mc._x, 400, 2, true, true) ;
    48   * tw.addGlobalEventListener( debug ) ;
    49   * }
    50   * @author eKameleon
    51   */
    52  class pegas.transitions.TweenLite extends Motion 
    53  {
    54  
    55  	/**
    56  	 * Creates a new TweenLite instance.
    57  	 * <p><b>Usage :</b></p>
    58  	 * {@code
    59  	 * var tw:TweenLite = new TweenLite( obj, prop:String, e:Function, b:Number, f:Number, d:Number , u:Boolean, auto:Boolean) ;
    60  	 * }
    61  	 */
    62  	function TweenLite( obj , prop:String , easing:Function , begin:Number , finish:Number , duration:Number , useSeconds:Boolean , auto:Boolean ) 
    63  	{
    64  		this.target     = obj        ;
    65  		this.duration   = duration   ;
    66  		this.useSeconds = useSeconds ;
    67  		this.setTweenEntry( prop , easing, begin, finish ) ;
    68  		if ( auto == true ) 
    69  		{
    70  			run() ;
    71  		}
    72  	}
    73  
    74  	/**
    75  	 * Returns the TweenEntry reference of this tween.
    76  	 * @return the TweenEntry reference of this tween.
    77  	 */
    78  	public function get tweenEntry():TweenEntry 
    79  	{
    80  		return _entry ;
    81  	}
    82  
    83  	/**
    84  	 * (read-write) Sets the model of this Tween object.
    85  	 */
    86  	public function set tweenEntry( entry:TweenEntry ):Void 
    87  	{
    88  		_entry = entry || new TweenEntry() ;
    89  	}
    90  
    91  	/**
    92  	 * Returns a shallow copy of this Tween object.
    93  	 * @return a shallow copy of this Tween object.
    94  	 */
    95  	public /*override*/ function clone() 
    96  	{
    97  		var t:TweenLite = new TweenLite(target) ;
    98  		t.duration   = duration   ;
    99  		t.useSeconds = useSeconds ;
   100  		t.tweenEntry = tweenEntry.clone() ;
   101  		return t ;
   102  	}
   103  	
   104  	/**
   105  	 * Instructs the tweened animation to continue tweening from its current animation point to a new finish and duration point.
   106  	 * @param finish A number indicating the ending value of the target object property that is to be tweened.
   107  	 * @param duration A number indicating the length of time or number of frames for the tween motion.
   108  	 * @param noRestart This optional flag is used to fix the restart process of the tween.
   109  	 */
   110  	public function continueTo(finish:Number, duration:Number, noRestart:Boolean ):Void 
   111  	{
   112  		_entry.begin  = target[_entry.prop] ;
   113  		_entry.finish = finish ;
   114  		if ( !isNaN(duration) )
   115  		{
   116  			this.duration = duration;
   117  		}
   118  		if ( noRestart == true  )
   119  		{
   120  			return ;
   121  		}
   122  		if ( running == false )
   123  		{
   124  			this.run();
   125  		}
   126  	}
   127  
   128  	/**
   129  	 * Set the TweenEntry property of this TweenLite object.
   130  	 * @param prop the property name of the object to change.
   131  	 * @param easing the easing function of the tween entry.
   132  	 * @param begin the begin value.
   133  	 * @param finish the finish value. 
   134  	 */
   135  	public function setTweenEntry( prop:String, easing:Function , begin:Number , finish:Number ):Void
   136  	{
   137  		if ( _entry == null )
   138  		{
   139  			_entry = new TweenEntry() ;
   140  		}
   141  		_entry.prop   = prop   ;
   142  		_entry.easing = easing ; 	
   143  		_entry.begin  = begin  ;
   144  		_entry.finish = finish ;
   145  	}
   146  
   147  	/**
   148  	 * Update the current Tween in the time.
   149  	 */
   150  	public /*override*/ function update():Void 
   151  	{
   152  		target[_entry.prop] = _entry.setPosition( _entry.getPosition(_time, _duration) ) ;
   153  		updateAfterEvent() ;
   154  		notifyChanged() ;
   155  	}
   156  	
   157  	/**
   158  	 * Instructs the tweened animation to play in reverse from its last direction of tweened property increments. 
   159  	 * If this method is called before a Tween object's animation is complete, the animation abruptly jumps to the end of its play and then plays in a reverse direction from that point.
   160  	 */
   161  	public function yoyo():Void 
   162  	{
   163  		continueTo( _entry.begin, getTime() );
   164  	}
   165  	
   166  	private var _entry:TweenEntry ;
   167  	
   168  }
