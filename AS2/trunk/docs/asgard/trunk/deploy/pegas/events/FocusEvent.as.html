    24  import vegas.events.DynamicEvent;
    25  import vegas.util.serialize.Serializer;
    26  
    27  /**
    28   * The FocusEvent class.
    29   * @author eKameleon
    30   */
    31  class pegas.events.FocusEvent extends DynamicEvent 
    32  {
    33  
    34  	/**
    35  	 * Creates a new FocusEvent instance.
    36  	 */
    37  	public function FocusEvent( type:String, target, p_keyCode:Number, p_relatedObject, p_shiftKey:Boolean, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number )
    38  	{
    39  		super(type, target, context, bubbles, eventPhase, time, stop) ;
    40  		keyCode = isNaN(p_keyCode) ? null : p_keyCode ;
    41  		relatedObject = p_relatedObject || null  ;
    42  		shiftKey = p_shiftKey || null  ;
    43  	}
    44  
    45  	/**
    46  	 * [static] Dispatched after a display object gains focus.
    47  	 */
    48  	static public var FOCUS_IN:String = "focusIn" ;
    49  	
    50  	/**
    51  	 * [static] Dispatched after a display object loses focus.
    52  	 */
    53  	static public var FOCUS_OUT:String = "focusOut" ;
    54  
    55  	/**
    56  	 * [static] Dispatched when the user attempts to change focus using keyboard navigation.
    57  	 */
    58  	static public var KEY_FOCUS_CHANGE:String = "keyFocusChange" ;
    59  	
    60  	/**
    61  	 * [static] Dispatched when the user attempts to change focus by using a pointer device.
    62  	 */
    63  	static public var MOUSE_FOCUS_CHANGE:String = "MouseFocusChange" ;
    64  
    65  	static private var __ASPF__ = _global.ASSetPropFlags(FocusEvent, null , 7, 7) ;
    66  
    67  	/**
    68  	 * The key code value of the key pressed to trigger a keyFocusChange event.
    69  	 */
    70  	public var keyCode:Number = null ;
    71  
    72  	/**
    73  	 * A reference to the complementary InteractiveObject instance that is affected by the change in focus.
    74  	 */	
    75  	public var relatedObject = null ;
    76  	
    77  	/**
    78  	 * Indicates whether the Shift key modifier is activated, in which case the value is true.
    79  	 */
    80  	public var shiftKey:Boolean = null ;
    81  		
    82  	/**
    83  	 * Returns a shallow copy of the object.
    84  	 */
    85  	public function clone() 
    86  	{
    87  		return new FocusEvent(getType(), getTarget()) ;
    88  	}
    89  
    90  	/**
    91  	 * This protected method is used by the toSource method.
    92  	 */
    93  	/*protected*/ private function _getParams():Array {
    94  		var ar:Array = super._getParams() ;
    95  		ar.splice(2, null, Serializer.toSource(keyCode)) ;
    96  		ar.splice(3, null, Serializer.toSource(relatedObject)) ;
    97  		ar.splice(4, null, Serializer.toSource(shiftKey)) ;
    98  		return ar ;
    99  	}
   100  
   101  }
