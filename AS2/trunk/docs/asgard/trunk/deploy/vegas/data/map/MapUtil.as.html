    24  import vegas.data.Map;
    25  
    26  /**
    27   * The {@code MapUtil} utility class is an all-static class with methods for working with map.
    28   * @author eKameleon
    29   */
    30  class vegas.data.map.MapUtil 
    31  {
    32  
    33  	/**
    34  	 * Returns the value of the specified key in the map with a Number representation.
    35  	 * @param map the map used by this method
    36  	 * @param key the key in the map to used to return the number value.
    37  	 * @return the value of the specified key in the map with a Number representation.
    38  	 */
    39  	static public function getNumber(map:Map, key):Number
    40  	{
    41          if (map != null) 
    42          {
    43              var answer = map.get(key) ;
    44              if (answer != null) 
    45              {
    46                  if (answer instanceof Number) 
    47                  {
    48                      return Number(answer) ;
    49                  }
    50                  else if (answer instanceof String) 
    51                  {
    52                      var r:Number = parseInt(answer) ;
    53  					return (r == isNaN) ? null : Number(r) ;
    54                  }
    55              }
    56          }
    57          return null ;
    58      }
    59  
    60  
    61  }
