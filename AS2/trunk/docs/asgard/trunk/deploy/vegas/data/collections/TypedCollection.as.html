    24  import vegas.core.ICloneable;
    25  import vegas.data.Collection;
    26  import vegas.data.iterator.Iterable;
    27  import vegas.data.iterator.Iterator;
    28  import vegas.errors.IllegalArgumentError;
    29  import vegas.util.AbstractTypeable;
    30  import vegas.util.serialize.Serializer;
    31  import vegas.util.TypeUtil;
    32  
    33  /**
    34   * TypedCollection is a wrapper for Collection instances that ensures that only values of a specific type can be added to the wrapped collection.
    35   * @author eKameleon
    36   */
    37  class vegas.data.collections.TypedCollection extends AbstractTypeable implements ICloneable, Iterable, Collection 
    38  {
    39  
    40  	/**
    41  	 * Creates a new TypedCollection.
    42  	 * @throws IllegalArgumentError if the specified collection in argument is {@code null} or {@code undefined} 
    43  	 */
    44  	public function TypedCollection(p_type:Function , co:Collection) 
    45  	{
    46  		super(p_type) ;
    47  		if (!co) 
    48  		{
    49  			throw new IllegalArgumentError("Argument 'co' must not be 'null' or 'undefined'.") ;
    50  		}
    51  		if (co.size() > 0) 
    52  		{
    53  			var it:Iterator = co.iterator() ;
    54  			while ( it.hasNext() ) 	validate(it.next()) ;
    55  		}
    56  		_co = co ;
    57  	}
    58  
    59  	/**
    60  	 * Removes all of the elements from this collection (optional operation).
    61  	 */
    62  	public function clear():Void 
    63  	{
    64  		_co.clear() ;
    65  	}
    66  	
    67  	/**
    68  	 * Returns a shallow copy of this collection.
    69  	 * @return a shallow copy of this collection.
    70  	 */
    71  	public function clone() 
    72  	{
    73  		return new TypedCollection(getType(), _co.clone()) ;
    74  	}
    75  
    76  	/**
    77  	 * Returns {@code true} if this collection contains the specified element.
    78  	 * @return {@code true} if this collection contains the specified element.
    79  	 */
    80  	public function contains(o):Boolean 
    81  	{
    82  		return _co.contains(o) ;
    83      }
    84  
    85  	/**
    86  	 * Returns the element from this collection at the passed index.
    87  	 */
    88  	public function get(id:Number) 
    89  	{
    90  		return _co.get(id) ;
    91  	}
    92  
    93  	/**
    94  	 * Inserts an elements into the Collection.
    95  	 */
    96      public function insert(o):Boolean 
    97      {
    98  		validate(o) ;
    99  		return _co.insert(o) ;
   100      }
   101  
   102  	/**
   103  	 * Returns {@code true} if this collection contains no elements.
   104  	 * @return {@code true} if the collection is empty else {@code false}.
   105  	 */
   106  	public function isEmpty():Boolean 
   107  	{
   108  		return _co.isEmpty() ;
   109  	}
   110  
   111  	/**
   112  	 * Returns an iterator over the elements in this collection.
   113  	 * @return an iterator over the elements in this collection.
   114  	 */
   115  	public function iterator():Iterator 
   116  	{
   117  		return _co.iterator() ;
   118  	}
   119  
   120  	/**
   121  	 * Removes a single instance of the specified element from this collection, if it is present (optional operation).
   122  	 */
   123      public function remove(o):Boolean 
   124      {
   125  		return _co.remove(o);
   126      }
   127  
   128  	/**
   129  	 * Sets the type of the ITypeable object.
   130  	 */
   131  	public function setType(type:Function):Void 
   132  	{
   133  		super.setType(type) ;
   134  		_co.clear() ;
   135  	}
   136  
   137  	/**
   138  	 * Returns the number of elements in this collection.
   139  	 */
   140  	public function size():Number 
   141  	{
   142  		return _co.size() ;
   143  	}
   144  	
   145  	/**
   146  	 * Returns an array containing all of the elements in this collection.
   147  	 * @return an array representation of all the elements in this wrapped collection.
   148  	 */
   149  	public function toArray():Array 
   150  	{
   151  		return _co.toArray() ;
   152  	}
   153  
   154  	/**
   155  	 * Returns the Eden representation of the object.
   156  	 * @return a string representing the source code of the object.
   157  	 */
   158  	public function toSource(indent:Number, indentor:String):String 
   159  	{
   160  		var sourceA:String = TypeUtil.toString(_type) ;
   161  		var sourceB:String = Serializer.toSource(_co) ;
   162  		return Serializer.getSourceOf(this, [sourceA, sourceB]) ;
   163  	}
   164  
   165  	/**
   166  	 * Returns the string representation of this instance.
   167  	 * @return the string representation of this instance.
   168  	 */
   169  	public function toString():String 
   170  	{
   171  		return _co.toString() ;
   172  	}
   173  	
   174  	/**
   175  	 * The internal collection of this wrapped Collection.
   176  	 */
   177  	private var _co:Collection ;
   178  	
   179  }
