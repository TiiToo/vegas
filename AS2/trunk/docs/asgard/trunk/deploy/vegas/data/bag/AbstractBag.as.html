    42  /**
    43   * This class provides a skeletal implementation of the {@code Bag} interface, to minimize the effort required to implement this interface.
    44   * <p>To implement a bag, the programmer needs only to extend this class and provide implementations for the cursor, insert and size methods. For supporting the removal of elements, the cursor returned by the cursor method must additionally implement its remove method.</p>
    45   * @author eKameleon
    46   */
    47  class vegas.data.bag.AbstractBag extends CoreObject implements Bag 
    48  {
    49  
    50  	/**
    51  	 * Creates a new AbstractBag instance.
    52  	 * @param m a Map reference.
    53  	 */
    54  	private function AbstractBag( m:Map ) 
    55  	{
    56  		if (m) 
    57  		{
    58  			_setMap(m) ;
    59  		}
    60  	}

    62  	/**
    63  	 * Removes all of the elements from this bag.
    64  	 */
    65  	public function clear():Void 
    66  	{
    67  		_mods ++ ;
    68  		_map.clear() ;
    69  		_total = 0 ;
    70  	}

    73  	 * Returns the shallow copy of this bag.
    74  	 * @return the shallow copy of this bag.

    82  	 * Returns {@code true} if this bag contains the object passed in argument.
    83  	 * @return {@code true} if this bag contains the object passed in argument.
    84  	 */
    85  	public function contains(o):Boolean 
    86  	{
    87  		return _map.containsKey(o);
    88  	}

    90  	/**
    91  	 * Returns {@code true} if thie bag contains all object in the passed collection in argument.
    92  	 * @return {@code true} if thie bag contains all object in the passed collection in argument.
    93  	 */
    94  	public function containsAll(c:Collection):Boolean 
    95  	{
    96  		return containsAllInBag(new HashBag(c)) ;
    97  	}

    99  	/**
   100  	 * Returns {@code true} if thie bag contains all object in the passed bag in argument.
   101  	 * @return {@code true} if thie bag contains all object in the passed bag in argument.
   102  	 */
   103  	public function containsAllInBag(b:Bag):Boolean 
   104  	{
   105  		var result:Boolean = true ;
   106  		var i:Iterator = b.uniqueSet().iterator() ;
   107          while (i.hasNext()) 
   108          {
   109  			var current = i.next();
   110              var contains:Boolean = getCount(current) >= b.getCount(current) ;
   111              result = result && contains ;
   112          }
   113          return result;
   114  	}

   125  	/**
   126  	 * Returns the count of the specified object passed in argument.
   127  	 */
   128  	public function getCount(o):Number 
   129  	{
   130          var result:Number = 0;
   131          var count:Number = MapUtil.getNumber(_map, o);
   132          if (count != null) 
   133          {
   134              result = count ;
   135          }
   136          return result;
   137  	}

   142  	public function getModCount():Number

   147  	/**
   148  	 * Add 1 copy of the given object to the bag and keep a count. 
   149  	 */	
   150  	public function insert(o):Boolean 
   151  	{
   152  		return insertCopies(o, 1) ;
   153  	}
   154  
   155  	/**
   156  	 * Insert all elements represented in the given collection.
   157  	 */
   158  	public function insertAll(c:Collection):Boolean 
   159  	{
   160  		var changed:Boolean = false;
   161          var i:Iterator = c.iterator() ;
   162  		var added:Boolean ;
   163          while (i.hasNext()) 
   164          {
   165  			added = insert(i.next());
   166              changed = changed || added ;
   167          }
   168          return changed;
   169  	}
   170  
   171  	/**
   172  	 * Add n copies of the given object to the bag and keep a count. 
   173  	 */
   174  	public function insertCopies(o, nCopies:Number):Boolean 
   175  	{
   176  		_mods++ ;
   177          if (nCopies > 0) 
   178          {
   179              var count:Number = nCopies + getCount(o) ;
   180              _map.put(o, new Int(count));
   181              _total += nCopies;
   182              return count == nCopies ;
   183          }
   184          else 
   185          {
   186              return false;
   187          }
   188  	}
   189  	
   190  	/**
   191  	 * Returns {@code true} if the bag is empty.
   192  	 * @return {@code true} if the bag is empty.
   193  	 */
   194  	public function isEmpty():Boolean 
   195  	{
   196  		return _map.isEmpty() ;
   197      }

   199  	/**
   200  	 * Returns the bag iterator.
   201  	 * @return the bag iterator.
   202  	 */
   203  	public function iterator():Iterator 
   204  	{
   205  		return new BagIterator(this, _extractList().iterator()) ;
   206  	}
   207  
   208  	/**
   209  	 * Removes the object in argument in the bag.
   210  	 */
   211  	public function remove(o):Boolean 
   212  	{
   213  		return removeCopies(o, getCount(o));
   214  	}
   215  
   216  	/**
   217  	 * (Violation) Removes all elements represented in the given collection, respecting cardinality.
   218  	 */
   219  	public function removeAll(c:Collection):Boolean 
   220  	{
   221          var result:Boolean = false ;
   222          if (c != null) 
   223          {
   224              var i:Iterator = c.iterator() ;
   225              while (i.hasNext()) 
   226              {
   227  				var next = i.next() ;
   228                  var changed:Boolean = removeCopies(next, 1) ;
   229                  result = result || changed ;
   230              }
   231          }
   232          return result ;
   233      }
   234  
   235  	/**
   236  	 * Removes the given number of occurrences from the bag.
   237  	 */
   238  	public function removeCopies(o, nCopies:Number):Boolean 
   239  	{
   240  		_mods++;
   241          var result:Boolean = false;
   242          var count:Number = getCount(o) ;
   243          if (nCopies <= 0) 
   244          {
   245          	result = false;
   246          }
   247          else if (count > nCopies) 
   248          {
   249              _map.put(o, new Int(count - nCopies)) ;
   250              result = true ;
   251              _total -= nCopies ;
   252          } 
   253          else // count > 0 && count <= i 
   254          {   
   255              result = (_map.remove(o) != null) ; // need to remove all
   256              _total -= count ;
   257          }
   258          return result;
   259  	}
   260  
   261  	/**
   262  	 * (Violation) Removes any members of the bag that are not in the given collection, respecting cardinality.
   263  	 */
   264  	public function retainAll(c:Collection):Boolean 
   265  	{
   266  		return retainAllInBag(new HashBag(c));
   267  	}
   268  
   269  	/**
   270  	 * (Violation) Removes any members of the bag that are not in the given bag.
   271  	 */
   272  	public function retainAllInBag(b:Bag):Boolean 
   273  	{
   274          var result:Boolean = false ;
   275          var excess:Bag = new HashBag() ;
   276          var i:Iterator = uniqueSet().iterator() ;
   277          while (i.hasNext()) 
   278          {
   279              var cur = i.next() ;
   280              var count1:Number = getCount(cur) ;
   281              var count2:Number = b.getCount(cur) ;
   282              if ( 1 <= count2 && count2 <= count1) 
   283              {
   284                  excess.insertCopies(cur, count1 - count2) ;
   285              }
   286              else 
   287              {
   288                  excess.insertCopies(cur, count1) ;
   289              }
   290          }
   291          if (!excess.isEmpty()) 
   292          {
   293          	result = removeAll(excess) ;
   294          }
   295          return result;
   296      }

   346  	private var _map:Map = null ;

   356  	private var _total:Number = 0 ;

   358  	/**
   359  	 * This method calculate the total value. 
   360  	 */
   361  	private function _calcTotalSize():Number 
   362  	{
   363          _total = _extractList().size() ;
   364          return _total ;
   365      }

   367  	/**
   368  	 * Returns a List.
   369  	 */
   370  	private function _extractList():List 
   371  	{
   372          var result:List = new ArrayList() ;
   373          var i:Iterator  = uniqueSet().iterator();
   374          while (i.hasNext()) 
   375          {
   376              var current = i.next() ;
   377  			var l:Number = getCount(current) ;
   378              while (--l > -1) 
   379              {
   380              	result.insert(current);
   381              }
   382          }
   383          return result ;
   384      }

   387  	 * Returns the internal map reference.
   388  	 */
   389  	private function _getMap():Map 
   390  	{
   391  		return _map ;
   392  	}

   394  	/**
   395  	 * Sets the internal map reference.
   396  	 * @throws IllegalArgumentError
   397  	 */
   398  	private function _setMap(m:Map):Void 
   399  	{
   400  		if (m == null || m.isEmpty() == false) 
   401  		{
   402              throw new IllegalArgumentError("In this bag, the map must be non-null and empty.") ;
   403          }
   404          _map = m ;
   405  	}
