    24  import vegas.events.AbstractCoreEventDispatcher;
    25  import vegas.events.BasicEvent;
    26  import vegas.events.Event;
    27  
    28  /**
    29   * This filter contains all constants and methods to sort the Arrays in the application.
    30   * <p><b>Example : </b></p>
    31   * {@code
    32   * import vegas.data.array.ArrayFilter ;
    33   * 
    34   * import vegas.events.Delegate ;
    35   * import vegas.events.Event ;
    36   * 
    37   * var debug:Function = function ():Void
    38   * {
    39   *     trace("filter : " + af.filter ) ;
    40   *     trace("is NONE               : " + af.isNone() )
    41   *     trace("is CASEINSENSITIVE    : " + af.isCaseInsensitive() ) ;
    42   *     trace("is DESCENDING         : " + af.isDescending() ) ;
    43   *     trace("is NUMERIC            : " + af.isNumeric() ) ;
    44   *     trace("is RETURNINDEXEDARRAY : " + af.isReturnIndexedArray() ) ;
    45   *     trace("is UNIQUESORT         : " + af.isUniqueSort() ) ;
    46   *     trace("---") ;
    47   * }
    48   * 
    49   * var change:Function = function ( e:Event ):Void
    50   * {
    51   *     trace( e ) ;
    52   * }
    53   * 
    54   * var af:ArrayFilter = ArrayFilter.getInstance() ;
    55   * af.addEventListener( ArrayFilter.CHANGE, new Delegate(this, change ) ) ;
    56   * 
    57   * debug() ;
    58   * 
    59   * af.setCaseInsensitive( true ) ;
    60   * af.setDescending( true ) ;
    61   * af.setNumeric( true ) ;
    62   * af.setReturnIndexedArray( true ) ;
    63   * af.setUniqueSort( true ) ;
    64   * debug() ;
    65   * 
    66   * af.setCaseInsensitive( false ) ;
    67   * debug() ;
    68   * 
    69   * af.setDescending( false ) ;
    70   * debug() ;
    71   * 
    72   * af.setNumeric( false ) ;
    73   * debug() ;
    74   * 
    75   * af.setReturnIndexedArray( false) ;
    76   * debug() ;
    77   * 
    78   * af.setUniqueSort( false ) ;
    79   * debug() ;
    80   * 
    81   * }
    82   * @author eKameleon
    83   */
    84  class vegas.data.array.ArrayFilter extends AbstractCoreEventDispatcher
    85  {
    86  
    87  	/**
    88  	 * Creates a new ArrayFilter instance.
    89  	 * @param bGlobal the flag to use a global event flow or a local event flow.
    90  	 * @param sChannel the name of the global event flow if the {@code bGlobal} argument is {@code true}.
    91  	 */
    92  	public function ArrayFilter(bGlobal:Boolean , sChannel:String  ) 
    93  	{
    94  		setGlobal( bGlobal , sChannel ) ;
    95  		_eChange = new BasicEvent( CHANGE , this ) ;
    96  		_filter = NONE ;
    97  	}
    98  	
    99     	/**
   100       * Specifies case-insensitive sorting for the Array class sorting methods. You can use this constant
   101  	 * for the <code>options</code> parameter in the <code>sort()</code> or <code>sortOn()</code> method. 
   102  	 * <p>The value of this constant is 1.</p>
   103  	 */
   104  	public static var CASEINSENSITIVE:Number = Array.CASEINSENSITIVE ;
   105  	
   106  	/**
   107  	 * The change event occurs when the filter value is changed.
   108  	 */
   109  	public static var CHANGE:String = "change" ;
   110  	
   111  	/**
   112       * Specifies descending sorting for the Array class sorting methods. 
   113   	 * You can use this constant for the <code>options</code> parameter in the <code>sort()</code>
   114   	 * or <code>sortOn()</code> method. 
   115   	 * <p>The value of this constant is 2.</p>
   116  	 */
   117  	public static var DESCENDING:Number = Array.DESCENDING ;
   118  
   119  	/**
   120  	 * Specifies the default numeric sorting value for the Array class sorting methods.
   121  	 * <p>The value of this constant is 0.</p>
   122  	 */
   123  	public static var NONE:Number = 0;
   124  
   125  	/**
   126       * Specifies numeric (instead of character-string) sorting for the Array class sorting methods. 
   127       * Including this constant in the <code>options</code>
   128   	 * parameter causes the <code>sort()</code> and <code>sortOn()</code> methods 
   129  	 * to sort numbers as numeric values, not as strings of numeric characters.  
   130       * Without the <code>NUMERIC</code> constant, sorting treats each array element as a 
   131   	 * character string and produces the results in Unicode order. 
   132   	 *
   133     	 * <p>For example, given the array of values <code>[2005, 7, 35]</code>, if the <code>NUMERIC</code> 
   134   	 * constant is <strong>not</strong> included in the <code>options</code> parameter, the 
   135     	 * sorted array is <code>[2005, 35, 7]</code>, but if the <code>NUMERIC</code> constant <strong>is</strong> included, 
   136     	 * the sorted array is <code>[7, 35, 2005]</code>. </p>
   137   	 * 
   138   	 * <p>This constant applies only to numbers in the array; it does 
   139     	 * not apply to strings that contain numeric data such as <code>["23", "5"]</code>.</p>
   140   	 * 
   141   	 * <p>The value of this constant is 16.</p>
   142  	 */
   143  	public static var NUMERIC:Number = Array.NUMERIC ;
   144  		
   145  	/**
   146   	 * Specifies that a sort returns an array that consists of array indices as a result of calling
   147   	 * the <code>sort()</code> or <code>sortOn()</code> method. You can use this constant
   148   	 * for the <code>options</code> parameter in the <code>sort()</code> or <code>sortOn()</code> 
   149   	 * method, so you have access to multiple views on the array elements while the original array is unmodified. 
   150   	 * <p>The value of this constant is 8.</p>
   151   	 */
   152  	public static var RETURNINDEXEDARRAY:Number = Array.RETURNINDEXEDARRAY ;
   153  
   154  	/**
   155     	 * Specifies the unique sorting requirement for the Array class sorting methods. 
   156       * You can use this constant for the <code>options</code> parameter in the <code>sort()</code> or <code>sortOn()</code> method. 
   157       * The unique sorting option terminates the sort if any two elements or fields being sorted have identical values. 
   158  	 * <p>The value of this constant is 4.</p>
   159  	 */
   160  	public static var UNIQUESORT:Number = Array.UNIQUESORT ;
   161  
   162  	/**
   163  	 * Returns the current filter value of this object.
   164  	 * @return the current filter value of this object.
   165  	 */
   166  	public function get filter():Number 
   167  	{
   168  		return getFilter() ;
   169  	}
   170  	
   171  	/**
   172  	 * Sets the current filter value of this object.
   173  	 */
   174  	public function set filter(n:Number):Void
   175  	{
   176  		setFilter(n) ;	
   177  	}
   178  
   179  	/**
   180  	 * Indicates if the change event is notify or not when the filter value change.
   181  	 */
   182  	public var useEvent:Boolean = true ;
   183  	
   184  	/**
   185  	 * Returns {@code true} if the filter number value contains the option number value.
   186  	 * @return {@code true} if the filter number value contains the option number value.
   187  	 */
   188  	static public function contains( nFilter:Number, nOption:Number ) : Boolean
   189  	{
   190  		return Boolean(nOption & nFilter) ;
   191  	}
   192  	
   193  	/**
   194  	 * Returns the event name of the change event of this object.
   195  	 * @return the event name of the change event of this object.
   196  	 */
   197  	public function getEventTypeCHANGE():String
   198  	{
   199  		return _eChange.getType() ;
   200  	}
   201  	
   202  	/**
   203  	 * Returns the singleton reference of this class.
   204  	 * @return the singleton reference of this class.
   205  	 */
   206  	static public function getInstance():ArrayFilter
   207  	{
   208  		if (_instance == null)
   209  		{
   210  			_instance = new ArrayFilter() ;	
   211  		}	
   212  		return _instance ;
   213  	}
   214  	
   215  	/**
   216  	 * Returns the current filter value of this object.
   217  	 * @return the current filter value of this object.
   218  	 */
   219  	public function getFilter():Number
   220  	{
   221  		return isNaN(_filter) ? NONE : _filter ;
   222  	}
   223  
   224  	/**
   225  	 * Returns {@code true} if the CASEINSENSITIVE option value exist in the current filter.
   226  	 * @return {@code true} if the CASEINSENSITIVE option value exist in the current filter.
   227  	 */
   228  	public function isCaseInsensitive():Boolean
   229  	{
   230  		return contains( getFilter() , CASEINSENSITIVE ) ;	
   231  	}
   232  	
   233  	/**
   234  	 * Returns {@code true} if the DESCENDING option value exist in the current filter.
   235  	 * @return {@code true} if the DESCENDING option value exist in the current filter.
   236  	 */
   237  	public function isDescending():Boolean
   238  	{
   239  		return contains( getFilter() , DESCENDING ) ;	
   240  	}
   241  	
   242  	/**
   243  	 * Returns {@code true} if the filter is NONE.
   244  	 * @return {@code true} if the filter is NONE.
   245  	 */
   246  	public function isNone():Boolean
   247  	{
   248  		return getFilter() == NONE ;
   249  	}
   250  
   251  	/**
   252  	 * Returns {@code true} if the NUMERIC option value exist in the current filter.
   253  	 * @return {@code true} if the NUMERIC option value exist in the current filter.
   254  	 */
   255  	public function isNumeric():Boolean
   256  	{
   257  		return contains( getFilter() , NUMERIC ) ;	
   258  	}
   259  	
   260  	/**
   261  	 * Returns {@code true} if the RETURNINDEXEDARRAY option value exist in the current filter.
   262  	 * @return {@code true} if the RETURNINDEXEDARRAY option value exist in the current filter.
   263  	 */
   264  	public function isReturnIndexedArray():Boolean
   265  	{
   266  		return contains( getFilter() , RETURNINDEXEDARRAY ) ;	
   267  	}
   268  	
   269  	/**
   270  	 * Returns {@code true} if the UNIQUESORT option value exist in the current filter.
   271  	 * @return {@code true} if the UNIQUESORT option value exist in the current filter.
   272  	 */
   273  	public function isUniqueSort():Boolean
   274  	{
   275  		return contains( getFilter() , UNIQUESORT ) ;	
   276  	}
   277  	
   278  	/**
   279  	 * Notify a change in this object.
   280  	 */
   281  	public function notifyChange():Void
   282  	{
   283  		if ( useEvent == true )
   284  		{
   285  			dispatchEvent( _eChange ) ;
   286  		}	
   287  	}
   288  	
   289  	/**
   290  	 * Sets the CASEINSENSITIVE option value in the current filter.
   291  	 */
   292  	public function setCaseInsensitive(b:Boolean):Void
   293  	{
   294  		var old:Number = _filter ;
   295  		_filter = b ? _filter | CASEINSENSITIVE : _filter & ~CASEINSENSITIVE ;
   296  		if (_filter != old)
   297  		{
   298  			notifyChange() ;
   299  		}	
   300  	}
   301  
   302  	/**
   303  	 * Sets the DESCENDING option value in the current filter.
   304  	 */
   305  	public function setDescending( b:Boolean ):Void
   306  	{
   307  		var old:Number = _filter ;
   308  		_filter = b ? _filter | DESCENDING : _filter & ~DESCENDING ;
   309  		if (_filter != old)
   310  		{
   311  			notifyChange() ;
   312  		}
   313  	}
   314  
   315  	/**
   316  	 * Sets the event name of the change event of this object.
   317  	 */
   318  	public function setEventTypeCHANGE( type:String ):Void
   319  	{
   320  		_eChange.setType( type || CHANGE ) ;
   321  	}
   322  
   323  	/**
   324  	 * Sets the current filter value of this object.
   325  	 */
   326  	public function setFilter( n:Number ):Void
   327  	{
   328  		var old:Number = _filter ;
   329  		_filter = isNaN(n) ? NONE : n ;
   330  		if (_filter != old)
   331  		{
   332  			notifyChange() ;
   333  		}	
   334  	}
   335  
   336  	/**
   337  	 * Sets the NONE option value in the current filter.
   338  	 */
   339  	public function setNone():Void
   340  	{
   341  		var old:Number = _filter ;
   342  		_filter = NONE ;
   343  		if (_filter != old)
   344  		{
   345  			notifyChange() ;
   346  		}	
   347  	}
   348  
   349  	/**
   350  	 * Sets the NUMERIC option value in the current filter.
   351  	 */
   352  	public function setNumeric( b:Boolean ):Void
   353  	{
   354  		var old:Number = _filter ;
   355  		_filter = b ? _filter | NUMERIC : _filter & ~NUMERIC ;
   356  		if (_filter != old)
   357  		{
   358  			notifyChange() ;
   359  		}	
   360  	}
   361  
   362  	/**
   363  	 * Sets the RETURNINDEXEDARRAY option value in the current filter.
   364  	 */
   365  	public function setReturnIndexedArray( b:Boolean ):Void
   366  	{
   367  		var old:Number = _filter ;
   368  		_filter = b ? _filter | RETURNINDEXEDARRAY : _filter & ~RETURNINDEXEDARRAY ;
   369  		if (_filter != old)
   370  		{
   371  			notifyChange() ;
   372  		}	
   373  	}
   374  	
   375  	/**
   376  	 * Sets the UNIQUESORT option value in the current filter.
   377  	 */
   378  	public function setUniqueSort( b:Boolean ):Void
   379  	{
   380  		var old:Number = _filter ;
   381  		_filter = b ? _filter | UNIQUESORT : _filter & ~UNIQUESORT ;
   382  		if (_filter != old)
   383  		{
   384  			notifyChange() ;
   385  		}	
   386  	}
   387  
   388  	/**
   389  	 * The internal singleton reference.
   390  	 */
   391  	static private var _instance:ArrayFilter ;
   392  	
   393  	/**
   394  	 * The internal change event reference.
   395  	 */
   396  	private var _eChange:Event ;
   397  	
   398  	/**
   399  	 * The filter value of this object.
   400  	 */
   401  	private var _filter:Number ;
   402  	
   403  }
