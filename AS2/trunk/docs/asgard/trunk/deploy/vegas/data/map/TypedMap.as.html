    24  import vegas.core.ICloneable;
    25  import vegas.core.IFormattable;
    26  import vegas.data.iterator.Iterator;
    27  import vegas.data.Map;
    28  import vegas.errors.IllegalArgumentError;
    29  import vegas.util.AbstractTypeable;
    30  import vegas.util.serialize.Serializer;
    31  import vegas.util.TypeUtil;

    33  /**
    34   * TypedMap is a wrapper for Map instances that ensures that only values of a specific type can be added to the wrapped Map.
    35   * @author eKameleon
    36   */
    37  class vegas.data.map.TypedMap extends AbstractTypeable implements ICloneable, Map, IFormattable 
    38  {
    39  
    40  	/**
    41  	 * Creates a new TypedMap instance.
    42  	 * @param fType the type of all values in this TypedMap.
    43  	 * @param m the map to wrapp.
    44  	 * @throws IllegalArgumentError if the specified map in argument is {@code null} or {@code undefined}.
    45  	 */
    46  	public function TypedMap( fType:Function , m:Map) 
    47  	{
    48  		
    49  		super( fType ) ;
    50  		if ( m == null) 
    51  		{
    52  			throw new IllegalArgumentError("TypedMap constructor failed, Argument 'map' must not be 'null' or 'undefined'.") ;
    53  		}
    54  		if ( m.size() > 0) 
    55  		{
    56  			var it = m.iterator() ;
    57  			while ( it.hasNext() ) 
    58  			{
    59  				validate(it.next()) ;
    60  			}
    61  		}
    62  		_map = m ;
    63  		trace(_map) ;
    64  	}
    65  
    66  	/**
    67  	 * Removes all mappings from this map (optional operation).
    68  	 */
    69  	public function clear():Void 
    70  	{
    71  		_map.clear() ;
    72  	}
    73  
    74  	/**
    75  	 * Returns a shallow copy of the map.
    76  	 * @return a shallow copy of the map.
    77  	 */
    78  	public function clone() 
    79  	{
    80  		return new TypedMap(_type, _map) ;
    81  	}
    82  
    83  	/**
    84  	 * Returns {@code true} if this map contains a mapping for the specified key.
    85  	 * @return {@code true} if this map contains a mapping for the specified key.
    86  	 */
    87  	public function containsKey(o):Boolean 
    88  	{
    89  		return _map.containsKey(o) ;
    90      }
    91  
    92  	/**
    93  	 * Returns {@code true} if this map maps one or more keys to the specified value.
    94  	 * @return {@code true} if this map maps one or more keys to the specified value.
    95  	 */
    96  	public function containsValue(o):Boolean 
    97  	{
    98  		return _map.containsValue(o) ;
    99      }
   100  
   101  	/**
   102  	 * Returns the value to which this map maps the specified key.
   103  	 * @return the value to which this map maps the specified key.
   104  	 */
   105  	public function get(key) 
   106  	{
   107  		return _map.get(key) ;
   108  	}
   109  
   110  	/**
   111  	 * Returns an array of all the keys in the map.
   112  	 * @return an array of all the keys in the map.
   113  	 */
   114  	public function getKeys():Array 
   115  	{
   116  		return _map.getKeys() ;
   117  	}
   118  
   119  	/**
   120  	 * Returns an array of all the values in the map.
   121  	 * @return an array of all the values in the map.
   122  	 */
   123  	public function getValues():Array 
   124  	{
   125  		return _map.getValues() ;
   126  	}
   127  
   128  	/**
   129  	 * Returns {@code true} if this map contains no key-value mappings.
   130  	 * @return {@code true} if this map contains no key-value mappings.
   131  	 */
   132  	public function isEmpty():Boolean 
   133  	{
   134  		return _map.isEmpty() ;
   135  	}
   136  
   137  	/**
   138  	 * Returns the values iterator of this map.
   139  	 * @return the values iterator of this map.
   140  	 */
   141  	public function iterator():Iterator 
   142  	{
   143  		return _map.iterator() ;
   144  	}
   145  
   146  	/**
   147  	 * Returns the keys iterator of this map.
   148  	 * @return the keys iterator of this map.
   149  	 */
   150  	public function keyIterator():Iterator 
   151  	{
   152  		return _map.keyIterator() ;
   153  	}
   154  
   155  	/**
   156  	 * Associates the specified value with the specified key in this map (optional operation).
   157  	 */
   158  	public function put(key, value) 
   159  	{
   160  		validate(value) ;
   161  		return _map.put(key, value) ;
   162  	}
   163  
   164  	/**
   165  	 * Copies all of the mappings from the specified map to this map (optional operation).
   166  	 */
   167  	public function putAll(m:Map):Void 
   168  	{
   169  		var it:Iterator = m.iterator() ;
   170  		while(it.hasNext()) {
   171  			validate(it.next()) ;
   172  		}
   173  		_map.putAll(m) ;
   174  	}
   175  
   176  	/**
   177  	 * Removes the mapping for this key from this map if it is present (optional operation).
   178  	 */
   179      public function remove(key) 
   180      {
   181  		return _map.remove(key) ;
   182      }
   183  
   184  	/**
   185  	 * Sets the type of the ITypeable object.
   186  	 */
   187  	/*override*/ public function setType(type:Function):Void 
   188  	{
   189  		super.setType(type) ;
   190  		_map.clear() ;
   191  	}
   192  
   193  	/**
   194  	 * Returns the number of key-value mappings in this map.
   195  	 * @return the number of key-value mappings in this map.
   196  	 */
   197  	public function size():Number 
   198  	{
   199  		return _map.size() ;
   200  	}
   201  
   202  	/**
   203  	 * Returns a Eden representation of the object.
   204  	 * @return a string representing the source code of the object.
   205  	 */
   206  	public function toSource(indent:Number, indentor:String):String 
   207  	{
   208  		var sourceA:String = TypeUtil.toString(_type) ;
   209  		var sourceB:String = Serializer.toSource(_map) ;
   210  		return Serializer.getSourceOf(this, [sourceA, sourceB]) ;
   211  	}
   212  
   213  	/**
   214  	 * Returns the string representation of this instance.
   215  	 * @return the string representation of this instance
   216  	 */
   217  	public function toString():String 
   218  	{
   219  		return _map.toString() ;
   220  	}
   221  
   222  	private var _map:Map ;
   223  	
   224  }
