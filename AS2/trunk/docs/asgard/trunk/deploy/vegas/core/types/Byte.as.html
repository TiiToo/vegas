    24  import vegas.core.types.Bit;
    25  import vegas.util.ConstructorUtil;
    26  
    27  /**
    28   * {@code Byte} is represents a byte value.
    29   * <p>{@code Byte} can be used for a different kind of formatting of a byte value.
    30   * It allows to access the value as bit, kilo-bit, mega-bit, giga-bit, tera-bit, byte, kilo-byte, mega-byte, giga-byte and tera-byte.
    31   * </p>
    32   * <p><b>Examples:</b></p>
    33   * {@code 
    34   *   import vegas.core.types.Byte ;
    35   *   trace(new Byte(1).toString()) ; // 1B
    36   *   trace(new Byte(1234).toString()) ; // 1.21KB
    37   *   trace(new Byte(15002344).toString()) ; // 14.31MB
    38   * }
    39   * @author eKameleon
    40   */
    41  class vegas.core.types.Byte extends Bit
    42  {
    43  
    44  	/**
    45  	 * Creates a new Byte instance.
    46  	 * @param bytes value in byte
    47  	 */
    48  	public function Byte( bytes:Number ) 
    49  	{
    50  		super( bytes * BYTE );
    51  	}
    52  
    53  	/**
    54  	 * Shortname of byte.
    55  	 */
    56  	public static var SHORT_BYTE:String = "B" ;
    57  
    58  	/**
    59  	 * Shortname of kilo-byte.
    60  	 */
    61  	public static var SHORT_KILO_BYTE:String = "KB";
    62  
    63  	/**
    64  	 * Shortname of mega-byte.
    65  	 */
    66  	public static var SHORT_MEGA_BYTE:String = "MB";
    67  
    68  	/**
    69  	 * Shortname of giga-byte.
    70  	 */
    71  	public static var SHORT_GIGA_BYTE:String = "GB";
    72  
    73  	/**
    74  	 * Shortname of tera-byte.
    75  	 */
    76  	public static var SHORT_TERA_BYTE:String = "TB";	
    77  
    78  	/**
    79  	 * Returns a Eden reprensation of the object.
    80  	 * @return a string representing the source code of the object.
    81  	 */
    82  	public function toSource(indent:Number, indentor:String):String
    83  	{
    84  		return "new " + ConstructorUtil.getPath(this) + "(" + (_bit  / BYTE) + "," + _comma  + ")" ;
    85  	}
    86  
    87  	/**
    88  	 * Returns a string representation of the object.
    89  	 */
    90  	public function toString(indent:Number, indentor:String):String 
    91  	{
    92  		if (_bit < Bit.KBYTE) 
    93  		{
    94  			return getBytes() + Byte.SHORT_BYTE ;
    95  		}
    96  		else if(_bit < Bit.MBYTE) 
    97  		{
    98  			return getKBytes() + Byte.SHORT_KILO_BYTE ;
    99  		}
   100  		else if(_bit < Bit.GBYTE) 
   101  		{
   102  			return getMegaBytes() + Byte.SHORT_MEGA_BYTE ;
   103  		}
   104  		else if(_bit < Bit.TBYTE) 
   105  		{
   106  			return getGigaBytes() + Byte.SHORT_GIGA_BYTE ;
   107  		}
   108  		else 
   109  		{
   110  			return getTeraBytes() + Byte.SHORT_TERA_BYTE ;
   111  		}
   112  	}
   113  	
   114  }
