    24  import vegas.data.map.HashMap;
    25  import vegas.errors.ArgumentsError;
    26  import vegas.errors.Warning;
    27  import vegas.logging.errors.InvalidCategoryError;
    28  import vegas.logging.ILogger;
    29  import vegas.logging.ITarget;
    30  import vegas.logging.LogLogger;
    31  import vegas.util.StringUtil;
    32  
    33  /**
    34   * Provides pseudo-hierarchical logging capabilities with multiple format and output options.
    35   * @author eKameleon
    36   */
    37  class vegas.logging.Log 
    38  {
    39  
    40  	/**
    41  	 * const The default categoty of the {@code ILogger} instances returns with the {@code getLogger} method.
    42  	 */
    43  	static public var DEFAULT_CATEGORY:String = "" ;
    44  	
    45  	/**
    46  	 * const The string representation of all the illegal characters.
    47  	 */
    48  	static public var ILLEGALCHARACTERS:String = "[]~$^&/\\(){}<>+=`!#%?,:;'\"@" ;
    49  
    50  	/**
    51  	 * The static field used when throws an Error when a character is invalid.
    52  	 */     
    53  	static public var INVALID_CHARS:String = "Categories can not contain any of the following characters : []~$^&/\\(){}<>+=`!#%?,:;'\"@" ;
    54        
    55  	/**
    56  	 * The static field used when throws an Error when the length of one character is invalid.
    57  	 */     
    58      static public var INVALID_LENGTH:String = "Categories must be at least one character in length." ;
    59  
    60      /**
    61  	 * The static field used when throws an Error when the specified target is invalid.
    62  	 */     
    63      static public var INVALID_TARGET:String = "Log, Invalid target specified." ;
    64  
    65  	/**
    66  	 * The static field used when throws a Warning when the specified target is invalid.
    67  	 */
    68  	static public var REMOVE_TARGET_FAILED:String = "The target is invalid, isn't register for the moment, you can't remove it" ;
    69  
    70  	/**
    71  	 * Allows the specified target to begin receiving notification of log events.
    72  	 * @param target specific target that should capture log events.
    73  	 */
    74  	static public function addTarget( target:ITarget ):Void 
    75  	{
    76  		if (target != null)
    77  		{
    78  			target.addLogger( _logger ) ;
    79  		}
    80  		else
    81  		{
    82  			throw new ArgumentsError( INVALID_TARGET );
    83  		}
    84  	}
    85  
    86  	/**
    87  	 * This method removes all of the current loggers from the cache.
    88  	 */
    89  	static public function flush():Void 
    90  	{
    91  		_categories.clear() ;
    92  	}
    93  	
    94  	/**
    95  	 * Returns the logger associated with the specified category.
    96  	 * If the category given doesn't exist a new instance of a logger will be returned and associated with that category. Categories must be at least one character in length and may not contain any blanks or any of the following characters: []~$^&\/(){}<>+=`!#%?,:;'"@
    97  	 * @param The String to check for illegal characters. The following characters are not valid: []~$^&\/(){}<>+=`!#%?,:;'"@ 
    98  	 * @return An instance of a logger object for the specified name. If the name doesn't exist, a new instance with the specified name is returned.
    99  	 * @throws InvalidCategoryError if the category specified is malformed.
   100  	 */
   101  	static public function getLogger(category:String, isQueue:Boolean):ILogger 
   102  	{
   103  		
   104  		checkCategory( category ) ;
   105  		
   106  		if( !_categories.containsKey(category) )
   107  		{
   108  			var logger:LogLogger = new LogLogger(category) ;
   109  			logger.isQueue = isQueue ;
   110  			logger.parent = _logger ; // bubbling event.
   111  			_categories.put(category, logger) ;
   112  		}
   113  		return _categories.get(category) ;
   114  	}
   115  
   116  	/**
   117  	 * This method checks the specified string value for illegal characters.
   118  	 * @param value The String to check for illegal characters. The following characters are not valid: []~$^&\/(){}<>+=`!#%?,:;'"@
   119  	 * @return {@code true} if there are any illegal characters found, false otherwise.
   120  	 */
   121  	static public function hasIllegalCharacters(value:String):Boolean 
   122  	{
   123  		return StringUtil.indexOfAny( value, ILLEGALCHARACTERS.split("") ) != -1 ;
   124  	}
   125  
   126  	/**
   127  	 * Stops the specified target from receiving notification of log events.
   128  	 * @param specific target that should capture log events.
   129  	 */
   130  	static public function removeTarget(target:ITarget):Void 
   131  	{
   132  		if( target != null )
   133  		{
   134  			target.removeLogger( _logger ) ;
   135  		}
   136  		else
   137  		{
   138  			throw new Warning( REMOVE_TARGET_FAILED ) ;	
   139  		}
   140  	}
   141  	
   142  	/**
   143  	 * Returns the string representation of this instance.
   144  	 * @return the string representation of this instance
   145  	 */
   146  	static public function toString():String 
   147  	{
   148  		return "[Log]" ;
   149  	}
   150  
   151  	/**
   152  	 * The internal categories HashMap.
   153  	 */
   154  	static private var _categories:HashMap = new HashMap() ;
   155  
   156  	/**
   157  	 * The internal logger.
   158  	 */
   159  	static private var _logger:LogLogger = new LogLogger() ;
   160  
   161  	/**
   162  	 * Sentinal value for the target log level to indicate no logging.
   163  	 */
   164  	static private var NONE:Number = Number.MAX_VALUE;
   165          
   166  	/**
   167  	 * The most verbose supported log level among registered targets.
   168  	 */
   169      static private var _targetLevel:Number = Number.MAX_VALUE ;
   170  
   171      /**
   172  	 * This method will ensure that a valid category string has been specified.
   173  	 * If the category is not valid an <code>InvalidCategoryError</code> will be thrown.
   174  	 * Categories can not contain any blanks or any of the following characters: []`*~,!#$%^&amp;()]{}+=\|'";?&gt;&lt;./&#64; 
   175  	 * or be less than 1 character in length.
   176  	 */
   177      static private function checkCategory(category:String):Void
   178  	{
   179              
   180  		if(category == null || category.length == 0)
   181  		{
   182  			throw new InvalidCategoryError( INVALID_LENGTH ) ;
   183  		}
   184  		
   185  		if( hasIllegalCharacters(category) || (category.indexOf("*") != -1))
   186          {
   187          	throw new InvalidCategoryError( INVALID_CHARS ) ;
   188  		}
   189              
   190  	}
   191  
   192  
   193  }
