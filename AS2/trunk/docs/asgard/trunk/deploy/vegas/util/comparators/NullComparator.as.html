    24  import vegas.core.CoreObject;
    25  import vegas.core.IComparator;
    26  import vegas.core.ISerializable;
    27  import vegas.util.comparators.ComparableComparator;
    28  import vegas.util.serialize.Serializer;
    29  
    30  /**
    31   * This comparator compare Null objects.
    32   * When comparing two non-null objects, the ComparableComparator is used if the nonNullComparator isnt' define.
    33   * <p><b>Example :</b></p>
    34   * {@code
    35   * import vegas.util.comparators.NullComparator;
    36   * 
    37   * var comp1:NullComparator = new NullComparator(null, true) ;
    38   * var comp2:NullComparator = new NullComparator(null, false) ;
    39   * 
    40   * var n = null ;
    41   * var o = {} ;
    42   * 
    43   * trace( comp1.compare(n, n) ) ; // 0
    44   * trace( comp1.compare(n, o) ) ; // 1
    45   * trace( comp1.compare(o, n) ) ; // -1
    46   * 
    47   * trace("----") ;
    48   * 
    49   * trace( comp2.compare(n, n) ) ; // 0
    50   * trace( comp2.compare(n, o) ) ; // -1
    51   * trace( comp2.compare(o, n) ) ; // 1
    52   * }
    53   * @author eKameleon
    54   */
    55  class vegas.util.comparators.NullComparator extends CoreObject implements IComparator
    56  {
    57  
    58  	/**
    59  	 * Creates a new NullComparator instance.
    60  	 * @param nonNullComparator the comparator to use when comparing two non-null objects.
    61  	 * @param nullsAreHigh a {@code true} value indicates that null should be compared as higher than a non-null object. A {@code false} value indicates that null should be compared as lower than a non-null object. 
    62  	 */
    63  	public function NullComparator( nonNullComparator:IComparator, nullsAreHigh:Boolean ) 
    64  	{
    65  		this.nonNullComparator = nonNullComparator || null ;
    66  		this.nullsAreHigh = nullsAreHigh || false  ;
    67  	}
    68  	
    69  	/**
    70  	 * Defines the comparator to use when comparing two non-null objects.
    71  	 */
    72  	public var nonNullComparator:IComparator = null ;
    73  
    74  	/**
    75  	 * Defines that null should be compared as higher than a non-null object. 
    76  	 */
    77  	public var nullsAreHigh:Boolean = false ;
    78  	
    79  	/**
    80  	 * Perform a comparison between two objects. 
    81  	 * If both objects are null, a 0 value is returned. 
    82  	 * If one object is null and the other is not, the result is determined on whether the Comparator was constructed to have nulls as higher or lower than other objects.
    83  	 * If neither object is null, an underlying comparator specified in the constructor (or the default) is used to compare the non-null objects.
    84  	 * The default IComparator used to compare two non-null objects is the ComparableComparator.
    85  	 * @param o1 the first 'null' object to compare.
    86  	 * @param o2 the second 'null' object to compare.
    87  	 * @return <p>
    88  	 * <li>-1 if o1 is "lower" than (less than, before, etc.) o2 ;</li>
    89  	 * <li> 1 if o1 is "higher" than (greater than, after, etc.) o2 ;</li>
    90  	 * <li> 0 if o1 and o2 are equal.</li>
    91  	 * </p>
    92  	 * @see ComparableComparator
    93  	 */
    94  	public function compare(o1, o2):Number 
    95  	{
    96  		if (o1 == null && o2 == null)
    97  		{
    98  			return 0 ;
    99  		}			
   100  		else if (o1 == null && o2 != null)
   101  		{
   102  			return nullsAreHigh ? 1 : -1 ;
   103  		}
   104  		else if (o1 != null && o2 == null)
   105  		{
   106  			return nullsAreHigh ? -1 : 1 ;
   107  		}
   108  		else
   109  		{
   110  			if ( nonNullComparator != null )
   111  			{
   112  				return nonNullComparator.compare(o1, o2) ;	
   113  			}
   114  			else
   115  			{
   116  				return ComparableComparator.getInstance().compare(o1, o2) ;
   117  			}
   118  		}
   119  	}
   120  
   121  	/**
   122  	 * Returns the singleton instance of this class.
   123  	 * Developers are encouraged to use the comparator returned from this method instead of constructing a new instance to reduce allocation and GC overhead when multiple comparable comparators may be used in the same application.
   124  	 * @return the singleton instance of this class.
   125  	 */
   126  	static public function getInstance():NullComparator
   127  	{
   128  		if (_instance == null)
   129  		{
   130  			_instance = new NullComparator() ;
   131  		}
   132  		return _instance ;	
   133  	}
   134  
   135  	/**
   136  	 * The internal static singleton of this class.
   137  	 */
   138  	static private var _instance:NullComparator ;
   139  	
   140  	
   141  	/**
   142  	 * Returns a Eden reprensation of the object.
   143  	 * @return a string representing the source code of the object.
   144  	 */
   145  	public function toSource(indent : Number, indentor : String):String 
   146  	{
   147  		var sources:Array = [
   148  			ISerializable(nonNullComparator).toSource() || "null" , 
   149  			Serializer.toSource( nullsAreHigh )
   150  		] ;
   151  		return Serializer.getSourceOf(this, sources ) ;
   152  	}
   153  	
   154  }
