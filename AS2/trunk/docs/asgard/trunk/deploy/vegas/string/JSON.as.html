    60  /**
    61   * JSON (JavaScript object Notation) is a lightweight data-interchange format.
    62   * <p>Serializer & deserializer in AS2.</p>
    63   * <p>More information in the official site : <a href="http://www.JSON.org/">http://www.JSON.org</a></p>
    64   * <p>Add Hexa Digits tool in deserialize method - <a href="http://www.burrrn.com/projects/eden.html">EDEN inspiration</a></p>
    65   * <p><b>Example :</b></p>
    66   * {@code
    67   * import vegas.string.JSON ;
    68   *   
    69   * // --- Init
    70   * 
    71   * var a:Array = [2, true, "hello"] ;
    72   * var o:Object = { prop1 : 1 , prop2 : 2 } ;
    73   * var s:String = "hello world" ;
    74   * var n:Number = 4 ;
    75   * var b:Boolean = true ;
    76   * 
    77   * trace("-> Serialize") ;
    78   * trace("- a : " + JSON.serialize( a ) )  ;
    79   * trace("- o : " + JSON.serialize( o ) )  ;
    80   * trace("- s : " + JSON.serialize( s ) )  ;
    81   * trace("- n : " + JSON.serialize( n ) )  ;
    82   * trace("- b : " + JSON.serialize( b ) )  ;
    83   * 
    84   * trace (" Deserialize") ;
    85   *  
    86   * var source:String = '[ {"prop1":0xFF0000, prop2:2, prop3:"hello", prop4:true} , 2, true, * 3, [3, 2] ]' ;
    87   *  
    88   * var o = JSON.deserialize(source) ;
    89   * for (var prop:String in o) 
    90   * {
    91   *     trace(prop + " : " + o[prop] + " -> typeof :: " + typeof(o[prop])) ;
    92   * }
    93   *  
    94   * trace ("*** JSONError") ;
    95   * 
    96   * var source:String = "[3, 2," ; // test1
    97   * // var source:String = '{"prop1":coucou"}' ; // test2
    98   * var o = JSON.deserialize(source) ;
    99   * for (var prop:String in o) 
   100   * {
   101   *     trace(prop + " : " + o[prop]) ;
   102   * }
   103   *  // see output.
   104   * }
   105   * @author eKameleon
   106   */
   107  class vegas.string.JSON 
   108  {
   109  
   110  	/**
   111  	 * Deserialize the string source representation and return the result object.
   112  	 */
   113  	static public function deserialize(source:String)
   114  	{
   115  
   116  		source = new String(source) ; // speed
   117  		var at:Number = 0 ;
   118          var ch:String = " ";
   119  		
   120  		var _isDigit:Function ;
   121  		var _isHexDigit:Function ;
   122  		var _white:Function ;
   123  		var _string:Function ;
   124  		var _next:Function ;
   125  		var _array:Function ;
   126  		var _key:Function ;
   127  		var _object:Function ;
   128  		var _number:Function ;
   129  		var _word:Function ;
   130  		var _value:Function ;
   131  		var _error:Function ;
   132  		
   133  		_isDigit = function( c:String )
   134  		{
   135      		return( ("0" <= c) && (c <= "9") );
   136      	} ;
   137  			
   138  		_isHexDigit = function( c:String ) 
   139  		{
   140      		return( _isDigit( c ) || (("A" <= c) && (c <= "F")) || (("a" <= c) && (c <= "f")) );
   141      	} ;
   142  				
   143          _error = function(m:String) 
   144          {
   145              throw new JSONError( m, at - 1 , source) ;
   146          } ;
   147  		
   148          _next = function() 
   149          {
   150              ch = source.charAt(at);
   151              at += 1;
   152              return ch;
   153          } ;
   154  		
   155          _white = function () 
   156          {
   157             while (ch) 
   158             {
   159                  if (ch <= " ") 
   160                  {
   161                      _next();
   162                  }
   163                  else if (ch == "/") 
   164                  {
   165                      switch (_next()) 
   166                      {
   167                          case "/":
   168                          {
   169                              while (_next() && ch != "\n" && ch != "r") {}
   170                              break;
   171  						}
   172                          case "*":
   173                          {
   174                              
   175                              _next() ;
   176                              
   177                              for (;;) 
   178                              {
   179                                  if (ch) 
   180                                  {
   181                                  
   182                                      if (ch == "*") 
   183                                      {
   184                                          if (_next() == "/") 
   185                                          {
   186                                              _next();
   187                                              break;
   188                                          }
   189                                      } 
   190                                      else 
   191                                      {
   192                                          _next();
   193                                      }
   194                                  
   195                                  } 
   196                                  else 
   197                                  {
   198                                      _error("Unterminated Comment");
   199                                  }
   200                              }
   201                              break;
   202                          }
   203                          default:
   204                          {
   205                              _error("Syntax Error");
   206                          }
   207                      }
   208                  } 
   209                  else 
   210                  {
   211                      break ;
   212                  }
   213              }
   214          };
   215  		
   216          _string = function () 
   217          {
   218              var i, s = "", t, u;
   219  			
   220  			var outer:Boolean = false;
   221  			
   222              if (ch == "\"" || ch == "'") 
   223              {
   224              	var outerChar:String = ch ;
   225  				while (_next()) 
   226  				{
   227                      if (ch == outerChar) 
   228                      {
   229                          _next();
   230                          return s ;
   231                      }
   232                      else if (ch == "\\") 
   233                      {
   234                          switch ( _next() ) 
   235                          {
   236                          	case "b":
   237                          	{
   238                              	s += "b";
   239                              	break;
   240                          	}
   241                          	case "f":
   242                          	{
   243  	                            s += "f";
   244      	                        break;
   245                          	}
   246          	                case "n":
   247          	                {
   248              	                s += "\n";
   249                  	            break;
   250          	                }
   251                      	    case "r":
   252                      	    {
   253                          	    s += "r";
   254                              	break;
   255                      	    }
   256                          	case "t":
   257                          	{
   258                              	s += "\t";
   259                              	break;
   260                          	}
   261                          	case "u":
   262                          	{
   263                              	u = 0;
   264                              	for (i = 0; i < 4; i += 1) 
   265                              	{
   266                                  	t = parseInt(_next(), 16);
   267                                  	if (!isFinite(t)) 
   268                                  	{
   269                                      	outer = true;
   270  										break;
   271                                  	}
   272                                  	u = u * 16 + t;
   273                              	}
   274  								if(outer) 
   275  								{
   276  									outer = false;
   277  									break;
   278  								}
   279                              	s += String.fromCharCode(u);
   280                              	break;
   281                          	}
   282                          	default:
   283                          	{
   284                              	s += ch;
   285                          	}
   286                          }
   287                      } 
   288                      else 
   289                      {
   290                          s += ch;
   291                      }
   292                  }
   293              }
   294              _error("Bad String");
   295          };
   296  		
   297          _array = function() 
   298          {
   299              var a = [];
   300              if (ch == "[") 
   301              {
   302                  _next();
   303                  _white();
   304                  if (ch == "]") 
   305                  {
   306                      _next();
   307                      return a;
   308                  }
   309                  while (ch) 
   310                  {
   311                      a.push(_value());
   312                      _white();
   313                      if (ch == "]") 
   314                      {
   315                          _next();
   316                          return a;
   317                      }
   318                      else if (ch != ",") 
   319                      {
   320                          break;
   321                      }
   322                      _next();
   323                      _white() ;
   324                  }
   325              }
   326              _error("Bad Array");
   327          };
   328          
   329  		_key = function()
   330  		{
   331  	    	var s = ch;
   332  	    	var outer:Boolean = false;
   333  		
   334  			var semiColon:Number   = source.indexOf(":", at);
   335  			var quoteIndex:Number  = source.indexOf("\"", at) ;
   336  			var squoteIndex:Number = source.indexOf("'", at) ;
   337  
   338  			if( (quoteIndex <= semiColon && quoteIndex > -1) || (squoteIndex <= semiColon && squoteIndex > -1))
   339  			{
   340  				var s = _string();
   341  				_white() ;
   342  				if(ch == ":")
   343  				{
   344  					return s;
   345  				}
   346  				else
   347  				{
   348  					_error("Bad key");
   349  				}
   350  			}
   351  	
   352  			while ( _next() ) //Use key handling 
   353       	  	{
   354              	if (ch == ":") 
   355              	{
   356                  	return s;
   357              	} 
   358              	if(ch <= " ")
   359              	{
   360  				            	
   361  				}
   362              	else
   363              	{
   364  	            	s += ch;
   365              	}
   366  	    	}
   367  	    	
   368  			_error("Bad key") ;
   369  		
   370  		} ;
   371  		
   372          _object = function () 
   373          {
   374              var k, o = {} ;
   375              if (ch == "{") 
   376              {
   377                  _next();
   378                  _white();
   379                  if (ch == "}") 
   380                  {
   381                      _next();
   382                      return o;
   383                  }
   384                  while (ch) 
   385                  {
   386                      
   387                      k = _key() ;
   388                      
   389                      _white() ;
   390                      
   391                      if (ch != ":") 
   392                      {
   393                          break;
   394                      }
   395                      
   396                      _next();
   397                      
   398                      o[k] = _value() ;
   399                      
   400                      _white();
   401                      
   402                      if (ch == "}") 
   403                      {
   404                          _next();
   405                          return o;
   406                      }
   407                      else if (ch != ",") 
   408                      {
   409                          break;
   410                      }
   411            
   412                      _next();
   413                      _white();
   414            
   415                  }
   416              }
   417              _error("Bad Object") ;
   418          };
   419  		
   420          _number = function () 
   421          {
   422              
   423              var n = "" ;
   424              var v:Number ;
   425  			var hex:String = "" ;
   426  			var sign:String = "" ;
   427  			
   428              if (ch == "-") 
   429              {
   430                  n = "-";
   431                  sign = n ;
   432                  _next();
   433              }
   434              
   435              if( ch == "0" ) 
   436              {
   437          		_next() ;
   438  				if( ( ch == "x") || ( ch == "X") ) 
   439  				{
   440              		_next();
   441              		while( _isHexDigit( ch ) ) 
   442              		{
   443                  		hex += ch ;
   444                  		_next();
   445                  	}
   446              		if( hex == "" ) 
   447              		{   
   448              			_error("mal formed Hexadecimal") ;
   449  					}
   450  					else 
   451  					{
   452  						return Number( sign + "0x" + hex ) ;
   453  					}
   454              	} 
   455              	else 
   456              	{
   457  	            	n += "0" ;
   458              	}
   459  			}
   460  				
   461              while ( _isDigit(ch) ) 
   462              {
   463                  n += ch;
   464                  _next();
   465              }
   466              
   467              if (ch == ".") 
   468              {
   469                  n += ".";
   470                  while (_next() && ch >= "0" && ch <= "9") 
   471                  {
   472                      n += ch;
   473                  }
   474              }
   475              v = 1 * n ;
   476              if (!isFinite(v)) 
   477              {
   478                  _error("Bad Number");
   479              }
   480              else 
   481              {
   482                  return v;
   483              }
   484          } ;
   485  		
   486          _word = function () 
   487          {
   488              switch (ch) 
   489              {
   490                  case "t":
   491                  {
   492                      if (_next() == "r" && _next() == "u" && _next() == "e") 
   493                      {
   494                          _next();
   495                          return true;
   496                      }
   497                      break;
   498                  }
   499                  case "f" :
   500                  {
   501                      if (_next() == "a" && _next() == "l" && _next() == "s" && _next() == "e") 
   502                      {
   503                          _next();
   504                          return false;
   505                      }
   506                      break;
   507  				}
   508                  case "n":
   509                  {
   510                      if (_next() == "u" && _next() == "l" && _next() == "l") 
   511                      {
   512                          _next();
   513                          return null;
   514                      }
   515                      break;
   516  				}
   517              }
   518              _error("Syntax Error");
   519          };
   520  		
   521          _value = function () 
   522          {
   523              _white();
   524              switch (ch) 
   525              {
   526              
   527                  case "{" :
   528                      return _object();
   529                  case "[" :
   530                      return _array();
   531                  case "\"" :
   532                  case "'" :
   533                      return _string();
   534                  case "-" :
   535                      return _number();
   536                  default :
   537                      return ch >= "0" && ch <= "9" ? _number() : _word();
   538              }
   539          };
   540  		
   541          return _value() ;
   542  
   543      }
   544  	
   545  	static public var BACK_SLASH:String = "\\" ;	
   546  
   547  	/**
   548  	 * Serialize the object and return this string representation.
   549  	 * @return the string serialize representation of an object.
   550  	 */
   551  	static public function serialize(o):String 
   552  	{
   553  		
   554          var c:String ;
   555          var i:Number ;
   556          var l:Number ;
   557  		var s:String = "" ;
   558  		var v ;
   559  		
   560  		var tof:String = typeof(o) ;
   561  		
   562          switch (tof) 
   563          {
   564  
   565  			case "boolean" :
   566  			{
   567  				return String(o);
   568  			}
   569  
   570  			case "number" :
   571  			{
   572  				return isFinite(o) ? String(o) : "null" ;
   573  			}
   574  
   575  			case "object":
   576  			{
   577  				if (o) 
   578  				{
   579  					if (o instanceof Array) 
   580  					{
   581  						
   582  						l = o.length ;
   583  						
   584  						for (i = 0 ; i < l ; ++i) {
   585  						
   586  							v = serialize(o[i]);
   587  							if (s) s += "," ;
   588  							s += v ;
   589  						}
   590  						return "[" + s + "]";
   591  						
   592  					} 
   593  					else if (typeof(o.toString) != "undefined") 
   594  					{
   595  						for (var prop:String in o) 
   596  						{
   597  							v = o[prop];
   598  							if ( (typeof(v) != "undefined") && (typeof(v) != "function") ) 
   599  							{
   600  								v = serialize(v);
   601  								if (s) 
   602  								{
   603  									s += "," ;
   604  								}
   605  								s += serialize(prop) + ":" + v ;
   606  							}
   607  						}
   608  						return "{" + s + "}" ;
   609  					}
   610  				}
   611  				return "null" ;
   612          	}
   613  			
   614  			case "string" :
   615  			{
   616  				l = o.length ;
   617  				s = "\"";
   618  				for (i = 0 ; i < l ; i += 1) 
   619  				{
   620  					c = o.charAt(i);
   621  					if (c >= " ") 
   622  					{
   623  						if (c == BACK_SLASH || c == "\"") 
   624  						{
   625  							s += BACK_SLASH ;
   626  						}
   627  						s += c;
   628  					} 
   629  					else 
   630  					{
   631  						switch (c) 
   632  						{
   633  							case "b" :
   634  							{
   635  								s += "\\b" ;
   636  								break;
   637  							}
   638  							case "f" :
   639  							{
   640  								s += "\\f" ;
   641  								break;
   642  							}
   643  							case "\n" :
   644  							{
   645  								s += "\\n" ;
   646  								break;
   647  							}
   648  							case "r" :
   649  							{
   650  								s += "\\r" ;
   651  								break;
   652  							}
   653  							case "\t" :
   654  							{
   655  								s += "\\t" ;
   656  								break;
   657  							}
   658  							default :
   659  							{
   660  								var code:Number = c.charCodeAt() ;
   661  								s += "\\u00" + (Math.floor(code / 16).toString(16)) + ((code % 16).toString(16)) ;
   662  							}
   663  						}
   664  					}
   665  				}
   666  				return s + "\"";
   667          	}
   668  
   669  			default :
   670  			{
   671  				return "null" ;
   672  			}
   673  
   674  			 			
   675          }
   676          
   677      }
