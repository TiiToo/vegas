    24  import vegas.core.types.Char;
    25  
    26  /**
    27   * ECMA 262 Unicode IFormat-Control Characters tools.
    28   * <p><b>Example :</b></p>
    29   * {@code
    30   * import vegas.string.UnicodeChar ;
    31   * import vegas.core.types.Char ;
    32   *   
    33   * trace (UnicodeChar.WHITE_SPACE_CHARS.length) ;
    34   *   
    35   * var str:String = UnicodeChar.TAB + "coucou" ;
    36   * trace (str) ;
    37   *  
    38   * var s:Char = new Char('\t') ;
    39   * var b:Boolean = UnicodeChar.isWhiteSpace(s) ;
    40   * trace ("isWhiteSpace : " + b) ;
    41   *  
    42   * var unicode:String = "0040" ;
    43   * var char:Char = UnicodeChar.toChar(unicode) ;
    44   * trace(">> " + char) ;
    45   * }
    46   * @author eKameleon
    47   * @see <a href='http://www.ecma-international.org/publications/files/ECMA-ST/Ecma-262.pdf'>ECMAScript 262 specifications</a>
    48   */
    49  class vegas.string.UnicodeChar 
    50  {
    51  
    52  	/**
    53  	 * Back Slash utf8 representation (special char).
    54  	 */
    55  	static public var BACK_SLASH:String = "u005C" ;	
    56  
    57  	/**
    58  	 * Back Space utf8 representation (special char).
    59  	 */
    60  	static public var BACK_SPACE:String = "u0008" ;
    61  	
    62  	/**
    63  	 * Simple Quote utf8 representation (special char).
    64  	 */
    65  	static public var SIMPLE_QUOTE:String = "u0027" ;
    66  	
    67  	/**
    68  	 * Double Quote utf8 representation (special char).
    69  	 */
    70  	static public var DOUBLE_QUOTE:String = "u0022" ;
    71  	
    72  	/**
    73  	 * Tab utf8 representation (whitespace).
    74  	 */
    75  	static public var TAB:String = "u0009" ;
    76  
    77  	/**
    78  	 * Vertical Tab utf8 representation (whitespace).
    79  	 */
    80  	static public var VT:String = "u000B" ;
    81  	
    82  	/**
    83  	 * Form Feed utf8 representation (whitespace).
    84  	 */
    85  	static public var FF:String = "u000C" ;
    86  	
    87  	/**
    88  	 * Space utf8 representation (whitespace).
    89  	 */
    90  	static public var SP:String = "u0020" ;
    91  
    92  	/**
    93  	 * No-break space utf8 representation (whitespace).
    94  	 */
    95  	static public var NBSP:String = "u00A0" ;
    96  
    97  	/**
    98  	 * Array with all whitespace characters. NB : USP no implement (Any other Unicode "space separator")
    99  	 */
   100  	static public var WHITE_SPACE_CHARS:Array = [ TAB, VT, FF, SP, NBSP ] ;
   101  	
   102  	/**
   103  	 * Line Feed utf8 representation (line terminators).
   104  	 */
   105  	static public var LF:String = "u000A" ;
   106  
   107  	/**
   108  	 * Carriage Return utf8 representation (line terminators).
   109  	 */
   110  	static public var CR:String = "u000D" ;
   111  
   112  	/**
   113  	 * Line Separator utf8 representation (line terminators).
   114  	 */
   115  	static public var LS:String = "u2028" ;
   116  
   117  	/**
   118  	 * Paragraph Separator utf8 representation (line terminators).
   119  	 */
   120  	static public var PS:String = "u2029" ;	
   121  
   122  	/**
   123  	 * Array with all line terminators characters
   124  	 */
   125  	static public var LINE_TERMINATOR_CHARS:Array = [ LF, CR, LS, PS ] ;
   126  	
   127  	
   128  	static private var __ASPF__ = _global.ASSetPropFlags(UnicodeChar, null , 7, 7) ;
   129  	
   130  	/**
   131  	 * Returns {@code true} of the specified character is a whitespace.
   132  	 * @return {@code true} of the specified character is a whitespace.
   133  	 */
   134  	static public function isWhiteSpace( char:String ):Boolean 
   135  	{
   136  		var c:Char = new Char(char) ;
   137  		var ar:Array = WHITE_SPACE_CHARS ;
   138  		var l:Number = ar.length ;
   139  		while (--l > -1) 
   140  		{
   141  			if (ar[l] == char) 
   142  			{
   143  				return true ;
   144  			}
   145  		}
   146  		return false ;
   147  	}
   148  
   149  	/**
   150  	 * Returns {@code true} of the specified character is a line terminator.
   151  	 * @return {@code true} of the specified character is a line terminator.
   152  	 */
   153  	static public function isLineTerminators( char:String ):Boolean 
   154  	{
   155  		var c:Char = new Char(char) ;
   156  		var ar:Array = LINE_TERMINATOR_CHARS ;
   157  		var l:Number = ar.length ;
   158  		while (--l > -1) 
   159  		{
   160  			if (ar[l] == char) 
   161  			{
   162  				return true ;
   163  			}
   164  		}
   165  		return false ;
   166  	}
   167  
   168  	/**
   169  	 * Converts a unicode representation and returns this char's string.
   170  	 * @return The char of the unicode representation.
   171  	 */
   172  	static public function toChar( unicode:String ):Char 
   173  	{
   174  		return new Char(String.fromCharCode(parseInt( unicode, 16))) ;
   175  	}
   176  
   177  	/**
   178  	 * Converts an unicode number value and returns this string representation.
   179  	 * @return The string representation of a unicode number.
   180  	 */
   181      static public function toUnicode(n:Number):String 
   182      {
   183  		var hex:String = n.toString(16) ;
   184  		while( hex.length < 4 )
   185  		{
   186  			hex = "0" + hex ;
   187  		}
   188  		return hex ;
   189  	}
   190  
   191  }
