    40  	/**
    41  	 * Constructs a new {@code BasicEvent} instance.
    42  	 * <p>
    43  	 * {@code
    44  	 *     var e:BasicEvent = new BasicEvent(type, target, context, [bubbles:Boolean, [eventPhase:Number, [time:Number, [stop:Boolean]]]]) ;
    45  	 * }
    46  	 * </p>
    47  	 * @param type the string type of the instance. 
    48  	 * @param target the target of the event.
    49  	 * @param context the optional context object of the event.
    50  	 * @param bubbles indicates if the event is a bubbling event.
    51  	 * @param eventPhase the current EventPhase of the event.
    52  	 * @param time this parameter is used in the Eden deserialization.
    53  	 * @param stop this parameter is used in the Eden deserialization.
    54  	 */
    55  	public function BasicEvent ( type:String , target, context, bubbles:Boolean, eventPhase:Number, time:Number, stop:Number ) 
    56  	{
    57  		
    58  		_type = (getType() != null) ? getType() : type ;
    59  		_context = (getContext() != null) ? getContext() : context ;
    60  		_target = (getTarget() != null) ? getTarget() : target ;
    61  		_bubbles = ( bubbles != null) ? bubbles : true ;
    62  		_eventPhase = isNaN( eventPhase) ? EventPhase.AT_TARGET : eventPhase ;
    63  		_time = ( time > 0) ? time : ( (new Date()).valueOf() ) ;
    64  		
    65  		this.stop = isNaN( stop) ? EventPhase.NONE : stop ;
    66  		
    67  	}

   264  	/**
   265  	 * Returns the string representation of this event.
   266  	 * @return the string representation of this event.
   267  	 */
   268  	public function toString():String 
   269  	{
   270  		var phase:Number = getEventPhase() ;
   271  		var name:String = ConstructorUtil.getName(this);
   272  		var txt:String = "[" + name ;
   273  		if (getType()) txt += " " + getType() ;
   274  		switch (phase) 
   275  		{
   276  			case EventPhase.CAPTURING_PHASE :
   277  			{
   278  				txt += ", CAPTURING" ;
   279  				break;
   280  			}
   281  			case EventPhase.AT_TARGET:
   282  			{
   283  				txt += ", AT TARGET" ;
   284  				break ;
   285  			}
   286  			case EventPhase.BUBBLING_PHASE:
   287  			{
   288  				txt += ", BUBBLING" ;
   289  				break ;
   290  			}
   291  			default :
   292  			{
   293  				txt += ", (inactive)" ;
   294  				break;
   295  			}
   296  		}
   297  		if (getBubbles() && phase != EventPhase.BUBBLING_PHASE) 
   298  		{
   299  			txt += ", bubbles" ;
   300  		}
   301  		if (isCancelled()) 
   302  		{
   303  			txt += ", can cancel" ;
   304  		}
   305  		txt += "]" ;
   306  		return txt ;
   307  	}
