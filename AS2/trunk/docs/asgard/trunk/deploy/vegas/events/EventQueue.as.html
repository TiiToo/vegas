    24  import vegas.core.CoreObject;
    25  import vegas.data.iterator.Iterator;
    26  import vegas.data.Queue;
    27  import vegas.data.queue.LinearQueue;
    28  import vegas.events.Event;
    29  import vegas.util.TypeUtil;
    30  
    31  /**
    32   * Internal class used in {@code EventDispatcher} class to bufferize the events if no EventListener are registered with the event type of the event.
    33   */
    34  class vegas.events.EventQueue extends CoreObject  
    35  {
    36  
    37  	/**
    38  	 * Creates a new EventQueue instance.
    39  	 */
    40  	public function EventQueue() 
    41  	{
    42  		_events = new LinearQueue() ;
    43  	}
    44  
    45  	/**
    46  	 * Enqueue an event in the buffer if no EventListener are registered in the EventListener.
    47  	 */
    48      public function enqueue( e:Event ):Void 
    49      {
    50          e.queueEvent() ;
    51          _events.enqueue(e);
    52      }
    53  
    54  	/**
    55  	 * Returns a queue with all events bufferized.
    56  	 * @return a queue with all events bufferized.
    57  	 */
    58      public function getQueuedEvents( ):Queue 
    59      {
    60      	var arg = arguments[0] ;
    61  		if ( TypeUtil.typesMatch( arg , String ) ) 
    62  		{
    63  			var q:LinearQueue = new LinearQueue() ;
    64  			var eventType:String = arguments[0] ;
    65  			var it:Iterator = _events.iterator() ;
    66  			while (it.hasNext()) 
    67  			{
    68  				var e:Event = it.next() ;
    69  				if (e.getType() == eventType) 
    70  				{
    71  					q.enqueue(e) ;
    72  				}
    73  			}
    74  			return q ;
    75  		}
    76  		else 
    77  		{
    78  			return _events ;
    79  		}
    80      }
    81      
    82      /**
    83       * The size of this buffer.
    84       */
    85  	public function size():Number 
    86  	{
    87  		return _events.size() ;
    88  	}
    89  
    90  	/**
    91  	 * Internal LinearQueue used by this this instance.
    92  	 */
    93      private var _events:LinearQueue = null ;
    94      
    95  }
